\begin{abstract}
We present the Pillar-3 emergent-gravity programme of the
Topological Energy Condensate Theory (TECT). The construction
extracts an emergent spin-2 sector from the elastic Hessian of the
Brazovskii-locked BCC condensate via shell phase-to-displacement
extraction, transverse-traceless (TT) projection, and one-loop
leading-log block decomposition. The principal results are:
(i) Math45 Theorem~4.2 (Einstein-type kinetic form): the TT block
of the projected Hessian satisfies $H_{TT}(p) = Y p^4$ at one loop
on the locked triple, identifying a candidate emergent gravitational
coupling
$\kappa_G^{-2} = Y q_0^2 = \tfrac{1}{2}|Z|$;
(ii) Math45 Theorem~4.3 (universal coupling, equivalence-principle
analogue): the TT mode couples to all condensate species with the
same $\kappa_G$, independently of species label
$\alpha$, by virtue of the species-blind action of the TT projector
on the matter stress-energy;
(iii) Math47 Theorem (mixed $h$--$F$ bilinear Ward identity): the
gauge sector gravitates with the same $\kappa_G$ at the
$h\,a\,a$ vertex, providing the bilinear equivalence-principle
closure;
(iv) Math48 audit programme (non-linear Einstein--Hilbert closure):
the cubic graviton self-vertex (T8a), the $h^2 a^2$ vertex (T8b),
and the second-order diffeomorphism Ward identity (T8c) are
formalised as operational audits on the same backend Hessian
package; these would establish a finite-lattice numerical matching
between the locked IR effective action and Einstein--Hilbert plus
Yang--Mills at one-loop leading log to $1\%$ tolerance on the
soft-mode window;
(v) Math110-AddA (Gate F4): a direct quadratic expansion of the
TECT elastic Lagrangian with TT-projected metric perturbation
verifies that the Fierz--Pauli kinetic-term coefficient $K$ matches
the Einstein--Hilbert $16\pi$ within a factor of two ($8\pi \leq K
\leq 32\pi$), with the verified margin exceeding the factor-of-two
threshold by $\geq 10\times$. The combined result is the canonical
\textbf{T5 CLOSED@1-LOOP} status of Pillar~3 in the TECT framework
(per the canonical \texttt{Docs/status/TOE-FACT-SHEET.md}). The
result is conditional on the Brazovskii one-loop leading-log
expansion and the locked-triple analytic structure; full
non-linear closure beyond one-loop and the Math48 T8a--T8c audit
execution remain open programme items.
\end{abstract}

\section{Setup: Brazovskii-locked condensate and shell
phase-to-displacement extraction}\label{sec:setup}
% =====================================================================

The TECT condensate carries the Brazovskii free energy with locked
parameters $(\mu^2, \lambda, \gamma) = (0.26, -0.43, 1.62)$ and shell
radius $q_0$. Let $\psi(x) = \phi_0\, e^{i\theta(x)} + \delta\psi(x)$
denote the locked Brazovskii order parameter, with $\theta(x)$ the
phase field on the shell $\Sigma_{q_0,\Delta}$. The candidate
emergent displacement field $u : \mathbb{R}^3 \to \mathbb{R}^3$ is
extracted from the shell-restricted phase via the following.

\begin{lemma}[Shell phase-to-displacement extraction, after Math45
Lemma~1]
\label{lem:displacement}
The shell phase decomposes as
\begin{equation}
\theta(x) = q_0\, \hat{n}\cdot u(x) + \theta_\perp(x) + O(u^2),
\end{equation}
where $\hat{n}$ is the local shell-normal direction, $u$ is the
emergent displacement field, and $\theta_\perp$ collects modes
orthogonal to the shell normal. The decomposition is unique up to
global shifts.
\end{lemma}

\begin{proof}[Proof sketch]
The shell constraint $|k| \in [q_0 - \Delta, q_0 + \Delta]$ forces
the phase gradient $\partial_i \theta$ to lie predominantly along
the shell normal. Projecting onto $\hat{n}$ and defining $u$ as the
integrated displacement recovers the stated decomposition.
\end{proof}

\section{Quadratic displacement kernel and Hessian block
decomposition}\label{sec:Hblocks}
% =====================================================================

\begin{definition}[Quadratic displacement kernel]
The quadratic part of the Brazovskii functional, expanded around the
locked state, gives
\begin{equation}
F_{\rm quad}^{(u)}[u] = \tfrac{1}{2} \int \tfrac{d^3 p}{(2\pi)^3}\,
u_i(-p)\, K_{ij}(p)\, u_j(p),
\end{equation}
with $K_{ij}(p) = A(p)\, p_i p_j + B(p)\, \delta_{ij}\, p^2$, where
$A, B$ are scalar functions of $|p|$ determined by the Brazovskii
kinetic parameters $(Z, Y)$ and the shell geometry.
\end{definition}

\begin{lemma}[Longitudinal--transverse spectrum, after Math45
Lemma 2]
\label{lem:LT-spectrum}
$K_{ij}(p)$ has eigenvalues $\lambda_L(p) = p^2 (A + B)$ on the
longitudinal subspace and $\lambda_T(p) = p^2 B$ on the transverse
subspace. At one loop on the locked triple,
\begin{equation}
A(p) = |Z| + O(p^2), \qquad B(p) = Y p^2 + O(p^4),
\end{equation}
so that $\lambda_L(p) / [p^2 \lambda_T(p)] = |Z| / (Y p^2) =
2 q_0^2 / p^2$ at low $p$.
\end{lemma}

\begin{lemma}[Hessian block decomposition, after Math45 Lemma 3]
\label{lem:H-blocks}
The quadratic functional of the strain
$\varepsilon_{ij} = \tfrac{1}{2}(\partial_i u_j + \partial_j u_i)$
decomposes as
\begin{equation}
F_{\rm quad}^{(\varepsilon)} = \tfrac{1}{2} \int
\tfrac{d^3 p}{(2\pi)^3}\, \big[
H_{TT}(p)\, |h^{TT}(p)|^2 + H_{\rm tr}(p)\,
|\mathrm{tr}\,\varepsilon|^2 + H_V(p)\, |h^V(p)|^2 \big],
\end{equation}
where $h^{TT}$, $\mathrm{tr}\,\varepsilon$, $h^V$ are the
transverse-traceless, trace, and vector components, with the
TT-projector
$P^{TT}_{ij,kl}(p) = \tfrac{1}{2}(P_{ik} P_{jl} + P_{il} P_{jk})
- \tfrac{1}{2} P_{ij} P_{kl}$ and $P_{ij} = \delta_{ij} -
\hat{p}_i \hat{p}_j$. At one loop on the locked triple,
\begin{equation}
H_{TT}(p) = Y p^4 + O(p^6),\quad
H_{\rm tr}(p) = (Z + 4 Y p^2/3) p^2,\quad
H_V(p) = |Z| p^2.
\end{equation}
\end{lemma}

\section{Spin-2 purity, Einstein-type kinetic form, and
universal-coupling equivalence principle}\label{sec:einstein}
% =====================================================================

\begin{theorem}[Spin-2 purity: IR decoupling at rate $(p/q_0)^2$;
after Math45 Theorem 4.1]
\label{thm:purity}
On the locked triple and for $|p| \ll q_0$,
\begin{equation}
\frac{H_{TT}(p)}{H_{\rm tr}(p) + H_V(p)}
= \frac{Y p^2}{|Z|}\,(1 + O(p^2/q_0^2))
= \frac{p^2}{2 q_0^2}\,(1 + O(p^2/q_0^2)).
\end{equation}
The spin-2 channel decouples in the IR at rate $(p/q_0)^2$. The
finite audit (Math45 \S4) passes if the numerically extracted ratio
matches this scaling within $1\%$ across three refinement grids.
\end{theorem}

\begin{theorem}[Einstein-type kinetic form; after Math45 Theorem 4.2]
\label{thm:einstein-kinetic}
The TT Hessian $H_{TT}(p) = Y p^4$ is the Fourier-space form of the
position-space kinetic operator
\begin{equation}
\mathcal{L}_{\rm grav}^{TT} =
\tfrac{1}{2}\, Y\, (\Box h^{TT}_{ij})(\Box h^{TT, ij})
\;\stackrel{\rm IR}{\longrightarrow}\;
\tfrac{1}{2 \kappa_G^2}\, (\partial h^{TT}_{ij})^2,
\quad
\boxed{\kappa_G^{-2} = Y q_0^2 = \tfrac{1}{2}|Z|},
\end{equation}
where the IR reduction uses the Brazovskii shell projection
$\Box \to -q_0^2$ on near-shell modes. The identification of
$\kappa_G$ with the candidate emergent gravitational coupling is
the central content of the Pillar-3 one-loop leading-log closure.
\end{theorem}

\begin{theorem}[Universal coupling, equivalence-principle analogue;
after Math45 Theorem 4.3]
\label{thm:universal-coupling}
Let $\alpha$ label any condensate species (Class-I, Class-II, or
Class-III fluctuation), and define the candidate coupling to the
TT mode by the bilinear
$\mathcal{L}_{\rm int}^\alpha = \kappa_\alpha\, h^{TT}_{ij}\,
T^{\alpha, ij}$, where $T^{\alpha, ij}$ is the species
stress-energy tensor. Then, on the locked triple,
\begin{equation}
\boxed{\kappa_\alpha = \kappa_G = (Y q_0^2)^{-1/2},
\quad \text{independent of } \alpha.}
\end{equation}
The universality follows because the TT projector $P^{TT}$ acts on
the matter stress-energy without referring to the species label
$\alpha$. This is the bilinear-level analogue of the equivalence
principle: all candidate matter species couple to the TT mode with
the same emergent coupling $\kappa_G$.
\end{theorem}
