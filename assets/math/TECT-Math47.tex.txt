\section*{Math47: Mixed metric--gauge bilinear response and the
equivalence-principle closure of $(\kappa_G,c_W,c_B)$}

\subsection*{0. Motivation}

The Math45 / Math46c audit extracts the graviton kinetic
coefficient $\kappa_G^{-2}=Yq_0^{2}=|Z|/2$ from the transverse-
traceless block of the projected Hessian with symmetric-tensor
probes acting on the locked condensate. The Math44 / Math46b
audit extracts the Yang--Mills coefficients
$c_W=1/(96\pi^{2})$, $c_B=1/(64\pi^{2})$ from gauge-plane-wave
probes of the one-loop $\tfrac12\mathrm{Tr}\log$. Neither audit,
taken separately, establishes that the emergent $SU(2)_W\times
U(1)_Y$ gauge fields couple to the emergent graviton with the
universal strength $\kappa_G$ demanded by the equivalence
principle. The universal coupling must be read off the
\emph{mixed} $\partial^{2}S_{\mathrm{eff}}/\partial\varepsilon_h
\,\partial\varepsilon_a$ bilinear vertex at $a=0,h=0$.

\subsection*{1. Mixed probe on the locked vacuum}

\begin{defn}[Mixed metric--gauge probe]
\label{defn:mixed-probe}
Fix a probe momentum $p$ with $|p|\ll q_0$, orthonormal triad
$(\hat p,\hat e_1,\hat e_2)$, and TT polarisation
$E\in\{E^{(TT1)},E^{(TT2)}\}$ (Math46c Def probe-basis). Fix a
gauge polarisation $(\hat{\epsilon}^{a}_\mu,\hat{\epsilon}^{0}_\mu)$
with $\hat\epsilon\cdot p=0$ and a second momentum $q$ with
$|q|\lesssim q_0$. Define the two-parameter deformation of the
locked package:
\begin{equation}
\Psi_{\varepsilon_h,\varepsilon_a}(x)
 \;:=\;
 \Psi_{\mathrm{lock}}\!\bigl((I+\varepsilon_h E\cos(p\!\cdot\!x))\,x\bigr)
 \,\mathrm{Pexp}\!\left[
   i\,\varepsilon_a\!\int^{x}\! a_\mu(y)\,dy^{\mu}
 \right],
\label{eq:mixed-def}
\end{equation}
with $a_\mu(y)=[\hat\epsilon^{a}_\mu T^{a}+
\hat\epsilon^{0}_\mu Y]\cos(q\cdot y)$.
\end{defn}

\begin{defn}[Mixed bilinear vertex]
\label{defn:Gamma3}
Let $S_{\mathrm{eff}}$ be the Math46b $\tfrac12\mathrm{Tr}\log$
effective action on the Brazovskii shell. The mixed bilinear
response at the locked vacuum is
\begin{equation}
\Gamma^{(h,a,a)}(p;q_1,q_2)
 \;:=\;
 \frac{1}{V}\,
 \frac{\partial^{3}S_{\mathrm{eff}}}
      {\partial\varepsilon_h\,\partial\varepsilon_a(q_1)\,
       \partial\varepsilon_a(q_2)}
 \Bigg|_{\varepsilon_h=\varepsilon_a=0},
\label{eq:Gamma3-def}
\end{equation}
evaluated with momentum conservation $p+q_1+q_2=0$.
\end{defn}

\begin{rmk}
$\Gamma^{(h,a,a)}$ is the on-shell three-point function between
one graviton polarisation and two gauge-probe polarisations;
it is the lowest-order vertex sensitive to \emph{how strongly}
the gauge sector gravitates.
\end{rmk}

\subsection*{2. Ward identity and vertex structure}

\begin{lem}[Diffeomorphism Ward identity]
\label{lem:Ward}
Under the infinitesimal diffeomorphism
$x^{\mu}\to x^{\mu}+\xi^{\mu}(x)$ with
$\xi^{\mu}(x)=\zeta^{\mu}\cos(p\!\cdot\!x)$, $\zeta\cdot p=0$,
the effective action satisfies
\begin{equation}
\delta_{\xi}S_{\mathrm{eff}}[a,h]
 \;=\;-\int\!d^{3}x\;\xi^{\mu}\,\partial^{\nu}T^{(a)}_{\mu\nu}[a]
 \;+\;O(\xi^{2}),
\end{equation}
where $T^{(a)}_{\mu\nu}[a]$ is the gauge-sector stress tensor
determined by Math44 \S 4 (and depending on $(c_W,c_B)$ through
its normalisation).
\end{lem}

\begin{proof}[Proof sketch]
The locked vacuum preserves three-translation invariance; the
diffeomorphism $\xi$ acts on $\Psi_{\mathrm{lock}}$ as a pure
coordinate reparametrisation and therefore leaves the
family-and-locking part of $S_{\mathrm{eff}}$ invariant. The
gauge-source part transforms covariantly; integrating by parts
and using on-shell gauge invariance yields the stated divergence
of $T^{(a)}_{\mu\nu}$.
\end{proof}

\begin{prop}[Forced vertex structure]
\label{prop:vertex}
The bilinear vertex (\ref{eq:Gamma3-def}) is forced by
Lem~\ref{lem:Ward} into the form
\begin{equation}
\Gamma^{(h,a,a)}_{\alpha,\hat\epsilon_1,\hat\epsilon_2}(p;q_1,q_2)
 \;=\;
 \kappa_G^{\mathrm{mix}}\,
 E^{(\alpha)}_{\mu\nu}\,
 T^{(a),\mu\nu}_{\hat\epsilon_1,\hat\epsilon_2}(q_1,q_2)
 \,+\,O(p^{2}),
\label{eq:vertex-form}
\end{equation}
with $T^{(a)}_{\mu\nu}$ the on-shell gauge stress tensor
normalised by $(c_W,c_B)$ of Math44, and $\kappa_G^{\mathrm{mix}}$
the effective gravitational coupling read off the mixed vertex.
\end{prop}

\begin{thm}[Equivalence-principle identity]
\label{thm:EP}
On the Math40 locked triple, at the Math43 matching scale
$\ell^{\star}=1$, with $(c_W,c_B)$ satisfying Math44 Thm
\textnormal{cWcB} and $Z_h=Yq_0^{2}=|Z|/2$ satisfying Math45
Thm \textnormal{kappaG},
\begin{equation}
\kappa_G^{\mathrm{mix}} \;=\; \kappa_G \;\equiv\;
\sqrt{2/|Z|}.
\label{eq:EP}
\end{equation}
\end{thm}

\begin{proof}
Equation~(\ref{eq:mixed-def}) places the gauge plane wave on
the $(I+\varepsilon_h E)$-deformed lattice. Expanding
$\tfrac12\mathrm{Tr}\log[-D_a^{2}+U''(\rho_D)]$ simultaneously
in $\varepsilon_h,\varepsilon_a$ and projecting onto the TT
sector:
\[
\frac{\partial^{3}S_{\mathrm{eff}}}
     {\partial\varepsilon_h\partial\varepsilon_a^{2}}
 =
 \frac{\partial^{2}}{\partial\varepsilon_a^{2}}
 \Bigl[\,
   \tfrac12\mathrm{Tr}\log\!\bigl(-D_a^{2}+U''(\rho_D)\bigr)
   \Bigr|_{g=I+\varepsilon_h E}
 \Bigr]_{\!\varepsilon_a=0}.
\]
The derivative with respect to the background metric is, by the
standard heat-kernel identity, the trace of the stress tensor
weighted by $E$:
\[
\frac{\partial S_{\mathrm{eff}}[a]}{\partial g^{\mu\nu}}
 = \tfrac12\,T^{(a)}_{\mu\nu}[a].
\]
Contracting with $E^{\mu\nu}$ and comparing to the pure-$h$
TT kinetic term $Z_h p^{2}\,(E\!\cdot\!E)$ of Math46c T2 gives
$\kappa_G^{\mathrm{mix}}=\kappa_G$ to $O(p^2)$ corrections.
\end{proof}

\subsection*{3. Operational audit T7}

\begin{defn}[Mixed-response coefficient]
\label{defn:kappa-mix}
Let $E=E^{(TT1)}$ and $\hat\epsilon_{1,2}$ be transverse gauge
polarisations. Then
\begin{equation}
\kappa_G^{\mathrm{mix}}
 \;:=\;
 \lim_{p,q\to 0}\,
 \frac{\Gamma^{(h,a,a)}_{TT,\hat\epsilon_1,\hat\epsilon_2}(p;q,-p-q)}
      {E^{\mu\nu}\,T^{(a)}_{\mu\nu,\hat\epsilon_1,\hat\epsilon_2}
         (q,-p-q)}.
\label{eq:kappa-mix-def}
\end{equation}
The denominator is evaluated with the Math46b-extracted
$(c_W,c_B)$, not with their Math44 target values (so T7 is not
circular).
\end{defn}

\begin{defn}[T7: equivalence-principle closure]
\label{defn:T7}
The \emph{equivalence-principle defect} is
\begin{equation}
\Delta_{\mathrm{EP}}
 \;:=\;
 \frac{\lvert\kappa_G^{\mathrm{mix}}-\kappa_G\rvert}
      {\kappa_G}
 \;\stackrel{!}{\le}\;10^{-2},
\qquad
\kappa_G\;=\;\sqrt{2/|Z|}.
\end{equation}
\end{defn}

\begin{prop}[T7 failure modes]
\label{prop:T7fail}
T7 is falsified on refined grids $N\in\{32,64,128\}$ if any of
the following occurs:
\begin{enumerate}
\item[(E1)] $\kappa_G^{\mathrm{mix}}(N)$ does not converge to
      $\kappa_G$ at rate $\ge O(N^{-2})$.
\item[(E2)] The mixed vertex has a non-Ward component
      $\Gamma^{(h,a,a)}-\kappa_G^{\mathrm{mix}}\,E\!\cdot\!T^{(a)}$
      whose Frobenius norm exceeds $10^{-2}$ of the leading
      term.
\item[(E3)] $\kappa_G^{\mathrm{mix}}$ extracted from the
      $SU(2)_W$ sector and from the $U(1)_Y$ sector differ by
      more than $1\%$ after polarisation averaging.
\end{enumerate}
Any of (E1)--(E3) refutes the equivalence principle as an
emergent TECT statement and therefore invalidates the joint
Math44/Math45 closure.
\end{prop}

\subsection*{4. Joint extractor interface}

T7 is not computable from either the Math46c C2 extractor or
the Math46b C3 extractor alone. A faithful Math47-compliant
\emph{joint} extractor (tag: C2--C3 mixed) must expose:

\begin{description}
\item[\textbf{J1.}] Load the same solver package (Math46c P1,
Math46b E1).
\item[\textbf{J2.}] Load the same actual backend adapter
(Math46c P2, Math46b E5).
\item[\textbf{J3.}] Construct the mixed probe family
$\{v_{E,p}\otimes a_\mu(q)\}$ by composing the geometric
affine deformation (Math46c Def tangent) with the path-ordered
gauge exponential (Math46b Eq.~$F_a$-def). At $O(\varepsilon_h
\varepsilon_a^{2})$ the composition reduces to FFT-resampled
plane waves in both $p$ and $q$.
\item[\textbf{J4.}] Compute
$\partial^{3}S_{\mathrm{eff}}/\partial\varepsilon_h
\partial\varepsilon_a^{2}$ by a three-point finite difference
with $\varepsilon_h=\varepsilon_a=10^{-5}$, using Hutchinson
stochastic-trace-log with $N_s\sim 32$ samples on the shell
$\Sigma_{q_0,\Delta}$.
\item[\textbf{J5.}] Extract $\kappa_G^{\mathrm{mix}}$ via
Def~\ref{defn:kappa-mix} using the Math46b-extracted
$(c_W,c_B)$ in the denominator, and emit $\Delta_{\mathrm{EP}}$
plus Ward-residual diagnostics per Prop~\ref{prop:T7fail}.
\end{description}

\subsection*{5. Consistency with Math40--Math46c}

\begin{thm}[Math47 closure with locked triple]
\label{thm:M47-closure}
Under Math40 locked $(\mu^{2},\lambda,\gamma)=(0.26,-0.43,+1.62)$,
$(Z,Y)=(-1,+\tfrac12)$, $\ell^{\star}=1$:
\begin{equation}
\kappa_G \;=\; \sqrt{2/|Z|}\;=\;\sqrt{2},
\qquad
\kappa_G^{\mathrm{mix}}/\kappa_G \;=\; 1 \pm O(N^{-2}).
\end{equation}
Conversely, if the Math46b audit yields $(c_W,c_B)$ at their
Math44 values and the Math46c audit yields $Z_h=|Z|/2$, but T7
reports $\Delta_{\mathrm{EP}}>10^{-2}$ in the continuum limit,
then one of the two individual audits is secretly absorbing a
compensating error (a ``non-minimal coupling'' artefact) and
both claims must be revisited.
\end{thm}

\subsection*{6. Cost budget}

Per probe pair $(p,q)$: one $O(\varepsilon^{0})$ + three
$O(\varepsilon)$ evaluations of $\tfrac12\mathrm{Tr}\log
L_{\mathrm{full}}$ with Hutchinson $N_s\sim 32$. Multiply by
$n_{\mathrm{pol}}=3$ (TT pair + two TT rotations of $p$) and
$n_q=3$ (three transverse gauge polarisations $\hat\epsilon$)
and $n_{\mathrm{grid}}=3$ ($N\in\{32,64,128\}$):
\begin{equation}
C_{\mathrm{T7}}
 \;\approx\;4\cdot N_s\cdot n_{\mathrm{pol}}\cdot n_q
 \cdot n_{\mathrm{grid}}\cdot C_{\mathrm{Hess}}(N)
 \;\approx\;1150\cdot C_{\mathrm{Hess}}(N).
\end{equation}
At $N=128$ on a single A100 this is $\sim 20$--$24$ h
wall-clock. Combined with Math46b ($\sim 3$--$4$ h) and Math46c
($\sim 0.5$ h), the full closure audit fits in a two-day
campaign on one A100.

\subsection*{7. What remains (Math47 residual)}

(a) The Hutchinson variance on a third-order derivative scales
worse than on the second-order derivative of Math46b; the
stochastic sample budget $N_s$ may need to be increased to
$\sim 64$--$128$ for the $1\%$ audit threshold. A variance
pilot at $N=32$ is cheap and should precede the $N=128$ run.

(b) The gauge stress tensor $T^{(a)}_{\mu\nu}$ on a curved
on-grid metric requires the covariant plaquette definition of
Math46b E5 evaluated on the deformed lattice
$(I+\varepsilon_h E)$; the leading-order expansion is well-
defined but the error cancellation at
$O(\varepsilon_h^{2},\varepsilon_a^{2})$ must be verified
empirically.

(c) Math47 establishes closure at the one-loop bilinear level
only. Non-linear $(h^{2} a^{2})$ and $(h a^{4})$ vertices --
graviton self-coupling and higher gauge--graviton mixing --
are deferred to Math48 (non-linear Einstein--Hilbert closure).
