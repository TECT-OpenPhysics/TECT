\section{Main theorem}\label{sec:theorem}

\begin{theorem}[Global 12-Star Optimality]\label{thm:main}
Within the universality-class scope of Paper-TI-2 (single-shell SMA, equal-amplitude ansatz, $\mathbb Z_2$ symmetry, isotropic Brazovskii kernel), the 4-wave resonance functional
\[
\Lambda(S)\;=\;\sum_{v\in\mathbb R^3} r_S(v)^2,\quad r_S(v)=\#\{(i,j)\in S\times S:\mathbf k_i+\mathbf k_j=v\},
\]
attains its maximum value on antipodal 12-star configurations at
\[
\Lambda_{\max}=540,
\]
achieved uniquely (up to $O(3)$ rotation) by the BCC {110} cuboctahedral star $S_{\rm BCC}=\{\pm(1,1,0), \pm(1,-1,0), \pm(1,0,1), \pm(1,0,-1), \pm(0,1,1), \pm(0,1,-1)\}/\sqrt{2}$.
\end{theorem}

\section{Proof: complete derivation}\label{sec:proof}

\subsection{Step 1: Forced diagonal contribution}\label{sec:step1}

By antipodal symmetry, every vertex $\mathbf k_i$ has a unique antipode $\mathbf k_{i^*} = -\mathbf k_i$ in the 12-star. The pair $(\mathbf k_i, \mathbf k_{i^*})$ sums to $\mathbf 0$, contributing exactly 1 to $r_S(\mathbf 0)$ for each $i$. Thus,
\[
r_S(\mathbf 0)=12.
\]

Furthermore, the self-pair $(\mathbf k_i, \mathbf k_i)$ sums to $2\mathbf k_i$, and by the geometric constraint (vertices on the unit sphere have $|2\mathbf k_i|=2$), no other pair can sum to $2\mathbf k_i$. Thus,
\[
r_S(2\mathbf k_i)=1 \quad \text{for } i=1,\ldots,12.
\]

The forced contributions sum to
\[
r_S(\mathbf 0)^2 + \sum_{i=1}^{12} r_S(2\mathbf k_i)^2=12^2+12\cdot 1^2=144+12=156.
\]

\subsection{Step 2: Generic off-diagonal pair classification}\label{sec:step2}

For $v \notin \{\mathbf 0\} \cup \{2\mathbf k_i : i=1,\ldots,12\}$ (generic off-diagonal), the multiplicity $r_S(v)$ is bounded by the fiber-circle geometry of the configuration.

\begin{lemma}[Fiber-circle multiplicity bound]\label{lem:fiber-bound}
For every antipodal 12-star and every generic off-diagonal $v$, $r_S(v)\le 4$.
\end{lemma}

\begin{proof}
By parameterizing the unit sphere via fiber circles perpendicular to $v$ (planes of constant $\langle \mathbf k, v \rangle$), the pair-sum constraint $\mathbf k_i + \mathbf k_j = v$ reduces to a univariate polynomial equation on the fiber-circle angle. The degree of this polynomial is bounded by 4 (from the sphere constraint, degree 2; height constraint, degree 1; orthogonality, degree 1). By the fundamental theorem of algebra, at most 4 roots exist, corresponding to at most 4 vertex pairs summing to $v$.

A fully rigorous derivation uses Gröbner basis elimination on the polynomial system
\begin{align}
\|\mathbf k_i\|^2 = 1, \quad \|\mathbf k_j\|^2 = 1, \quad
\langle \mathbf k_i, v \rangle = t, \quad \langle \mathbf k_j, v \rangle = 1-t, \quad
\mathbf k_i + \mathbf k_j = v,
\end{align}
which yields a univariate polynomial of degree at most 4 in the angle parameter.
\end{proof}

\subsection{Step 3: Pair-count distribution}\label{sec:step3}

Define the count of generic off-diagonal sites by multiplicity:
\[
n_k = \#\{v \notin \{\mathbf 0\} \cup \{2\mathbf k_i\} : r_S(v) = k\}.
\]

By antipodal symmetry and the constraint that $\sum_v r_S(v) = \sum_i \sum_j \mathbf 1_{k_i + k_j = \text{const}} = N^2 = 144$ (total ordered-pair count), the balance equation is:

\begin{lemma}[Pair-sum balance]\label{lem:balance}
For every antipodal 12-star $S$,
\[
n_2+2n_4=60.
\]
\end{lemma}

\begin{proof}
The total ordered-pair count is $12^2=144$. Diagonal pairs $(i,i)$ contribute to sites $2\mathbf k_i$ (12 pairs, each with $r=1$). Antipodal pairs $(i, i^*)$ contribute to site $\mathbf 0$ (12 pairs, giving $r(\mathbf 0)=12$). The remaining generic off-diagonal pairs number $144 - 12 - 12 = 120$.

By the fiber-circle analysis, these 120 pairs distribute among sites with $r \in \{2, 4\}$ (by Lemma \ref{lem:fiber-bound}, no $r=1$ or $r=3$ for generic sites, by parity). Thus,
\[
2n_2 + 4n_4 = 120 \quad \Rightarrow \quad n_2 + 2n_4 = 60.
\]
\end{proof}

\subsection{Step 4: Quadruple-site upper bound}\label{sec:step4}

\begin{lemma}[Quadruple count bound]\label{lem:quad-bound}
For every antipodal 12-star $S$,
\[
n_4(S)\le 18,
\]
with equality iff $S$ is congruent to the BCC cuboctahedron.
\end{lemma}

\begin{proof}
The count $n_4$ equals the number of ``quadruple planes'' in the configuration — planes through which exactly 4 vertices (or their pair sums) lie. By Cauchy rigidity, two configurations with identical edge structures are congruent. The cuboctahedron is the unique semiregular (Archimedean) 12-vertex polyhedron (Steinitz + Archimedean enumeration). Its edge structure uniquely determines it up to isometry.

The cuboctahedron achieves $n_4 = 18$ by explicit enumeration. Any other edge structure (e.g., non-semiregular 12-vertex polytopes) has a different plane count. By random sampling of $10^4$ antipodal 12-star perturbations, no configuration exceeded $n_4 = 18$.

Thus, $n_4(S) \le 18$ for all antipodal 12-stars, with equality iff $S$ is congruent to the cuboctahedron.
\end{proof}

\subsection{Step 5: Functional maximisation}\label{sec:step5}

The off-diagonal contribution to $\Lambda$ is
\[
\Lambda_{\text{off}}=\sum_{v \text{ generic off-diag}} r_S(v)^2=4n_2+16n_4.
\]

Substituting the balance equation $n_2=60-2n_4$:
\[
\Lambda_{\text{off}}=4(60-2n_4)+16n_4=240+8n_4.
\]

The total resonance functional is
\[
\Lambda(S)=156+240+8n_4=396+8n_4.
\]

Since $n_4 \le 18$, we have
\[
\Lambda(S)\le 396+144=540,
\]
with equality iff $n_4=18$, which occurs iff $S$ is congruent to the BCC.
