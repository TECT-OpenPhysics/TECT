\subsection{Brazovskii free energy and existence window}

Adopting the Math56-Addendum form, the TECT BCC condensate near the
Brazovskii first-order transition is governed by
\begin{equation}
  \mathcal{F}[\Psi]
   = \int d^3x\,\Bigl[
      \tfrac{1}{2}\mu^2 |\Psi|^2
    + \tfrac{Z}{2}|\nabla\Psi|^2
    + \tfrac{Y}{2}|\nabla^2\Psi|^2
    + \lambda |\Psi|^4
    + \tfrac{5}{2}\gamma |\Psi|^6\Bigr],
  \label{eq:F-brazovskii}
\end{equation}
with the Math56-Addendum Leibler--Wickham normalisations $K_4 = 1$,
$K_6 = 5/2$. The existence window for non-trivial BCC extrema is
\begin{equation}
  \mu^2 \le r_c^{\mathrm{meta}} = \frac{2\lambda^2}{15\gamma}
  = 1.52\times 10^{-2},
\end{equation}
at $(\lambda, \gamma) = (-0.43, 1.62)$. The Brazovskii shell
wavenumber is $q_0 = \sqrt{-Z/(2Y)}$.

\subsection{Lattice symmetry and BCC Brillouin zone}

The TECT condensate selects the BCC lattice in reciprocal space as
its modulation wavevector lattice (TECT-Math10--14). The symmetry
group of the condensate is the cubic point group $O_h$ (order 48),
generated by the reflections in the coordinate planes and the
cyclic-$120^\circ$ rotations about the body-diagonal axes. The first
Brillouin zone $\mathrm{BZ}_{\mathrm{BCC}}$ is the Wigner--Seitz cell
of the primitive reciprocal lattice, which is an FCC lattice; the
resulting polyhedron is a \emph{truncated octahedron} with 14 faces
(8 hexagons + 6 squares) and volume $(2\pi)^3/(a^3/2) =
2\,(2\pi/a)^3$ per primitive cell of the direct BCC lattice of
spacing $a$. Under $O_h$ the BZ is invariant; under $SO(3)$ it is
not.

\subsection{Objective}

We must establish:
(i) the unique cubic-harmonic $O_h$-equivariant operator that is
orthogonal to \emph{every} $SO(3)$-scalar (not merely to the trace),
and
(ii) the IR fate of this operator's coupling under an RG flow whose
momentum-shell integration is taken over $\mathrm{BZ}_{\mathrm{BCC}}$,
not over a sphere.

\section{Exact $O_h$-equivariant decomposition of the rank-4 operator}
% =====================================================================

\subsection{Irreducible-representation content of $\mathrm{Sym}^4(\mathbb{R}^3)$}

Let $V = \mathbb{R}^3$ carry the vector representation of $O(3)$.
Under $SO(3) \subset O(3)$ the totally symmetric rank-4 tensor
representation decomposes into spherical harmonics of weight $L$,
\begin{equation}
  \mathrm{Sym}^4(V)
  \cong L = 0 \oplus L = 2 \oplus L = 4,
  \qquad \dim 15 = 1 + 5 + 9.
  \label{eq:SOsym4}
\end{equation}
Under $O_h \subset O(3)$, the branching rules \cite{Koster,Bradley}
give
\begin{equation}
  L = 0 \;\mapsto\; A_{1g},
  \qquad
  L = 2 \;\mapsto\; E_g \oplus T_{2g},
  \qquad
  L = 4 \;\mapsto\; A_{1g} \oplus E_g \oplus T_{1g} \oplus T_{2g}.
  \label{eq:Oh-branching}
\end{equation}

\begin{lemma}[Unique cubic-harmonic $A_{1g}$]
\label{lem:uniqueA1g}
Within $\mathrm{Sym}^4(V)$, the $A_{1g}$ irreducible representations
of $O_h$ appear with multiplicity two: once in $L = 0$ (the
full-rotation scalar $\delta_{(ij}\delta_{kl)}$) and once in $L = 4$
(the cubic-harmonic scalar). The latter is spanned by the single
tensor
\begin{equation}
  \Delta^{(K_4)}_{ijkl}
  := \delta_{ijkl} -
     \tfrac{1}{5}\bigl(\delta_{ij}\delta_{kl}
                   + \delta_{ik}\delta_{jl}
                   + \delta_{il}\delta_{jk}\bigr),
  \label{eq:K4-tensor}
\end{equation}
where $\delta_{ijkl} = 1$ if $i = j = k = l$, else $0$.
$\Delta^{(K_4)}_{ijkl}$ is the unique rank-4 totally symmetric tensor
that is (a) $O_h$-invariant, (b) traceless under all
$SO(3)$-contractions, and (c) normalised so that its
$\delta_{ijkl}$-component has unit coefficient.
\end{lemma}

\begin{proof}
Symmetry: $\delta_{ijkl}$ is totally symmetric, and the three
$\delta\delta$ terms are the full symmetrisation
$\delta_{(ij}\delta_{kl)} = \tfrac{1}{3}\bigl(\delta_{ij}\delta_{kl}
+ \delta_{ik}\delta_{jl} + \delta_{il}\delta_{jk}\bigr)$.
$O_h$-invariance follows because both $\delta_{ijkl}$ and the
permutation-symmetrised Kronecker tensor are diagonal in the
cubic axes.

Trace-freeness: contracting with $\delta^{ij}$,
\begin{align}
  \delta^{ij}\Delta^{(K_4)}_{ijkl}
  &= \delta^{ij}\delta_{ijkl}
    - \tfrac{1}{5}\delta^{ij}\bigl(\delta_{ij}\delta_{kl}
                  + \delta_{ik}\delta_{jl} + \delta_{il}\delta_{jk}\bigr) \\
  &= \delta_{kl}
    - \tfrac{1}{5}\bigl(3\delta_{kl} + \delta_{kl} + \delta_{kl}\bigr)
   = \delta_{kl} - \delta_{kl} = 0.
\end{align}
The subtraction coefficient $\tfrac{1}{5}$ (equivalently,
$\tfrac{3}{5}$ of the $\delta_{(ij}\delta_{kl)}$-projector with the
$\tfrac{1}{3}$-factor built in) is forced by the requirement
$\delta^{ij}\delta_{ijkl} - c \cdot \delta^{ij}(\delta_{ij}\delta_{kl}
+ 2\,\mathrm{perms}) = 0$, i.e. $\delta_{kl} - c \cdot 5 \delta_{kl} =
0$, giving $c = 1/5$.

Uniqueness: any other $O_h$-invariant rank-4 symmetric tensor can be
written as $\alpha\,\delta_{(ij}\delta_{kl)} + \beta\,\delta_{ijkl} +
(\text{terms in } T_{1g}, T_{2g}, E_g, \text{ etc.})$. The
trace-free projection fixes $\alpha$ in terms of $\beta$ up to
normalisation, yielding Eq.~\eqref{eq:K4-tensor} uniquely up to
rescaling.
\end{proof}

\begin{remark}[Correction to v2]
TECT-Math\_IR\_Bound-rigorous-v2 Eq.~(2) defined the cubic operator
with an isotropic-subtraction coefficient of $1/3$ rather than the
correct $3/5$ appearing below in
Eq.~\eqref{eq:Oc4-correct-v3} (obtained by contracting
$\Delta^{(K_4)}_{ijkl}$ with four $\partial\Psi$ factors; the $1/5$
of Eq.~\eqref{eq:K4-tensor} becomes $3/5$ after the full
$(ij,kl)$-permutation sum on the symmetric fields). The v2 operator
therefore retained a residual $A_{1g}$ (isotropic) admixture of
magnitude $\tfrac{3}{5} - \tfrac{1}{3} = \tfrac{4}{15}$ times
$[(\nabla\Psi)^2]^2$ and consequently mixed with the rotational
invariant under RG flow. The v3 expression below is the genuine
cubic-harmonic-$A_{1g}$ operator.
\end{remark}

\subsection{The correct cubic-anisotropy operator}

\begin{definition}[v3 cubic-anisotropy operator]
\label{def:Oc4v3}
Let $\Psi \in C^\infty(\mathbb{R}^3, \mathbb{C})$. The v3
$O_h$-equivariant cubic-anisotropy operator is
\begin{equation}
  \mathcal{O}^{(c),\mathrm{v3}}_4[\Psi]
  := \Delta^{(K_4)}_{ijkl}
     (\partial^i \Psi^*)(\partial^j \Psi)(\partial^k \Psi^*)(\partial^l \Psi)
  = \sum_{i=1}^{3}|\partial_i \Psi|^4
   - \tfrac{3}{5}\bigl[\,|\nabla\Psi|^2\,\bigr]^2.
  \label{eq:Oc4-correct-v3}
\end{equation}
\end{definition}

\begin{proposition}[Representation-theoretic non-mixing at an
$SO(3)$-equivariant fixed point]
\label{prop:SO3-orthog}
Let $\mathcal{V}_6$ denote the finite-dimensional vector space of
local, translation-invariant, derivative operators of mass dimension
$\le 6$ in the field $\Psi$, spanned by the operator basis
$\{(\nabla\Psi)^4,\ \mathcal{O}^{(c),\mathrm{v3}}_4,\
|\nabla^2\Psi|^2,\ |\Psi|^2|\nabla\Psi|^2,\ \dots\}$. Under the
natural $SO(3)$-action on $\mathcal{V}_6$, the space decomposes into
irreducible $SO(3)$-representations
\begin{equation}
  \mathcal{V}_6
  = \bigoplus_{L = 0,2,4} \mathcal{V}_6^{(L)}.
\end{equation}
The operator $\mathcal{O}^{(c),\mathrm{v3}}_4$ defined in
Eq.~\eqref{eq:Oc4-correct-v3} lies in the $L = 4$ sector
$\mathcal{V}_6^{(4)}$ (cubic-harmonic $A_{1g}\subset L=4$ of $O_h$);
the isotropic operators $(\nabla\Psi)^4,\,|\nabla^2\Psi|^2,\,
|\Psi|^2|\nabla\Psi|^2$ lie in $\mathcal{V}_6^{(0)}$. Consequently,
for any $SO(3)$-equivariant linear map $\Gamma:\mathcal{V}_6 \to
\mathcal{V}_6$ (in particular, the linearised RG map at an
$SO(3)$-invariant fixed point, modulo the explicit $O_h$-breaking
discussed in §4),
\begin{equation}
  \Gamma\,\mathcal{V}_6^{(4)} \subset \mathcal{V}_6^{(4)},
  \qquad
  \Gamma\,\mathcal{V}_6^{(0)} \subset \mathcal{V}_6^{(0)}.
  \label{eq:block-diag}
\end{equation}
Equivalently, the $L = 4$ cubic-harmonic coupling $g^{(c)}$ does not
mix with any $L = 0$ isotropic coupling under an $SO(3)$-equivariant
linearised RG flow.
\end{proposition}

\begin{proof}
The decomposition $\mathcal{V}_6 = \bigoplus_L \mathcal{V}_6^{(L)}$
is Schur's lemma applied to the finite-dimensional
$SO(3)$-representation on local operators of bounded mass dimension.
The assignment $\mathcal{O}^{(c),\mathrm{v3}}_4 \in
\mathcal{V}_6^{(4)}$ follows from Lemma~\ref{lem:uniqueA1g}, which
identifies $\Delta^{(K_4)}_{ijkl}$ as the unique rank-4 symmetric
tensor in the cubic-harmonic $A_{1g}$ irreducible component of
$L = 4$ under the branching \eqref{eq:Oh-branching}; the isotropic
operators transform trivially under $SO(3)$ and therefore populate
$\mathcal{V}_6^{(0)}$. The block-diagonality \eqref{eq:block-diag}
is an immediate consequence of Schur's lemma for
$SO(3)$-equivariant maps between distinct irreducible components.
\end{proof}

\begin{remark}[Scope of the non-mixing statement]
\label{rem:SO3-orthog-scope}
Proposition~\ref{prop:SO3-orthog} is a representation-theoretic
statement about operator-space decomposition; it is \emph{not} a
pointwise or integral orthogonality of the operator densities
themselves. Indeed, for a non-symmetric configuration (e.g.\
$\Psi(x) = x_1$), the product
$\mathcal{O}^{(c),\mathrm{v3}}_4[\Psi]\cdot(\nabla\Psi)^4$ has
non-vanishing integral over $\Omega\subset\mathbb{R}^3$, because the
density product mixes $L = 4$ harmonic content with $L = 0$ content
at the level of field configurations. What
Proposition~\ref{prop:SO3-orthog} states is the weaker (and exactly
required) fact that the RG mixing matrix is block-diagonal across
$L$-sectors at any $SO(3)$-equivariant fixed point. The BCC cubic
lattice breaks $SO(3)$ to $O_h$ explicitly; this reduces Schur's
lemma to block-diagonality across $O_h$-irreps rather than
$SO(3)$-irreps, and thereby allows a non-zero
$A_{1g}(L=4) \leftrightarrow A_{1g}(L=0)$ mixing of parametrically
small size $(\Delta_{\mathrm{BZ}}/q_0)^{2}$. This residual mixing
is what Theorem~\ref{thm:eta-bound} quantifies; it is \emph{not}
suppressed to zero by Proposition~\ref{prop:SO3-orthog}.
\end{remark}

\section{Canonical dimensions: Gaussian and Brazovskii fixed points}

\subsection{Gaussian fixed point (reprise of v2)}

\begin{lemma}[Gaussian dimension]
\label{lem:dim-gauss-v3}
At the Gaussian fixed point in $d = 3$ spatial dimensions with
$[\Psi] = (d-2)/2 = 1/2$, the operator
$\mathcal{O}^{(c),\mathrm{v3}}_4$ has canonical dimension
$[\mathcal{O}^{(c),\mathrm{v3}}_4] = 4 + 4\cdot\tfrac{1}{2} = 6$ and
its coupling has dimension $[g^{(c)}] = d - 6 = -3$. The operator is
IR-irrelevant at Gaussian level.
\end{lemma}

(The v2 proof transfers unchanged; the $3/5$ coefficient does not
affect the power counting.)

\subsection{Brazovskii fixed point: anisotropic scaling}
\label{sec:brazovskii-v3}

Near the shell $|k| = q_0$, let $\delta k_\parallel, \delta k_\perp$
denote the radial and tangential components of
$\vec k - q_0 \hat r$. The Brazovskii dispersion
$\omega(k) = \mu^2 + Z k^2 + Y k^4$ at $Z = -2Yq_0^2$ reduces
near $|k| = q_0$ to
\begin{equation}
  \omega(q_0 + \delta k)
  = \mu^2 + Y\bigl(2 q_0\,\delta k_\parallel
           + \delta k_\perp^2 + \delta k_\parallel^2\bigr)^{2}.
  \label{eq:braz-disp-v3}
\end{equation}

\begin{lemma}[Anisotropic scaling]
\label{lem:braz-aniso-v3}
The Brazovskii fixed point is characterised by the anisotropic
scaling
\begin{equation}
  [\delta k_\perp] = \mu, \qquad [\delta k_\parallel] = \mu^2,
  \label{eq:aniso-scal}
\end{equation}
under which $\omega \sim Y\mu^4$ is consistent (both
$2 q_0 \delta k_\parallel \sim \mu^2$ and
$\delta k_\perp^2 \sim \mu^2$). The effective dimension at the
shell point is
\begin{equation}
  d_{\mathrm{eff}}
  = (d - 1)\cdot[\delta k_\perp]^{-1} + 1\cdot[\delta k_\parallel]^{-1}
  = (d-1) + 2 = 2 + 2 = 4.
\end{equation}
\end{lemma}

\begin{proof}
Set the two leading terms of \eqref{eq:braz-disp-v3} to the same
scale: $2 q_0 \delta k_\parallel \sim \delta k_\perp^2$ gives
\eqref{eq:aniso-scal}. The dimension count is the integration
measure $d^{d-1}\!\delta k_\perp\, d\delta k_\parallel$ scaling as
$\mu^{d-1}\cdot\mu^{2} = \mu^{d+1}$, i.e.\ $\mu^4$ at $d = 3$.
\end{proof}

\begin{lemma}[Brazovskii field dimension]
\label{lem:phi-dim-v3}
With the propagator $G(k) = 1/\omega(k)$ scaling as $\mu^{-4}$ and
the two-point function $\langle\Psi(x)\Psi(0)\rangle = \int
d^3k\,G(k)\,e^{ikx}$,
\begin{equation}
  [\Psi]_B = \tfrac{1}{2}([k^d] - [\omega]) = \tfrac{1}{2}(4 - 4) = 0.
\end{equation}
\end{lemma}

\begin{lemma}[Cubic coupling dimension at Brazovskii FP]
\label{lem:g-dim-v3}
The v3 cubic-anisotropy operator has Brazovskii scaling dimension
\begin{equation}
  [\mathcal{O}^{(c),\mathrm{v3}}_4]_B
  = 4\cdot[\partial_i]_{B,\mathrm{avg}} + 4\cdot[\Psi]_B
  = 4\cdot[\partial_i]_{B,\mathrm{avg}} + 0.
\end{equation}
Each partial $\partial_i$ decomposes along cubic axes into a
generic direction on the shell; its projection onto
$\{\partial_\perp, \partial_\parallel\}$ is generically non-trivial.
In the IR limit $\mu \to 0$, $[\partial_\perp] = \mu \gg
[\partial_\parallel] = \mu^2$, so the dominant scaling is governed
by $\partial_\perp$ and
$[\mathcal{O}^{(c),\mathrm{v3}}_4]_B \to 4\mu$. The coupling then
satisfies
\begin{equation}
  [g^{(c)}]_B
  = d_{\mathrm{eff}} - [\mathcal{O}^{(c),\mathrm{v3}}_4]_B
  = 4 - 4 = 0
  \quad (\text{tangential sector, marginal})
\end{equation}
\begin{equation}
  [g^{(c)}]_B
  = 4 - 4\cdot 2 = -4
  \quad (\text{radial sector, strongly irrelevant}).
\end{equation}
\end{lemma}

\begin{proof}
Direct substitution. The tangential marginality is consistent with
the fact that Brazovskii is its own critical dimension
($d_{\mathrm{eff}} = d_c = 4$), and derivative operators on the
shell sit at the edge of relevance.
\end{proof}

\begin{remark}[Marginality $\Rightarrow$ 1-loop is decisive]
The tangential sector carries $[g^{(c)}]_B = 0$, meaning the
tree-level RG predicts neither relevance nor irrelevance for the
cubic-anisotropy coupling. The 1-loop anomalous dimension
$\eta^{(c)}$ therefore determines the IR fate: if $\eta^{(c)} > 0$,
the operator is marginally relevant; if $\eta^{(c)} < 0$, it is
marginally irrelevant. The computation in §4 is therefore
load-bearing for the Lorentz-emergence claim.
\end{remark}

\section{Brillouin-zone momentum-shell 1-loop computation}
% =====================================================================

\subsection{Shell setup}

Let $b > 1$ be an RG rescaling factor. The Wilsonian momentum-shell
is the BZ region
\begin{equation}
  S_b := \bigl\{\vec k \in \mathrm{BZ}_{\mathrm{BCC}} :
     \Lambda/b \le \|\vec k - q_0 \hat n(\vec k)\| \le \Lambda\bigr\},
  \label{eq:shell-def}
\end{equation}
where $\hat n(\vec k)$ is the closest Brazovskii-shell direction
to $\vec k$ (well-defined for $\vec k$ within a thin annulus of
$|\vec k| = q_0$). The RG step integrates out modes in $S_b$.

\subsection{1-loop correction to the cubic coupling}

The 1-loop correction to $g^{(c)}$ comes from the triangle diagram
with two external $\Psi$ propagators, one internal loop closed by
the $\lambda |\Psi|^4$ vertex. Its amputated kernel contracted
against $\Delta^{(K_4)}_{ijkl}$ gives
\begin{equation}
  \delta g^{(c)}
  = -\tfrac{1}{2}\lambda^2\,\int_{S_b} \frac{d^3 k}{(2\pi)^3}
     \frac{\Delta^{(K_4)}_{ijkl}\, k_i k_j k_k k_l}{[\omega(k)]^{2}},
  \label{eq:1loop-correction}
\end{equation}
where the factor $\tfrac{1}{2}$ is the symmetry factor of the
triangle diagram.

\begin{lemma}[Vanishing of the BZ-averaged integrand on the
spherical part]
\label{lem:sphere-vanish}
Let $S_b^{\mathrm{sph}}$ denote the spherical-shell approximation to
$S_b$ in which the BZ is replaced by $\{|\vec k| \le q_0 +
\Lambda\}$. Then
\begin{equation}
  \int_{S_b^{\mathrm{sph}}}
   \frac{d^3 k}{(2\pi)^3}
   \frac{\Delta^{(K_4)}_{ijkl}\, k_i k_j k_k k_l}{[\omega(k)]^{2}}
   = 0.
\end{equation}
\end{lemma}

\begin{proof}
On a sphere, $\int d\Omega\,k_i k_j k_k k_l / |\vec k|^4$ is the
standard symmetric rank-4 integral
$\tfrac{|\vec k|^4}{15}(\delta_{ij}\delta_{kl} + \delta_{ik}\delta_{jl}
+ \delta_{il}\delta_{jk}) = |\vec k|^4\,\delta_{(ij}\delta_{kl)}$.
Contracting with $\Delta^{(K_4)}_{ijkl}$ gives zero by
Lemma~\ref{lem:uniqueA1g}.
\end{proof}

\begin{remark}[Non-vanishing contribution is purely from BZ
boundary]
Lemma~\ref{lem:sphere-vanish} shows that the 1-loop correction to
$g^{(c)}$ vanishes in the isotropic approximation. The non-zero
contribution at the BCC Brazovskii fixed point arises entirely from
the \emph{polyhedral} deviation of the truncated-octahedron BZ from
a sphere. This is the technical point where v2 missed a factor of
order $(\Delta_{\mathrm{BZ}}/q_0)^2$, with
$\Delta_{\mathrm{BZ}}/q_0 \le 0.13$ the fractional BZ radius
variation between the square-face centres (closest) and
hexagonal-face vertices (farthest) of the truncated octahedron.
\end{remark}

\subsection{Non-vanishing contribution from BZ boundary corrugation}

\begin{theorem}[1-loop $\eta^{(c)}$ bound]
\label{thm:eta-bound}
Under the BCC Brillouin-zone momentum-shell RG
\eqref{eq:shell-def}, the 1-loop anomalous dimension of the cubic
coupling $g^{(c)}$ satisfies
\begin{equation}
  \eta^{(c)}_{\mathrm{1\mbox{-}loop}}
  = -\,\tfrac{1}{2}\lambda^2\,\mathcal{I}_{\mathrm{BZ}}
       + O(\lambda^3),
  \qquad
  \mathcal{I}_{\mathrm{BZ}}
  := \int_{\mathrm{BZ}_{\mathrm{BCC}}^{\mathrm{corrug}}}
    \frac{d^3 k}{(2\pi)^3}
    \frac{\Delta^{(K_4)}_{ijkl}\, k_i k_j k_k k_l}{[\omega(k)]^{2}},
  \label{eq:eta-1loop}
\end{equation}
with $\mathrm{BZ}_{\mathrm{BCC}}^{\mathrm{corrug}}$ denoting the
symmetric-difference region between the BCC BZ and its inscribed
sphere of radius $q_0 + \Lambda$. The magnitude of
$\mathcal{I}_{\mathrm{BZ}}$ is bounded by
\begin{equation}
  |\mathcal{I}_{\mathrm{BZ}}|
  \le C_{\mathrm{geom}}\,
     \Bigl(\frac{\Delta_{\mathrm{BZ}}}{q_0}\Bigr)^{2}
     \cdot \frac{1}{[\omega(q_0)]^{2}\,(q_0 a)^3}
     \cdot q_0^{4},
  \label{eq:I-bound}
\end{equation}
with $C_{\mathrm{geom}} \le 1$ a dimensionless geometric constant
determined by the truncated-octahedron first Fourier mode on the
sphere. Consequently,
\begin{equation}

  \label{eq:eta-bound-explicit}
\end{equation}
with $\omega_0 = \omega(q_0) = \mu^2$ the shell-minimum gap. The
sign is determined below and is negative for the parameter range of
the TECT Brazovskii fixed point.
\end{theorem}

\begin{proof}
The full BZ integral decomposes as
$\int_{\mathrm{BZ}} = \int_{\mathrm{sph}} + \int_{\mathrm{corrug}}$,
with the sphere-inscribed part $\int_{\mathrm{sph}}$ vanishing by
Lemma~\ref{lem:sphere-vanish}. The residual is supported on the
symmetric difference, which lies in the annulus
$q_0 - \Delta_{\mathrm{BZ}} \le |\vec k| \le q_0 +
\Delta_{\mathrm{BZ}}$ restricted to the 8 hexagonal-face-corner
octants of the BCC BZ; its volume is
$\sim \Delta_{\mathrm{BZ}}/q_0 \cdot (q_0)^3 \cdot (q_0 a)^{-1}$
(the last factor converting from primitive to conventional BCC
volume normalisation). The integrand $k^4/\omega^2$ is bounded by
$q_0^4/\omega_0^{2}$ on the annulus. The remaining factor of
$(\Delta_{\mathrm{BZ}}/q_0)$ comes from the $\Delta^{(K_4)}$-weighted
angular average over the corrugation region, which vanishes to
leading order by $O_h$ symmetry and contributes first at second
order in the corrugation amplitude. Combining yields
\eqref{eq:I-bound}.

The sign of $\eta^{(c)}_{\mathrm{1\mbox{-}loop}}$ is governed by the
sign of the projection coefficient
\begin{equation}
  c_4 := \int_{\mathrm{BZ}_{\mathrm{BCC}}^{\mathrm{corrug}}}
        \frac{d^3 k}{(2\pi)^3}\,
        \frac{\Delta^{(K_4)}_{ijkl}\,k_i k_j k_k k_l}{[\omega(k)]^2},
  \label{eq:c4-def}
\end{equation}
together with the universal triangle prefactor $-\tfrac{1}{2}\lambda^2
\le 0$. Explicit evaluation of the angular kernel
$\Delta^{(K_4)}_{ijkl}\hat k_i\hat k_j\hat k_k\hat k_l$ at two
extremal directions gives
\begin{align}
  \hat k = \hat e_1 &: \quad \Delta^{(K_4)}_{ijkl}\hat k_i\hat k_j
    \hat k_k\hat k_l = 1 - \tfrac{3}{5} = \tfrac{2}{5} > 0, \\
  \hat k = (1,1,1)/\sqrt 3 &: \quad \Delta^{(K_4)}_{ijkl}\hat k_i\hat k_j
    \hat k_k\hat k_l = \tfrac{1}{3} - \tfrac{3}{5} = -\tfrac{4}{15}<0.
\end{align}
The \emph{sign of $c_4$} is therefore not determined by evaluation at
extremal directions alone; it is the outcome of a signed, weighted
volume integral over the truncated-octahedron BZ corrugation region.
The conclusion $\eta^{(c)}<0$ therefore requires either (i) the
rigorous exact BZ angular integral with interval arithmetic, or
(ii) a geometric region-measure dominance inequality showing that
the cubic-axis-proximal corrugation volume out-weights the
body-diagonal-proximal corrugation volume under the
$\omega(k)^{-2}$ kernel; both routes are explicitly developed in the
v4 completion plan
\texttt{TECT-Math\_IR\_Bound-v4-outline.tex.txt}. Within the present
note we therefore record the sign determination as \emph{conditional}
on the v4 projection-coefficient theorem; the bound
Eq.~\eqref{eq:eta-bound-explicit} itself is independent of the sign
and stands unconditionally.
\end{proof}

\paragraph{Erratum E1 (operator structure for complex fields).}
Definition~\ref{def:Oc4v3} writes the cubic-anisotropy operator in
closed form as
$\sum_{i=1}^{3}|\partial_i \Psi|^4 - \tfrac{3}{5}[|\nabla\Psi|^2]^2$.
This closed form is correct only for \emph{real} scalars. For a
$U(1)$-covariant complex field $\Psi \in C^\infty(\mathbb{R}^3,
\mathbb{C})$ — the case actually declared in
Def.~\ref{def:Oc4v3} — the ordered contraction
$\Delta^{(K_4)}_{ijkl}(\partial^i\Psi^*)(\partial^j\Psi)
(\partial^k\Psi^*)(\partial^l\Psi)$
produces three distinct $U(1)$-invariant rank-$4$ densities under the
three Wick pairings $(\delta_{ij}\delta_{kl},
\delta_{ik}\delta_{jl}, \delta_{il}\delta_{jk})$:
\begin{align}
  \delta_{ij}\delta_{kl} &\;\longrightarrow\; |\nabla\Psi|^4, \\
  \delta_{ik}\delta_{jl} &\;\longrightarrow\; |(\nabla\Psi)^2|^2, \\
  \delta_{il}\delta_{jk} &\;\longrightarrow\; |\nabla\Psi|^4,
\end{align}
where $(\nabla\Psi)^2 \equiv \sum_i (\partial_i\Psi)^2 \in
\mathbb{C}$ and $|\nabla\Psi|^2 \equiv \sum_i (\partial_i\Psi^*)
(\partial_i\Psi) \in \mathbb{R}_{\ge 0}$. The correct complex-field
expression is therefore
\begin{equation}
  \mathcal{O}^{(c),\mathrm{v3,\mathbb{C}}}_4[\Psi]
  = \sum_{i=1}^{3}|\partial_i\Psi|^4
   - \tfrac{1}{5}\bigl[\,2|\nabla\Psi|^4 + |(\nabla\Psi)^2|^2\,\bigr].
  \label{eq:Oc4-complex-v31}
\end{equation}
Eq.~\eqref{eq:Oc4-correct-v3} recovers Eq.~\eqref{eq:Oc4-complex-v31}
when $\Psi \in \mathbb{R}$, in which case
$|(\nabla\Psi)^2|^2 = |\nabla\Psi|^4$ and the bracket collapses to
$3|\nabla\Psi|^4$, yielding the $3/5$ coefficient. The entire §3--§5
RG analysis is unaffected because (a) the $O_h$-equivariance is
enforced by $\Delta^{(K_4)}_{ijkl}$ alone; (b) the Brazovskii
condensate develops a definite phase below the first-order
transition, and in the phase-gauged representation $\Psi = \rho
e^{i\theta}$ the tangential fluctuation $\delta\rho$ is a real scalar
for which Eq.~\eqref{eq:Oc4-correct-v3} is the correct operator;
(c) the phase sector $\theta$ decouples from the cubic-anisotropy
analysis by Ward identity. Eq.~\eqref{eq:Oc4-complex-v31} supersedes
Eq.~\eqref{eq:Oc4-correct-v3} for gauge-invariant expressions;
Eq.~\eqref{eq:Oc4-correct-v3} remains valid for the
condensate-amplitude mode and is the form used throughout §3--§5.

\paragraph{Erratum E2 (Cor.~\ref{cor:eta-numerical} arithmetic).}
Corollary~\ref{cor:eta-numerical} quotes $|\eta^{(c)}_{\mathrm{1
\mbox{-}loop}}| \approx 7 \times 10^{-4}$. Direct substitution into

\paragraph{Erratum E3 (false integral orthogonality in Prop.~2).}
