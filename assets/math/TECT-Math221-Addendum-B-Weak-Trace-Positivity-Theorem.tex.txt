\section*{§3. Universal Trace-of-Squares-Positive Theorem (Lemma A.1)}

\subsection*{3.1 Statement}

\textbf{Lemma A.1 (Trace of Squares for Hermitian Operators)}.
Let $V$ be a finite-dimensional Hilbert space (real or complex) of
dimension $\dim(V) = n$. Let $T: V \to V$ be a Hermitian linear operator
(i.e., $T^\dagger = T$).

Then
\begin{equation}
\mathrm{Tr}_V(T^2) = \mathrm{Tr}_V(T^\dagger T) = \sum_{i=1}^n \lambda_i^2 \geq 0,
\end{equation}
where $\{\lambda_i\}_{i=1}^n$ are the eigenvalues of $T$ (real, since $T$
is Hermitian). Equality $\mathrm{Tr}_V(T^2) = 0$ holds iff all $\lambda_i = 0$,
i.e., iff $T = 0$ as an operator.

Conversely, $T \neq 0 \Rightarrow \mathrm{Tr}_V(T^2) > 0$.

\subsection*{3.2 Proof}

For a Hermitian $T$, the eigendecomposition is:
\begin{equation}
T = \sum_{i=1}^n \lambda_i \, |i\rangle\langle i|,
\end{equation}
where $\lambda_i \in \mathbb{R}$ and $\{|i\rangle\}$ form an orthonormal
basis. Thus:
\begin{equation}
T^2 = \sum_{i=1}^n \lambda_i^2 \, |i\rangle\langle i|.
\end{equation}
Taking the trace:
\begin{equation}
\mathrm{Tr}(T^2) = \sum_{i=1}^n \lambda_i^2 \geq 0.
\end{equation}
Equality holds iff $\lambda_i = 0$ for all $i$, i.e., $T = 0$. \hfill$\square$

\subsection*{3.3 Application to Lie-algebra generators}

Let $\rho: G \to U(V)$ be a unitary representation of a compact Lie group
$G$ on finite-dimensional $V$. Then every generator $X \in \mathfrak{g}$
(the Lie algebra) satisfies $\rho(X) = T$ with $T$ Hermitian (in the
convention $\rho(\exp[\epsilon X]) = \exp[i \epsilon T]$).

By Lemma A.1, $X \in \ker(\rho)$ (the kernel of the representation map)
iff $\rho(X) = 0$ iff $\mathrm{Tr}_{V}(\rho(X)^2) = 0$.

Therefore:
\begin{equation}
\mathrm{Tr}_{V}(\rho(X)^2) > 0 \quad \Leftrightarrow \quad X \notin \ker(\rho).
\end{equation}
