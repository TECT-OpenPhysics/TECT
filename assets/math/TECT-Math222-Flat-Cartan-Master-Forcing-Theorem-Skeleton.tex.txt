\section*{1. Statement of theorem}

Let $X = \mathbb{CP}^2$. Let $L_\chi \to X$ be the $U(1)_\chi$ line
bundle and $E_5 \to X$ the $SU(5)$ vector bundle. Write the Chern
classes as
\begin{equation}
c_1(L_\chi) = k\, h, \qquad c_2(E_5) = n\, h^2, \qquad k, n \in \mathbb{Z},
\end{equation}
where $h \in H^2(\mathbb{CP}^2, \mathbb{Z})$ is the positive generator.

\textbf{Theorem (Flat-Cartan Master Forcing — Math222 v1.0)}.
Assume the TECT effective free energy decomposes as
\begin{equation}
\mathcal F_{\rm total}(k,n) \;=\; \mathcal F_{\rm top}(k,n) \;+\; \Gamma_{\rm rest}(k,n),
\end{equation}
where the topological-stiffness sector is
\begin{equation}
\mathcal F_{\rm top}
\;=\;
\frac{\kappa_\chi}{2}\int_X F_\chi \wedge \star F_\chi
\;+\;
\frac{\kappa_5}{2}\int_X \mathrm{tr}(F_5 \wedge \star F_5),
\end{equation}
and assume the following two hypotheses:

\textbf{Hypothesis (H1) — positive curvature stiffness}:
\begin{equation}
\kappa_\chi > 0 \quad \text{and} \quad \kappa_5 > 0.
\end{equation}

\textbf{Hypothesis (H2) — relative no-negative-offset bound}:
\begin{equation}
\Gamma_{\rm rest}(k,n) - \Gamma_{\rm rest}(0,0)
\;\geq\;
-\epsilon \, \mathcal F_{\rm top}(k,n),
\qquad
0 \leq \epsilon < 1.
\end{equation}

Then the unique global minimiser of $\mathcal F_{\rm total}$ over the
discrete moduli of Chern sectors $(k,n) \in \mathbb{Z}^2$ is
\begin{equation}
(k_*, n_*) = (0, 0).
\end{equation}
Equivalently,
\begin{equation}
c_1(U(1)_\chi) \;=\; 0 \quad \text{and} \quad c_2(E_{SU(5)}) \;=\; 0.
\end{equation}
Since $c_2(E_5) = 0$ places the $SU(5)$ bundle in the topologically
trivial sector over the present $\mathbb{CP}^2$ base, one may choose
a trivial transition atlas. The identity transition functions lie in
the Cartan torus $T \subset SU(5)$. In this precise sense the
topological minimiser admits a flat-Cartan representative.
\hfill$\square$

\section{2. Proof}

\subsection{2.1 $U(1)_\chi$ topological lower bound}

For a connection $A_\chi$ on $L_\chi$ with $c_1(L_\chi) = k h$, the
Chern-Weil representative satisfies
\begin{equation}
\left[\frac{F_\chi}{2\pi}\right] = k h \quad \text{in } H^2(X, \mathbb{R}).
\end{equation}
By the Hodge theorem on the closed Kähler manifold $X = \mathbb{CP}^2$,
the harmonic representative $h_{\rm harm}$ minimises the $L^2$-norm
within the cohomology class. Therefore
\begin{equation}
\int_X F_\chi \wedge \star F_\chi
\;=\;
\| F_\chi \|_{L^2}^2
\;\geq\;
\| 2\pi k\, h_{\rm harm} \|_{L^2}^2
\;=\;
(2\pi)^2 k^2 \| h_{\rm harm} \|^2.
\end{equation}
Since $X = \mathbb{CP}^2$ is compact and $h$ is non-trivial, the
harmonic representative satisfies $\| h_{\rm harm} \|^2 > 0$ (explicit
value: $\| h_{\rm harm} \|^2 = 1$ in the Fubini-Study normalisation
$\int_X h^2 = 1$). Combining with $\kappa_\chi > 0$ from (H1):
\begin{equation}
\mathcal F_\chi(k)
\;:=\;
\frac{\kappa_\chi}{2} \int_X F_\chi \wedge \star F_\chi
\;\geq\;
2\pi^2 \kappa_\chi k^2 \| h_{\rm harm} \|^2.
\end{equation}
Therefore
\begin{equation}
k \neq 0 \quad \Longrightarrow \quad \mathcal F_\chi(k) > 0. \tag{2.1.1}
\end{equation}

\subsection{2.2 $SU(5)$ topological lower bound (norm-form, trace-convention safe)}

Decompose the curvature 2-form into self-dual and anti-self-dual parts:
\begin{equation}
F_5 \;=\; F_5^+ \;+\; F_5^-, \qquad \star F_5^\pm \;=\; \pm F_5^\pm.
\end{equation}
The Yang-Mills $L^2$-norm decomposes orthogonally:
\begin{equation}
\| F_5 \|_{L^2}^2
\;:=\;
\int_X \langle F_5, F_5 \rangle\, d\mathrm{vol}
\;=\;
\| F_5^+ \|_{L^2}^2 \;+\; \| F_5^- \|_{L^2}^2.
\end{equation}
The trace-pairing $\int_X \mathrm{tr}(F_5 \wedge F_5)$ is, in any fixed
positive-definite trace normalisation on $\mathfrak{su}(5)$,
\begin{equation}
\int_X \mathrm{tr}(F_5 \wedge F_5)
\;=\;
C_{\rm tr} \left( \| F_5^+ \|_{L^2}^2 - \| F_5^- \|_{L^2}^2 \right),
\qquad
C_{\rm tr} > 0,
\end{equation}
where $C_{\rm tr}$ is a fixed positive convention-dependent constant
(its precise value does not enter the conclusion). Consequently
\begin{equation}
\| F_5 \|_{L^2}^2
\;\geq\;
\left| C_{\rm tr}^{-1} \int_X \mathrm{tr}(F_5 \wedge F_5) \right|.
\end{equation}
By Chern-Weil theory, in the same normalisation,
\begin{equation}
c_{\rm CW} \int_X \mathrm{tr}(F_5 \wedge F_5) \;=\; n,
\qquad
c_{\rm CW} > 0,
\end{equation}
so combining,
\begin{equation}
\| F_5 \|_{L^2}^2 \;\geq\; \frac{|n|}{C_{\rm tr}\, c_{\rm CW}}
\;=:\;
\widetilde C_5 \, |n|,
\qquad
\widetilde C_5 > 0.
\end{equation}
Combining with $\kappa_5 > 0$ from (H1):
\begin{equation}
\mathcal F_5(n)
\;:=\;
\frac{\kappa_5}{2} \, \| F_5 \|_{L^2}^2
\;\geq\;
\frac{1}{2} \, \kappa_5 \, \widetilde C_5 \, |n|
\;=:\;
C_5 \, \kappa_5 \, |n|,
\qquad
C_5 > 0.
\end{equation}
The numerical value $\widetilde C_5 = 8\pi^2$ in the standard trace
normalisation $\mathrm{tr}(T^a T^b) = \tfrac{1}{2}\delta^{ab}$ for the
fundamental of $SU(5)$ (so that $C_5 = 4\pi^2$), but the skeleton
theorem requires only $C_5 > 0$ as a fixed positive constant. Therefore
\begin{equation}
n \neq 0 \quad \Longrightarrow \quad \mathcal F_5(n) > 0. \tag{2.2.1}
\end{equation}

\subsection{2.3 Combined topological positivity}

For every $(k,n) \neq (0,0)$, by (2.1.1) or (2.2.1):
\begin{equation}
\mathcal F_{\rm top}(k,n) \;=\; \mathcal F_\chi(k) + \mathcal F_5(n) \;>\; 0.
\tag{2.3.1}
\end{equation}
At $(k,n) = (0,0)$, the trivial connection (zero curvature) gives
$\mathcal F_{\rm top}(0,0) = 0$.

\subsection{2.4 Total free-energy comparison}

Using (H2):
\begin{align}
\mathcal F_{\rm total}(k,n) - \mathcal F_{\rm total}(0,0)
&= \mathcal F_{\rm top}(k,n) + \Gamma_{\rm rest}(k,n) - \Gamma_{\rm rest}(0,0) \\
&\geq \mathcal F_{\rm top}(k,n) - \epsilon\, \mathcal F_{\rm top}(k,n) \\
&= (1 - \epsilon)\, \mathcal F_{\rm top}(k,n).
\end{align}
Since $1 - \epsilon > 0$ and $\mathcal F_{\rm top}(k,n) > 0$ for
$(k,n) \neq (0,0)$, it follows that
\begin{equation}
\mathcal F_{\rm total}(k,n) \;>\; \mathcal F_{\rm total}(0,0)
\qquad \forall (k,n) \neq (0,0).
\end{equation}

Therefore the unique global minimiser is $(k_*, n_*) = (0,0)$.
Equivalently $c_1(U(1)_\chi) = 0$ and $c_2(E_{SU(5)}) = 0$, which is
the flat-Cartan sector. \hfill$\square$

\section{4. Devil's-advocate self-test (CLAUDE.md §6.3.1 + §6.3.4)}

\subsection*{Structural objections}

\textbf{$\alpha^{222}$ — Is the assumption $X = \mathbb{CP}^2$ load-bearing?}

\textbf{ADDRESSED}: The Hodge bound (2.1) relies on $X$ being a closed
oriented Riemannian manifold; the Yang-Mills self-dual bound (2.2)
relies on $X$ being 4-dimensional and oriented. $\mathbb{CP}^2$ is
canonical for TECT (Math162 atlas). The proof generalises to any
closed oriented 4-manifold with $H^2(X,\mathbb{Z}) \neq 0$; the
specific value $\| h_{\rm harm} \|^2 = 1$ is Fubini-Study normalisation.

\textbf{$\beta^{222}$ — Is (H2) ever satisfiable for $\epsilon \geq 1$?}

\textbf{ADDRESSED}: If $\epsilon \geq 1$, the proof (§2.4) does NOT
go through; $\mathcal F_{\rm total}(k,n)$ could be lower than
$\mathcal F_{\rm total}(0,0)$. The strict inequality $\epsilon < 1$
is therefore essential. This is exactly the content of Lemma B that
must be derived from TECT microstructure.

\textbf{$\gamma^{222}$ — Is the $\Gamma_{\rm rest}$ vs $\Gamma_{\rm eff}$
distinction important?}

\textbf{CRITICAL — DOCUMENTED}: yes. If one writes
$\Delta\Gamma_{\rm eff}[A] \geq -\epsilon \mathcal F_{\rm top}[A]$ where
$\Gamma_{\rm eff} = \mathcal F_{\rm top} + \Gamma_{\rm rest}$, the
inequality reduces to $\mathcal F_{\rm top}[A] + \Delta\Gamma_{\rm rest}[A] \geq -\epsilon \mathcal F_{\rm top}[A]$,
i.e. $\Delta\Gamma_{\rm rest}[A] \geq -(1+\epsilon)\mathcal F_{\rm top}[A]$,
which is a WEAKER statement that does NOT force the global minimum at
$(0,0)$. The reviewer's audit of Math220 (Issue 1) is exactly this point.

\subsection*{Quantitative sanity checks (§6.3.4 BINDING)}

\textbf{Check 1 — Dimensional}: $[\mathcal F_{\rm top}] = [\mathcal F_{\rm total}] = [\Gamma_{\rm rest}] = $ energy. $[\epsilon] = 1$. $[\kappa_\chi] = [\kappa_5] = $ inverse length squared (in natural units). All sides of (H2) and (2.4) have identical dimensions. \textbf{PASS}.

\textbf{Check 2 — Limit case $\epsilon \to 0$}: the bound (H2) reduces to $\Delta\Gamma_{\rm rest} \geq 0$. The conclusion $\mathcal F_{\rm total}(k,n) > \mathcal F_{\rm total}(0,0)$ becomes strict inequality with the full topological gap $\mathcal F_{\rm top}(k,n)$. \textbf{Self-consistent}.

\textbf{Check 3 — Limit case $\epsilon \to 1^-$}: the bound (2.4) gives $\mathcal F_{\rm total}(k,n) - \mathcal F_{\rm total}(0,0) \geq (1-\epsilon)\mathcal F_{\rm top}(k,n) \to 0^+$. The strict inequality $\mathcal F_{\rm total}(k,n) > \mathcal F_{\rm total}(0,0)$ is preserved (since $\mathcal F_{\rm top}(k,n) > 0$ strictly), but the gap shrinks to zero. This is the boundary of the forcing regime. \textbf{Self-consistent}.

\textbf{Check 4 — Special integer values}: at $(k,n) = (1,0)$: $\mathcal F_{\rm top} = 2\pi^2 \kappa_\chi > 0$; gap $\geq 2\pi^2 \kappa_\chi (1-\epsilon)$. At $(k,n) = (0,1)$: $\mathcal F_{\rm top} = 4\pi^2 \kappa_5 > 0$; gap $\geq 4\pi^2 \kappa_5 (1-\epsilon)$. Both order-of-magnitude estimates plausible at canonical TECT $\kappa$ values. \textbf{Magnitude PASS}.
