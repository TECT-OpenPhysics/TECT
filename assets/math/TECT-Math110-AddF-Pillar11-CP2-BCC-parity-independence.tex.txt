\subsection*{1. CP$^2$ complex conjugation and orientation reversal}

\paragraph{Definition.} On homogeneous coordinates,
$$
\mathcal{C}:\;[z_1:z_2:z_3]\;\longmapsto\;[\bar z_1:\bar z_2:\bar z_3].
$$
This is well-defined on $\mathbb{CP}^2$ because the projective
equivalence $z\sim e^{i\chi}z$ implies
$\bar z\sim e^{-i\chi}\bar z$.  Clearly $\mathcal{C}^2=\mathrm{id}$.

\paragraph{Action on Fubini-Study form.}
\begin{lemma}\label{lem:C-FS}
$\mathcal{C}^*\omega_{\rm FS}=-\omega_{\rm FS}$.
\end{lemma}

\begin{proof}
$\mathcal{C}$ is an anti-holomorphic involution: it preserves the
Fubini-Study Riemannian metric but reverses the complex structure $J$
(by definition of anti-holomorphicity, $\mathcal{C}_*J=-J\mathcal{C}_*$).
The Kähler form is $\omega_{\rm FS}=g(J\cdot,\cdot)$, so under
$\mathcal{C}$:
$$
\mathcal{C}^*\omega_{\rm FS}(X,Y)
=\omega_{\rm FS}(\mathcal{C}_*X,\mathcal{C}_*Y)
=g(J\mathcal{C}_*X,\mathcal{C}_*Y)
=g(-\mathcal{C}_*JX,\mathcal{C}_*Y)
=-g(JX,Y)=-\omega_{\rm FS}(X,Y).
$$
\end{proof}

\paragraph{Consequence.} The integer symplectic charge
$$
n[\Sigma]=\frac{1}{2\pi}\int_{\Sigma}\omega_{\rm FS}\in\mathbb{Z}
$$
satisfies $n[\mathcal{C}(\Sigma)]=-n[\Sigma]$.  Hence $\mathcal{C}$
generates a $\mathbb{Z}_2^{\mathcal{C}}$ symmetry that flips the
elementary action sector $\eta=\pm 1$.

\subsection*{2. BCC spatial parity}

\paragraph{Definition.} $\mathcal{P}_{\rm BCC}:x\mapsto -x$ on real
space, equivalently $k\mapsto -k$ on the BCC reciprocal star.

\paragraph{Action on TECT condensate.} Writing
$$
\Psi(x)=\sum_{a=1}^{3}\bigl[z_a\Phi_a(x)+\bar z_a\bar\Phi_a(x)\bigr]
$$
with three plane-family modes $\Phi_a$ ($a\in\{xy,xz,yz\}$) satisfying
the BCC structural identity
$$
\Phi_a(-x)=\bar\Phi_a(x),\qquad\text{(BCC reality condition)}
$$
the action is
$$
\mathcal{P}_{\rm BCC}\Psi(x)=\Psi(-x)
=\sum_a\bigl[z_a\bar\Phi_a(x)+\bar z_a\Phi_a(x)\bigr].
$$

\subsection*{3. Independence of $\mathcal{C}$ and $\mathcal{P}_{\rm BCC}$}

\begin{theorem}[Independence and commutativity]\label{thm:indep}
Under the BCC reality condition $\Phi_a(-x)=\bar\Phi_a(x)$ and the
generic CP$^2$ amplitude assumption $z_a\ne\bar z_a$:
\begin{enumerate}
\item $\mathcal{C}\ne\mathcal{P}_{\rm BCC}$ as transformations on the
TECT state space;
\item $\mathcal{C}\mathcal{P}_{\rm BCC}=\mathcal{P}_{\rm BCC}\mathcal{C}$;
\item $\langle\mathcal{C},\mathcal{P}_{\rm BCC}\rangle\simeq\mathbb{Z}_2^{\mathcal{C}}\times\mathbb{Z}_2^{\rm BCC}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(1) Distinctness.}  $\mathcal{C}$ swaps the internal amplitudes
$z_a\leftrightarrow\bar z_a$; $\mathcal{P}_{\rm BCC}$ swaps the mode
functions $\Phi_a\leftrightarrow\bar\Phi_a$.  These coincide only on
the real slice $z_a=\bar z_a$, which has codimension three in
$\mathbb{CP}^2$ (real locus = $\mathbb{RP}^2\subset\mathbb{CP}^2$) and
is not the generic operating point.

\textbf{(2) Commutativity.}  $\mathcal{C}$ acts on the discrete
amplitude indices, $\mathcal{P}_{\rm BCC}$ on the continuous spatial
argument; they act on disjoint factors of the state space, hence
commute.

\textbf{(3) Group structure.}  $\mathcal{C}^2=\mathcal{P}_{\rm BCC}^2=\mathrm{id}$,
$[\mathcal{C},\mathcal{P}_{\rm BCC}]=0$, $\mathcal{C}\ne\mathcal{P}_{\rm BCC}$
$\Rightarrow$ Klein four-group $\mathbb{Z}_2\times\mathbb{Z}_2$.
\end{proof}

\subsection*{4. Refined four-sector cancellation theorem}

\begin{theorem}[Refined Pillar 11 four-sector cancellation]\label{thm:P11-refined}
Let the residual sector label be
$(\eta,\pi)\in\mathbb{Z}_2^{\mathcal{C}}\times\mathbb{Z}_2^{\rm BCC}$
with $\eta=\pm 1$ the CP$^2$ symplectic-orientation charge and
$\pi=\pm 1$ the BCC spatial-parity charge.  Assume the residual scalar
contribution factorises as
$$
\lambda_{\eta,\pi}=\eta\,\lambda_{\pi}.
$$
Then
$$
\lambda_{\rm tot}=\sum_{\eta=\pm 1}\sum_{\pi=\pm 1}\lambda_{\eta,\pi}
=\sum_{\pi=\pm 1}(+1+(-1))\lambda_\pi=0.
$$
\end{theorem}

\begin{proof}
The $\eta$-sum cancels at fixed $\pi$ for every $\pi$.  Hence
$\lambda_{\rm tot}=0$ identically, regardless of the relation between
$\lambda_+$ and $\lambda_-$.  If additionally BCC parity enforces
$\lambda_+=\lambda_-$ (lattice-inversion-symmetric residuals), the
cancellation is sector-symmetric.
\end{proof}

\paragraph{Comment on factorisation hypothesis.}  The factorisation
$\lambda_{\eta,\pi}=\eta\lambda_\pi$ is the analogue of
``CP-odd integrand'' in standard QFT: any local Lagrangian that is
odd under $\mathcal{C}$ produces a residual with $\eta$-linear
dependence.  This is a structural property of the elastic
Lagrangian + CP$^2$ projection.
