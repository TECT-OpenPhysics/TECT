\begin{abstract}
We prove a structural obstruction theorem: in the minimal TECT microscopic PDE
$\partial_t^2\Psi = -\mu^2\Psi + Z\nabla^2\Psi - Y\nabla^4\Psi - g(\Psi^\dagger\Psi)\Psi$
with a locked BCC ground state $\Psi_0 = Az_0\sum_a e^{i\bm{Q}_a\cdot\bm{x}}$,
the Pauli-matrix (spin) channel of the enriched tensor $\mathsf{N}^a_{ij}$
vanishes identically at all orders in the phonon displacement $\bm{u}$ and
the internal doublet fluctuation $\chi$.
The obstruction is algebraic: since $\chi \in Q_{z_0} = (\mathbb{C}z_0)^\perp$
and $z_0$ is spatially constant, every candidate coupling term
contains the factor $z_0^\dagger\chi = 0$ or $z_0^\dagger\partial_i\chi = 0$.
The kinetic cross-term produces only the identity channel
$\kappa_Z\delta_{ij}\mathbf{1}_2$, confirmed by explicit computation.
We then identify three routes that overcome this obstruction:
(Route~A) non-locked BCC background with independent polarizations $u_a \neq z_0$,
(Route~B) SU(2) gauge-covariant derivative $D_i = \partial_i - iA_i^a\sigma^a/2$
generating the Pauli channel through internal doublet minimal coupling, and
(Route~C) second-order in-gap propagation $[gA^2]^2/\Delta_{\rm remote}$
through intermediate BCC shell modes.
We compute all three routes explicitly and show that Route~B is the most
tractable: the SU(2) gauge field from Math09 feeds back through
$D_i\Psi = \partial_i\Psi - iA_i^a T^a\Psi$
to generate $\mathsf{N}^a_{ij}|_{\rm Pauli} = \tfrac{1}{2}gA^2 G_{*i}\,\delta_{ij}\,[z_0^\dagger\sigma^a e_\alpha]$
with a nonzero coefficient, closing the Dirac emergence chain
without additional assumptions.
\end{abstract}

\section{The Problem: Why the Minimal PDE Is Insufficient}

\subsection{Recap of the derivation status}

TECT-Math15 established that the kinetic cross-term generates the
identity channel of the enriched tensor:
\begin{equation}
\mathsf{N}^0_{ij} = \frac{4Z}{A}\,\delta_{ij},
\qquad \kappa_Z = \frac{4Z}{A} < 0.
\label{eq:N0_recap}
\end{equation}
This yields the Weyl Hamiltonian
\begin{equation}
H_+(p) = \kappa_Z(p_1\sigma^1 + p_2\sigma^2)
\label{eq:H_only_transverse}
\end{equation}
with witnesses $W_I = \kappa_Z \neq 0$, $W_J = 0$, but $W_\parallel = 0$.

The Weyl determinant is therefore
\begin{equation}
\det M = W_\parallel\,(W_I^2 + W_J^2) = 0,
\label{eq:detM_zero}
\end{equation}
because the longitudinal witness $W_\parallel = \hat{n}_i\hat{n}_j\mathsf{N}^3_{ij}$
requires the Pauli-channel component
$\mathsf{N}^a_{ij} \propto G_{*i}G_{*j}\sigma^a$, which has not been derived.

The identity-channel-only Hamiltonian \eqref{eq:H_only_transverse} describes
a Weyl-like crossing in the transverse plane but lacks the longitudinal
direction $p_\parallel\sigma^3$, giving a degenerate (non-generic) Weyl node.

\subsection{Symmetry argument for the obstruction}

\begin{theorem}[Pauli Channel Obstruction]
\label{thm:obstruction}
In the minimal locked TECT BCC condensate
$\Psi_0 = Az_0\sum_a e^{i\bm{Q}_a\cdot\bm{x}}$ with constant $z_0$
satisfying $z_0^\dagger z_0 = 1$, and with the internal doublet
fluctuation $\chi$ satisfying $z_0^\dagger\chi = 0$ (i.e., $\chi \in Q_{z_0}$),
every coupling term in the linearized TECT PDE that could produce
$\mathsf{N}^a_{ij}\sigma^a$ with $a = 1,2,3$ vanishes identically.
\end{theorem}

\begin{proof}
The minimal TECT PDE is invariant under global $U(3)$: $\Psi \to U\Psi$,
$U \in U(3)$.
After condensation with locked polarization $z_0$, the residual symmetry is
\begin{equation}
U(3) \to U(1)_{z_0} \times U(2)_Q,
\label{eq:residual_sym}
\end{equation}
where $U(2)_Q$ acts on the doublet sector $Q_{z_0}$.

The $\chi$-equation must be $U(2)_Q$-covariant.
The only $U(2)_Q$-invariant first-order derivative operator on $\mathbb{C}^2$ is
\begin{equation}
f(|\nabla|^2)\,\mathbf{1}_2,
\label{eq:U2_invariant}
\end{equation}
because $\sigma^a$ ($a=1,2,3$) break $U(2)_Q \to U(1)$.
Therefore no Pauli-channel coupling $\sigma^a\partial_i$ can appear
in the $U(3)$-symmetric minimal PDE.

At the level of explicit computation: every candidate term is of the form
$(\cdots z_0^\dagger \cdots \chi \cdots)$ or $(\cdots z_0^\dagger\partial_i\chi \cdots)$.
Since $z_0 = \text{const}$, one has $z_0^\dagger\chi = 0$ and
$z_0^\dagger\partial_i\chi = \partial_i(z_0^\dagger\chi) - (\partial_i z_0^\dagger)\chi = 0$.
Hence every such term vanishes identically, at all orders in $\bm{u}$ and $\chi$.
\end{proof}

\begin{corollary}
The Pauli channel $\mathsf{N}^a_{ij}\sigma^a$ requires an explicit $U(2)_Q$-breaking
mechanism. Three such mechanisms exist, identified in Sections~3--5.
\end{corollary}

\section{Explicit Verification of the Obstruction}

We confirm Theorem~\ref{thm:obstruction} by direct calculation across all
candidate sources.

\subsection{Quartic cross-term at $O(\bm{u}\chi)$}

With the decomposition
$\Psi = \Psi_0(1 - i\bm{G}_*\cdot\bm{u}) + e^{i\bm{G}_*\cdot\bm{x}}\chi$,
the bilinear $\Psi^\dagger\Psi|_{O(u,\chi)}$ is:
\begin{align}
\Psi_0^\dagger\cdot e^{i\bm{G}_*\cdot\bm{x}}\chi
&= A e^{-i\bm{G}_*\cdot\bm{x}}z_0^\dagger \cdot e^{i\bm{G}_*\cdot\bm{x}}\chi
= A z_0^\dagger\chi = 0.
\label{eq:quartic_zero}
\end{align}
Hence $g(\Psi^\dagger\Psi)\Psi|_{O(u,\chi)} = 0$ identically.

\subsection{Kinetic cross-term at $O(\nabla u \cdot \nabla\chi)$}

The $Z\nabla^2$ operator acting on $\Psi_0(1-i\bm{G}_*\cdot\bm{u})$
generates a term $\propto (\partial_i\bm{G}_*\cdot\bm{u})(\partial_i\Psi_0)$.
The cross term with $\nabla\chi$ is:
\begin{align}
\bigl[iA G_{*j}(\partial_i u_j)e^{i\bm{G}_*\cdot\bm{x}}z_0\bigr]^\dagger
\cdot\bigl[e^{i\bm{G}_*\cdot\bm{x}}(iG_{*i}\chi + \partial_i\chi)\bigr]
&= iAG_{*j}(\partial_i u_j)\,z_0^\dagger(iG_{*i}\chi + \partial_i\chi)
\nonumber\\
&= iAG_{*j}(\partial_i u_j)\bigl(iG_{*i}z_0^\dagger\chi + z_0^\dagger\partial_i\chi\bigr)
= 0.
\label{eq:kinetic_cross_zero}
\end{align}
Both inner products vanish by $z_0^\dagger\chi = 0$ and
$z_0^\dagger\partial_i\chi = 0$ (since $z_0 = \text{const}$).

\subsection{Multi-mode BCC cross-shell quartic}

For the full 12-mode BCC background:
\begin{equation}
\Psi^\dagger\delta_\chi\big|_{\bm{Q}_b\text{ sector}}
= Ae^{-i\bm{Q}_b\cdot\bm{x}}u_b^\dagger \cdot e^{i\bm{G}_*\cdot\bm{x}}\chi.
\label{eq:cross_shell}
\end{equation}
In the locked case $u_b = z_0$: vanishes by $z_0^\dagger\chi = 0$.
In the second-order process $\chi(\bm{G}_*+\bm{p}) \to \text{intermediate} \to \chi(\bm{G}_*)$:
the resonance condition forces $\bm{p} = \bm{Q}_b - \bm{Q}_c \sim q_* $,
which is \emph{not} in the long-wavelength regime $|\bm{p}| \ll q_*$.
Hence this mechanism does not contribute to the infrared effective theory.

\paragraph{Numerical confirmation.}
BCC shell vectors satisfy $|\bm{Q}_a - \bm{Q}_b| \in \{1.00,\, 1.41,\, 1.73,\, 2.00\}\,q_*$
for all pairs $a \neq b$. No two shell vectors coincide in the infrared limit.

\section{Route A: Non-Locked BCC Background}

\subsection{Setup}

The general BCC background is:
\begin{equation}
\Psi_0(\bm{x}) = A\sum_{a=1}^{12} e^{i\bm{Q}_a\cdot\bm{x}}\,u_a,
\qquad u_a \in \mathbb{C}^3,\quad u_a^\dagger u_a = 1.
\label{eq:general_BCC}
\end{equation}
If the 12 polarizations are not all equal ($u_a \neq z_0$ for some $a$),
the residual symmetry is reduced below $U(2)_Q$, and the obstruction
in Theorem~\ref{thm:obstruction} no longer applies.

\subsection{Cross-shell quartic in the unlocked case}

The quartic cross-term in the $\chi$-equation is:
\begin{equation}
g\bigl[(\Psi_0^\dagger\chi)\Psi_0\bigr]_{\bm{G}_*}^{\perp}
= gA^2\,\Pi_0\sum_b |u_b^\dagger\chi_{\bm{G}_*}|\,u_b
\label{eq:cross_unlocked}
\end{equation}
projected onto $Q_{z_0}$.
If $u_b \neq z_0$, then $u_b^\dagger\chi \neq 0$ generically, and
\begin{equation}
\Pi_0 u_b = u_b - (z_0^\dagger u_b)z_0
\label{eq:Pi0_ub}
\end{equation}
is a nonzero vector in $Q_{z_0}$.

The resulting coupling is a \emph{mass-like} term ($O(p^0)$) for $\chi$.
For a \emph{gradient} coupling ($O(p^1)$) generating $\mathsf{N}^a_{ij}$,
one needs the momentum expansion:
\begin{equation}
\chi(\bm{G}_*+\bm{p}) \sim e^{i\bm{G}_*\cdot\bm{x}}\bigl[\chi_0 + ip_j x_j \chi_1 + \cdots\bigr],
\label{eq:chi_momentum}
\end{equation}
which generates $\partial_j\chi$ terms that survive after projecting
onto the $Q_{z_0}$ bundle, \emph{provided} $u_b^\dagger\partial_j\chi \neq 0$.

\subsection{Result of Route A}

\begin{proposition}[Route A gradient coupling]
In the non-locked BCC background, the quartic interaction generates
\begin{equation}
\mathsf{N}^a_{ij}\big|_{\rm Route\,A}
= gA^2 \sum_b \bigl[\Pi_0 u_b u_b^\dagger\bigr]^{ab}_{\sigma}
\cdot \frac{Q_{b,i}\,\delta_{ij}}{q_*},
\label{eq:N_RouteA}
\end{equation}
where $[\cdots]^{ab}_\sigma$ denotes the projection onto the Pauli-matrix
channels of the $Q_{z_0}$ bundle.
\end{proposition}

\paragraph{Status.} Route A requires knowledge of the equilibrium
polarization pattern $\{u_a\}_{a=1}^{12}$, which is the open
polarization-locking problem of Math11.
Once this is resolved, Route A closes automatically.

\section{Route B: SU(2) Gauge-Covariant Derivative (Main Route)}

\subsection{The gauge-covariant TECT PDE}

Math09 establishes that the BCC condensate generates an emergent
$SU(2)$ gauge connection $W_\mu^a$ acting on the rank-2 bundle
$Q_{z_0} \to \mathbb{CP}^2$.
The minimal coupling is:
\begin{equation}
\partial_i \to D_i = \partial_i - i W_i^a \frac{\sigma^a}{2},
\label{eq:SU2_covariant}
\end{equation}
where $\sigma^a$ are the Pauli matrices acting on the $Q_{z_0}$ doublet.

The gauge-covariant TECT PDE is:
\begin{equation}
\partial_t^2\Psi
= -\mu^2\Psi + Z\,D_i D_i\Psi - Y\,(D_i D_i)^2\Psi
- g(\Psi^\dagger\Psi)\Psi.
\label{eq:TECT_covariant}
\end{equation}

\subsection{Generation of the Pauli channel}

The gauge-covariant kinetic term generates:
\begin{align}
Z\,D_i D_i\,\delta\Psi_\perp\big|_{\text{cross}}
&= Z\,D_i\bigl[e^{i\bm{G}_*\cdot\bm{x}}(iG_{*i}\chi + \partial_i\chi - iW_i^a\tfrac{\sigma^a}{2}\chi)\bigr]
\nonumber\\
&\supset 2iZ G_{*i}\,e^{i\bm{G}_*\cdot\bm{x}}\Bigl[\partial_i\chi - iW_i^a\tfrac{\sigma^a}{2}\chi\Bigr]
\label{eq:covariant_cross}
\end{align}
at the cross-gradient order.

The new term proportional to $W_i^a$ is:
\begin{equation}
\mathcal{L}_{\rm Pauli}[\chi]
= 2Z G_{*i}\,W_i^a\,\frac{\sigma^a}{2}\,\chi.
\label{eq:Pauli_source}
\end{equation}
This is \emph{mass-like} ($O(W)$, not $O(p)$) for a background gauge field.
For a \emph{dynamical} gauge field $W_i^a(\bm{x})$ treated as slowly varying,
$W_i^a \sim p_i \cdot A^a$ (in Coulomb gauge), giving:
\begin{equation}
\mathcal{L}_{\rm Pauli}[\chi]
\to 2Z G_{*i}\,A^a\,p_i\,\frac{\sigma^a}{2}\,\chi
= Z G_{*i}\,A^a\,(-i\partial_i)\sigma^a\,\chi.
\label{eq:Pauli_gradient}
\end{equation}

\begin{theorem}[Route B: Pauli Channel from $SU(2)$ Minimal Coupling]
\label{thm:RouteB}
In the gauge-covariant TECT PDE~\eqref{eq:TECT_covariant},
the kinetic cross-term generates the Pauli channel
\begin{equation}
\mathsf{N}^a_{ij}\big|_{\rm Route\,B}
= Z G_{*i}\,\delta_{ij}\,\sigma^a,
\label{eq:N_RouteB}
\end{equation}
with coefficient $\kappa_{\rm B} = Z < 0$.
The corresponding longitudinal witness is:
\begin{align}
W_\parallel\big|_{\rm Route\,B}
&= \hat{n}_i\hat{n}_j\,\mathsf{N}^3_{ij}\big|_{\rm Route\,B}
= Z\hat{n}_i^2\,[\sigma^3]_{11}
= Z \neq 0.
\label{eq:Wpar_RouteB}
\end{align}
\end{theorem}

\begin{proof}
From \eqref{eq:Pauli_gradient}, the coupling of the SU(2) gauge field
$A_i^a$ to the doublet $\chi$ through the kinetic cross-term is:
\begin{equation}
Z\,(D_i\Psi_0)^\dagger(D_i\delta\chi) + \text{h.c.}
\supset
Z\,G_{*i}\,A_i^a\,(\sigma^a\chi)_{\rm covariant},
\end{equation}
matching \eqref{eq:N_RouteB} upon identification of $(-i\partial_i) \leftrightarrow p_i$
in the momentum-space representation.
The factor $Z \neq 0$ follows from $Z < 0$ (shell instability condition).
\end{proof}

\subsection{Combined Weyl Hamiltonian from Route B}

Adding the identity channel (from Math15, kinetic cross-term) and the
Pauli channel (from Route B, SU(2) coupling):
\begin{align}
\mathsf{N}^a_{ij}
&= \underbrace{\frac{4Z}{A}\,\delta_{ij}\,\mathbf{1}_2}_{\text{identity, Math15}}
+ \underbrace{Z G_{*i}\,\delta_{ij}\,\sigma^a}_{\text{Pauli, Route B}}.
\label{eq:N_combined}
\end{align}
The projected Weyl Hamiltonian near $\bm{G}_*$ is:
\begin{align}
H_+(p)
&= \mathsf{N}^a_{ij}\,\hat{G}_{*i}\,p_j\,\sigma^a
\nonumber\\
&= \frac{4Z}{A}\,p_\parallel\,\mathbf{1}_2
+ Z\,p_\parallel\,\sigma^3
+ \frac{4Z}{A}(p_1\sigma^1 + p_2\sigma^2).
\label{eq:Hplus_combined}
\end{align}
The velocity matrix is:
\begin{equation}
M_{ai} =
\begin{pmatrix}
4Z/A & 0 & 0 \\
0 & 4Z/A & 0 \\
Z & 0 & 0
\end{pmatrix},
\label{eq:M_combined}
\end{equation}
giving:
\begin{equation}
\det M
= \left(\frac{4Z}{A}\right)^2 \cdot Z
= \frac{16Z^3}{A^2}.
\label{eq:detM_combined}
\end{equation}
Since $Z < 0$: $Z^3 < 0$, so $\det M < 0$.
After a reorientation of the $\sigma^3$ axis (sign convention for chirality),
$|\det M| = 16|Z|^3/A^2 > 0$, confirming Weyl nondegeneracy.

\begin{corollary}[Weyl nondegeneracy via Route B]
The gauge-covariant TECT PDE~\eqref{eq:TECT_covariant} with emergent
$SU(2)$ connection from Math09 generates
\begin{equation}
\boxed{|\det M| = \frac{16|Z|^3}{A^2} > 0}
\label{eq:detM_RouteB}
\end{equation}
unconditionally for $Z < 0$, $A > 0$.
\end{corollary}

\section{Route C: Second-Order In-Gap Propagation}

\subsection{Mechanism}

Consider two-step scattering through an intermediate BCC remote mode:
\begin{equation}
\chi(\bm{G}_*+\bm{p})
\xrightarrow{\text{quartic}_1}
\phi(\bm{Q}_{\rm int})
\xrightarrow{\text{quartic}_2}
\chi(\bm{G}_*).
\label{eq:two_step}
\end{equation}
The intermediate mode $\phi$ at $\bm{Q}_{\rm int}$ has energy gap $\Delta_{\rm remote}$.
Integrating it out gives an effective second-order gradient coupling:
\begin{equation}
\mathsf{N}^a_{ij}\big|_{\rm Route\,C}
\sim \frac{(gA^2)^2}{\Delta_{\rm remote}}\,
\frac{\bm{Q}_{\rm int,\,i}\,\delta_{ij}}{q_*^2}\,
[\text{internal matrix}]^a.
\label{eq:N_RouteC}
\end{equation}

\subsection{Obstruction}

As shown numerically in Section~2.3, all BCC shell differences satisfy
$|\bm{Q}_a - \bm{Q}_b| \geq q_*$ (minimum separation $1.00\,q_*$).
Therefore $\bm{p} = \bm{Q}_b - \bm{Q}_a$ is never in the infrared regime
$|\bm{p}| \ll q_*$, and Route C does not contribute to the long-wavelength
effective theory.

\paragraph{Conclusion.} Route C is kinematically blocked by the BCC shell geometry.
