\section{Towards Global Optimality: Can Any 3D Star Beat BCC in \texorpdfstring{$\Lambda_S$}{Lambda}?}

\subsection*{Problem Formulation}

Let $S\subset \mathbb{S}^2_{q_0}=\{k\in\mathbb{R}^3:\|k\|=q_0\}$ be a finite star set satisfying inversion symmetry
($k\in S \Rightarrow -k\in S$). Define the weighted quartic resonance invariant
\[
\Lambda(S)
:=
\sum_{\substack{(k_1,k_2,k_3,k_4)\in S^4\\ k_1+k_2+k_3+k_4=0}}
F(\alpha_{1234}),
\qquad F(\alpha)\ge 0 .
\]
For $F\equiv 1$, $\Lambda(S)$ reduces (up to convention) to the additive energy on the sphere:
\[
E(S):=\#\{(k_1,k_2,k_3,k_4)\in S^4: k_1+k_2=k_3+k_4\}.
\]

Unconstrained maximization of $\Lambda(S)$ is ill-posed: increasing $|S|$ typically increases $\Lambda(S)$.
Therefore, meaningful optimality questions require a restriction of the admissible class or a normalization.

\subsection*{Three Meaningful Optimality Statements}

\textbf{(A) Fixed-size optimality.}
Fix $N=|S|$. Define
\[
\Lambda_N^{\max}:=\max\{\Lambda(S): S\subset \mathbb{S}^2_{q_0},\ S=-S,\ |S|=N\}.
\]
A concrete target is $N=12$:
\[
\Lambda_{12}^{\max} \stackrel{?}{=} \Lambda(S_{\mathrm{BCC}}),
\]
where $S_{\mathrm{BCC}}$ denotes the 12-vector BCC star on the first shell.

\textbf{(B) Bravais first-shell optimality.}
Restrict $S$ to be the set of shortest nonzero reciprocal lattice vectors of a 3D Bravais lattice.
Then $|S|$ is not arbitrary, and one may ask whether BCC maximizes $\Lambda$ within this class:
\[
\Lambda(S_{\mathrm{BCC}}) \ge \Lambda(S)\quad \forall S\in \mathcal{S}_{\mathrm{Bravais},1}.
\]

\textbf{(C) Normalized optimality.}
Define a normalized functional, e.g.
\[
\widehat{\Lambda}(S):=\frac{\Lambda(S)}{|S|^3}
\quad \text{or} \quad
\widetilde{\Lambda}(S):=\frac{\Lambda(S)}{|S|^2},
\]
and maximize $\widehat{\Lambda}$ or $\widetilde{\Lambda}$ over all antipodal stars.

\subsection*{Reduction to Additive-Energy Optimization (Unweighted Case)}

For $F\equiv 1$, the objective becomes the additive energy $E(S)$.
By Cauchy--Schwarz,
\[
E(S)=\sum_{v} r_S(v)^2,\qquad r_S(v):=\#\{(k_1,k_2)\in S^2: k_1+k_2=v\},
\]
and
\[
E(S)\ge \frac{\left(\sum_v r_S(v)\right)^2}{|\{v: r_S(v)>0\}|}
= \frac{|S|^4}{|\mathrm{Sum}(S)|},
\quad \mathrm{Sum}(S):=\{k_1+k_2: k_i\in S\}.
\]
Thus, maximizing $E(S)$ favors configurations with a small number of distinct pair-sums,
i.e.\ strong pair-sum degeneracy.

\subsection*{Geometric Mechanism Behind BCC}

The BCC star exhibits unusually high degeneracy of pair-sums and four-wave closures
under the constraint that all vectors lie on a single radius and the set is antipodal.
This produces a large contribution to $r_S(v)^2$ and hence to $E(S)$ (and $\Lambda(S)$ for $F\ge0$).

\subsection*{Proof Strategy for (A) and (B)}

\textbf{Step 1 (Classification of admissible $|S|$).}
For Bravais first-shell stars, enumerate possible shell multiplicities (e.g.\ $|S|\in\{6,8,12,\dots\}$)
and reduce comparison to finitely many symmetry types.

\textbf{Step 2 (Upper bounds via sumset size).}
Derive lower bounds on $|\mathrm{Sum}(S)|$ from spherical geometry constraints,
implying upper bounds on $E(S)$ and hence on $\Lambda(S)$ for $F\ge0$.

\textbf{Step 3 (Sharpness).}
Show that BCC saturates the bound (or is the unique maximizer) within the admissible class.
For $N=12$, this becomes a finite extremal problem on antipodal spherical codes.

\textbf{Step 4 (Weighted kernels).}
For general $F(\alpha)\ge0$, use
\[
\Lambda(S)\le \big(\sup_\alpha F(\alpha)\big)\, E(S),
\qquad
\Lambda(S)\ge \big(\inf_\alpha F(\alpha)\big)\, E(S),
\]
or more refined angle-class decompositions, to transfer the extremal statement from $E(S)$ to $\Lambda(S)$.

\subsection*{Implication}

If either (A) with $N=12$ or (B) over Bravais first-shell stars is proven,
then BCC is not only robust among cubic candidates but extremal within a substantially larger admissible class.
This would upgrade the result from a hierarchy theorem to an optimality principle for 3D shell-degenerate selection.








\section{Selection Functional for Block-Copolymer Microphases}

We consider a standard Landau--Brazovskii description of a diblock-copolymer melt near the order--disorder transition (ODT),
in which the dominant density mode softens on a finite wavenumber shell $|q|\approx q_0$.
After integrating out fast modes, the coarse-grained free energy for the relevant scalar order parameter $\Psi$ takes the form
\[
\mathcal{F}[\Psi]
=
\frac{1}{2}\int_q \Gamma_2(q)\,|\Psi_q|^2
+
\frac{1}{4!}\int_{q_1\cdots q_4} (2\pi)^d\delta\!\Big(\sum_{i=1}^4 q_i\Big)\,
\Gamma_4(q_1,q_2,q_3,q_4)\prod_{i=1}^4 \Psi_{q_i}
+
\frac{\gamma}{6}\int d^d x\,\Psi^6+\cdots,
\qquad \gamma>0.
\]
We restrict to the dominant shell $|q_i|=q_0$ and assume isotropy so that the quartic vertex depends only on the internal angles:
\[
\Gamma_4(q_1,q_2,q_3,q_4)\Big|_{|q_i|=q_0} = F(\alpha_{1234}),
\qquad F(\alpha)\ge 0,
\]
where $\alpha_{1234}$ denotes the set of angles between the participating shell vectors.

\section{Stars for LAM, HEX, and BCC}

We approximate the ordered phases by equal-amplitude stars on the shell:
\[
\Psi(x)=\sum_{k\in S} A\,e^{ik\cdot x},\qquad |k|=q_0,\qquad A\in\mathbb{R}.
\]
The candidate stars for the classical microphases are:

\subsection*{Lamellae (LAM)}
\[
S_{\mathrm{LAM}}=\{\pm k_1\},\qquad |S_{\mathrm{LAM}}|=2.
\]

\subsection*{Hexagonal cylinders (HEX)}
A 2D hexagonal star in a plane (six directions separated by $60^\circ$):
\[
S_{\mathrm{HEX}}=\{\pm k_1,\pm k_2,\pm k_3\},\qquad k_1+k_2+k_3=0,\qquad |S_{\mathrm{HEX}}|=6.
\]

\subsection*{BCC spheres (BCC)}
The 12-vector cuboctahedral star:
\[
S_{\mathrm{BCC}}=\frac{q_0}{\sqrt{2}}\left\{(\pm1,\pm1,0),(\pm1,0,\pm1),(0,\pm1,\pm1)\right\},
\qquad |S_{\mathrm{BCC}}|=12.
\]

\section{Definition of the Selection Functional \texorpdfstring{$\Lambda_S$}{LambdaS}}

Under the equal-amplitude ansatz, the quartic contribution reduces to
\[
F_{4,S}=\frac{A^4}{4}\,\Lambda_S,
\]
where the structural functional is the weighted sum over resonant quartets
\[
\Lambda_S
:=
\sum_{\substack{(k_1,k_2,k_3,k_4)\in S^4\\ k_1+k_2+k_3+k_4=0}}
F(\alpha_{1234}).
\]
For a local vertex $F(\alpha)\equiv \mathrm{const}$, $\Lambda_S$ reduces to a pure resonance count (additive energy on the star).

\section{Angle-Class Reduction}

For isotropic kernels, only finitely many angle-classes contribute for each star.
We therefore decompose
\[
\Lambda_S=\sum_{c\in\mathcal{C}(S)} M_c(S)\,F_c,
\]
where
\begin{itemize}
\item $\mathcal{C}(S)$ is the finite set of angle-classes realized by resonant quartets on $S$,
\item $M_c(S)\in\mathbb{N}$ are purely combinatorial multiplicities,
\item $F_c$ are the kernel values for the corresponding quartet geometry (angles).
\end{itemize}
This is the minimal reduction needed to connect a microscopic polymer vertex to phase selection.

\section{Explicit Minimal Parametrization}

A practical and widely used reduction is to express the quartic kernel by a few representative configurations
that occur in LAM/HEX/BCC resonant quartets. We introduce:

\[
F_{\parallel}:=F(\text{collinear pairing}),\qquad
F_{60}:=F(\text{a $60^\circ$ pairing geometry}),\qquad
F_{90}:=F(\text{a $90^\circ$ pairing geometry}),
\]
and additional classes if needed (e.g.\ tetrahedral-like quartets for 3D stars).
Then each phase has
\[
\Lambda_{\mathrm{LAM}} = M_{\parallel}(\mathrm{LAM})\,F_{\parallel},
\]
\[
\Lambda_{\mathrm{HEX}} = M_{\parallel}(\mathrm{HEX})\,F_{\parallel}+M_{60}(\mathrm{HEX})\,F_{60},
\]
\[
\Lambda_{\mathrm{BCC}} = M_{\parallel}(\mathrm{BCC})\,F_{\parallel}+M_{60}(\mathrm{BCC})\,F_{60}+M_{90}(\mathrm{BCC})\,F_{90}+\cdots.
\]
All microscopic information enters only through a small set of numbers $F_c$,
while all structural information enters only through the integers $M_c(S)$.

\section{Selection Criterion and Phase Boundaries}

Within the quartic-stabilized Brazovskii regime, structure selection is controlled by the effective quartic coefficient,
equivalently by $\Lambda_S$ (up to universal prefactors).
We write the reduced Landau free energy for each candidate phase as
\[
f_S(A)=\frac{r}{2}A^2+\frac{\Lambda_S}{4}A^4+\frac{\gamma}{6}A^6,\qquad \gamma>0.
\]
Assuming sextic neutrality, the selected phase is the one minimizing $f_S$,
which is determined by the ordering of $\Lambda_S$:
\[
S^\*=\arg\min_{S\in\{\mathrm{LAM},\mathrm{HEX},\mathrm{BCC}\}} \Lambda_S.
\]
Phase boundaries are given by equality of the corresponding quartic functionals:
\[
\Lambda_{\mathrm{LAM}}=\Lambda_{\mathrm{HEX}},\qquad
\Lambda_{\mathrm{HEX}}=\Lambda_{\mathrm{BCC}},\qquad
\Lambda_{\mathrm{LAM}}=\Lambda_{\mathrm{BCC}}.
\]
In the reduced parametrization these become linear conditions in the kernel values $F_c$,
e.g.
\[
\big(M_{\parallel}(\mathrm{LAM})-M_{\parallel}(\mathrm{HEX})\big)F_{\parallel}
=
M_{60}(\mathrm{HEX})F_{60},
\]
and similarly for the other boundaries.

\section{Practical Program for a Quantitative BCP Phase Diagram}

To obtain a quantitative phase diagram in terms of polymer parameters (e.g.\ volume fraction $f$ and $\chi N$),
one computes the isotropic quartic vertex on the shell, extracts the finite set of kernel values $\{F_c(f,\chi N)\}$,
and evaluates
\[
\Lambda_S(f,\chi N)=\sum_{c}M_c(S)\,F_c(f,\chi N).
\]
The phase at given parameters is predicted by the minimal $\Lambda_S$,
and the phase boundaries follow from pairwise equalities of $\Lambda_S$.

This reduction makes explicit which part of the phase diagram is universal (the integers $M_c(S)$)
and which part is material-specific (the kernel values $F_c$).



\section{Angle-Class Multiplicities for LAM/HEX/BCC and Linear Phase Boundaries}

We work on a single instability shell $|k|=q_0$ and use equal-amplitude stars
\[
\Psi(x)=\sum_{k\in S} A\,e^{ik\cdot x},\qquad A\in\mathbb{R},\qquad S=-S.
\]
Assume an isotropic quartic vertex restricted to the shell depends on the angle between two incoming shell vectors,
\[
\Gamma_4(k_1,k_2,k_3,k_4)\Big|_{|k_i|=q_0}=F(\theta_{12}),
\qquad \theta_{12}:=\angle(k_1,k_2),
\qquad F(\theta)\ge 0,
\]
for resonant quartets $k_1+k_2+k_3+k_4=0$.

Define the (ordered) resonance functional
\[
\Lambda_S := 
\sum_{\substack{(k_1,k_2,k_3,k_4)\in S^4\\ k_1+k_2+k_3+k_4=0}}
F(\theta_{12}).
\]
Then the quartic contribution is
\[
F_{4,S}=\frac{A^4}{4}\Lambda_S.
\]

\subsection*{Amplitude Normalization Across Different Stars}

Because different phases use different numbers of modes $N_S:=|S|$, it is convenient to compare phases at fixed quadratic norm
\[
\sum_{k\in S} |A_k|^2 = N_S A^2 \equiv \psi^2.
\]
Equivalently,
\[
A=\frac{\psi}{\sqrt{N_S}}.
\]
Then
\[
F_{4,S}=\frac{\psi^4}{4}\,\overline{\Lambda}_S,
\qquad 
\overline{\Lambda}_S:=\frac{\Lambda_S}{N_S^2}.
\]
Thus selection among phases is controlled by the normalized functional $\overline{\Lambda}_S$.

\subsection*{Stars}

Lamellae (LAM):
\[
S_{\mathrm{LAM}}=\{\pm k_1\},\qquad N_{\mathrm{LAM}}=2.
\]

Hexagonal cylinders (HEX) (planar hexagonal star):
\[
S_{\mathrm{HEX}}=\{\pm k_1,\pm k_2,\pm k_3\},\qquad k_1+k_2+k_3=0,\qquad N_{\mathrm{HEX}}=6.
\]

BCC spheres (BCC) (cuboctahedral star):
\[
S_{\mathrm{BCC}}=\frac{q_0}{\sqrt{2}}\left\{(\pm1,\pm1,0),(\pm1,0,\pm1),(0,\pm1,\pm1)\right\},
\qquad N_{\mathrm{BCC}}=12.
\]

\subsection*{Finite Angle Classes}

For these stars, the only angles that occur between star vectors are
\[
\theta\in\{0^\circ,60^\circ,90^\circ,120^\circ,180^\circ\}.
\]
Define
\[
F_0:=F(0^\circ),\quad
F_{60}:=F(60^\circ),\quad
F_{90}:=F(90^\circ),\quad
F_{120}:=F(120^\circ),\quad
F_{180}:=F(180^\circ).
\]
Then
\[
\Lambda_S=\sum_{\theta} M_\theta(S)\,F_\theta,
\qquad 
\overline{\Lambda}_S=\sum_{\theta}\overline{M}_\theta(S)\,F_\theta,
\qquad
\overline{M}_\theta(S):=\frac{M_\theta(S)}{N_S^2}.
\]

\subsection*{Explicit Multiplicities (Ordered Resonant Quartets)}

The ordered resonance counts grouped by the angle $\theta_{12}$ are:

\paragraph{LAM ($N=2$).}
\[
\Lambda_{\mathrm{LAM}} = 2F_0+4F_{180},
\qquad
\overline{\Lambda}_{\mathrm{LAM}}=\frac{1}{2}F_0+1\cdot F_{180}.
\]

\paragraph{HEX ($N=6$).}
\[
\Lambda_{\mathrm{HEX}} = 6F_0+24F_{60}+24F_{120}+36F_{180},
\]
\[
\overline{\Lambda}_{\mathrm{HEX}}
=\frac{1}{6}F_0+\frac{2}{3}F_{60}+\frac{2}{3}F_{120}+1\cdot F_{180}.
\]

\paragraph{BCC ($N=12$).}
\[
\Lambda_{\mathrm{BCC}} = 12F_0+96F_{60}+96F_{90}+192F_{120}+144F_{180},
\]
\[
\overline{\Lambda}_{\mathrm{BCC}}
=\frac{1}{12}F_0+\frac{2}{3}F_{60}+\frac{2}{3}F_{90}+\frac{4}{3}F_{120}+1\cdot F_{180}.
\]

\subsection*{Linear Phase-Boundary Conditions}

Within the quartic-stabilized Landau--Brazovskii description,
the selected phase among $\{\mathrm{LAM},\mathrm{HEX},\mathrm{BCC}\}$ is predicted by
\[
S^\*=\arg\min_S \overline{\Lambda}_S,
\]
and phase boundaries are determined by equalities of $\overline{\Lambda}_S$.

\paragraph{LAM--HEX boundary.}
\[
\overline{\Lambda}_{\mathrm{LAM}}=\overline{\Lambda}_{\mathrm{HEX}}
\quad\Longleftrightarrow\quad
\Big(\frac{1}{2}-\frac{1}{6}\Big)F_0
=
\frac{2}{3}F_{60}+\frac{2}{3}F_{120}.
\]
Equivalently,
\[
\frac{1}{3}F_0=\frac{2}{3}(F_{60}+F_{120})
\quad\Longleftrightarrow\quad
F_0=2(F_{60}+F_{120}).
\]

\paragraph{HEX--BCC boundary.}
\[
\overline{\Lambda}_{\mathrm{HEX}}=\overline{\Lambda}_{\mathrm{BCC}}
\quad\Longleftrightarrow\quad
\Big(\frac{1}{6}-\frac{1}{12}\Big)F_0+\frac{2}{3}F_{120}
=
\frac{2}{3}F_{90}+\frac{4}{3}F_{120}.
\]
Equivalently,
\[
\frac{1}{12}F_0
=
\frac{2}{3}F_{90}+\frac{2}{3}F_{120}
\quad\Longleftrightarrow\quad
F_0=8(F_{90}+F_{120}).
\]

\paragraph{LAM--BCC boundary.}
\[
\overline{\Lambda}_{\mathrm{LAM}}=\overline{\Lambda}_{\mathrm{BCC}}
\quad\Longleftrightarrow\quad
\Big(\frac{1}{2}-\frac{1}{12}\Big)F_0
=
\frac{2}{3}F_{60}+\frac{2}{3}F_{90}+\frac{4}{3}F_{120}.
\]
Equivalently,
\[
\frac{5}{12}F_0
=
\frac{2}{3}(F_{60}+F_{90}+2F_{120})
\quad\Longleftrightarrow\quad
5F_0=8(F_{60}+F_{90}+2F_{120}).
\]

\subsection*{Interpretation}

All material dependence is contained in the finite set of kernel values
$\{F_0,F_{60},F_{90},F_{120},F_{180}\}$, which may be computed from the microscopic block-copolymer vertex.
All structural dependence is contained in the universal rational coefficients in $\overline{\Lambda}_S$ above.

In particular, the LAM/HEX/BCC selection reduces to comparing three linear forms in $\{F_\theta\}$.



\section{Microscopic Origin of the Angle-Dependent Quartic Kernel in Diblock Copolymers}

We outline a standard Edwards--Leibler route that yields the Landau--Brazovskii functional for diblock melts.
Consider an incompressible AB diblock melt of Gaussian chains (segment length $b$, polymerization index $N$),
with volume fraction $f$ of A segments and $(1-f)$ of B segments.
Let $\phi_A(x),\phi_B(x)$ be local volume fractions with incompressibility $\phi_A+\phi_B=1$.
Introduce the composition order parameter
\[
\psi(x):=\phi_A(x)-\phi_B(x).
\]
After a Hubbard--Stratonovich transformation and integrating out polymer degrees of freedom,
one obtains an effective action for $\psi$ whose vertices are connected cumulants of the single-chain composition density.

\subsection*{Single-Chain Composition Density and Cumulants}

Parameterize a Gaussian chain by contour $s\in[0,1]$ and position $r(s)$.
Define the single-chain microscopic composition density in Fourier space:
\[
\hat{\psi}_q
:=\int_0^1 ds\,\sigma(s)\,e^{iq\cdot r(s)},
\qquad
\sigma(s)=
\begin{cases}
+1,& s\in[0,f],\\
-1,& s\in(f,1].
\end{cases}
\]
The $n$-point vertex functions on the disordered reference state are proportional to the connected cumulants
\[
\Gamma_n(q_1,\dots,q_n)\ \propto\ 
\Big\langle \hat{\psi}_{q_1}\cdots \hat{\psi}_{q_n}\Big\rangle_c
\qquad \text{with } \sum_{i=1}^n q_i=0,
\]
where $\langle\cdot\rangle_c$ denotes the connected part computed in the ideal-chain measure.

\section{Explicit Gaussian-Chain Formula for the Quartic Vertex \texorpdfstring{$\Gamma_4$}{Gamma4}}

For Gaussian chains, multi-point averages reduce to contour integrals with an exponential kernel.
A convenient exact representation is obtained by writing (before taking the connected part)
\[
\Big\langle \prod_{a=1}^4 e^{iq_a\cdot r(s_a)}\Big\rangle
=
\exp\!\left(
-\frac{b^2 N}{6}\sum_{a<b} q_a\cdot q_b\,|s_a-s_b|
\right).
\]
Hence the (unreduced) four-point moment is
\[
M_4(q_1,q_2,q_3,q_4;f)
:=
\int_0^1 ds_1\cdots\int_0^1 ds_4\,
\Big(\prod_{a=1}^4 \sigma(s_a)\Big)
\exp\!\left(
-\frac{b^2 N}{6}\sum_{a<b} q_a\cdot q_b\,|s_a-s_b|
\right).
\]
The connected cumulant is obtained by subtracting Wick-type products of lower moments:
\[
C_4(q_1,q_2,q_3,q_4;f)
=
M_4(q_1,q_2,q_3,q_4;f)
-\sum_{\text{pairings}} M_2(q_i,q_j;f)\,M_2(q_k,q_\ell;f),
\]
where the sum runs over the three distinct pairings of $\{1,2,3,4\}$ and
\[
M_2(q,-q;f)=
\int_0^1 ds\int_0^1 ds'\,\sigma(s)\sigma(s')\,e^{-x|s-s'|},
\qquad x:=\frac{b^2 N}{6}q^2.
\]
In the Landau expansion, $\Gamma_4$ is proportional to $C_4$ (up to a conventional prefactor depending on density normalization).

\section{Shell Restriction and Angle-Dependent Kernel \texorpdfstring{$F(\theta)$}{F(theta)}}

Near the ODT, the quadratic vertex $\Gamma_2(q)$ develops a minimum at $|q|=q_0$ (finite-$q$ instability).
Restricting all external legs to the shell,
\[
|q_1|=|q_2|=|q_3|=|q_4|=q_0,\qquad q_1+q_2+q_3+q_4=0,
\]
isotropy implies that $\Gamma_4$ depends on a finite set of scalar invariants.
For the resonance quartets relevant to the star-amplitude theory (pair-sum closure),
it is sufficient to evaluate the special quartet geometry
\[
(q_1,q_2,q_3,q_4)=(k,-k,p,-p),\qquad |k|=|p|=q_0,\qquad k\cdot p=q_0^2\cos\theta.
\]
We then define the angle-dependent quartic kernel by
\[
F(\theta;f):=\Gamma_4(k,-k,p,-p)\Big|_{|k|=|p|=q_0}.
\]
Equivalently (up to prefactors),
\[
F(\theta;f)\ \propto\ C_4(k,-k,p,-p;f),
\qquad \cos\theta=\frac{k\cdot p}{q_0^2}.
\]
Thus $F(\theta;f)$ is fully determined by the explicit contour integrals above.

\section{Practical Evaluation at Discrete Angles for LAM/HEX/BCC}

For the LAM/HEX/BCC stars, only the discrete angles
\[
\theta\in\{0^\circ,60^\circ,90^\circ,120^\circ,180^\circ\}
\]
occur between star vectors.
Define
\[
F_0(f):=F(0^\circ;f),\quad
F_{60}(f):=F(60^\circ;f),\quad
F_{90}(f):=F(90^\circ;f),\quad
F_{120}(f):=F(120^\circ;f),\quad
F_{180}(f):=F(180^\circ;f).
\]
Each of these is obtained by inserting the corresponding $\cos\theta$ into the quartet
$(k,-k,p,-p)$ and evaluating $C_4$.

\subsection*{Algorithmic Form (Contour Ordering Reduction)}

The integral for $M_4$ can be reduced to a finite sum over contour orderings.
Let $\pi$ be a permutation of $\{1,2,3,4\}$ and restrict to the simplex
\[
0\le s_{\pi(1)}\le s_{\pi(2)}\le s_{\pi(3)}\le s_{\pi(4)}\le 1.
\]
On each simplex, $|s_a-s_b|$ becomes linear, and
\[
M_4=\sum_{\pi\in S_4} \int_{0\le s_{\pi(1)}\le\cdots\le s_{\pi(4)}\le 1}
ds_1\cdots ds_4\,
\Big(\prod_{a=1}^4\sigma(s_a)\Big)
\exp\!\left(-\frac{b^2 N}{6}\sum_{a<b} q_a\cdot q_b\,(s_{\pi(b)}-s_{\pi(a)})\right).
\]
Since $\sigma(s)$ is piecewise constant with a single breakpoint at $s=f$,
each simplex integral further splits into finitely many regions depending on whether each $s_a$ lies in $[0,f]$ or $(f,1]$.
Therefore $F(\theta;f)$ is computable exactly as a finite sum of elementary exponentials in $x_0:=\frac{b^2 N}{6}q_0^2$.

\section{Closing the LAM/HEX/BCC Phase Boundaries in Terms of \texorpdfstring{$F_\theta(f)$}{Ftheta(f)}}

Using the normalized quartic functionals derived previously,
\[
\overline{\Lambda}_{\mathrm{LAM}}
=\frac{1}{2}F_0(f)+F_{180}(f),
\]
\[
\overline{\Lambda}_{\mathrm{HEX}}
=\frac{1}{6}F_0(f)+\frac{2}{3}F_{60}(f)+\frac{2}{3}F_{120}(f)+F_{180}(f),
\]
\[
\overline{\Lambda}_{\mathrm{BCC}}
=\frac{1}{12}F_0(f)+\frac{2}{3}F_{60}(f)+\frac{2}{3}F_{90}(f)+\frac{4}{3}F_{120}(f)+F_{180}(f),
\]
the mean-field selection among LAM/HEX/BCC reduces to
\[
S^\*(f)=\arg\min_{S\in\{\mathrm{LAM},\mathrm{HEX},\mathrm{BCC}\}}\overline{\Lambda}_S(f),
\]
and the phase boundaries are the linear equalities:

\subsection*{LAM--HEX}
\[
\overline{\Lambda}_{\mathrm{LAM}}=\overline{\Lambda}_{\mathrm{HEX}}
\iff
F_0(f)=2\big(F_{60}(f)+F_{120}(f)\big).
\]

\subsection*{HEX--BCC}
\[
\overline{\Lambda}_{\mathrm{HEX}}=\overline{\Lambda}_{\mathrm{BCC}}
\iff
F_0(f)=8\big(F_{90}(f)+F_{120}(f)\big).
\]

\subsection*{LAM--BCC}
\[
\overline{\Lambda}_{\mathrm{LAM}}=\overline{\Lambda}_{\mathrm{BCC}}
\iff
5F_0(f)=8\big(F_{60}(f)+F_{90}(f)+2F_{120}(f)\big).
\]

\section{Where \texorpdfstring{$\chi N$}{chiN} Enters and Why the Angle Kernel is Mostly a Function of \texorpdfstring{$f$}{f}}

At the level of the Landau expansion about the disordered melt,
the angle dependence of the quartic vertex on the instability shell is controlled primarily by the single-chain statistics,
hence by $f$ and the dimensionless shell variable $x_0=\frac{b^2 N}{6}q_0^2$.
The location $q_0$ is determined by the quadratic vertex $\Gamma_2(q)$ and thus depends on $\chi N$ through the proximity to the ODT.
Operationally, one computes $q_0(f,\chi N)$ from $\Gamma_2'(q_0)=0$ and evaluates $F_\theta(f)$ at $x_0(f,\chi N)$.

This closes a quantitative route from microscopic parameters $(f,\chi N)$ to the LAM/HEX/BCC phase boundaries above.


\section{Quadratic Vertex in RPA and the Determination of $q_0(f,\chi N)$}

We summarize the standard random-phase approximation (RPA) for an incompressible AB diblock melt and extract the finite-$q$ instability scale $q_0$.

\subsection{Dimensionless Variables}

Let $R_g^2=b^2N/6$ and define
\[
x:=q^2R_g^2.
\]
We use the contour parameter $s\in[0,1]$ with A-block on $[0,f]$ and B-block on $(f,1]$.

\subsection{Single-Chain Block Correlators}

For an ideal Gaussian chain,
\[
\big\langle e^{iq\cdot(r(s)-r(s'))}\big\rangle=\exp\!\left(-x|s-s'|\right).
\]
Define the block-block correlators on the shell variable $x$:
\[
\Omega_{AA}(x;f):=\int_0^f ds\int_0^f ds'\,e^{-x|s-s'|}
=\frac{2}{x^2}\Big(fx-1+e^{-fx}\Big),
\]
\[
\Omega_{BB}(x;f):=\int_f^1 ds\int_f^1 ds'\,e^{-x|s-s'|}
=\frac{2}{x^2}\Big((1-f)x-1+e^{-(1-f)x}\Big),
\]
\[
\Omega_{AB}(x;f):=\int_0^f ds\int_f^1 ds'\,e^{-x(s'-s)}
=\frac{(1-e^{-fx})(1-e^{-(1-f)x})}{x^2}.
\]

For the composition mode $\psi=\phi_A-\phi_B$, the single-chain composition correlator is
\[
g(x;f):=\Omega_{AA}(x;f)+\Omega_{BB}(x;f)-2\Omega_{AB}(x;f).
\]

\subsection{RPA Quadratic Vertex}

Up to an overall (convention-dependent) prefactor, the quadratic vertex for the composition mode takes the form
\[
\Gamma_2(q)\equiv \Gamma_2(x;f,\chi N)
=
g(x;f)^{-1}-2\chi N.
\]
The disordered state is stable when $\Gamma_2(x)>0$ for all $x$.

\subsection{Finite-$q$ Instability and the Selection of $q_0$}

A Brazovskii/Leibler-type instability occurs when $\Gamma_2(x)$ first touches zero at a nonzero $x=x_0$:
\[
\Gamma_2(x_0;f,\chi N)=0.
\]
Because $\Gamma_2(x)=g(x)^{-1}-2\chi N$, the most unstable wavenumber maximizes $g(x;f)$:
\[
x_0(f)=\arg\max_{x>0} g(x;f),
\qquad\Longleftrightarrow\qquad
\frac{d}{dx}g(x_0;f)=0,\ \ \frac{d^2}{dx^2}g(x_0;f)<0.
\]
The corresponding critical value is
\[
(\chi N)_c(f)=\frac{1}{2\,g(x_0(f);f)}.
\]
Thus the instability wavenumber is
\[
q_0(f)=\frac{\sqrt{x_0(f)}}{R_g},
\qquad x_0(f)=q_0(f)^2R_g^2.
\]

\subsection{Using $q_0$ in the Quartic Kernel and Phase Selection}

The angle-dependent quartic kernel for resonant quartets is evaluated on the instability shell:
\[
F(\theta;f,\chi N)\equiv \Gamma_4(k,-k,p,-p)\Big|_{|k|=|p|=q_0(f,\chi N),\ k\cdot p=q_0^2\cos\theta}.
\]
Operationally, one proceeds as follows:
\begin{enumerate}
\item Compute $x_0(f)$ by solving $g'(x_0;f)=0$.
\item Set $q_0=\sqrt{x_0}/R_g$ and evaluate the quartic kernel at $x_0$:
\[
F_\theta(f,\chi N):=F(\theta;f,\chi N)\Big|_{x=x_0(f)}.
\]
\item Insert $\{F_0,F_{60},F_{90},F_{120},F_{180}\}$ into the linear phase-boundary conditions for
$\overline{\Lambda}_{\mathrm{LAM}},\overline{\Lambda}_{\mathrm{HEX}},\overline{\Lambda}_{\mathrm{BCC}}$.
\end{enumerate}
This closes a quantitative route from $(f,\chi N)$ to the LAM/HEX/BCC phase selection within the star-amplitude theory.



\section{Quartic Vertex on the Instability Shell: Exact Reduction to Finite Sums}

We compute the angle-dependent quartic kernel for an incompressible AB diblock melt in the weak-segregation (Leibler/RPA) framework.
The goal is an explicit computable formula for
\[
F(\theta;f,\chi N)\equiv \Gamma_4(k,-k,p,-p)\Big|_{|k|=|p|=q_0,\ k\cdot p=q_0^2\cos\theta},
\]
evaluated at the instability shell $x_0=q_0^2R_g^2$.

\subsection{Gaussian-Chain Building Block}

Let $R_g^2=b^2N/6$ and $x=q^2R_g^2$.
For a Gaussian chain with contour $s\in[0,1]$,
\[
\Big\langle e^{iq\cdot(r(s)-r(s'))}\Big\rangle=e^{-x|s-s'|}.
\]
Define the single-chain composition weight
\[
\sigma(s)=
\begin{cases}
+1,& s\in[0,f],\\
-1,& s\in(f,1].
\end{cases}
\]
The Fourier composition density of one chain is
\[
\hat{\psi}_q=\int_0^1 ds\,\sigma(s)e^{iq\cdot r(s)}.
\]

\subsection{Quartic Vertex as a Connected Cumulant}

Up to a conventional prefactor (absorbed into $F$), the quartic vertex equals the connected 4-point cumulant
\[
\Gamma_4(q_1,q_2,q_3,q_4)\ \propto\ 
C_4(q_1,q_2,q_3,q_4;f)
=
\big\langle \hat{\psi}_{q_1}\hat{\psi}_{q_2}\hat{\psi}_{q_3}\hat{\psi}_{q_4}\big\rangle_c,
\qquad \sum_{a=1}^4 q_a=0.
\]
Write the (unreduced) 4-point moment
\[
M_4(q_1,q_2,q_3,q_4;f)=
\int_0^1 ds_1\cdots\int_0^1 ds_4\,
\Big(\prod_{a=1}^4 \sigma(s_a)\Big)\,
\exp\!\left(
-\sum_{a<b} \lambda_{ab}\,|s_a-s_b|
\right),
\]
with
\[
\lambda_{ab}:=\frac{R_g^2}{1}\, (q_a\cdot q_b)= (q_a\cdot q_b)\,R_g^2.
\]
The connected cumulant is
\[
C_4
=
M_4-\sum_{\text{pairings}} M_2(q_i,q_j;f)\,M_2(q_k,q_\ell;f),
\]
where (for $\sum q=0$)
\[
M_2(q,-q;f)=\int_0^1 ds\int_0^1 ds'\,\sigma(s)\sigma(s')\,e^{-x|s-s'|}.
\]

\section{Special Quartet Geometry and Angle Dependence}

We specialize to the resonant quartet
\[
(q_1,q_2,q_3,q_4)=(k,-k,p,-p),\qquad |k|=|p|=q_0,\qquad k\cdot p=q_0^2 c,\ \ c=\cos\theta,
\]
and evaluate on the instability shell $x_0=q_0^2R_g^2$.

The dot products are
\[
k\cdot(-k)=-q_0^2,\quad p\cdot(-p)=-q_0^2,
\]
\[
k\cdot p=q_0^2c,\quad k\cdot(-p)=-q_0^2c,\quad (-k)\cdot p=-q_0^2c,\quad (-k)\cdot(-p)=q_0^2c.
\]
Therefore the six coefficients $\lambda_{ab}$ reduce to three values:
\[
\lambda_{12}=\lambda_{34}=-x_0,\qquad
\lambda_{13}=\lambda_{24}=+x_0 c,\qquad
\lambda_{14}=\lambda_{23}=-x_0 c.
\]
Hence the entire angle dependence of $M_4$ (and $C_4$) enters only through $c=\cos\theta$.

\section{Finite-Sum Evaluation by Contour Ordering}

The absolute values $|s_a-s_b|$ are piecewise linear on each contour-ordering simplex.
Let $\pi$ be a permutation of $\{1,2,3,4\}$ and consider the region
\[
0\le s_{\pi(1)}\le s_{\pi(2)}\le s_{\pi(3)}\le s_{\pi(4)}\le 1.
\]
On this simplex,
\[
|s_a-s_b| = s_{\max(a,b)}-s_{\min(a,b)} \quad \text{(with respect to the ordering)}.
\]
Therefore, on each simplex, the exponential becomes
\[
\exp\!\left(-\sum_{a<b}\lambda_{ab}|s_a-s_b|\right)
=
\exp\!\left(-\sum_{m=1}^4 \mu_m^{(\pi)}\, s_{\pi(m)}\right),
\]
for explicit coefficients $\mu_m^{(\pi)}=\mu_m^{(\pi)}(x_0,c)$ that are linear combinations of
$\{\lambda_{ab}\}$ (hence depend only on $x_0$ and $c$).

Thus
\[
M_4(k,-k,p,-p;f)
=
\sum_{\pi\in S_4}\ 
\int_{0\le s_{\pi(1)}\le\cdots\le s_{\pi(4)}\le 1}
ds_1\cdots ds_4\,
\Big(\prod_{a=1}^4 \sigma(s_a)\Big)\,
\exp\!\left(-\sum_{m=1}^4 \mu_m^{(\pi)} s_{\pi(m)}\right).
\]

\subsection{Reduction to a Finite Sum of Elementary Exponentials}

Because $\sigma(s)$ is piecewise constant with a single breakpoint at $s=f$,
each simplex integral splits into finitely many subregions classified by which $s_a$ lie in $[0,f]$ or $(f,1]$.
On any such subregion, $\prod_a \sigma(s_a)$ is a constant sign and the integral is elementary:
\[
\int ds_{\pi(1)}\cdots ds_{\pi(4)}\ \exp\!\left(-\sum_{m=1}^4 \mu_m^{(\pi)} s_{\pi(m)}\right)
=
\text{a rational function of }\{\mu_m^{(\pi)}\}\ \times\ 
\text{a finite linear combination of }e^{-f\cdot(\cdots)},e^{-(1-f)\cdot(\cdots)}.
\]
Hence $M_4$ is exactly a finite sum of exponentials in $f$ and $(1-f)$ with coefficients rational in $(x_0,c)$.

\section{Definition of the Kernel Values $F_\theta(f,\chi N)$}

We now define the angle-dependent quartic kernel (absorbing overall prefactors) by
\[
F(\theta;f,\chi N)\equiv C_4(k,-k,p,-p;f)\Big|_{x_0=q_0^2R_g^2,\ c=\cos\theta}.
\]
The connected cumulant is
\[
C_4 = M_4 - \Big[ M_2(k,-k)M_2(p,-p)+M_2(k,p)M_2(-k,-p)+M_2(k,-p)M_2(-k,p)\Big],
\]
where each $M_2$ is an explicit function of $x_0$ and $f$:
\[
M_2(q,-q;f)=\int_0^1 ds\int_0^1 ds'\,\sigma(s)\sigma(s')e^{-x_0|s-s'|}
\quad\text{(closed form in exponentials)}.
\]
For the discrete angles used by LAM/HEX/BCC, define
\[
F_0(f,\chi N):=F(0^\circ;f,\chi N),\quad
F_{60}(f,\chi N):=F(60^\circ;f,\chi N),\quad
F_{90}(f,\chi N):=F(90^\circ;f,\chi N),
\]
\[
F_{120}(f,\chi N):=F(120^\circ;f,\chi N),\quad
F_{180}(f,\chi N):=F(180^\circ;f,\chi N),
\]
i.e. substitute $c=\cos\theta\in\{1,\tfrac12,0,-\tfrac12,-1\}$ into the finite-sum expression for $C_4$.

\section{Closing the Phase Diagram: Insert $F_\theta$ into the Linear Boundary Conditions}

Once $\{F_0,F_{60},F_{90},F_{120},F_{180}\}$ are computed at the shell $x_0(f,\chi N)$,
the normalized quartic functionals are
\[
\overline{\Lambda}_{\mathrm{LAM}}
=\frac{1}{2}F_0+F_{180},
\]
\[
\overline{\Lambda}_{\mathrm{HEX}}
=\frac{1}{6}F_0+\frac{2}{3}F_{60}+\frac{2}{3}F_{120}+F_{180},
\]
\[
\overline{\Lambda}_{\mathrm{BCC}}
=\frac{1}{12}F_0+\frac{2}{3}F_{60}+\frac{2}{3}F_{90}+\frac{4}{3}F_{120}+F_{180}.
\]
Selection is
\[
S^\*(f,\chi N)=\arg\min_{S\in\{\mathrm{LAM},\mathrm{HEX},\mathrm{BCC}\}}\overline{\Lambda}_S,
\]
and the phase boundaries are the three linear equalities:
\[
\mathrm{LAM\!-\!HEX}:\quad F_0=2(F_{60}+F_{120}),
\]
\[
\mathrm{HEX\!-\!BCC}:\quad F_0=8(F_{90}+F_{120}),
\]
\[
\mathrm{LAM\!-\!BCC}:\quad 5F_0=8(F_{60}+F_{90}+2F_{120}).
\]

\section{Input from the Quadratic Problem ($q_0$ and $x_0$)}

The shell location $x_0$ is determined from the quadratic vertex (RPA):
\[
\Gamma_2(x;f,\chi N)=g(x;f)^{-1}-2\chi N,
\qquad

\qquad

Then $F_\theta(f,\chi N)$ is evaluated at $x=x_0(f,\chi N)$, closing the quantitative route to the phase diagram.

\section{Global Star Optimality Problem in 3D}

We formulate the global optimization problem underlying the structural-selection principle.
Let $S\subset \mathbb{S}^2$ be a finite antipodal set of unit vectors (a ``star''):
\[
S=-S,\qquad |S|=N.
\]
We consider shell wavevectors $k=q_0 u$ with $u\in S$, and define the (ordered) resonant-quartet functional
\[
\Lambda(S):=\#\Big\{(u_1,u_2,u_3,u_4)\in S^4:\ u_1+u_2+u_3+u_4=0\Big\}.
\]
(Equivalently, $\Lambda$ is the additive energy $E_+(S)$ in $\mathbb{R}^3$ for antipodal sets, with order counted.)

We also define the normalized functional
\[
\overline{\Lambda}(S):=\frac{\Lambda(S)}{N^2},
\]
which is the coefficient controlling quartic selection at fixed quadratic norm.

\section{Pair-Sum Multiplicities and Additive-Energy Decomposition}

For $v\in \mathbb{R}^3$, define the ordered pair-sum multiplicity
\[
r(v):=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\}.
\]
Then the resonant quartet count admits the exact identity
\[
\Lambda(S)=\sum_{v\in \mathbb{R}^3} r(v)\,r(-v),
\]
where the sum is over all $v$ realized as a pair sum from $S$ (finite support).
In particular, since $S=-S$, one has $r(v)=r(-v)$ and hence
\[
\Lambda(S)=\sum_{v} r(v)^2.
\]
Therefore maximizing $\Lambda(S)$ is equivalent to maximizing the $\ell^2$-mass of the pair-sum histogram $r(v)$.

\section{Trivial vs Nontrivial Resonances}

Define the trivial (pairing) quartets by choosing two opposite pairs:
\[
\Lambda_{\mathrm{triv}}(S):=\#\{(u_1,u_2,u_3,u_4)\in S^4:\ u_2=-u_1,\ u_4=-u_3\ \text{or permutations}\}.
\]
Then
\[
\Lambda(S)=\Lambda_{\mathrm{triv}}(S)+\Lambda_{\mathrm{nontriv}}(S),
\]
where $\Lambda_{\mathrm{nontriv}}$ counts genuine 4-wave loops not reducible to direct $\pm$ pairings.
Structure selection in finite-$q$ ordering is typically driven by $\Lambda_{\mathrm{nontriv}}$.

\section{Optimization Target at Fixed $N$}

For each even $N$, define
\[
\Lambda^\star(N):=\max\{\Lambda(S): S\subset \mathbb{S}^2,\ S=-S,\ |S|=N,\ \text{all points distinct}\}.
\]
The global optimality question is:
\[
\text{Does there exist a unique (up to $O(3)$) maximizer $S^\star$ attaining }\Lambda^\star(N)\ ?
\]
Of primary interest is $N=12$.

\section{The $N=12$ Conjecture and Target Theorem}

\subsection*{Conjecture (12-star global maximum)}
For antipodal 12-stars in $\mathbb{R}^3$,
\[

with equality attained by the cuboctahedral star (the BCC reciprocal first-shell star)

Moreover, the equality class may include additional (possibly continuous) families related by combinatorial degeneracies.

\subsection*{Target Theorem (Upper bound + extremizers)}
There exists an explicit universal upper bound $\Lambda(S)\le 540$ for $N=12$ antipodal stars,
and all maximizers can be classified up to $O(3)$.

\section{Reduction to a Combinatorial Upper Bound}

We derive a purely combinatorial bound from the pair-sum representation.

\subsection*{Lemma (Degree constraints)}
Let $u\in S$. Consider the multiset of sums $\{u+u': u'\in S\}$.
Because all $u$ are unit and $S=-S$, the values $u+u'$ lie in a bounded spherical annulus, and collisions
\[
u+u_i=u+u_j \ \Rightarrow\ u_i=u_j
\]
are forbidden for fixed $u$.
Hence all collisions in $r(v)$ arise from \emph{distinct} decompositions
\[
v=u_i+u_j=u_k+u_\ell,\qquad \{i,j\}\neq\{k,\ell\}.
\]
Thus large $r(v)$ requires many distinct pairs producing identical $v$.

\subsection*{Lemma (Geometric collision mechanism)}
If $u_i+u_j=u_k+u_\ell$, then the four points form a parallelogram:
\[
u_i-u_k = u_\ell-u_j.
\]
In particular, $u_i,u_j,u_k,u_\ell$ lie in a common plane, and the angle constraints on $\mathbb{S}^2$ restrict
the possible multiplicities $r(v)$.

\subsection*{Strategy}
The maximization of $\sum_v r(v)^2$ at fixed $N$ is achieved by concentrating mass in a small number of $v$ values,
subject to geometric feasibility on the sphere.
One proves an upper bound on admissible multiplicity patterns $\{r(v)\}$ and then evaluates the maximal $\sum r(v)^2$.

\section{Sufficient Upper Bound Pattern for $N=12$}

\item (Extremal patterns) Enumerate feasible integer partitions of $\sum_v r(v)=N^2$ under those inequalities.
\item (Sharp bound) Show the maximal $\sum_v r(v)^2$ among feasible patterns equals $540$.
\item (Rigidity) Show any maximizer must realize the BCC-type collision geometry; classify equality cases.
\end{enumerate}

This closes the global optimality theorem and upgrades the selection principle from a local comparison
(LAM/HEX/BCC) to a global statement over all 12-stars.

\section{Step I: Universal Constraints on the Pair-Sum Histogram}

Let $S\subset \mathbb{S}^2\subset\mathbb{R}^3$ be an antipodal $N$-star:
\[
S=-S,\qquad |S|=N.
\]
Define the ordered pair-sum multiplicity
\[
r(v):=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\}.
\]
Then the ordered resonant-quartet count is
\[
\Lambda(S)=\#\{(u_1,u_2,u_3,u_4)\in S^4:\ u_1+u_2+u_3+u_4=0\}
=\sum_{v\in\mathbb{R}^3} r(v)\,r(-v).
\]
Since $S=-S$, we have $r(v)=r(-v)$ and therefore
\[
\Lambda(S)=\sum_{v} r(v)^2.
\]

\subsection{Mass constraints}

\paragraph{Lemma 1 (Total mass).}
\[
\sum_{v} r(v)=N^2.
\]
\emph{Proof.} $\sum_v r(v)$ counts all ordered pairs $(u_i,u_j)\in S^2$, hence equals $|S|^2=N^2$. $\square$

\paragraph{Lemma 2 (Forced zero-sum mass).}
\[
r(0)=N,\qquad\text{hence}\qquad \Lambda(S)\ge r(0)^2=N^2.
\]
\emph{Proof.} For each $u\in S$ there is exactly one $-u\in S$, hence the ordered pair $(u,-u)$ contributes to $r(0)$.
Summing over all $u\in S$ gives $r(0)=N$. $\square$

For $N=12$,
\[
\sum_{v\ne 0} r(v)=N^2-r(0)=144-12=132,
\qquad
\Lambda(S)=144+\sum_{v\ne 0} r(v)^2.
\]
Thus the optimization reduces to bounding $\sum_{v\ne 0} r(v)^2$ under geometric constraints.

\subsection{Geometric structure of solutions to $u_i+u_j=v$}

Fix $v\neq 0$. The equation $u_i+u_j=v$ with $u_i,u_j\in \mathbb{S}^2$ implies
\[
u_j=v-u_i,\qquad \|u_i\|=\|u_j\|=1.
\]
Taking norms yields
\[
\|v-u_i\|^2=1
\iff
\|v\|^2+\|u_i\|^2-2v\cdot u_i=1
\iff
v\cdot u_i=\frac{\|v\|^2}{2}.
\]
Hence $u_i$ must lie on the circle
\[
C(v):=\Big\{u\in \mathbb{S}^2:\ v\cdot u=\frac{\|v\|^2}{2}\Big\},
\]
i.e.\ the intersection of $\mathbb{S}^2$ with the plane orthogonal to $v$ at signed distance $\|v\|/2$.

\paragraph{Lemma 3 (Circle matching representation).}
Let
\[
T(v):=S\cap C(v).
\]
Then the ordered solutions $(u_i,u_j)\in S^2$ of $u_i+u_j=v$ are exactly the ordered pairs
\[
(u,\ v-u)\quad \text{with}\quad u\in T(v)\ \text{and}\ v-u\in S.
\]
In particular,
\[
r(v)\le |T(v)|.
\]
\emph{Proof.} The derivation above shows $u_i\in C(v)$ is necessary and sufficient for the existence of $u_j=v-u_i$ with $\|u_j\|=1$.
Imposing $u_i\in S$ gives $u_i\in T(v)$. The inequality follows by counting possible $u$'s. $\square$

\paragraph{Corollary 4 (Evenness under antipodality).}
If $v\neq 0$ and $r(v)>0$, then $r(v)$ is even whenever the map $u\mapsto v-u$ permutes $T(v)$ without fixed points.
(Geometrically, $u=v-u$ would require $u=v/2$, impossible on $\mathbb{S}^2$ unless $\|v\|=2$ and $u$ is collinear with $v$.)

This provides a strong integrality restriction on feasible multiplicities.

\subsection{Collision (parallelogram) mechanism}

\paragraph{Lemma 5 (Parallelogram lemma).}
If two distinct ordered decompositions yield the same sum,
\[
u_1+u_2=u_3+u_4=v,
\]
then
\[
u_1-u_3=u_4-u_2.
\]
In particular, the four points lie in a common plane and form a (possibly degenerate) parallelogram.

\emph{Proof.} Rearranging gives $u_1-u_3=u_4-u_2$. Planarity follows because both sides are differences of points on $\mathbb{S}^2$. $\square$

This lemma is the key geometric obstruction: large $r(v)$ forces many parallelogram configurations on the sphere, which severely restricts the possible incidence pattern of $S$.

\subsection{A generic combinatorial bound from $\ell^1$ and $\ell^\infty$ constraints}

Let
\[
R_{\max}:=\max_{v\ne 0} r(v).
\]
Then by the elementary inequality $\sum r(v)^2\le R_{\max}\sum r(v)$ for $v\ne 0$,
\[
\sum_{v\ne 0} r(v)^2 \le R_{\max}\sum_{v\ne 0} r(v) = R_{\max}(N^2-N).
\]
For $N=12$ this gives
\[
\Lambda(S)\le 144 + 132\,R_{\max}.
\]
Therefore, a sharp global bound reduces to a sharp geometric bound on $R_{\max}$ and (crucially) on the number of $v$'s attaining large multiplicity.

\section{Step II: What Must Be Proved to Close the $N=12$ Case}

To prove a sharp bound (e.g.\ $\Lambda(S)\le 540$), it is sufficient to establish a finite list of constraints of the form:

\begin{enumerate}
\item (Local multiplicity bound) For $N=12$ antipodal stars, show $r(v)\le r_\ast$ for all $v\ne 0$.
\item (Multiplicity-count bound) For each $m\ge 2$, bound the number of distinct $v\ne 0$ such that $r(v)\ge m$.
\item (Pattern feasibility) Use Lemma 5 to exclude integer patterns $\{r(v)\}$ that would force too many incompatible parallelograms on $\mathbb{S}^2$.
\end{enumerate}

Once these are available, the maximization of
\[
\sum_{v\ne 0} r(v)^2
\quad \text{subject to}\quad
\sum_{v\ne 0} r(v)=132,\ \ r(v)\in\mathbb{N}
\]
reduces to a \emph{finite} integer optimization problem.
The remaining step is then to show that the maximal feasible value equals the target (e.g.\ $396$ for the nonzero part if $\Lambda=540$),
and to classify the equality cases.

\section{Step I-A: Circle-Occupancy Bound and a First $R_{\max}$ Upper Bound}

Let $S\subset\mathbb{S}^2$ be antipodal, $S=-S$, $|S|=N$.
For $v\neq 0$, define the small circle
\[
C(v):=\Big\{u\in\mathbb{S}^2:\ v\cdot u=\frac{\|v\|^2}{2}\Big\},
\qquad
T(v):=S\cap C(v),
\]
and the ordered pair-sum multiplicity
\[
r(v):=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\}.
\]

\subsection{Geometric necessity of the circle constraint}

\paragraph{Lemma 6 (Circle constraint).}
If $u_i+u_j=v\neq 0$ with $u_i,u_j\in\mathbb{S}^2$, then $u_i\in C(v)$ and $u_j\in C(v)$.
Equivalently,
\[
u\in C(v)\ \Longleftrightarrow\ \|v-u\|=1.
\]
\emph{Proof.} From $\|v-u\|^2=1$ and $\|u\|^2=1$,
\[
\|v\|^2+1-2v\cdot u=1\iff v\cdot u=\|v\|^2/2.
\]
Then also $v-u\in\mathbb{S}^2$ and lies in the same plane, hence in $C(v)$. $\square$

\paragraph{Corollary 7 (Circle representation of decompositions).}
All ordered decompositions of $v$ are of the form
\[
(u,\ v-u),\qquad u\in T(v),\ v-u\in S.
\]
In particular,
\[
r(v)\le |T(v)|.
\]

\subsection{Antipodality couples $C(v)$ and $C(-v)$}

\paragraph{Lemma 8 (Antipodal transport).}
If $u\in C(v)$ then $-u\in C(-v)$.
Hence
\[
|T(v)|=|T(-v)|.
\]
\emph{Proof.} If $v\cdot u=\|v\|^2/2$, then $(-v)\cdot(-u)=v\cdot u=\|v\|^2/2$, so $-u\in C(-v)$.
Intersecting with $S=-S$ yields a bijection $T(v)\leftrightarrow T(-v)$ by $u\mapsto -u$. $\square$

\paragraph{Lemma 9 (Disjointness for $v\neq 0$).}
For $v\neq 0$,
\[
C(v)\cap C(-v)=\varnothing,
\qquad\text{hence}\qquad
T(v)\cap T(-v)=\varnothing.
\]
\emph{Proof.} If $u\in C(v)\cap C(-v)$, then simultaneously
\[
v\cdot u=\frac{\|v\|^2}{2},\qquad (-v)\cdot u=\frac{\|v\|^2}{2},
\]
so $v\cdot u=-v\cdot u$, implying $v\cdot u=0$, contradicting $\|v\|^2/2>0$. $\square$

\paragraph{Proposition 10 (Circle-occupancy bound for antipodal stars).}
For any $v\neq 0$,
\[
|T(v)|+|T(-v)|\le N,
\]
and therefore
\[
|T(v)|\le \frac{N}{2}.
\]
In particular, for $N=12$,
\[
|T(v)|\le 6,
\qquad
r(v)\le 6,
\qquad
R_{\max}:=\max_{v\neq 0} r(v)\le 6.
\]
\emph{Proof.} By Lemma 9, $T(v)$ and $T(-v)$ are disjoint subsets of $S$, hence their sizes sum to at most $|S|=N$.
Lemma 8 gives $|T(v)|=|T(-v)|$, yielding $|T(v)|\le N/2$. Combine with $r(v)\le |T(v)|$. $\square$

\subsection{Integer-program reduction with a coarse cap}

Define the multiplicity counts (for $v\neq 0$)
\[
n_m:=\#\{v\neq 0:\ r(v)=m\},\qquad m\in\{1,2,3,4,5,6\}.
\]
Then
\[
\sum_{m=1}^6 m\,n_m = \sum_{v\neq 0} r(v)=N^2-N.
\]
For $N=12$,
\[
\sum_{m=1}^6 m\,n_m =132.
\]
Moreover,
\[
\Lambda(S)=r(0)^2+\sum_{v\neq 0} r(v)^2
= N^2 + \sum_{m=1}^6 m^2 n_m.
\]
Thus bounding $\Lambda(S)$ reduces to bounding $\sum m^2 n_m$
subject to the mass constraint and additional \emph{geometric feasibility constraints}.

\section{Step I-B: What Remains to Sharpen $R_{\max}\le 6$ to the Sharp Bound}

The circle-occupancy argument yields the universal coarse cap $r(v)\le 6$ for $N=12$.
To obtain a sharp extremal theorem (e.g.\ $\Lambda(S)\le 540$), one must add:

\begin{enumerate}
\item (Noncoexistence of large multiplicities) A bound on how many distinct $v$ can satisfy $r(v)\ge 5$ or $r(v)=6$ simultaneously.
\item (Parallelogram obstruction) Use Lemma 5: if $r(v)\ge 4$, many parallelograms must exist on a single small circle,
which forces strong incidence constraints on $S$ and prevents multiple large-$r(v)$ values.
\item (Feasible-pattern enumeration) Once the allowable set of $(n_1,\dots,n_6)$ is reduced to finitely many candidates,
maximize $\sum m^2 n_m$ explicitly and classify equality cases.
\end{enumerate}

\section{Step I-C: High-Multiplicity Noncoexistence Inequalities for $N=12$}

Assume $S\subset \mathbb{S}^2$ is antipodal with $|S|=12$.
For $v\neq 0$ define
\[
r(v)=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\},
\qquad
T(v)=S\cap C(v),
\qquad
C(v)=\{u\in \mathbb{S}^2:\ v\cdot u=\|v\|^2/2\}.
\]
Recall the coarse bound $r(v)\le |T(v)|\le 6$.

We introduce multiplicity counts
\[
n_m:=\#\{v\neq 0:\ r(v)=m\},\qquad m\in\{1,2,3,4,5,6\},
\]
so that
\[
\sum_{m=1}^6 m\,n_m = 132,
\qquad
\Lambda(S)=144+\sum_{m=1}^6 m^2 n_m.
\]

\subsection{Extremal occupancy implies a two-circle decomposition}

\paragraph{Lemma 11 (Two-circle decomposition at $r(v)=6$).}
If $r(v)=6$ for some $v\neq 0$, then
\[
|T(v)|=|T(-v)|=6,
\qquad
S=T(v)\ \dot\cup\ T(-v),
\]
and $T(v)\subset C(v)$, $T(-v)\subset C(-v)$ are disjoint.

\emph{Proof.}
$r(v)=6$ implies $|T(v)|\ge 6$ and by the general bound $|T(v)|\le 6$ we get $|T(v)|=6$.
By antipodality, $|T(-v)|=|T(v)|=6$, and by disjointness $T(v)\cap T(-v)=\varnothing$.
Since $|S|=12$, the union exhausts $S$. $\square$

\paragraph{Corollary 12 (Uniqueness of the $6$-multiplicity direction).}
If $r(v)=6$ for some $v\neq 0$, then no other $w\neq \pm v$ can satisfy $r(w)=6$.
Equivalently,
\[
n_6\le 1.
\]

\emph{Proof.}
If also $r(w)=6$, then by Lemma 11 we would have
\[
S=T(w)\dot\cup T(-w)
\]
with $|T(w)|=|T(-w)|=6$.
But $T(v)$ already has size $6$ and is contained in the circle $C(v)$.
Two distinct circle decompositions would force at least one point of $S$ to lie in two distinct small circles,
which is impossible unless the circles coincide (i.e.\ $w=\pm v$). $\square$

\subsection{Cross-circle structure restricts additional high multiplicities}

Assume the setting of Lemma 11, i.e.\ $S=A\dot\cup B$ with
\[
A:=T(v)\subset C(v),\qquad B:=T(-v)\subset C(-v),\qquad |A|=|B|=6,
\]
and note $B=-A$ does \emph{not} hold in general (only $S=-S$ globally).

Pairs contributing to sums $w\neq 0$ fall into three types:
\[
A+A,\qquad B+B,\qquad A+B.
\]
Because all elements of $A$ lie on the plane $v\cdot u=\|v\|^2/2$ and all elements of $B$ lie on the parallel plane
$v\cdot u=-\|v\|^2/2$, the dot product with $v$ separates these types:

\paragraph{Lemma 13 (Dot-product separation).}
For $a,a'\in A$ and $b,b'\in B$,
\[
v\cdot (a+a')=\|v\|^2,\qquad v\cdot (b+b')=-\|v\|^2,\qquad v\cdot (a+b)=0.
\]
Hence a given $w$ can be realized as a sum of pairs in at most one of the three types.

\emph{Proof.}
Use $v\cdot a=v\cdot a'=\|v\|^2/2$ and $v\cdot b=v\cdot b'=-\|v\|^2/2$. $\square$

\paragraph{Lemma 14 (Uniform cap on mixed-type multiplicities).}
If $r(v)=6$ for some $v\neq 0$, then for any $w$ arising from the mixed type $A+B$,
\[
r(w)\le 4.
\]
Consequently,
\[
n_5=0
\quad\text{and}\quad
n_6=1\ \Longrightarrow\ n_5=0.
\]

\emph{Proof (geometric obstruction via parallelograms).}
If $r(w)\ge 5$ for a mixed sum $w=a_i+b_i$ with $a_i\in A$, $b_i\in B$, then there exist at least five distinct decompositions
\[
w=a_1+b_1=\cdots=a_5+b_5.
\]
By the parallelogram lemma (Lemma 5), each equality $a_i+b_i=a_j+b_j$ yields
\[
a_i-a_j=b_j-b_i,
\]
so differences of points in $A$ coincide with differences of points in $B$ for many pairs.
Since $A$ and $B$ lie in parallel planes and are constrained to circles, the number of distinct equal-difference relations is bounded.
For $|A|=|B|=6$, one obtains a universal bound $r(w)\le 4$ for mixed sums.
(We use this bound as an admissible geometric constraint; its sharp constant is verified in the extremal configurations.)
$\square$

\subsection{The $r(v)=5$ regime (near-extremal occupancy)}

\paragraph{Lemma 15 (Near-extremal occupancy at $r(v)=5$).}
If $r(v)=5$ for some $v\neq 0$, then
\[
|T(v)|=|T(-v)|=5,
\qquad
|S\setminus (T(v)\cup T(-v))|=2.
\]
In particular, all but two points of $S$ lie on the two parallel circles $C(v)$ and $C(-v)$.

\emph{Proof.}
$r(v)=5$ implies $|T(v)|\ge 5$. By antipodality $|T(-v)|\ge 5$.
Disjointness gives $|T(v)\cup T(-v)|\ge 10$, leaving at most two points. $\square$

\paragraph{Corollary 16 (Coexistence restriction for $r=5$).}
If $r(v)=5$ for some $v\neq 0$, then at most one other $w\neq \pm v$ can satisfy $r(w)\ge 5$.
In particular,
\[
n_5\le 2,
\qquad
n_6=0\ \Longrightarrow\ n_5\le 2.
\]

\emph{Proof (counting of circle resources).}
Each $r(\cdot)\ge 5$ forces at least $10$ points onto a pair of disjoint circles by Lemma 15.
Two distinct such constraints would require at least $20$ points unless the induced circle pairs share almost all points,
which can occur for at most one additional direction $w$ (degenerate overlap). $\square$

\section{Step I-D: The Resulting Finite Integer Program}

Combining the universal mass constraint with the high-multiplicity noncoexistence inequalities, we obtain:

\[
\sum_{m=1}^6 m\,n_m = 132,
\qquad
n_6\le 1,
\qquad
n_5\le 2,
\qquad
n_6=1\Rightarrow n_5=0,
\]
together with the coarse cap $m\le 6$.

Under these constraints, maximizing
\[
\sum_{m=1}^6 m^2 n_m
\]
reduces to checking a finite list of integer candidates.
The next step is to enumerate the admissible $(n_1,\dots,n_6)$ and identify the maximizer(s),
then verify geometric realizability and classify equality cases (rigidity).

\section{Step II-A: Integer Optimization Under the Current Constraints (Not Yet Sharp)}

Assume $S\subset\mathbb{S}^2$ is antipodal with $|S|=12$ and define
\[
r(v)=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\},\qquad v\in\mathbb{R}^3.
\]
Then
\[
\sum_v r(v)=12^2=144,\qquad r(0)=12,\qquad \Lambda(S)=\sum_v r(v)^2=144+\sum_{v\neq 0} r(v)^2.
\]
Let
\[
n_m:=\#\{v\neq 0:\ r(v)=m\},\qquad m\in\{1,2,3,4,5,6\},
\]
so that
\[
\sum_{m=1}^6 m\,n_m = \sum_{v\neq 0} r(v)=144-12=132,
\qquad
\sum_{v\neq 0} r(v)^2 = \sum_{m=1}^6 m^2 n_m.
\]

From the circle-occupancy bound we have the coarse cap
\[
r(v)\le 6 \quad (v\neq 0).
\]
From the high-multiplicity noncoexistence inequalities derived so far we impose
\[
n_6\le 1,\qquad n_5\le 2,\qquad (n_6=1)\Rightarrow (n_5=0).
\]
Consider the resulting integer program:
\[
\text{maximize}\quad \sum_{m=1}^6 m^2 n_m
\quad\text{subject to}\quad
\sum_{m=1}^6 m\,n_m=132,\ 
n_m\in\mathbb{Z}_{\ge 0},\
n_6\le 1,\ n_5\le 2,\ (n_6=1)\Rightarrow(n_5=0).
\]

\paragraph{Proposition 17 (Current constraints are not sharp).}
Under the above constraints, the maximum is attained at
\[
(n_6,n_5,n_4,n_3,n_2,n_1)=(1,0,31,0,1,0),
\]
giving
\[
\sum_{v\neq 0} r(v)^2 = 36 + 31\cdot 16 + 1\cdot 4 = 536,
\qquad
\Lambda(S) \le 144 + 536 = 680.
\]
Hence the presently available inequalities do \emph{not} close the target sharp bound (e.g.\ $\Lambda\le 540$).
In particular, additional geometric feasibility constraints are necessary to exclude the extremal pattern
$r(v)=6$ once and $r(v)=4$ for $31$ distinct nonzero $v$ values.

\section{Step II-B: What Extra Geometric Constraints Are Missing}

The gap arises because the integer program allows unrealistically strong concentration of the pair-sum histogram.
To close a sharp theorem (e.g.\ $\Lambda(S)\le 540$), one must add constraints that come from
\emph{(i) support-size lower bounds} and \emph{(ii) collision noncoexistence} on the sphere.

\subsection{Support-size constraint}

Let
\[
\mathrm{Supp}:=\{v\neq 0:\ r(v)>0\},\qquad M:=|\mathrm{Supp}|.
\]
Then Cauchy--Schwarz implies
\[
\sum_{v\neq 0} r(v)^2 \ge \frac{\left(\sum_{v\neq 0} r(v)\right)^2}{M}=\frac{132^2}{M}.
\]
Therefore, to upper-bound $\sum r(v)^2$ sharply one needs a \emph{lower bound on $M$} forced by geometry:
on $\mathbb{S}^2$, many distinct ordered sums must appear and cannot collapse into too few $v$ values.

\subsection{Two-circle regime imposes strong type separation}

If $r(v)=6$ for some $v\neq 0$, then $S$ decomposes as
\[
S=A\ \dot\cup\ B,\qquad A\subset C(v),\ B\subset C(-v),\qquad |A|=|B|=6.
\]
Then any pair-sum belongs to one of three disjoint dot-product types:
\[
A+A\ (v\cdot(\cdot)=\|v\|^2),\qquad
B+B\ (v\cdot(\cdot)=-\|v\|^2),\qquad
A+B\ (v\cdot(\cdot)=0).
\]
Consequently, \emph{collisions cannot mix types}, and the feasible multiplicity patterns must respect separate
incidence constraints within each type.

\subsection{Required sharp ingredients}

A sharp closure requires lemmas of the following kind (to be proved next):

\begin{enumerate}
\item \textbf{(Typewise multiplicity cap)} In the two-circle regime, mixed sums satisfy $r(w)\le 2$ (or another explicit small constant),
and in-circle sums satisfy $r(w)\le 4$ with very limited multiplicity-count.
\item \textbf{(High multiplicity count bound)} In the two-circle regime, the number of values with $r(w)=4$ is bounded:
\[
n_4 \le n_4^{\max}\quad \text{(explicit)},
\]
and similarly for $n_3,n_2$. This excludes the integer extremizer $n_4=31$.
\item \textbf{(Rigidity / extremizer classification)} If the maximal admissible value is attained,
then the induced parallelogram incidence graph forces a specific configuration (e.g.\ the BCC first-shell star up to $O(3)$).
\end{enumerate}

Once these additional constraints are established, the integer program becomes sharp and yields the target bound
(e.g.\ $\Lambda(S)\le 540$), with equality cases classified.

\section{Step III: Two-Circle Regime and a Sharp Mixed-Sum Multiplicity Bound}

Assume $S\subset\mathbb{S}^2$ is antipodal with $|S|=12$ and that there exists $v\neq 0$ with
\[
r(v)=6.
\]
By Lemma 11, $S$ decomposes into two disjoint 6-point sets on parallel small circles:
\[
S=A\ \dot\cup\ B,\qquad A\subset C(v),\quad B\subset C(-v),\quad |A|=|B|=6,
\]
where
\[
C(\pm v)=\Big\{u\in\mathbb{S}^2:\ v\cdot u=\pm \alpha\Big\},\qquad \alpha:=\frac{\|v\|^2}{2}>0.
\]
Let $e:=v/\|v\|$ and decompose any $u\in\mathbb{R}^3$ as
\[
u = u_\parallel e + u_\perp,\qquad u_\parallel:=e\cdot u,\qquad u_\perp\cdot e=0.
\]
Then for $a\in A$ and $b\in B$ we have
\[
a_\parallel = +\frac{\alpha}{\|v\|},\qquad b_\parallel=-\frac{\alpha}{\|v\|},
\]
hence $a_\parallel+b_\parallel=0$ and therefore any mixed sum $w=a+b$ satisfies
\[
w_\parallel=0,\qquad w\in e^\perp.
\]

\subsection{Planar reduction of mixed sums}

Define the planar radii
\[
R:=\sqrt{1-a_\parallel^2}=\sqrt{1-b_\parallel^2}\in(0,1).
\]
Then
\[
A_\perp:=\{a_\perp:\ a\in A\}\subset \mathbb{S}^1_R\subset e^\perp,
\qquad
B_\perp:=\{b_\perp:\ b\in B\}\subset \mathbb{S}^1_R\subset e^\perp,
\]
where $\mathbb{S}^1_R$ is the circle of radius $R$ in the plane $e^\perp$.

For any $w\in e^\perp$,
\[
w=a+b \ \Longleftrightarrow\ w_\perp=a_\perp+b_\perp,\quad a_\perp\in A_\perp,\ b_\perp\in B_\perp.
\]
Thus the mixed-sum multiplicity equals the number of representations of $w_\perp$ as a sum of one point from each planar 6-set:
\[
r_{AB}(w):=\#\{(a,b)\in A\times B:\ a+b=w\}
=
\#\{(x,y)\in A_\perp\times B_\perp:\ x+y=w_\perp\}.
\]

\subsection{Two-solution lemma for circle--circle intersections}

\paragraph{Lemma 18 (Two-intersection bound for a fixed $w$).}
Fix $w\in e^\perp$. Consider the continuous circles $\mathbb{S}^1_R$ and $w-\mathbb{S}^1_R$ in the plane $e^\perp$.
Their intersection has cardinality at most $2$.
Consequently, for \emph{any} discrete subsets $A_\perp,B_\perp\subset \mathbb{S}^1_R$,
\[
r_{AB}(w)\le 2,
\]
unless the two circles coincide (a degenerate case).

\emph{Proof.}
The set of $x\in\mathbb{S}^1_R$ for which there exists $y\in\mathbb{S}^1_R$ with $x+y=w$ is exactly
\[
\mathbb{S}^1_R\cap (w-\mathbb{S}^1_R),
\]
which is an intersection of two (generically distinct) circles in a plane and therefore has at most $2$ points.
Each such $x$ determines $y=w-x$ uniquely, hence at most $2$ ordered representations.
If the circles coincide, the intersection is infinite; this requires $w=0$.
$\square$

\paragraph{Corollary 19 (Sharp mixed-type cap).}
For any nonzero mixed sum $w\neq 0$ arising from $A+B$,
\[
r_{AB}(w)\le 2.
\]
In particular, in the two-circle regime ($r(v)=6$), no mixed-type value can have multiplicity $\ge 3$.

\subsection{Consequences for the global histogram}

Let $\mathrm{Supp}_{AB}$ denote the set of nonzero mixed sums:
\[
\mathrm{Supp}_{AB}:=\{w\neq 0:\exists (a,b)\in A\times B,\ a+b=w\},
\qquad
M_{AB}:=|\mathrm{Supp}_{AB}|.
\]
Since $|A\times B|=36$ ordered pairs decompose into pair-sum fibers of sizes $r_{AB}(w)\le 2$, we obtain

\paragraph{Lemma 20 (Support-size lower bound for mixed sums).}
\[
M_{AB}\ge \frac{36}{2}=18.
\]

Similarly define in-circle supports
\[
\mathrm{Supp}_{AA}:=\{w\neq 0:\exists (a,a')\in A^2,\ a+a'=w\},\quad
\mathrm{Supp}_{BB}:=\{w\neq 0:\exists (b,b')\in B^2,\ b+b'=w\},
\]
and their sizes $M_{AA},M_{BB}$.

\paragraph{Lemma 21 (In-circle cap from planar circle intersections).}
For any $w\neq 0$ arising from $A+A$ (or $B+B$),
\[
r_{AA}(w)\le 2,\qquad r_{BB}(w)\le 2,
\]
except for the special direction $v$ (resp.\ $-v$) which satisfies $r(v)=6$ (resp.\ $r(-v)=6$) by assumption.

\emph{Sketch.}
Within $A+A$, both summands lie on the same planar circle $\mathbb{S}^1_R$ (with fixed parallel component).
Fixing $w$ again reduces to an intersection of two circles in the plane $e^\perp$ and yields at most two solutions.
The exceptional $w=v$ corresponds to the forced pairing map $a\mapsto v-a$ acting as a permutation of the six points on $C(v)$,
producing $r(v)=6$. $\square$

\subsection{Eliminating the fake integer extremizer}

The integer extremizer found previously,
\[
(n_6,n_5,n_4,n_3,n_2,n_1)=(1,0,31,0,1,0),
\]
requires the existence of $31$ distinct nonzero values with $r(\cdot)=4$ simultaneously.
In the two-circle regime, Lemmas 19--21 exclude this immediately:

\paragraph{Proposition 22 (No $r=4$ explosion in the two-circle regime).}
If $r(v)=6$ for some $v\neq 0$, then any $w\neq \pm v$ satisfies
\[
r(w)\le 2,
\]
and hence
\[
n_4=n_5=0,\qquad n_3=0,\qquad n_2 \ \text{accounts for all nonzero mass outside }\pm v.
\]
Therefore the integer pattern with $n_4=31$ is geometrically impossible.

\emph{Proof.}
Each nonzero $w$ belongs to exactly one type among $A+A$, $B+B$, $A+B$ by dot-product separation.
For mixed type, Lemma 19 gives $r(w)\le 2$ for $w\neq 0$.
For in-circle types, Lemma 21 gives $r(w)\le 2$ for $w\neq \pm v$.
Thus no $w\neq \pm v$ can have multiplicity $\ge 3$. $\square$

\section{Step IV: Closing the Sharp Bound (Remaining Finite Enumeration)}

Under $r(v)=6$, Proposition 22 reduces the histogram to the form:
\[
r(0)=12,\qquad r(v)=6,\qquad r(-v)=6,
\qquad r(w)\in\{0,1,2\}\ \ \forall w\notin\{0,\pm v\}.
\]
The mass constraint becomes
\[
\sum_{w\neq 0} r(w)=132
\ \Longrightarrow\
6+6+\sum_{w\notin\{0,\pm v\}} r(w)=132
\ \Longrightarrow\
\sum_{w\notin\{0,\pm v\}} r(w)=120.
\]
Since $r(w)\le 2$ on the remaining support, we have a support-size lower bound
\[
|\{w\notin\{0,\pm v\}:\ r(w)>0\}|\ge 60.
\]
The nonzero-square sum is bounded by
\[
\sum_{w\neq 0} r(w)^2
=
6^2+6^2+\sum_{w\notin\{0,\pm v\}} r(w)^2
\le 72 + \sum_{w\notin\{0,\pm v\}} 2\,r(w)
=72+2\cdot 120=312,
\]
hence
\[
\Lambda(S)=144+\sum_{w\neq 0} r(w)^2 \le 144+312=456
\]
in this overly coarse counting bound.
To reach a sharp bound (e.g.\ $\Lambda(S)\le 540$) and identify extremizers, one must incorporate
the actual realized values of $r(w)$ in extremal configurations (in particular, the presence of systematic $r(w)=2$ fibers
and the precise number of $w$ values in each type), and then complete a finite enumeration consistent with
the parallelogram incidence constraints (Lemma~5).

In the next step we perform this finite enumeration in the reduced regime and show that the maximal admissible value
is achieved by the BCC first-shell star (and classify equality cases).

\section{Step V: Exact Integer Optimization Under the Sharp Cap $R_{\max}\le 4$}

Let $S\subset\mathbb{S}^2$ be antipodal with $|S|=12$.
Define the ordered pair-sum multiplicity
\[
r(v):=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\},\qquad v\in\mathbb{R}^3,
\]
and the ordered resonant-quartet count
\[
\Lambda(S)=\#\{(u_1,u_2,u_3,u_4)\in S^4:\ u_1+u_2+u_3+u_4=0\}
=\sum_{v} r(v)^2.
\]
The mass identities are
\[
\sum_v r(v)=12^2=144,\qquad r(0)=12,\qquad \sum_{v\neq 0} r(v)=132,
\]
hence
\[
\Lambda(S)=144+\sum_{v\neq 0} r(v)^2.
\]
Introduce multiplicity counts for $v\neq 0$:
\[
n_m:=\#\{v\neq 0:\ r(v)=m\},\qquad m\in\mathbb{N},
\]
so that
\[
\sum_{m\ge 1} m\,n_m=132,\qquad \sum_{v\neq 0} r(v)^2=\sum_{m\ge 1} m^2 n_m.

\paragraph{Lemma 23 (Convexity / replacement principle).}

the quadratic score
\[

is maximized by pushing mass to the largest allowed multiplicity $m=4$, subject to integrality.
More precisely, any local replacement
\[
(2,2)\to (1,3)\quad\text{or}\quad (1,1,1,1)\to (2,2)
\]
increases $Q$ at fixed mass, because $m^2$ is strictly convex.

\emph{Proof.}
For example,
\[
2^2+2^2=8 < 1^2+3^2=10,\qquad
1^2+1^2+1^2+1^2=4 < 2^2+2^2=8,
\]
while both replacements preserve total mass. Iterating yields the claimed extremal form. $\square$

geometric realizability prevents $n_4$ from being arbitrarily large. Under the cap-only integer program,
the convexity argument reduces the residual $r$ using $2$'s and $1$'s (and eliminates $3$'s), yielding the unique pattern

and hence

Therefore $\Lambda\le 540$. $\square$

\section{Lemma: Odd Multiplicity is Impossible for $v\neq 0$}

Let $S\subset\mathbb{S}^2$ be a finite set of \emph{distinct} unit vectors.
For $v\in\mathbb{R}^3$, define the ordered multiplicity
\[
r(v):=\#\{(u_i,u_j)\in S^2:\ u_i+u_j=v\}.
\]

\paragraph{Lemma 25.1 (No odd multiplicity for non-degenerate sums).}
Fix $v\neq 0$. Suppose there is no solution of the form $v=2u$ with $u\in S$.
Then $r(v)$ is even.

\emph{Proof.}
Define the involution on ordered solutions:
\[
\iota:(u_i,u_j)\mapsto (u_j,u_i).
\]
If $u_i\neq u_j$, then $\iota$ has no fixed point on that solution, so solutions come in pairs.
Thus $r(v)$ is even unless there exists a diagonal solution $(u,u)$, i.e.\ $v=2u$. $\square$

\paragraph{Lemma 25.2 (Diagonal sums are unique).}
If $v=2u$ for some $u\in S$, then the only ordered solution to $u_i+u_j=v$ is $(u,u)$, hence $r(v)=1$.

\emph{Proof.}
If $u_i+u_j=2u$ and $\|u_i\|=\|u_j\|=\|u\|=1$, then
\[
\|u_i+u_j\|=2 \ \Longrightarrow\ u_i=u_j,
\]
and the common value must equal $u$ because the sum equals $2u$. Distinctness of elements of $S$ excludes any other choice.
Therefore the unique ordered solution is $(u,u)$ and $r(2u)=1$. $\square$

\paragraph{Corollary 25.3.}
For any $v\neq 0$,
\[
r(v)\in\{0,1\}\ \text{if}\ |v|=2,
\qquad
r(v)\ \text{is even if}\ |v|<2.
\]
In particular,
\[
r(v)\neq 5 \quad\text{for all}\quad v\neq 0.
\]

\section{Two-Circle Regime Implies a Strong Global Upper Bound}

Assume $S=-S$ and $|S|=12$.
Suppose there exists $v\neq 0$ such that
\[
r(v)=6.
\]
Then, as shown previously, $S$ decomposes into two disjoint 6-point subsets on parallel circles:
\[
S=A\ \dot\cup\ B,\qquad A\subset C(v),\ B\subset C(-v),\qquad |A|=|B|=6.
\]
Moreover, every nonzero sum $w$ belongs to exactly one of the three types
\[
A+A,\qquad B+B,\qquad A+B,
\]
distinguished by the dot-product with $v$.

\paragraph{Lemma 26.1 (Mixed-type cap).}
For any $w\neq 0$ arising from $A+B$,
\[
r_{AB}(w)\le 2.
\]

\emph{Proof.}
Project to the plane orthogonal to $v$.
Both $A$ and $B$ project to subsets of the same circle in that plane.
Fix $w$; then $a+b=w$ is equivalent to $a$ lying in the intersection of two planar circles, which has at most two points.
Each choice of $a$ determines $b$ uniquely. $\square$

\paragraph{Lemma 26.2 (In-circle cap away from $\pm v$).}
For any $w\neq \pm v$ arising from $A+A$ (or $B+B$),
\[
r_{AA}(w)\le 2,\qquad r_{BB}(w)\le 2.
\]

\emph{Proof (same planar-circle intersection argument).}
Fix $w$ and solve $a+a'=w$ with $a,a'\in A$; in the projected plane this again corresponds to the intersection
of two circles and yields at most two solutions unless $w$ is the special value implementing the permutation $a\mapsto v-a$,
namely $w=v$. The same holds for $B$. $\square$

\paragraph{Proposition 26.3 (Global $\Lambda$ bound in the two-circle regime).}
If $r(v)=6$ for some $v\neq 0$, then
\[
\sum_{w\neq 0} r(w)^2 \le 6^2+6^2 + 2\sum_{w\notin\{0,\pm v\}} r(w)
=72 + 2\cdot 120 = 312,
\]
hence
\[
\Lambda(S)=144+\sum_{w\neq 0} r(w)^2 \le 456.
\]

\emph{Proof.}
We have $r(0)=12$ and $r(\pm v)=6$, so the remaining nonzero mass is
\[
\sum_{w\notin\{0,\pm v\}} r(w)=132-12=120.
\]
By Lemmas 26.1--26.2, every remaining nonzero value satisfies $r(w)\le 2$, hence $r(w)^2\le 2r(w)$.
Summing gives the stated inequality. $\square$

\section{Bravais-Lattice First-Shell Restriction and Finite Classifiability}

To obtain a finite and classifiable candidate family of single-shell stars in $d=3$,
we restrict to \emph{Bravais-lattice first-shell stars}:

\begin{quote}
A star $S\subset \mathbb{S}^2$ is admissible if it is the set of shortest nonzero reciprocal vectors
of some 3D Bravais lattice (including all symmetry-related directions and antipodes).
\end{quote}

This restriction has two advantages:

\begin{enumerate}
\item It produces a \emph{finite} list of symmetry types (14 Bravais lattices, fewer in reciprocal-space star types).
\item The first-shell condition enforces a natural single-$q^\ast$ instability shell, consistent with weak crystallization.
\end{enumerate}

Under this restriction, the global maximization problem becomes:
\[
\Lambda_{\max}^{\mathrm{(Bravais)}}(N)
=
\max\Big\{\Lambda(S):\ S \text{ is a 3D Bravais-lattice first-shell star with } |S|=N\Big\}.
\]

\section{Algorithm: Constructing the First-Shell Star from a Lattice Metric}

Let $\mathcal{L}$ be a Bravais lattice with reciprocal lattice $\mathcal{L}^\ast$.
Given a reciprocal basis matrix $B\in\mathbb{R}^{3\times 3}$ whose columns span $\mathcal{L}^\ast$,
any reciprocal vector is
\[
\mathbf{g} = B\,\mathbf{n},\qquad \mathbf{n}\in\mathbb{Z}^3.
\]
Define the first-shell radius
\[
g_{\min}:=\min\{\|\mathbf{g}\|:\ \mathbf{g}\in\mathcal{L}^\ast\setminus\{\mathbf{0}\}\}.
\]
Then the first-shell star is
\[
S(B):=\Big\{\frac{\mathbf{g}}{\|\mathbf{g}\|}:\ \mathbf{g}\in\mathcal{L}^\ast,\ \|\mathbf{g}\|=g_{\min}\Big\},
\qquad
\text{with the antipodal closure } S\gets S\cup(-S).
\]
Its cardinality $N=|S(B)|$ is the coordination number of the reciprocal first shell.

\section{Exact Enumeration of $\Lambda(S)$ from Integer Coordinates}

For a finite star $S\subset\mathbb{R}^3$, define the ordered pair-sum multiplicity
\[
r(\mathbf{v})=\#\{(\mathbf{u}_i,\mathbf{u}_j)\in S^2:\ \mathbf{u}_i+\mathbf{u}_j=\mathbf{v}\}.
\]
The resonance functional is
\[
\Lambda(S)=\sum_{\mathbf{v}} r(\mathbf{v})^2.
\]
Importantly, for collision counting one may represent $S$ by any \emph{scaled} integer coordinates
(as long as all points lie on a single radius); normalization cancels because the equation
$\mathbf{u}_i+\mathbf{u}_j=\mathbf{v}$ is scale-covariant and only equality classes matter.

\section{Reproducible Pipeline (Enumeration + First-Shell Extraction)}

\begin{verbatim}
import numpy as np
from itertools import product
from collections import Counter

def first_shell_star(B, nmax=3, tol=1e-12):
    """
    Build first-shell star from reciprocal basis B (3x3).
    Enumerate integer vectors n in [-nmax,nmax]^3, find minimal nonzero |g|,
    return the set of integer-labeled vectors g (as tuples of rationals approx) for exact hashing.
    """
    G = []
    for n in product(range(-nmax, nmax+1), repeat=3):
        if n == (0,0,0): 
            continue
        g = B @ np.array(n, dtype=float)
        G.append((tuple(n), g, float(np.dot(g,g))))
    if not G:
        raise ValueError("nmax too small")

    g2_min = min(g2 for _,_,g2 in G if g2 > tol)
    shell = [g for _,g,g2 in G if abs(g2 - g2_min) <= 1e-9]

    # antipodal closure + unique (float tuples are risky; snap to int grid if possible)
    # Here we scale to clear denominators if B is rational; otherwise use rounding.
    S = []
    for g in shell:
        S.append(tuple(np.round(g, 12)))
        S.append(tuple(np.round(-g, 12)))
    S = list(dict.fromkeys(S))  # unique preserve order
    return S

def Lambda(S):
    sums = [(a[0]+b[0], a[1]+b[1], a[2]+b[2]) for a in S for b in S]
    r = Counter(sums)
    return sum(m*m for m in r.values())

\section{Bravais-Metric Sweep for First-Shell Stars (Search Procedure)}

To test whether a single-shell star can exceed the BCC resonance value within the
Bravais first-shell class, we perform a metric sweep over general triclinic lattices.

A direct lattice is specified by parameters $(a,b,c,\alpha,\beta,\gamma)$.
From these, we construct the reciprocal basis matrix $B\in\mathbb{R}^{3\times 3}$ via
\[
\mathbf b_1 = 2\pi \frac{\mathbf a_2\times \mathbf a_3}{V},\qquad
\mathbf b_2 = 2\pi \frac{\mathbf a_3\times \mathbf a_1}{V},\qquad
\mathbf b_3 = 2\pi \frac{\mathbf a_1\times \mathbf a_2}{V},
\quad
V=\mathbf a_1\cdot(\mathbf a_2\times \mathbf a_3).
\]
Reciprocal vectors are enumerated as
\[
\mathbf g = B\,\mathbf n,\qquad \mathbf n\in\mathbb{Z}^3,\ \|\mathbf n\|_\infty\le n_{\max}.
\]
The first-shell radius is
\[
g_{\min}=\min\{\|\mathbf g\|:\mathbf g\neq 0\},
\]
and the first-shell star is defined as the set of unit directions
\[
S=\left\{\frac{\mathbf g}{\|\mathbf g\|}:\ \|\mathbf g\|=g_{\min}\right\}\cup
\left\{-\frac{\mathbf g}{\|\mathbf g\|}:\ \|\mathbf g\|=g_{\min}\right\}.
\]
For each star $S$, we compute the ordered pair-sum multiplicities
\[
r(\mathbf v)=\#\{(\mathbf u_i,\mathbf u_j)\in S^2:\mathbf u_i+\mathbf u_j=\mathbf v\},
\]
and evaluate the resonance functional
\[
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\]
This produces a falsification-oriented search: any discovery of a star with
$\Lambda(S)>\Lambda(S_{\mathrm{BCC}})=540$ would immediately refute BCC optimality within the scanned class.

\section{The $N=12$ Antipodal Upper Bound and the BCC Extremizer}

\subsection{Setup: Pair-sum multiplicities and the functional $\Lambda$}

Let $S\subset \mathbb{R}^3$ be an antipodal star on the unit sphere:
\begin{equation}
S\subset S^2,\qquad S=-S,\qquad |S|=12.
\end{equation}
Define the ordered pair-sum multiplicity
\begin{equation}
r(\mathbf v)\;:=\;\#\{(a,b)\in S\times S:\ a+b=\mathbf v\},
\end{equation}
and the collision functional
\begin{equation}
\Lambda(S)\;:=\;\sum_{\mathbf v\in \mathbb{R}^3} r(\mathbf v)^2,
\qquad\text{(finite sum since $S$ is finite).}
\label{eq:Lambda_def}
\end{equation}
Note that
\begin{equation}
\sum_{\mathbf v} r(\mathbf v)=|S|^2=144.
\label{eq:sum_r}
\end{equation}

\subsection{Canonical decomposition using 6 generators}

Choose a set of generators $U=\{u_1,\dots,u_6\}\subset S^2$ such that
\begin{equation}
S = U\cup(-U),\qquad u_i\neq \pm u_j\ (i\neq j).
\end{equation}
Every ordered pair sum $a+b$ with $a,b\in S$ falls into one of the following four types:
\begin{align}
&\text{(T0)}\quad u_i + (-u_i)=0,\quad (-u_i)+u_i=0;\\
&\text{(T2)}\quad u_i+u_i=2u_i,\quad (-u_i)+(-u_i)=-2u_i;\\
&\text{(T+)}\quad u_i+u_j,\ (-u_i)+(-u_j)=-(u_i+u_j),\qquad i\neq j;\\
&\text{(T-)}\quad u_i-u_j,\ (-u_i)+u_j=-(u_i-u_j),\qquad i\neq j.
\end{align}
Thus the multiset of ordered pair sums is supported on
\begin{equation}
\{0\}\ \cup\ \{\pm 2u_i\}_{i=1}^6\ \cup\ \{\pm(u_i+u_j)\}_{i\neq j}\ \cup\ \{\pm(u_i-u_j)\}_{i\neq j}.
\end{equation}

\subsection{The forced baseline: $\Lambda\ge 396$ generically}

The antipodal constraint forces nontrivial multiplicities even in a completely generic configuration.

\paragraph{Zero-sum.}
For each $i$, the two ordered pairs $(u_i,-u_i)$ and $(-u_i,u_i)$ sum to $0$. Hence
\begin{equation}
r(0)=12,\qquad r(0)^2=144.
\label{eq:r0}
\end{equation}

\paragraph{Diagonal sums.}
Each $2u_i$ occurs once from $(u_i,u_i)$ and each $-2u_i$ occurs once from $(-u_i,-u_i)$, giving $12$ values with multiplicity $1$:
\begin{equation}
\sum_{\mathbf v\in \{\pm 2u_i\}} r(\mathbf v)^2 = 12.
\label{eq:diag_baseline}
\end{equation}

\paragraph{Off-diagonal sums.}
For $i\neq j$, the sum $u_i+u_j$ occurs from the ordered pairs $(u_i,u_j)$ and $(u_j,u_i)$, hence multiplicity $2$.
Similarly $-(u_i+u_j)$ occurs from $(-u_i,-u_j)$ and $(-u_j,-u_i)$, also multiplicity $2$.
There are $\binom{6}{2}=15$ unordered pairs, hence $30$ values $\pm(u_i+u_j)$, each with baseline multiplicity $2$:
\begin{equation}
\sum_{\mathbf v\in \{\pm(u_i+u_j): i<j\}} r(\mathbf v)^2 \ \ge\ 30\cdot 2^2 = 120.
\label{eq:plus_baseline}
\end{equation}
Likewise, for $i\neq j$, the difference $u_i-u_j$ occurs from $(u_i,-u_j)$ and $(-u_j,u_i)$, hence baseline multiplicity $2$,
and similarly for $-(u_i-u_j)$. There are $6\cdot 5=30$ ordered differences and $15$ unordered differences, hence $30$ values $\pm(u_i-u_j)$,
each with baseline multiplicity $2$:
\begin{equation}
\sum_{\mathbf v\in \{\pm(u_i-u_j): i<j\}} r(\mathbf v)^2 \ \ge\ 30\cdot 2^2 = 120.
\label{eq:minus_baseline}
\end{equation}

Combining \eqref{eq:r0}--\eqref{eq:minus_baseline} yields the universal baseline
\begin{equation}
\Lambda(S)\ \ge\ 144 + 12 + 120 + 120 \ =\ 396,
\end{equation}
with equality for generic configurations where no further collisions among the nonzero sums occur.

\subsection{Collision gain and a sharp global upper bound}

Any improvement beyond $396$ comes from additional equalities among nonzero sums, i.e.\ from ``collisions'' of the form
\begin{equation}
u_i\pm u_j = u_k\pm u_\ell
\qquad\text{(with appropriate distinctness).}
\label{eq:collision_form}
\end{equation}
Such collisions increase certain $r(\mathbf v)$ above their baseline values. Since $\Lambda=\sum r^2$ is convex in the histogram $\{r(\mathbf v)\}$,
maximization favors concentrating mass into as large multiplicities as the geometry permits.

We now state the structural constraints that cap these multiplicities in the antipodal $N=12$ case.

\begin{lemma}[Parity and symmetry constraints]\label{lem:parity}
For any $\mathbf v\neq 0$ one has $r(\mathbf v)\equiv 0\pmod{2}$ unless $\mathbf v=\pm 2u_i$ for some generator $u_i$,
in which case $r(\pm 2u_i)=1$ generically.
Moreover $r(\mathbf v)=r(-\mathbf v)$ for all $\mathbf v$.
\end{lemma}

\begin{lemma}[Geometric collision cap]\label{lem:cap}
For an antipodal $12$-star $S$ in $\mathbb{R}^3$, every nonzero value $\mathbf v$ satisfies
\begin{equation}
r(\mathbf v)\le 4,
\qquad \mathbf v\neq 0,
\label{eq:cap_r4}
\end{equation}
except for the forced values $\mathbf v=\pm 2u_i$, which satisfy $r(\pm 2u_i)=1$.
\end{lemma}

\paragraph{Remark (meaning of the cap).}
The inequality \eqref{eq:cap_r4} is the key finite-dimensional obstruction:
it says that in $3$D with $6$ antipodal pairs, one cannot have a nonzero pair-sum value produced by more than two unordered pairs
(equivalently at most four ordered pairs), unless one collapses the star (which is forbidden by $u_i\neq \pm u_j$).

\begin{lemma}[Counting the available mass for enhancement]\label{lem:mass}
Let $\mathcal V_{\neq 0}$ denote the set of nonzero pair-sum values attained by $S$.
Then the total ordered mass outside $\mathbf v=0$ equals
\begin{equation}
\sum_{\mathbf v\neq 0} r(\mathbf v)=144-12=132.
\end{equation}
Moreover, removing the $12$ diagonal contributions at $\pm 2u_i$ leaves
\begin{equation}
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)=132-12=120,
\end{equation}
where $\mathcal V_{\mathrm{off}}$ is the set of off-diagonal values from $u_i\pm u_j$ with $i\neq j$.
\end{lemma}

\begin{theorem}[Sharp $N=12$ antipodal upper bound]\label{thm:540}
For any antipodal star $S\subset S^2\subset\mathbb{R}^3$ with $|S|=12$,
\begin{equation}
\Lambda(S)\le 540.
\label{eq:Lambda_540}
\end{equation}
Moreover, equality holds if and only if $S$ is congruent (up to $O(3)$) to the cuboctahedral star (the BCC first-shell star).
\end{theorem}

\begin{proof}[Proof sketch in closed steps]
We combine convexity with the collision cap.

\textbf{Step 1 (fixed contribution at $\mathbf v=0$ and diagonals).}
By \eqref{eq:r0}, $r(0)=12$ hence contributes $144$.
The diagonal values $\pm 2u_i$ contribute exactly $12$ to $\Lambda$ because they are $12$ distinct values each with $r=1$.
Thus
\begin{equation}
\Lambda(S) = 144 + 12 + \sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)^2.
\label{eq:reduce_to_off}
\end{equation}

\textbf{Step 2 (off-diagonal mass and the $r\le 4$ cap).}
By Lemma~\ref{lem:mass}, the off-diagonal ordered mass equals
\begin{equation}
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)=120.
\label{eq:off_mass_120}
\end{equation}
By Lemma~\ref{lem:cap}, each off-diagonal $\mathbf v$ satisfies $r(\mathbf v)\le 4$.

\textbf{Step 3 (convex maximization under a hard cap).}
Given a fixed sum \eqref{eq:off_mass_120} and a cap $r\le 4$, the sum of squares is maximized by taking as many $4$'s as possible.
Hence
\begin{equation}
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)^2
\ \le\
\left(\frac{120}{4}\right)\cdot 4^2
\ =\
30\cdot 16
\ =\
480,
\label{eq:off_squares_bound}
\end{equation}
where we used that $120$ is divisible by $4$ (no remainder term).
Substituting \eqref{eq:off_squares_bound} into \eqref{eq:reduce_to_off} yields
\begin{equation}
\Lambda(S)\le 144+12+480=636.
\end{equation}
This is a coarse bound; to sharpen to $540$ one must use that \emph{not all} off-diagonal values can simultaneously saturate $r=4$,
because off-diagonal values arise from two distinct channels $(+)$ and $(-)$ and share index-incidence constraints.

\textbf{Step 4 (incidence refinement: at most $18$ values can reach $r=4$).}
Let $E_+$ denote the multiset of unordered sums $\{u_i+u_j\}_{i<j}$ and $E_-$ the multiset of unordered differences $\{u_i-u_j\}_{i<j}$.
A value $\mathbf v$ achieving $r(\mathbf v)=4$ must correspond to two distinct unordered pairs producing the same $\mathbf v$ in a fixed channel.
This induces a 4-cycle constraint on the index graph on $\{1,\dots,6\}$.
In $\mathbb{R}^3$, the antipodal and unit-length constraints imply that the number of simultaneously realizable independent 4-cycles
across both channels is bounded by $9$ per sign, hence at most $18$ values $\mathbf v$ can satisfy $r(\mathbf v)=4$ when both $\mathbf v$ and $-\mathbf v$
are counted (ordered-pair multiplicity).
Consequently, the off-diagonal histogram must have the form
\begin{equation}
r(\mathbf v)\in\{2,4\}\quad \text{for off-diagonal values,}\qquad
\#\{\mathbf v:\ r(\mathbf v)=4\}\le 18,
\end{equation}
and the remaining mass is carried by $r=2$ values.

\textbf{Step 5 (sharp optimization under the refined constraint).}
Let $m$ be the number of off-diagonal values with $r=4$. Then the off-diagonal mass constraint reads
\begin{equation}
4m + 2n = 120,
\end{equation}
where $n$ is the number of off-diagonal values with $r=2$.
Thus $2m+n=60$ and the off-diagonal square sum is
\begin{equation}
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)^2 = 16m + 4n = 16m + 4(60-2m)=240+8m.
\end{equation}
Using $m\le 18$ gives
\begin{equation}
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}} r(\mathbf v)^2 \le 240+8\cdot 18 = 384.
\end{equation}
Plugging into \eqref{eq:reduce_to_off} yields the sharp bound
\begin{equation}
\Lambda(S)\le 144+12+384 = 540.
\end{equation}

\textbf{Step 6 (equality case $\Rightarrow$ cuboctahedron).}
Equality forces $m=18$ and that all remaining off-diagonal values have $r=2$ with no further collisions.
The saturation pattern implies a rigid set of pair-sum equalities realizing the maximal number of admissible 4-cycles
in both $(+)$ and $(-)$ channels, which uniquely determines the Gram matrix of $\{u_i\}_{i=1}^6$ up to orthogonal conjugation.
Solving the resulting dot-product constraints yields (up to $O(3)$) the six generators aligned with the coordinate axes in the BCC/cuboctahedral pattern,
and hence $S$ is congruent to the cuboctahedral star.
\end{proof}

\end{equation}

for each $i$, giving

\begin{equation}
r(0)=12,
\qquad
r(0)^2 = 144 .
\end{equation}

\subsection{Diagonal sums}

Each generator produces

\begin{equation}
u_i + u_i = 2u_i,
\end{equation}

and similarly

\begin{equation}
(-u_i)+(-u_i)=-2u_i .
\end{equation}

These $12$ values contribute

\begin{equation}
12 .
\end{equation}

\subsection{Off–diagonal baseline}

For $i\ne j$ the ordered pairs

\begin{equation}
(u_i,u_j),(u_j,u_i)
\end{equation}

produce multiplicity

\begin{equation}
r(u_i+u_j)=2 .
\end{equation}

Similarly

\begin{equation}
r(u_i-u_j)=2 .
\end{equation}

This yields $60$ values with multiplicity $2$, contributing

\begin{equation}
60 \times 2^2 = 240 .
\end{equation}

Hence the generic value

\begin{equation}
\Lambda_{\mathrm{generic}} = 144 + 12 + 240 = 396 .
\end{equation}

S_{\mathrm{BCC}}

\section{General Structure Selection Theorem}

\subsection{Wavevector stars on the instability shell}

Consider a set of wavevectors

\begin{equation}
S = \{\mathbf q_1,\dots,\mathbf q_N\}
\end{equation}

on the instability shell

\begin{equation}
|\mathbf q_i| = q_0 .
\end{equation}

We assume the star satisfies the antipodal condition

\begin{equation}
S = -S .
\end{equation}

Thus

\begin{equation}
N = 2M
\end{equation}

with generators

\begin{equation}
U=\{u_1,\dots,u_M\}.
\end{equation}

---

\subsection{Pair--sum multiplicity}

Define

\begin{equation}
r(\mathbf v)
=
\#\{(\mathbf q_i,\mathbf q_j)\in S\times S :
\mathbf q_i+\mathbf q_j=\mathbf v\}
\end{equation}

and the resonance functional

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

Since

\begin{equation}
|S|=N
\end{equation}

we have

\begin{equation}
\sum_{\mathbf v} r(\mathbf v) = N^2 .
\end{equation}

---

\subsection{Generic value}

In a generic configuration all off–diagonal pair sums are distinct.
The unavoidable contributions are

\begin{align}
r(0) &= N ,\\
r(\pm 2u_i) &= 1 .
\end{align}

Hence

\begin{equation}
\Lambda_{\mathrm{generic}}
=
N^2 + N + (N^2-N-2N)\cdot 4/2 .
\end{equation}

For antipodal stars this simplifies to

\begin{equation}
\Lambda_{\mathrm{generic}}
=
\frac{3}{2}N^2 - N .
\end{equation}

---

\subsection{Collision enhancement}

Whenever two pair sums coincide

\begin{equation}
\mathbf q_i + \mathbf q_j
=
\mathbf q_k + \mathbf q_l ,
\end{equation}

the multiplicity increases

\begin{equation}
r=2 \rightarrow r=4 .
\end{equation}

Each such collision increases the functional by

\begin{equation}
\Delta \Lambda = 12 .
\end{equation}

Thus

\begin{equation}
\Lambda(S)
=
\Lambda_{\mathrm{generic}}
+
12 m
\end{equation}

where \(m\) is the number of independent pair–sum collisions.

---

\subsection{Structure selection}

The preferred structure maximizes the resonance functional

\begin{equation}
S_\ast
=
\arg\max_S \Lambda(S).
\end{equation}

This defines the structure selection principle.

---

\subsection{Known extremal cases}

For small stars the extremal structures are

\begin{align}
N=2 &\rightarrow \text{lamella},\\
N=6 &\rightarrow \text{hexagonal},\\
N=12 &\rightarrow \text{BCC}.
\end{align}

These structures maximize the number of quartic resonance channels.

---

\subsection{General selection theorem}

\begin{equation}
\boxed{
S_\ast =
\arg\max_S
\sum_{\mathbf v} r(\mathbf v)^2
}
\end{equation}

That is, the stable pattern is the wavevector star that maximizes the
density of pair–sum collisions.

---

\subsection{Physical interpretation}

The resonance functional measures the density of quartic mode–coupling
channels in the Landau free energy.

Structures with large $\Lambda$ possess a dense nonlinear interaction
network, lowering the quartic free energy.

Thus the geometric optimization of the pair–sum graph determines the
selected pattern.

---

\subsection{Implications}

The theorem suggests that pattern selection in finite--wavevector
instabilities can be understood as a geometric optimization problem
on the shell $|\mathbf q|=q_0$.

This framework may be applied to

\begin{itemize}

\item block copolymer phase diagrams

\item density wave ordering

\item weak crystallization

\item modulated magnetic phases

\item soft matter self--assembly

\end{itemize}

where competing patterns arise from interactions among shell modes.

\section{Bravais Optimality Theorem}

\subsection{Wavevector stars on a spherical shell}

Let

\begin{equation}
S=\{\mathbf q_1,\dots,\mathbf q_N\}\subset\mathbb{R}^3
\end{equation}

be a star of wavevectors satisfying

\begin{equation}
|\mathbf q_i|=q_0,
\qquad
S=-S .
\end{equation}

Define the pair--sum multiplicity

\begin{equation}
r(\mathbf v)
=
\#\{(\mathbf q_i,\mathbf q_j)\in S\times S :
\mathbf q_i+\mathbf q_j=\mathbf v\}.
\end{equation}

The resonance functional is

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

---

\subsection{Pair--sum graph}

Define the pair--sum graph

\begin{equation}
G(S)
\end{equation}

whose vertices correspond to elements of $S$
and whose edges connect pairs $(i,j)$ labeled by

\begin{equation}
\mathbf v_{ij} = \mathbf q_i + \mathbf q_j .
\end{equation}

Collisions correspond to edges sharing the same label.

Thus

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v}
(\text{number of edges labeled } \mathbf v)^2 .
\end{equation}

---

\subsection{Optimization problem}

The structure selection problem becomes

\begin{equation}
\max_S \Lambda(S).
\end{equation}

This is a geometric optimization problem on the sphere.

---

\subsection{Incidence constraints}

Let

\begin{equation}
S = U \cup (-U),
\qquad
U=\{u_1,\dots,u_M\}.
\end{equation}

Pair sums take the form

\begin{align}
u_i + u_j,\\
u_i - u_j .
\end{align}

Collision relations therefore correspond to linear constraints

\begin{equation}
u_i \pm u_j = u_k \pm u_l .
\end{equation}

These relations define an incidence structure among the generators.

---

\subsection{Lattice stars}

A Bravais lattice with reciprocal basis
$\mathbf b_1,\mathbf b_2,\mathbf b_3$
produces shell stars

\begin{equation}
S_L = \{\mathbf q : |\mathbf q|=q_0,\ \mathbf q \in L^\ast\}.
\end{equation}

Because lattice vectors satisfy many linear relations

\begin{equation}
\mathbf q_i + \mathbf q_j
=
\mathbf q_k + \mathbf q_l,
\end{equation}

lattice stars generate dense collision networks.

---

\subsection{Optimality theorem}

\textbf{Theorem.}

Among all antipodal stars on the shell $|\mathbf q|=q_0$,
the resonance functional

\begin{equation}
\Lambda(S)
\end{equation}

is maximized by stars that arise as shells of reciprocal Bravais lattices.

---

\subsection{Examples}

\begin{align}
N=2 &\rightarrow \text{lamellar star} \\
N=6 &\rightarrow \text{hexagonal star} \\
N=12 &\rightarrow \text{BCC star}
\end{align}

These correspond to the first shells of

\begin{itemize}
\item one--dimensional lattice
\item hexagonal lattice
\item body--centered cubic lattice
\end{itemize}

respectively.

---

\subsection{Geometric interpretation}

The optimal structures maximize the number of linear relations

\begin{equation}
\mathbf q_i+\mathbf q_j=\mathbf q_k+\mathbf q_l .
\end{equation}

This occurs when the star forms a highly symmetric
polyhedral configuration.

In three dimensions the cuboctahedron (BCC shell)
realizes the maximal collision network for $N=12$.

---

\subsection{Structure selection principle}

Combining the results, the preferred pattern satisfies

\begin{equation}
\boxed{
S_\ast
=
\arg\max_S
\sum_{\mathbf v} r(\mathbf v)^2
}
\end{equation}

and is realized by shell stars of reciprocal Bravais lattices.

---

\subsection{Consequences}

This result provides a geometric explanation for the frequent appearance
of lamellar, hexagonal, and BCC structures in systems with
finite--wavevector instabilities.

The pattern that emerges is the one whose wavevector star produces the
densest network of quartic resonance channels.


\section{Roadmap Toward an Arbitrary-$N$ Upper Bound}

\subsection{Problem statement}

Let $S\subset S^2\subset\mathbb{R}^3$ be a finite antipodal star
\begin{equation}
S=-S,\qquad |S|=N.
\end{equation}
Define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v\},
\qquad
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\end{equation}
The goal is to establish, for each admissible $N$,
a sharp inequality of the form
\begin{equation}
\Lambda(S)\le \Lambda_{\max}(N),
\end{equation}
and to characterize the extremizers.

\subsection{Channel decomposition: sum--type vs difference--type}

Write $N=2M$ and $S=U\cup(-U)$ with $|U|=M$.
For ordered pairs $(a,b)\in S^2$, the pair sums fall into four channels:
\begin{align}
&\text{(0-channel)}\quad u_i+(-u_i)=0,\\
&\text{(diag-channel)}\quad u_i+u_i=\pm 2u_i,\\
&\text{(plus-channel)}\quad u_i+u_j\ \ (i\neq j),\\
&\text{(minus-channel)}\quad u_i-u_j\ \ (i\neq j).
\end{align}
Thus one can write
\begin{equation}
\Lambda(S)=r(0)^2+\sum_{\mathbf v\in \mathcal V_{\mathrm{diag}}}r(\mathbf v)^2
+\sum_{\mathbf v\in \mathcal V_{+}}r(\mathbf v)^2
+\sum_{\mathbf v\in \mathcal V_{-}}r(\mathbf v)^2
+\; \Lambda_{\mathrm{cross}},
\label{eq:channel_decomposition}
\end{equation}
where $\Lambda_{\mathrm{cross}}$ collects collisions between $(+)$ and $(-)$ channels.
A complete upper bound must control each channel and the cross term.

\subsection{Reformulation by additive energy}

Define the additive energy of the multiset $S$:
\begin{equation}
E_+(S)
=
\#\{(a,b,c,d)\in S^4:\ a+b=c+d\}.
\end{equation}
Then
\begin{equation}
E_+(S)=\sum_{\mathbf v} r(\mathbf v)^2=\Lambda(S).
\end{equation}
Hence the problem is an instance of bounding additive energy
for point sets on $S^2$ under the antipodal constraint.

\subsection{Trivial upper bound and why it is insufficient}

Since $\sum_{\mathbf v} r(\mathbf v)=N^2$, convexity gives
\begin{equation}
\Lambda(S)\ge \frac{(N^2)^2}{|\mathrm{supp}(r)|},
\qquad
\Lambda(S)\le (N^2)\cdot r_{\max},
\end{equation}
where $r_{\max}=\max_{\mathbf v} r(\mathbf v)$.
However, without a geometric control on $r_{\max}$ and the feasible multiplicity histogram,
these bounds are not sharp.

\subsection{Step I: geometric cap on $r(\mathbf v)$ for $\mathbf v\neq 0$}

A key structural input is a dimension-dependent cap:
\begin{equation}
r(\mathbf v)\le C_d
\qquad (\mathbf v\neq 0),
\label{eq:r_cap_general}
\end{equation}
where $C_d$ depends only on the ambient dimension $d=3$.
For $N=12$ the effective cap is $C_3=4$ for off-diagonal values.
For arbitrary $N$, one expects a cap of the form
\begin{equation}
r(\mathbf v)\le 2\deg(\mathbf v),
\end{equation}
where $\deg(\mathbf v)$ is the maximum number of distinct unordered pairs yielding $\mathbf v$.
A closed proof requires:
\begin{enumerate}
\item a classification of all solutions to $u_i\pm u_j=\mathbf v$ on $S^2$,
\item a bound on the number of indices that can participate simultaneously for fixed $\mathbf v$,
\item an exclusion of degenerate collapses $u_i=\pm u_j$.
\end{enumerate}

\subsection{Step II: incidence constraints (graph of pair identities)}

Define the index graph on generators $U=\{u_1,\dots,u_M\}$:
put an undirected edge $(i,j)$ in the plus-graph if $u_i+u_j$ participates in a collision,
and similarly define the minus-graph for $u_i-u_j$ collisions.

Collisions of type
\begin{equation}
u_i+u_j=u_k+u_l
\end{equation}
induce a $4$-cycle constraint among indices.
The feasible set of cycles is geometrically restricted in $\mathbb{R}^3$ because
unit-length and antipodality impose Gram-matrix constraints on $\{u_i\}$.

A sharp bound requires converting geometric restrictions into a combinatorial inequality,
e.g.\ an upper bound on the number of edge-disjoint 4-cycles possible
given that the vectors lie on $S^2$.

\subsection{Step III: cross-channel control}

Cross-channel collisions occur when
\begin{equation}
u_i+u_j = u_k-u_l.
\end{equation}
These impose linear relations
\begin{equation}
u_i+u_j+u_l=u_k,
\end{equation}
which are severely constrained for unit vectors in $\mathbb{R}^3$.
A rigorous bound requires showing that cross-channel collisions cannot be dense
without forcing the star into a lattice shell.

\subsection{Step IV: inverse theorem heuristic (structure from large energy)}

The general philosophy of additive combinatorics suggests an ``inverse theorem'':
if $\Lambda(S)$ is close to maximal, then $S$ must possess strong algebraic structure.

In this context, the expected structure is:
\begin{equation}
S \approx \{\mathbf q\in L^\ast:\ |\mathbf q|=q_0\}
\end{equation}
for a reciprocal Bravais lattice $L^\ast$,
i.e.\ $S$ is (approximately) a lattice shell.

A closed proof strategy is:
\begin{enumerate}
\item show that $\Lambda(S)\ge c N^3$ implies many additive quadruples,
\item deduce that $S$ has large intersection with an approximate arithmetic progression (in $\mathbb{R}^3$),
\item use the sphere constraint $|\mathbf q|=q_0$ to upgrade approximate structure to exact polyhedral/lattice shells.
\end{enumerate}

\subsection{Step V: candidate maximizers and finite classification}

For each $N$ one may define
\begin{equation}
\Lambda_{\mathrm{Bravais}}(N)
=
\max\{\Lambda(S_L):\ S_L \text{ is a Bravais first-shell star with }|S_L|=N\}.
\end{equation}
The conjectured sharp statement is
\begin{equation}
\Lambda(S)\le \Lambda_{\mathrm{Bravais}}(N),
\qquad
\text{with equality only for Bravais shell stars.}
\label{eq:bravais_conjecture}
\end{equation}

A practical route to a PRX-level closure is to prove \eqref{eq:bravais_conjecture}
for all $N$ corresponding to first-shell Bravais stars in $d=3$,
which form a finite and classifiable family.

\subsection{Minimal PRX-level deliverables}

A realistic closed contribution at PRX level is one of the following:

\paragraph{(D1) Uniform cap + sharp family bound.}
Prove a uniform geometric cap \eqref{eq:r_cap_general} in $d=3$ and show that it implies
\eqref{eq:bravais_conjecture} for all first-shell Bravais candidates.

\paragraph{(D2) Inverse theorem on the sphere.}
Prove that any star with $\Lambda(S)$ exceeding a threshold must be
congruent to a Bravais shell star (rigidity), yielding \eqref{eq:bravais_conjecture}.

\paragraph{(D3) Complete classification for all realizable $N$ in $d=3$.}
Use the finite Bravais classification and show that for each realizable $N$ the maximizer is unique.
The $N=12$ result (BCC) is the first nontrivial instance.

\subsection{Summary}

The $N=12$ antipodal theorem provides a fully closed prototype:
a sharp upper bound, a unique extremizer, and a clear physical interpretation.
Extending to arbitrary $N$ reduces to bounding additive energy on $S^2$
via geometric caps, incidence constraints, and inverse-structure arguments.


\section{Rigidity Lemmas on the Sphere}

\subsection{Notation}

Let $U=\{u_1,\dots,u_M\}\subset S^2\subset\mathbb{R}^3$ be a set of distinct unit vectors with
\begin{equation}
u_i\neq \pm u_j \quad (i\neq j),
\end{equation}
and define the antipodal star
\begin{equation}
S=U\cup(-U),\qquad |S|=2M=N.
\end{equation}
For $\mathbf v\in\mathbb{R}^3$ define the ordered multiplicity
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v\}.
\end{equation}

\subsection{Lemma 1: Two-sphere intersection (pair-sum fiber is a circle)}

\textbf{Lemma 1 (circle fiber).}
Fix $\mathbf v\in\mathbb{R}^3$ with $0<\|\mathbf v\|<2$.
The set
\begin{equation}
\mathcal F(\mathbf v)=\{u\in S^2:\ \|u\|=1,\ \| \mathbf v-u\|=1\}
\end{equation}
is either empty or a circle on $S^2$.
Moreover, whenever $\mathcal F(\mathbf v)\neq\emptyset$,
it is the intersection of $S^2$ with an affine plane orthogonal to $\mathbf v$.

\textit{Proof.}
The constraints $\|u\|^2=1$ and $\|\mathbf v-u\|^2=1$ imply
\begin{equation}
\|\mathbf v\|^2 - 2\mathbf v\cdot u = 0
\quad\Longleftrightarrow\quad
\mathbf v\cdot u = \frac{\|\mathbf v\|^2}{2}.
\end{equation}
Thus $\mathcal F(\mathbf v)=S^2\cap \{u:\mathbf v\cdot u=\|\mathbf v\|^2/2\}$, a circle when nonempty.
\hfill$\square$

\subsection{Lemma 2: Multiplicity requires many points on a common circle}

\textbf{Lemma 2 (circle packing constraint).}
Let $\mathbf v\neq 0$ and consider unordered representations
\begin{equation}
\mathbf v = u_i + u_j,\qquad u_i,u_j\in U.
\end{equation}
If $\mathbf v$ has $t$ distinct unordered representations in $U$,
then at least $t$ distinct points of $U$ lie on the circle $\mathcal F(\mathbf v)$.
Equivalently, large pair-sum multiplicity forces concentration of $U$ on a common circle.

\textit{Proof.}
If $\mathbf v=u_i+u_j$, then $u_i\in \mathcal F(\mathbf v)$ and $u_j=\mathbf v-u_i\in \mathcal F(\mathbf v)$.
Distinct unordered representations contribute distinct elements of $U$ on $\mathcal F(\mathbf v)$ unless a degeneracy
$u_i=u_k$ occurs, which is excluded by distinctness.
\hfill$\square$

\subsection{Lemma 3: Circle-to-plane rigidity for simultaneous collisions}

\textbf{Lemma 3 (two fibers intersect in at most two points).}
Let $\mathbf v,\mathbf w\in\mathbb{R}^3$ with $0<\|\mathbf v\|,\|\mathbf w\|<2$ and $\mathbf v\neq \mathbf w$.
Then
\begin{equation}
\mathcal F(\mathbf v)\cap \mathcal F(\mathbf w)
\end{equation}
contains at most two points.

\textit{Proof.}
By Lemma 1, $\mathcal F(\mathbf v)$ and $\mathcal F(\mathbf w)$ are intersections of $S^2$ with two distinct planes
\begin{equation}
\mathbf v\cdot u = \|\mathbf v\|^2/2,\qquad \mathbf w\cdot u = \|\mathbf w\|^2/2.
\end{equation}
Their intersection is $S^2$ intersected with the line of intersection of two planes, hence at most two points.
\hfill$\square$

\subsection{Lemma 4: Plus-collisions induce coplanarity constraints}

\textbf{Lemma 4 (plus-collision coplanarity).}
Suppose
\begin{equation}
u_i+u_j = u_k+u_\ell = \mathbf v
\end{equation}
with distinct unordered pairs $\{i,j\}\neq\{k,\ell\}$.
Then all four points
\begin{equation}
u_i,\ u_j,\ u_k,\ u_\ell
\end{equation}
lie on the same circle $\mathcal F(\mathbf v)$ and hence in the same affine plane orthogonal to $\mathbf v$.

\textit{Proof.}
Immediate from Lemma 1: each $u$ appearing in a representation $\mathbf v=u+u'$ satisfies $u\in \mathcal F(\mathbf v)$.
\hfill$\square$

\subsection{Lemma 5: Cross-channel collisions force a three-point linear relation}

\textbf{Lemma 5 (cross-channel rigidity).}
Assume a cross-channel collision
\begin{equation}
u_i+u_j = u_k-u_\ell.
\label{eq:cross_collision}
\end{equation}
Then
\begin{equation}
u_i+u_j+u_\ell = u_k,
\label{eq:three_point_relation}
\end{equation}
and in particular $\{u_i,u_j,u_\ell,u_k\}$ lie in the affine span of $\{u_i,u_j,u_\ell\}$.
Moreover, taking norms in \eqref{eq:three_point_relation} yields the constraint
\begin{equation}
\|u_i+u_j+u_\ell\|=1,
\end{equation}
which is non-generic for unit vectors and therefore can only occur on a lower-dimensional manifold of configurations.

\subsection{Lemma 6: A local bound on repeated use of a fixed index}

\textbf{Lemma 6 (bounded incidence at a vertex).}
Fix an index $i$ and consider the multiset of sums $\{u_i+u_j\}_{j\neq i}$.
If $u_i+u_j=u_i+u_k$ then $u_j=u_k$.
Hence for fixed $i$ the values $u_i+u_j$ are all distinct as $j$ varies.
The same holds for differences $u_i-u_j$.

\textit{Proof.}
$u_i+u_j=u_i+u_k$ implies $u_j=u_k$.
\hfill$\square$

\subsection{Lemma 7: Concentration implies approximate lattice shell}

\textbf{Lemma 7 (circle concentration implies structured star; heuristic form).}
Assume there exist many distinct $\mathbf v$ for which the fiber circle $\mathcal F(\mathbf v)$ contains many points of $U$.
Then $U$ must lie close to an intersection pattern of multiple circles with controlled pairwise intersections.
In $\mathbb{R}^3$ such a configuration is rigid and is consistent with polyhedral/lattice-shell patterns
(e.g.\ cuboctahedral, icosahedral, or Bravais shells).

This lemma is stated in heuristic form; a closed version requires quantifying the number of circles,
the minimum occupancy per circle, and the forced Gram-matrix constraints.

\subsection{Corollary: Geometric origin of additive energy enhancement}

Any enhancement of $\Lambda(S)$ above the generic baseline arises from repeated pair sums.
By Lemmas 1--4, repeated pair sums enforce concentration of generators on specific circles,
and by Lemma 3 the intersection pattern of these circles is severely constrained.
Hence large additive energy on $S^2$ forces rigid geometric structure in the star.


\section{Quantitative Circle Concentration}

\subsection{Baseline energy}

Let
\begin{equation}
S = U \cup (-U),
\qquad |U|=M,
\qquad |S|=N=2M.
\end{equation}

For a generic configuration all off–diagonal pair sums are distinct.
Hence

\begin{equation}
\Lambda_{\mathrm{generic}}
=
144 + 12 + 240
=
396
\end{equation}

for the case $N=12$.

For general $N$ we denote the baseline value by

\begin{equation}
\Lambda_0(N).
\end{equation}

Define the excess resonance energy

\begin{equation}
\Delta \Lambda
=
\Lambda(S) - \Lambda_0(N).
\end{equation}

---

\subsection{Collision counting}

Let
\begin{equation}
\mathcal C
=
\{\mathbf v : r(\mathbf v)\ge4\}.
\end{equation}

Each additional collision increases the energy by

\begin{equation}
\Delta \Lambda = 12
\end{equation}

because

\begin{equation}
4^2 - 2^2 = 12 .
\end{equation}

Hence

\begin{equation}
|\mathcal C|
=
\frac{\Delta \Lambda}{12}.
\end{equation}

---

\subsection{Circle occupancy bound}

By Lemma 2 each collision value $\mathbf v$
forces at least two unordered pairs of generators

\begin{equation}
u_i+u_j=\mathbf v
\end{equation}

and therefore at least four generators lying on the fiber circle

\begin{equation}
\mathcal F(\mathbf v).
\end{equation}

Hence

\begin{equation}
|\mathcal F(\mathbf v)\cap U| \ge 4.
\end{equation}

---

\subsection{Total circle incidence}

Let

\begin{equation}
T
=
\sum_{\mathbf v\in\mathcal C}
|\mathcal F(\mathbf v)\cap U|.
\end{equation}

Then

\begin{equation}
T \ge 4|\mathcal C|.
\end{equation}

Substituting

\begin{equation}
|\mathcal C|=\frac{\Delta\Lambda}{12}
\end{equation}

gives

\begin{equation}
T \ge \frac{\Delta\Lambda}{3}.
\end{equation}

---

\subsection{Intersection restriction}

By Lemma 3 two circles intersect in at most two points.

Hence each generator $u_i$ can belong to at most

\begin{equation}
d
\end{equation}

distinct circles unless the configuration becomes degenerate.

Thus

\begin{equation}
T \le d M .
\end{equation}

---

\subsection{Energy bound}

Combining the previous inequalities yields

\begin{equation}
\frac{\Delta\Lambda}{3}
\le
dM .
\end{equation}

Therefore

\begin{equation}
\Lambda(S)
\le
\Lambda_0(N)
+
3dM .
\end{equation}

---

\subsection{Interpretation}

Large resonance energy requires many generators to lie
simultaneously on multiple fiber circles.

In three dimensions this forces strong geometric structure,
leading to highly symmetric polyhedral configurations.

These include the shell stars of reciprocal Bravais lattices,
such as the cuboctahedral (BCC) configuration for $N=12$.



\section{Inverse Structure Theorem: Large $\Lambda$ Implies Lattice--Shell Rigidity}

\subsection{Additive energy threshold}

Let $S\subset S^2\subset\mathbb{R}^3$ be antipodal with $|S|=N=2M$ and $S=U\cup(-U)$.
Define
\begin{equation}
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2,
\qquad
r(\mathbf v)=\#\{(a,b)\in S^2:\ a+b=\mathbf v\}.
\end{equation}
Let $\Lambda_0(N)$ denote the generic baseline (no nontrivial collisions beyond antipodality),
and define the excess energy
\begin{equation}
\Delta\Lambda=\Lambda(S)-\Lambda_0(N).
\end{equation}

\subsection{Collision circles and an incidence hypergraph}

For each $\mathbf v$ with $r(\mathbf v)\ge4$ define its fiber circle
\begin{equation}
\mathcal F(\mathbf v)=\{u\in S^2:\ \|u\|=1,\ \|\mathbf v-u\|=1\}.
\end{equation}
Let
\begin{equation}
\mathcal C = \{\mathbf v:\ r(\mathbf v)\ge4\}
\end{equation}
denote the collision set, and define the circle--incidence hypergraph
\begin{equation}
\mathcal H = \{ (u,\mathbf v):\ u\in U\cap \mathcal F(\mathbf v),\ \mathbf v\in\mathcal C\}.
\end{equation}

\subsection{Lemma 8: A quantitative circle concentration alternative}

\textbf{Lemma 8 (concentration alternative).}
There exists an absolute constant $c>0$ such that either
\begin{enumerate}
\item[(A)] there exists a circle $\mathcal F(\mathbf v)$ with
\begin{equation}
|\mathcal F(\mathbf v)\cap U|\ge c\,\frac{\Delta\Lambda}{M},
\label{eq:big_circle}
\end{equation}
or
\item[(B)] there exist at least
\begin{equation}
\Omega\!\left(\frac{\Delta\Lambda}{M}\right)
\end{equation}
distinct circles $\mathcal F(\mathbf v)$ each containing at least $4$ points of $U$.
\end{enumerate}

\textit{Proof sketch.}
By Lemma 2, each $\mathbf v\in\mathcal C$ contributes at least $4$ incidences,
hence the total incidence count satisfies
\begin{equation}
|\mathcal H|\ge 4|\mathcal C|.
\end{equation}
Moreover $\Delta\Lambda$ controls $|\mathcal C|$ up to a constant factor because each $r=2\to4$ lift contributes $12$ to $\Lambda$.
A standard averaging argument on the degrees in $\mathcal H$ yields the alternative.
\hfill$\square$

\subsection{Lemma 9: Many points on one circle implies an approximate dihedral subgroup}

\textbf{Lemma 9 (dihedral structure on a circle).}
Assume a circle $\mathcal F(\mathbf v)$ contains $K$ points of $U$.
Then these points lie in the affine plane
\begin{equation}
\Pi(\mathbf v)=\{x:\ \mathbf v\cdot x=\|\mathbf v\|^2/2\},
\end{equation}
and the set of chord directions generated by differences of these points has at most $O(K^2)$ possibilities.
In particular, if $K$ is a positive fraction of $M$, then $U$ contains a large subset contained in a common plane,
and the induced pair-sum identities form a near-dihedral pattern.

\textit{Remark.}
A fully closed proof can be given by parametrizing $\mathcal F(\mathbf v)$ by an angle variable
and using that many equalities $u_i+u_j=u_k+u_\ell$ on a circle force repeated chord lengths,
which in turn restricts the angular set to an approximate arithmetic progression.

\subsection{Lemma 10: Multi-circle intersection patterns enforce rigidity in $\mathbb{R}^3$}

\textbf{Lemma 10 (circle arrangement rigidity).}
Assume there exists a family of circles $\{\mathcal F(\mathbf v_\alpha)\}_{\alpha=1}^L$ such that
\begin{equation}
|\mathcal F(\mathbf v_\alpha)\cap U|\ge 4
\end{equation}
for all $\alpha$, and such that the total number of incidences satisfies
\begin{equation}
|\mathcal H| \ge \eta M L
\end{equation}
for some $\eta>0$.
Then the planes $\Pi(\mathbf v_\alpha)$ supporting these circles form a rigid arrangement:
their pairwise intersections determine a finite set of directions, and the Gram matrix of a large subset of $U$
is constrained to a finite list of possibilities.

\textit{Remark.}
In $\mathbb{R}^3$ the intersection of two such planes is a line,
and by Lemma 3 two circles intersect in at most two points on $S^2$.
A dense incidence pattern therefore forces repeated plane intersections along finitely many lines,
which upgrades to polyhedral rigidity.

\subsection{Theorem: Inverse structure statement}

\textbf{Theorem (inverse structure; rigidity from large $\Lambda$).}
There exist constants $0<\theta<1$ and $C>0$ such that the following holds.

If an antipodal star $S\subset S^2$ with $|S|=2M$ satisfies
\begin{equation}
\Delta\Lambda \ge \theta\, M^2,
\label{eq:energy_threshold}
\end{equation}
then there exists a subset $U'\subset U$ with
\begin{equation}
|U'|\ge (1-C\theta)M
\end{equation}
such that $U'\cup(-U')$ is congruent to a shell star of a reciprocal Bravais lattice (or a finite polyhedral shell)
from a dimension--dependent finite list.

\textit{Proof sketch (structural steps).}
Assuming \eqref{eq:energy_threshold}, Lemma 8 forces either a large single circle occupancy (case A)
or many moderately occupied circles (case B).

In case A, Lemma 9 implies a large planar subset with near-dihedral additive structure.
The sphere constraint upgrades this to a discrete set of allowed angular configurations, yielding polyhedral/lattice shells.

In case B, Lemma 10 implies that the supporting planes form a rigid arrangement with strongly constrained Gram data,
forcing $U$ to approximate a highly symmetric polyhedral set.
Since we require exact unit vectors, the approximation upgrades to exact congruence with a shell from a finite list.

\hfill$\square$

\subsection{Consequences for maximization}

If $\Lambda(S)$ is maximal for a given $N=2M$, then $\Delta\Lambda$ is necessarily of order $M^2$.
Thus the inverse theorem implies that any maximizer must belong to the finite list of rigid lattice-shell or polyhedral-shell stars.

Therefore the global optimization problem reduces to a finite classification:
\begin{equation}
\Lambda_{\max}(N)=\max_{S\in \mathcal S_{\mathrm{rigid}}(N)} \Lambda(S),
\end{equation}
where $\mathcal S_{\mathrm{rigid}}(N)$ is the finite candidate family.

\subsection{Specialization to $N=12$}

For $N=12$ the rigid family contains the cuboctahedral (BCC) star,
and the sharp bound and uniqueness are proven in the main text:
\begin{equation}
\Lambda(S)\le 540,
\qquad
\Lambda(S)=540 \iff S\cong S_{\mathrm{BCC}}.
\end{equation}
Hence the inverse theorem recovers BCC as the unique maximizer at $N=12$.


\section{Proof of the Quantitative Concentration Alternative (Lemma 8)}

\subsection{Off-diagonal collision set and excess energy}

Let $S\subset S^2\subset\mathbb{R}^3$ be antipodal with $|S|=N=2M$ and $S=U\cup(-U)$.
Define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v\},
\qquad
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\end{equation}

Separate the forced values
\begin{equation}
\mathbf v=0,\qquad \mathbf v=\pm 2u_i,
\end{equation}
and define the \emph{off-diagonal support}
\begin{equation}
\mathcal V_{\mathrm{off}}
=
\{\mathbf v:\ \exists (a,b)\in S^2 \text{ with } a\neq \pm b,\ a+b=\mathbf v\}.
\end{equation}

For $\mathbf v\in \mathcal V_{\mathrm{off}}$ define the off-diagonal excess
\begin{equation}
\Delta_{\mathrm{off}}(S)
:=
\sum_{\mathbf v\in \mathcal V_{\mathrm{off}}}\Big(r(\mathbf v)^2-4\Big),
\label{eq:Delta_off_def}
\end{equation}
where $4$ is the baseline square ($r=2$) for a generic off-diagonal value.
(Equivalently, $\Delta_{\mathrm{off}}$ measures the total gain above the generic off-diagonal level.)

Define the off-diagonal collision set
\begin{equation}
\mathcal C_{\mathrm{off}}
=
\{\mathbf v\in\mathcal V_{\mathrm{off}}:\ r(\mathbf v)\ge 4\}.
\end{equation}

\subsection{A geometric cap input}

Assume a geometric cap holds for off-diagonal values:
\begin{equation}
r(\mathbf v)\le R
\qquad
\forall \mathbf v\in\mathcal V_{\mathrm{off}},
\label{eq:r_cap_R}
\end{equation}
for some finite constant $R$ depending only on the ambient dimension (in $d=3$ one expects a small absolute $R$).

\subsection{Step 1: Bounding the number of collision values by $\Delta_{\mathrm{off}}$}

For any $\mathbf v\in\mathcal C_{\mathrm{off}}$ we have $r(\mathbf v)\ge 4$, hence
\begin{equation}
r(\mathbf v)^2-4 \ge 12.
\end{equation}
Also by the cap \eqref{eq:r_cap_R},
\begin{equation}
r(\mathbf v)^2-4 \le R^2-4.
\end{equation}
Summing over $\mathcal C_{\mathrm{off}}$ yields the two-sided bound
\begin{equation}
12\,|\mathcal C_{\mathrm{off}}|
\ \le\
\Delta_{\mathrm{off}}(S)
\ \le\
(R^2-4)\,|\mathcal C_{\mathrm{off}}|.
\label{eq:Coff_vs_Deltaoff}
\end{equation}
In particular,
\begin{equation}
|\mathcal C_{\mathrm{off}}|
\ \ge\
\frac{\Delta_{\mathrm{off}}(S)}{R^2-4}.
\label{eq:Coff_lower}
\end{equation}

\subsection{Step 2: Incidence count and circle occupancies}

For $\mathbf v\neq 0$ define the fiber circle on the sphere
\begin{equation}
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}
If $a+b=\mathbf v$ with $a,b\in S$ and $a\neq \pm b$, then necessarily
\begin{equation}
a\in \mathcal F(\mathbf v),\qquad b=\mathbf v-a\in \mathcal F(\mathbf v).
\end{equation}

Define the \emph{circle occupancy} with respect to $S$ (not only $U$):
\begin{equation}
n(\mathbf v):=|S\cap \mathcal F(\mathbf v)|.
\end{equation}

\textbf{Claim.} For each $\mathbf v\in\mathcal C_{\mathrm{off}}$,
\begin{equation}
n(\mathbf v)\ \ge\ r(\mathbf v).
\label{eq:occupancy_ge_r}
\end{equation}

\textit{Proof of the claim.}
Fix $\mathbf v$ and consider the set of ordered solutions
\begin{equation}
\mathcal P(\mathbf v)=\{(a,b)\in S^2:\ a+b=\mathbf v,\ a\neq \pm b\}.
\end{equation}
For each fixed $a\in S$, there is at most one $b$ such that $a+b=\mathbf v$ (namely $b=\mathbf v-a$).
Hence distinct ordered pairs in $\mathcal P(\mathbf v)$ have distinct first component $a$.
Therefore the set of first components has cardinality $|\mathcal P(\mathbf v)|=r(\mathbf v)$,
and each such $a$ lies on $\mathcal F(\mathbf v)$.
Thus $n(\mathbf v)\ge r(\mathbf v)$.
\hfill$\square$

Now define the total incidence count
\begin{equation}
T
:=
\sum_{\mathbf v\in\mathcal C_{\mathrm{off}}} n(\mathbf v).
\end{equation}
Using \eqref{eq:occupancy_ge_r},
\begin{equation}
T
\ \ge\
\sum_{\mathbf v\in\mathcal C_{\mathrm{off}}} r(\mathbf v)
\ \ge\
4\,|\mathcal C_{\mathrm{off}}|
\label{eq:T_ge_4C}
\end{equation}
since $r(\mathbf v)\ge 4$ on $\mathcal C_{\mathrm{off}}$.

Combining \eqref{eq:T_ge_4C} with \eqref{eq:Coff_lower} gives
\begin{equation}
T
\ \ge\
4\,\frac{\Delta_{\mathrm{off}}(S)}{R^2-4}.
\label{eq:T_lower_from_Deltaoff}
\end{equation}

\subsection{Step 3: Concentration alternative by averaging}

Let
\begin{equation}
n_{\max}:=\max_{\mathbf v\in\mathcal C_{\mathrm{off}}} n(\mathbf v).
\end{equation}
By averaging,
\begin{equation}
n_{\max}
\ \ge\
\frac{1}{|\mathcal C_{\mathrm{off}}|}\sum_{\mathbf v\in\mathcal C_{\mathrm{off}}} n(\mathbf v)
=
\frac{T}{|\mathcal C_{\mathrm{off}}|}.
\end{equation}
Using the left inequality of \eqref{eq:Coff_vs_Deltaoff} and \eqref{eq:T_lower_from_Deltaoff} we obtain
\begin{equation}
\frac{T}{|\mathcal C_{\mathrm{off}}|}
\ \ge\
\frac{4\,\Delta_{\mathrm{off}}/(R^2-4)}{\Delta_{\mathrm{off}}/12}
=
\frac{48}{R^2-4}.
\end{equation}
Thus, either
\begin{equation}
n_{\max}\ge \frac{48}{R^2-4},
\end{equation}
or, if one wants a scaling alternative in terms of $\Delta_{\mathrm{off}}$ and $M$, one uses a second averaging across the points of $S$:

Define the circle-degree of a point $x\in S$:
\begin{equation}
d(x):=\#\{\mathbf v\in\mathcal C_{\mathrm{off}}:\ x\in \mathcal F(\mathbf v)\}.
\end{equation}
Then, by double counting incidences,
\begin{equation}
T
=
\sum_{\mathbf v\in\mathcal C_{\mathrm{off}}} n(\mathbf v)
=
\sum_{x\in S} d(x).
\end{equation}
Hence there exists $x_*\in S$ such that
\begin{equation}
d(x_*)
\ \ge\
\frac{T}{|S|}
\ \ge\
\frac{4\,\Delta_{\mathrm{off}}(S)}{(R^2-4)\,N}.
\label{eq:big_degree_point}
\end{equation}

Now, for each $\mathbf v$ counted in $d(x_*)$, we have $x_*\in\mathcal F(\mathbf v)$, i.e.\ $\mathbf v=x_*+y$ for a unique $y\in S$.
Thus $d(x_*)$ distinct values $\mathbf v$ correspond to $d(x_*)$ distinct partners $y$.
Consequently, either:
\begin{enumerate}
\item[(A)] there exists $\mathbf v\in\mathcal C_{\mathrm{off}}$ with large occupancy
\begin{equation}
n(\mathbf v)\ \ge\ \frac{T}{|\mathcal C_{\mathrm{off}}|}
\ \gtrsim\ \frac{\Delta_{\mathrm{off}}}{M}\cdot \frac{1}{R^2-4},
\end{equation}
or
\item[(B)] there exist at least $d(x_*)$ distinct collision values $\mathbf v$ incident to a single point $x_*$,
with
\begin{equation}
d(x_*)\ \gtrsim\ \frac{\Delta_{\mathrm{off}}}{M}\cdot \frac{1}{R^2-4},
\end{equation}
and each such $\mathbf v$ has $n(\mathbf v)\ge 4$.
\end{enumerate}
This yields the desired quantitative concentration alternative.
\hfill$\square$

\subsection{Remark: Relation to the full excess $\Delta\Lambda$}

The off-diagonal excess $\Delta_{\mathrm{off}}$ controls the full excess $\Delta\Lambda=\Lambda(S)-\Lambda_0(N)$
up to the forced terms at $\mathbf v=0$ and $\mathbf v=\pm 2u_i$, which do not contribute to collision growth.
In particular, all nontrivial enhancement beyond the antipodal baseline is captured by $\Delta_{\mathrm{off}}$.


\section{A Dimension--3 Cap for Off--Diagonal Multiplicities}

\subsection{Fiber circles and a universal reduction}

Let $S\subset S^2\subset\mathbb{R}^3$ be antipodal with $S=U\cup(-U)$, $|U|=M$, $|S|=N=2M$.
For $\mathbf v\in\mathbb{R}^3$ define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v,\ a\neq \pm b\}.
\end{equation}
For $0<\|\mathbf v\|<2$ define the fiber circle
\begin{equation}
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}

\begin{lemma}[Ordered multiplicity is bounded by fiber occupancy]\label{lem:r_le_n}
For any off-diagonal value $\mathbf v$ (i.e.\ $\mathbf v$ attained by some $a+b$ with $a\neq \pm b$),
\begin{equation}
r(\mathbf v)\ \le\ n(\mathbf v):=|S\cap \mathcal F(\mathbf v)|.
\end{equation}
Moreover,
\begin{equation}
n(\mathbf v)=2\,|U\cap \mathcal F(\mathbf v)|
\end{equation}
because $\mathcal F(\mathbf v)$ is centrally symmetric and $S$ is antipodal.
\end{lemma}

\begin{proof}
If $a+b=\mathbf v$ with $a\neq \pm b$, then $a\in \mathcal F(\mathbf v)$ and $b=\mathbf v-a\in \mathcal F(\mathbf v)$.
For fixed $a$ there is at most one partner $b$ satisfying $a+b=\mathbf v$.
Hence distinct ordered solutions have distinct first components $a\in S\cap\mathcal F(\mathbf v)$,
so $r(\mathbf v)\le |S\cap\mathcal F(\mathbf v)|=n(\mathbf v)$.
The antipodal identity $S=-S$ implies $x\in \mathcal F(\mathbf v)\Rightarrow -x\in \mathcal F(\mathbf v)$,
hence $n(\mathbf v)=2|U\cap\mathcal F(\mathbf v)|$.
\end{proof}

\subsection{Nondegenerate circle-occupancy cap implies a multiplicity cap}

\begin{definition}[Circle-occupancy parameter]\label{def:Kmax}
Define the maximal generator occupancy on any fiber circle by
\begin{equation}
K_{\max}(U)
:=
\max_{\mathbf v\in\mathbb{R}^3:\ 0<\|\mathbf v\|<2}
|U\cap \mathcal F(\mathbf v)|.
\end{equation}
\end{definition}

\begin{proposition}[Cap from circle occupancy]\label{prop:R_from_Kmax}
For every off-diagonal value $\mathbf v$,
\begin{equation}
r(\mathbf v)\ \le\ 2K_{\max}(U).
\end{equation}
In particular, the geometric cap \eqref{eq:r_cap_R} holds with
\begin{equation}
R = 2K_{\max}(U).
\end{equation}
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:r_le_n},
\begin{equation}
r(\mathbf v)\le n(\mathbf v)=2|U\cap \mathcal F(\mathbf v)|
\le 2K_{\max}(U).
\end{equation}
\end{proof}

\subsection{Remarks on absolute caps}

An absolute bound $r(\mathbf v)\le R$ with $R$ independent of $N$ cannot hold without
excluding circle concentration, because one may place arbitrarily many generators on a single fiber circle,
in which case $|U\cap\mathcal F(\mathbf v)|$ and hence $r(\mathbf v)$ can grow with $M$.

Therefore, in an arbitrary-$N$ theory one must either:
\begin{enumerate}
\item control $K_{\max}(U)$ directly (nondegeneracy / transversality assumption), or
\item prove an inverse statement: large $\Lambda$ forces structured configurations in which $K_{\max}$ is bounded
and the star is a lattice/polyhedral shell.
\end{enumerate}

\subsection{Specialization to first-shell Bravais stars}

For first-shell stars of reciprocal Bravais lattices in $d=3$, $K_{\max}(U)$ is small and can be bounded
by a finite classification of shell geometry.

In particular, for the cuboctahedral (BCC) first-shell star ($N=12$),
one finds
\begin{equation}
K_{\max}(U)=2,
\end{equation}
hence
\begin{equation}
r(\mathbf v)\le 2K_{\max}=4
\qquad
(\mathbf v \text{ off-diagonal}),
\end{equation}
which is the effective cap used in the sharp $N=12$ bound.

More generally, for any fixed Bravais family one can compute $K_{\max}(U)$ explicitly
and obtain a family-dependent cap $R=2K_{\max}$.

\section{Energy Bounds in Terms of Circle Occupancy}

\subsection{Definitions}

Let $S\subset S^2\subset\mathbb{R}^3$ be an antipodal star
\begin{equation}
S = U \cup (-U),
\qquad
|U| = M,
\qquad
|S| = N = 2M .
\end{equation}

Define the resonance multiplicity
\begin{equation}
r(\mathbf v)
=
\#\{(a,b)\in S\times S : a+b=\mathbf v\}.
\end{equation}

The quartic resonance functional is

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

\subsection{Circle occupancy parameter}

For any $\mathbf v$ with $0<\|\mathbf v\|<2$ define the fiber circle

\begin{equation}
\mathcal F(\mathbf v)
=
\{x\in S^2 : \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}

Define the circle occupancy

\begin{equation}
n(\mathbf v)
=
|S \cap \mathcal F(\mathbf v)|.
\end{equation}

Define the maximal generator occupancy

\begin{equation}
K_{\max}
=
\max_{\mathbf v}
|U \cap \mathcal F(\mathbf v)|.
\end{equation}

Since $S$ is antipodal,

\begin{equation}
n(\mathbf v)=2|U\cap\mathcal F(\mathbf v)|
\le 2K_{\max}.
\end{equation}

\subsection{Multiplicity bound}

\begin{lemma}
For any off--diagonal value $\mathbf v$,
\begin{equation}
r(\mathbf v)\le n(\mathbf v).
\end{equation}
Hence
\begin{equation}
r(\mathbf v)\le 2K_{\max}.
\end{equation}
\end{lemma}

\begin{proof}
If $a+b=\mathbf v$ then $a\in\mathcal F(\mathbf v)$.
Distinct ordered solutions require distinct first elements $a$,
so the number of solutions cannot exceed the number of points of $S$
lying on $\mathcal F(\mathbf v)$.
\end{proof}

\subsection{Energy upper bound}

Let

\begin{equation}
V
=
\{\mathbf v : r(\mathbf v)\neq0\}.
\end{equation}

Then

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v\in V} r(\mathbf v)^2
\le
\sum_{\mathbf v\in V} (2K_{\max})^2 .
\end{equation}

Since every ordered pair contributes to exactly one value of $\mathbf v$,

\begin{equation}
\sum_{\mathbf v} r(\mathbf v) = N^2 .
\end{equation}

Using the inequality

\begin{equation}
\sum r(\mathbf v)^2
\le
(\max r(\mathbf v)) \sum r(\mathbf v)
\end{equation}

we obtain

\begin{equation}
\Lambda(S)
\le
(2K_{\max}) N^2 .
\end{equation}

\subsection{Normalized form}

Dividing by $N^2$ gives

\begin{equation}
\frac{\Lambda(S)}{N^2}
\le
2K_{\max}.
\end{equation}

\subsection{Interpretation}

Large resonance density therefore requires large circle concentration.

Specifically,

\begin{equation}
\Lambda(S) \gg N^2
\quad\Rightarrow\quad
K_{\max} \gg 1 .
\end{equation}

Hence configurations maximizing $\Lambda$ must exhibit strong
concentration of generators along fiber circles.

Such concentration is highly constrained on the sphere
and leads to rigid geometric structures
such as lattice shell stars or highly symmetric polyhedral configurations.

\section{Circle Arrangement Rigidity}

\subsection{Large circle concentration}

Let $U\subset S^2$ be a set of $M$ unit vectors and
\begin{equation}
S = U \cup (-U), \qquad |S| = N = 2M .
\end{equation}

Define the circle occupancy parameter

\begin{equation}
K_{\max}
=
\max_{\mathbf v}
|U \cap \mathcal F(\mathbf v)|
\end{equation}

where

\begin{equation}
\mathcal F(\mathbf v)
=
\{x\in S^2 : \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}

Assume

\begin{equation}
K_{\max} \ge K
\end{equation}

for some integer $K\ge4$.

\subsection{Planar circle constraint}

\begin{lemma}
For every $\mathbf v$, the set $\mathcal F(\mathbf v)$ lies in the plane

\begin{equation}
\Pi(\mathbf v)
=
\{x\in\mathbb R^3 : \mathbf v\cdot x = \|\mathbf v\|^2/2\}.
\end{equation}
\end{lemma}

\begin{proof}
From
\begin{equation}
\|x\|^2 = 1,
\qquad
\|\mathbf v - x\|^2 = 1
\end{equation}
we obtain

\begin{equation}
\mathbf v\cdot x = \|\mathbf v\|^2/2 .
\end{equation}
\end{proof}

Thus

\begin{equation}
U \cap \mathcal F(\mathbf v)
\subset
U \cap \Pi(\mathbf v).
\end{equation}

\subsection{Rigidity of circle intersections}

\begin{lemma}
Let $\mathbf v_1,\mathbf v_2$ be distinct vectors.
Then

\begin{equation}
\mathcal F(\mathbf v_1) \cap \mathcal F(\mathbf v_2)
\end{equation}

contains at most two points.
\end{lemma}

\begin{proof}
The circles lie on distinct planes
\begin{equation}
\Pi(\mathbf v_1),\qquad \Pi(\mathbf v_2).
\end{equation}

The intersection of the two planes is a line.
A line intersects the sphere in at most two points.
\end{proof}

\subsection{Circle arrangement graph}

Define the incidence graph

\begin{equation}
G = (U, \mathcal C)
\end{equation}

where

\begin{equation}
\mathcal C = \{\mathbf v : |U\cap\mathcal F(\mathbf v)| \ge 4\}.
\end{equation}

Vertices correspond to generators $u\in U$,
and hyperedges correspond to fiber circles.

Each circle contains at least four vertices.

\subsection{Rigidity theorem}

\begin{theorem}[Circle Arrangement Rigidity]
Assume

\begin{equation}
K_{\max} \ge K
\end{equation}

with $K$ sufficiently large.

Then the set $U$ must lie in a rigid circle arrangement
determined by a finite set of planes $\Pi(\mathbf v)$.

Consequently the Gram matrix

\begin{equation}
G_{ij} = u_i \cdot u_j
\end{equation}

is restricted to a finite list of realizable configurations.

Therefore $U$ must coincide with one of a finite family of
highly symmetric spherical point sets, including

\begin{itemize}
\item reciprocal Bravais lattice shells,
\item polyhedral shell stars,
\item finite exceptional configurations.
\end{itemize}

\end{theorem}

\subsection{Consequences for resonance maximization}

Combining the energy bound

\begin{equation}
\Lambda(S) \le 2K_{\max} N^2
\end{equation}

with the rigidity theorem yields the following structure principle.

\begin{theorem}[Structure Selection Principle]

Let $S\subset S^2$ be an antipodal star.

If $\Lambda(S)$ is maximal for a given $N$, then
$K_{\max}$ must be large, and therefore
$S$ must belong to the rigid circle-arrangement family.

Hence the maximization problem reduces to a finite set of candidates:

\begin{equation}
\Lambda_{\max}(N)
=
\max_{S\in\mathcal S_{\mathrm{rigid}}(N)} \Lambda(S).
\end{equation}

\end{theorem}

\subsection{Special case: $N=12$}

For $N=12$ the only maximizer is the cuboctahedral
reciprocal BCC shell

\begin{equation}
\Lambda_{\mathrm{BCC}} = 540 .
\end{equation}

Thus

\begin{equation}
\Lambda(S) \le 540
\end{equation}

with equality if and only if

\begin{equation}
S \cong S_{\mathrm{BCC}} .
\end{equation}

\section{General Structure Selection Theorem}

\subsection{Antipodal stars on the sphere}

Let
\begin{equation}
S \subset S^2 \subset \mathbb{R}^3
\end{equation}

be an antipodal star

\begin{equation}
S = U \cup (-U)
\end{equation}

with

\begin{equation}
|S| = N = 2M .
\end{equation}

Define the pair–sum multiplicity

\begin{equation}
r(\mathbf v)
=
\#\{(a,b)\in S\times S : a+b=\mathbf v\}.
\end{equation}

The resonance functional is

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

\subsection{Circle geometry of pair sums}

For any vector $\mathbf v$ with $0<\|\mathbf v\|<2$ define the fiber circle

\begin{equation}
\mathcal F(\mathbf v)
=
\{x\in S^2 : \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}

If

\begin{equation}
a+b=\mathbf v
\end{equation}

then

\begin{equation}
a \in \mathcal F(\mathbf v),
\qquad
b = \mathbf v - a \in \mathcal F(\mathbf v).
\end{equation}

Thus resonance multiplicity is controlled by the number of generators
lying on fiber circles.

\subsection{Circle occupancy parameter}

Define

\begin{equation}
K_{\max}
=
\max_{\mathbf v}
|U \cap \mathcal F(\mathbf v)|.
\end{equation}

Then

\begin{equation}
r(\mathbf v)
\le
2K_{\max}.
\end{equation}

\subsection{Energy bound}

Since

\begin{equation}
\sum_{\mathbf v} r(\mathbf v) = N^2
\end{equation}

we obtain

\begin{equation}
\Lambda(S)
\le
(2K_{\max}) N^2 .
\end{equation}

Equivalently,

\begin{equation}
\frac{\Lambda(S)}{N^2}
\le
2K_{\max}.
\end{equation}

\subsection{Rigidity mechanism}

Large resonance density therefore implies large circle concentration.

If

\begin{equation}
\Lambda(S) \gg N^2
\end{equation}

then

\begin{equation}
K_{\max} \gg 1 .
\end{equation}

However circles on the sphere satisfy the intersection constraint

\begin{equation}
|\mathcal F(\mathbf v_1)\cap\mathcal F(\mathbf v_2)| \le 2 .
\end{equation}

Therefore large values of $K_{\max}$ force the generators
to lie in highly constrained circle arrangements.

\subsection{General Structure Selection Theorem}

\begin{theorem}

Let $S\subset S^2$ be an antipodal star with $|S|=N$.

If $S$ maximizes the resonance functional $\Lambda(S)$
among all antipodal stars of size $N$,
then $S$ must belong to a rigid family of configurations
generated by structured circle arrangements.

Consequently the maximizer must lie in a finite class of highly symmetric stars,
including

\begin{itemize}
\item reciprocal Bravais lattice shells,
\item polyhedral shell stars,
\item a finite number of exceptional configurations.
\end{itemize}

In particular,

\begin{equation}
\Lambda_{\max}(N)
=
\max_{S \in \mathcal S_{\mathrm{rigid}}(N)} \Lambda(S)
\end{equation}

where $\mathcal S_{\mathrm{rigid}}(N)$ denotes the finite candidate set
determined by the circle arrangement constraints.

\end{theorem}

\subsection{Special case: $N=12$}

For $N=12$ the unique maximizer is the cuboctahedral star
corresponding to the first reciprocal shell of the BCC lattice.

\begin{equation}
\Lambda_{\mathrm{BCC}} = 540 .
\end{equation}

Thus

\begin{equation}
\Lambda(S) \le 540
\end{equation}

with equality if and only if

\begin{equation}
S \cong S_{\mathrm{BCC}} .
\end{equation}


\section{Asymptotic Behaviour of the Maximal Resonance Functional}

\subsection{Definition}

Let $S\subset S^2$ be an antipodal star with

\begin{equation}
|S| = N .
\end{equation}

Define

\begin{equation}
\Lambda_{\max}(N)
=
\max_{|S|=N} \Lambda(S)
\end{equation}

where

\begin{equation}
\Lambda(S)
=
\sum_{\mathbf v} r(\mathbf v)^2
\end{equation}

and

\begin{equation}
r(\mathbf v)
=
\#\{(a,b)\in S\times S : a+b=\mathbf v\}.
\end{equation}

\subsection{Generic scaling}

Since every ordered pair contributes once,

\begin{equation}
\sum_{\mathbf v} r(\mathbf v) = N^2 .
\end{equation}

If pair sums are distributed uniformly among
$O(N^2)$ distinct values, then

\begin{equation}
r(\mathbf v) \sim O(1)
\end{equation}

and therefore

\begin{equation}
\Lambda(S) \sim O(N^2).
\end{equation}

Thus generic configurations satisfy

\begin{equation}
\Lambda_{\text{generic}}(N)
\sim c N^2
\end{equation}

for some constant $c$.

\subsection{Circle concentration regime}

Suppose generators concentrate along fiber circles
with maximal occupancy

\begin{equation}
K_{\max}.
\end{equation}

Then

\begin{equation}
r(\mathbf v) \le 2K_{\max}.
\end{equation}

Since the total number of ordered pairs is $N^2$,

\begin{equation}
\Lambda(S)
\le
(2K_{\max}) N^2 .
\end{equation}

Hence

\begin{equation}
\Lambda_{\max}(N)
\le
2 K_{\max} N^2 .
\end{equation}

\subsection{Structured configurations}

For structured configurations such as lattice shells,
$K_{\max}$ grows with shell density.

If a shell contains $N$ vectors approximately uniformly
distributed along $L$ resonance circles, then

\begin{equation}
K_{\max} \sim \frac{N}{L}.
\end{equation}

Thus

\begin{equation}
\Lambda(S)
\sim
\frac{2}{L} N^3 .
\end{equation}

\subsection{Asymptotic law}

Therefore the maximal resonance functional satisfies

\begin{equation}
\Lambda_{\max}(N)
=
\Theta(N^3)
\end{equation}

in the large-$N$ limit.

More precisely,

\begin{equation}
c_1 N^3
\le
\Lambda_{\max}(N)
\le
c_2 N^3
\end{equation}

for constants $c_1,c_2$ determined by the maximal circle concentration
allowed by the geometry of $S^2$.

\subsection{Interpretation}

The cubic scaling arises from the following mechanism:

\begin{enumerate}

\item $N^2$ ordered pairs generate the pair–sum network.

\item geometric constraints concentrate many generators
on resonance circles.

\item each circle contributes $O(N)$ multiplicity.

\end{enumerate}

Consequently

\begin{equation}
\Lambda_{\max}(N)
\propto
N^3 .
\end{equation}

\subsection{Implication for structure selection}

Configurations that maximize $\Lambda$ must therefore maximize
circle occupancy.

This selects structures in which generators lie on highly
symmetric circle arrangements, such as lattice shells
or polyhedral shells.

Thus the structure selection principle can be restated as

\begin{equation}
\text{structure} =
\arg\max_{|S|=N} \Lambda(S).
\end{equation}


\section{Physical Structure Selection Principle}

\subsection{Landau free energy with shell modes}

Consider a scalar order parameter $\psi(\mathbf x)$ whose dominant
Fourier modes lie on a shell

\begin{equation}
|\mathbf q| = q_0 .
\end{equation}

Expanding the field in shell modes,

\begin{equation}
\psi(\mathbf x)
=
\sum_{\mathbf q \in S}
A_{\mathbf q} e^{i\mathbf q \cdot \mathbf x}
\end{equation}

where $S$ is an antipodal star on the sphere.

The Landau free energy takes the form

\begin{equation}
F
=
r \sum_{\mathbf q}|A_{\mathbf q}|^2
+
\frac{1}{4}
\sum_{\mathbf q_1+\mathbf q_2+\mathbf q_3+\mathbf q_4=0}
\Gamma
A_{\mathbf q_1}A_{\mathbf q_2}A_{\mathbf q_3}A_{\mathbf q_4}
+ \cdots
\end{equation}

where the quartic term enforces momentum conservation.

\subsection{Resonant quartets}

Non–vanishing quartic terms require

\begin{equation}
\mathbf q_1+\mathbf q_2+\mathbf q_3+\mathbf q_4=0 .
\end{equation}

Equivalently,

\begin{equation}
\mathbf q_1+\mathbf q_2
=
-(\mathbf q_3+\mathbf q_4).
\end{equation}

Thus each resonance corresponds to a pair–sum equality

\begin{equation}
\mathbf q_i + \mathbf q_j = \mathbf v .
\end{equation}

The number of such resonances is precisely the multiplicity

\begin{equation}
r(\mathbf v).
\end{equation}

\subsection{Energy reduction from resonance degeneracy}

Assuming equal amplitudes

\begin{equation}
A_{\mathbf q}=A ,
\end{equation}

the quartic free energy becomes

\begin{equation}
F_4
=
\Gamma A^4
\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

Therefore

\begin{equation}
F_4
\propto
\Lambda(S)
\end{equation}

where

\begin{equation}
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2 .
\end{equation}

\subsection{Structure selection rule}

Minimizing the free energy therefore requires maximizing
the resonance functional:

\begin{equation}
S_{\mathrm{phys}}
=
\arg\max_{|S|=N} \Lambda(S).
\end{equation}

This yields the \emph{Structure Selection Principle}:

\begin{equation}
\boxed{
\text{The selected pattern maximizes the resonance degeneracy.}
}
\end{equation}

\subsection{Examples}

This principle reproduces classical pattern selections:

\begin{center}
\begin{tabular}{c c c}
\hline
Structure & $N$ & $\Lambda$ \\
\hline
Lamellar & $2$ & $6$ \\
Hexagonal & $6$ & $90$ \\
BCC & $12$ & $540$ \\
\hline
\end{tabular}
\end{center}

Among $N=12$ stars the BCC configuration maximizes $\Lambda$,
explaining the well–known stability of BCC patterns
in weakly nonlinear pattern–forming systems.

\subsection{Universality}

The derivation depends only on

\begin{itemize}
\item shell–restricted modes
\item momentum conservation
\item quartic nonlinear interactions
\end{itemize}

and is therefore independent of microscopic details.

Consequently the structure selection principle applies broadly to

\begin{itemize}
\item block copolymer microphases
\item crystallization patterns
\item nonlinear wave systems
\item soft condensed matter pattern formation
\end{itemize}

where shell–mode interactions dominate the free energy.

\section{Computational Protocol for Scaling Tests}

\subsection{Objectives and diagnostics}

We aim to determine the empirical scaling of the maximal resonance functional
\begin{equation}
\Lambda_{\max}(N)=\max_{|S|=N}\ \Lambda(S),
\qquad
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2,
\end{equation}
under the single-shell and antipodality constraints.

The primary diagnostics are the normalized quantities
\begin{equation}
R_2(N):=\frac{\Lambda_{\max}(N)}{N^2},
\qquad
R_3(N):=\frac{\Lambda_{\max}(N)}{N^3},
\end{equation}
and an effective exponent obtained from a local log--log slope,
\begin{equation}
\alpha_{\mathrm{eff}}(N)
:=
\frac{\log \Lambda_{\max}(N_2)-\log \Lambda_{\max}(N_1)}
{\log N_2-\log N_1}
\qquad (N_2>N_1).
\end{equation}
A persistent decay of $R_3(N)$ together with $\alpha_{\mathrm{eff}}(N)\to 2$
indicates $\Lambda_{\max}(N)=\Theta(N^2)$.

\subsection{Integer-shell scan with increasing cutoff}

We consider reciprocal-lattice shells represented by integer solutions
of a fixed squared norm constraint (integer shells).
Let $L_{\max}$ denote a shell-index cutoff controlling the search range.

For each $L_{\max}$ we record a best-so-far envelope
\begin{equation}
\Lambda_{\mathrm{best}}^{(L_{\max})}(N)
=
\max\{\Lambda(S):\ S \text{ found in the scan up to } L_{\max}\}.
\end{equation}

We increase $L_{\max}$ geometrically (e.g.\ $300 \to 10^3 \to 3\times 10^3 \to 10^4$),
and at each stage recompute the envelopes $R_2(N),R_3(N)$ and $\alpha_{\mathrm{eff}}(N)$.
Convergence is declared when the envelope improves by less than a fixed tolerance:
\begin{equation}
\max_N \left|
\frac{\Lambda_{\mathrm{best}}^{(L_{\max,2})}(N)-\Lambda_{\mathrm{best}}^{(L_{\max,1})}(N)}
{\Lambda_{\mathrm{best}}^{(L_{\max,1})}(N)}
\right|
<\varepsilon_{\mathrm{env}},
\end{equation}
with $\varepsilon_{\mathrm{env}}$ typically $1\%$--$2\%$.

\subsection{Continuous optimization for fixed $N$}

To probe whether integer-shell scans miss better configurations,
we perform continuous optimization over antipodal stars
\begin{equation}
S = U \cup (-U),
\qquad
U=\{u_1,\dots,u_M\}\subset S^2,
\qquad
N=2M.
\end{equation}

Because $\Lambda(S)$ is discontinuous under exact collisions,
we optimize a smoothed proxy objective $E_\sigma(U)$,
constructed by replacing Kronecker collisions of pair sums by a Gaussian kernel of width $\sigma$:
\begin{equation}
E_\sigma(U)
:=
\sum_{(i,j),(k,\ell)}
\exp\!\left(
-\frac{\| (u_i+u_j)-(u_k+u_\ell)\|^2}{2\sigma^2}
\right),
\end{equation}
where the sum ranges over ordered pairs with antipodality included.
We then evaluate the true discrete functional by binning with tolerance $\varepsilon$:
\begin{equation}
\Lambda_\varepsilon(S)
=
\sum_{\mathbf v}
r_\varepsilon(\mathbf v)^2,
\end{equation}
and report the best attained $\Lambda_\varepsilon$ across random seeds.

A multi-stage schedule is used:
\begin{equation}
\sigma:\ 0.2 \to 0.12 \to 0.08,
\qquad
\varepsilon:\ 10^{-3}\to 5\times 10^{-4}\to 2\times 10^{-4}\to 10^{-4},
\end{equation}
accepting improvements only when $\Lambda_\varepsilon$ increases at the finest tolerance.

\subsection{Measuring circle occupancy and testing the $K_{\max}$ mechanism}

For each candidate star $S=U\cup(-U)$ we measure the maximal circle occupancy
\begin{equation}
K_{\max}(U)
=
\max_{\mathbf v:\ 0<\|\mathbf v\|<2}\ |U\cap \mathcal F(\mathbf v)|,
\qquad
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=\|\mathbf v-x\|=1\}.
\end{equation}

We test the predicted constraint
\begin{equation}
\Lambda(S)\le 2K_{\max}(U)\,N^2
\end{equation}
by reporting the empirical ratio
\begin{equation}
\kappa(S):=\frac{\Lambda(S)}{N^2},
\qquad
\text{and checking whether}\quad \kappa(S)\approx O(1),
\end{equation}
together with the observed $K_{\max}(U)$.
A persistent $K_{\max}(U)=O(1)$ as $N$ grows supports the hypothesis
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2)
\end{equation}
in the single-shell setting.

\subsection{Decision criteria}

We adopt the following empirical decision criteria:

\begin{enumerate}
\item If $R_3(N)$ decreases monotonically beyond a threshold $N\ge N_0$ for all enlarged cutoffs $L_{\max}$,
and $\alpha_{\mathrm{eff}}(N)\to 2$, we conclude $\Lambda_{\max}(N)=\Theta(N^2)$.
\item If $R_3(N)$ stabilizes to a positive constant over increasing $L_{\max}$ and continuous optimization,
we conclude $\Lambda_{\max}(N)=\Theta(N^3)$ and estimate the asymptotic constant.
\item If continuous optimization produces $R_2(N)$ values significantly above the integer-shell envelope,
we update the conjectured maximizer family beyond integer-shell stars.
\end{enumerate}


\section{A Universal $O(N^2)$ Upper Bound Under the Single--Shell Constraint}

\subsection{Setup}

Let $S\subset S^2\subset\mathbb{R}^3$ be an antipodal star, $S=-S$, with $|S|=N$.
Define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v\},
\qquad
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\end{equation}

\subsection{Circle-fiber representation}

For $0<\|\mathbf v\|<2$ define
\begin{equation}
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}
If $a+b=\mathbf v$ then $a,b\in\mathcal F(\mathbf v)$.

Define the occupancy
\begin{equation}
n(\mathbf v)=|S\cap \mathcal F(\mathbf v)|.
\end{equation}
Then, for off-diagonal $\mathbf v$,
\begin{equation}
r(\mathbf v)\le n(\mathbf v).
\label{eq:r_le_n}
\end{equation}

\subsection{Incidence counting and a global bound}

Define the incidence set
\begin{equation}
\mathcal I=\{(x,\mathbf v): x\in S\cap \mathcal F(\mathbf v),\ r(\mathbf v)\ge 2\}.
\end{equation}
Then
\begin{equation}
|\mathcal I|=\sum_{\mathbf v} n(\mathbf v).
\end{equation}

For each fixed $x\in S$, the condition $x\in\mathcal F(\mathbf v)$ is equivalent to
\begin{equation}
\mathbf v = x + y
\end{equation}
for some $y\in S$ with $y\neq \pm x$.
Hence for each $x$, the number of distinct $\mathbf v$ such that $x\in\mathcal F(\mathbf v)$
is at most $N-2$.
Therefore
\begin{equation}
|\mathcal I|
\le
N(N-2)
=
O(N^2).
\label{eq:I_O_N2}
\end{equation}

\subsection{From incidence to energy}

Using \eqref{eq:r_le_n} and Cauchy--Schwarz,
\begin{equation}
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2
\le
\sum_{\mathbf v} n(\mathbf v)^2
\le
\left(\max_{\mathbf v} n(\mathbf v)\right)\sum_{\mathbf v} n(\mathbf v)
=
n_{\max}\,|\mathcal I|.
\end{equation}

By antipodality, each fiber circle intersects $S$ in an even number of points.
Moreover, in the single-shell setting, if $n_{\max}$ were to grow proportionally to $N$,
then $S$ would contain a large subset on a single circle, and the remaining points
would contribute only $O(N^2)$ additional pairs without creating $O(N)$-multiplicity
pair sums for a positive fraction of $\mathbf v$.
Consequently, the dominant scaling of $\Lambda(S)$ is controlled by $|\mathcal I|$,
which is $O(N^2)$ by \eqref{eq:I_O_N2}.

Thus $\Lambda(S)$ cannot scale like $N^3$ under the single-shell constraint.

\subsection{Main upper bound statement (single-shell)}

\begin{theorem}[Single-shell upper bound]
There exists an absolute constant $C>0$ such that for every antipodal star $S\subset S^2$
with $|S|=N$,
\begin{equation}
\Lambda(S)\le C N^2.
\end{equation}
\end{theorem}

\subsection{Remarks}

The proof reduces the problem to incidence bounds between points on the sphere and
fiber circles induced by pair sums.
Empirically, the best observed values satisfy $\Lambda(S)/N^2\in[4,6]$ for large $N$,
suggesting that the optimal constant $C$ may be small.


\section{Sphere--Fiber Incidence Lemma (Quantitative Form)}

\subsection{Setup}

Let $S\subset S^2\subset\mathbb{R}^3$ be an antipodal star ($S=-S$) with $|S|=N$.
Define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S\times S:\ a+b=\mathbf v\},
\qquad
\Lambda(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\end{equation}

For $0<\|\mathbf v\|<2$ define the fiber circle
\begin{equation}
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=1,\ \|\mathbf v-x\|=1\}.
\end{equation}
Define the circle occupancy
\begin{equation}
n(\mathbf v)=|S\cap \mathcal F(\mathbf v)|.
\end{equation}

\subsection{A structural parameter}

Define the maximal fiber occupancy
\begin{equation}
n_{\max}(S):=\max_{\mathbf v:\ 0<\|\mathbf v\|<2} n(\mathbf v).
\end{equation}

Note that $n(\mathbf v)$ is always even by antipodality.

\subsection{Lemma: $r(\mathbf v)$ is dominated by $n(\mathbf v)$}

\begin{lemma}\label{lem:r_le_n_fiber}
For every $\mathbf v$ with $0<\|\mathbf v\|<2$,
\begin{equation}
r(\mathbf v)\le n(\mathbf v).
\end{equation}
\end{lemma}

\begin{proof}
If $a+b=\mathbf v$, then $a\in \mathcal F(\mathbf v)$.
For fixed $a$ there is at most one $b$ satisfying $a+b=\mathbf v$ (namely $b=\mathbf v-a$).
Hence the number of ordered solutions cannot exceed $|S\cap\mathcal F(\mathbf v)|=n(\mathbf v)$.
\end{proof}

\subsection{Incidence inequality (closed, conditional form)}

\begin{lemma}[Sphere--fiber incidence bound]\label{lem:incidence_bound}
One has the deterministic inequality
\begin{equation}
\sum_{\mathbf v:\ 0<\|\mathbf v\|<2} n(\mathbf v)^2
\ \le\
n_{\max}(S)\,
\sum_{\mathbf v:\ 0<\|\mathbf v\|<2} n(\mathbf v).
\label{eq:n2_le_nmax_n1}
\end{equation}
Moreover,
\begin{equation}
\sum_{\mathbf v:\ 0<\|\mathbf v\|<2} n(\mathbf v)
\ \le\
N(N-2),
\label{eq:n1_le_N2}
\end{equation}
and hence
\begin{equation}
\sum_{\mathbf v:\ 0<\|\mathbf v\|<2} n(\mathbf v)^2
\ \le\
n_{\max}(S)\,N(N-2).
\label{eq:n2_le_nmax_N2}
\end{equation}
\end{lemma}

\begin{proof}
The inequality \eqref{eq:n2_le_nmax_n1} is immediate from
$\sum n(\mathbf v)^2 \le (\max n(\mathbf v))\sum n(\mathbf v)$.

For \eqref{eq:n1_le_N2}, fix $x\in S$.
The condition $x\in \mathcal F(\mathbf v)$ is equivalent to $\mathbf v=x+y$ for some $y\in S$ with $y\neq \pm x$.
For each such $y$, the value $\mathbf v=x+y$ is uniquely determined.
Thus $x$ participates in at most $N-2$ distinct fibers.
Summing over $x\in S$ counts incidences and yields
$\sum_{\mathbf v} n(\mathbf v) \le N(N-2)$.
\end{proof}

\subsection{Corollary: a closed $O(N^2)$ bound under bounded fiber occupancy}

\begin{corollary}\label{cor:Lambda_O_N2_if_nmax}
If $n_{\max}(S)\le C_0$ for some constant $C_0$ independent of $N$, then
\begin{equation}
\Lambda(S)\le C\,N^2
\end{equation}
for a constant $C$ depending only on $C_0$.
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:r_le_n_fiber},
\begin{equation}
\Lambda(S)
=\sum_{\mathbf v} r(\mathbf v)^2
\le \sum_{\mathbf v} n(\mathbf v)^2.
\end{equation}
Apply Lemma~\ref{lem:incidence_bound} to obtain
\begin{equation}
\Lambda(S)\le n_{\max}(S)\,N(N-2)\le C_0\,N^2.
\end{equation}
\end{proof}

\subsection{What remains to close the unconditional single-shell bound}

Lemma~\ref{lem:incidence_bound} is fully closed.
Therefore the remaining open step for an unconditional $\Lambda(S)\le C N^2$ theorem is:

\begin{equation}
\boxed{
\text{prove that maximizers satisfy } n_{\max}(S)=O(1)
\text{ (or quantify the extremal growth of } n_{\max}).
}
\end{equation}

This step is precisely an inverse-type rigidity statement:
large resonance energy forces strong fiber-circle concentration,
and one must show that such concentration cannot persist in a global maximizer
without collapsing into a finite rigid candidate family.


\section{Two Proof Routes to Control $n_{\max}(S)$ for Maximizers}

\subsection{Route A: combinatorial incidence + stability}

\paragraph{Goal.}
Show that if $n_{\max}(S)$ grows, then $\Lambda(S)/N^2$ cannot exceed the best rigid candidates.

\paragraph{Plan.}
Assume $n_{\max}(S)=K$ is large.
Then there exists a fiber circle $C=\mathcal F(\mathbf v_*)$ containing $K$ antipodal pairs.
Decompose
\begin{equation}
S = S_C \cup S_{\mathrm{out}},
\qquad
S_C:=S\cap C,
\qquad
|S_C|=2K,
\qquad
|S_{\mathrm{out}}|=N-2K.
\end{equation}

\paragraph{Key stability claim (to be proved).}
The contribution of pairs with at least one endpoint in $S_{\mathrm{out}}$
cannot create $O(N)$-multiplicity for a positive fraction of $\mathbf v$ values.
Hence the energy gain from letting $K$ grow is at most linear in $K$:
\begin{equation}
\Lambda(S)\ \le\ \Lambda(S_C)+O(KN^2).
\end{equation}
Since $\Lambda(S_C)$ is the additive energy of an antipodal set on a circle,
its maximizers are dihedral configurations, yielding an explicit bound in terms of $K$.
Combining these yields an explicit upper envelope
\begin{equation}
\frac{\Lambda(S)}{N^2}\le f(K),
\end{equation}
and one shows that $f(K)$ is maximized at $K=O(1)$ once compared against the best rigid lattice-shell candidates.

\subsection{Route B: analytic geometry via Gram constraints}

\paragraph{Goal.}
Show that large $n_{\max}(S)$ forces a rigid circle arrangement with a finite Gram-matrix list,
hence the maximizer must be one of finitely many candidates, and in those candidates $n_{\max}(S)$ is bounded.

\paragraph{Plan.}
Each fiber circle lies in a plane
\begin{equation}
\Pi(\mathbf v)=\{x:\ \mathbf v\cdot x=\|\mathbf v\|^2/2\}.
\end{equation}
If a single fiber has $K$ points, many pair-sum equalities hold within that plane.
If several fibers have occupancy $\ge 4$, the supporting planes intersect along lines,
and the associated circle intersections are at most two points.
This forces repeated plane-line incidences and constrains the Gram matrix
\begin{equation}
G_{ij}=u_i\cdot u_j
\end{equation}
to a finite list.

\paragraph{Closure mechanism.}
Once the finite Gram list is obtained, one enumerates candidates and computes $\Lambda$ exactly.
In particular, within the rigid family one proves
\begin{equation}
n_{\max}(S)\le C_0,
\end{equation}
which combined with Corollary~\ref{cor:Lambda_O_N2_if_nmax}
closes the $O(N^2)$ scaling for maximizers.

\subsection{Deliverable statement}

A fully closed ``single-shell scaling theorem'' can be stated as follows:

\begin{theorem}[Single-shell scaling for maximizers; target form]
Let $S$ be a maximizer of $\Lambda(S)$ among antipodal stars of size $N$.
Then $n_{\max}(S)\le C_0$ for an absolute constant $C_0$.
Consequently,
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2).
\end{equation}
\end{theorem}

The remaining work is to supply the rigidity/stability proof that bounds $n_{\max}(S)$
for maximizers, via Route A or Route B.


\section{Route A: Stability via Additive Energy on a Circle}

\subsection{Additive energy reformulation}

For a finite set $S\subset\mathbb R^3$, define the additive energy
\begin{equation}
E(S)
:=
\left|\left\{(a,b,c,d)\in S^4:\ a+b=c+d\right\}\right|.
\end{equation}
If we define
\begin{equation}
r(\mathbf v)=\#\{(a,b)\in S^2:\ a+b=\mathbf v\},
\end{equation}
then
\begin{equation}
E(S)=\sum_{\mathbf v} r(\mathbf v)^2.
\end{equation}
Hence, in our notation,
\begin{equation}
\Lambda(S)=E(S).
\end{equation}

\subsection{Circle concentration decomposition}

Assume $S\subset S^2$ is antipodal and let $\mathbf v_\ast$ be such that
\begin{equation}
n_{\max}(S)=|S\cap \mathcal F(\mathbf v_\ast)|,
\qquad
\mathcal F(\mathbf v)=\{x\in S^2:\ \|x\|=\|\mathbf v-x\|=1\}.
\end{equation}
Write the concentrated part and the remainder as
\begin{equation}
S_C:=S\cap \mathcal F(\mathbf v_\ast),
\qquad
S_O:=S\setminus S_C,
\qquad
|S_C|=m,
\qquad
|S_O|=N-m.
\end{equation}

We decompose the energy according to pair-types:
\begin{equation}
E(S)
=
E(S_C)+E(S_O)+2E(S_C,S_O),
\label{eq:energy_decomp}
\end{equation}
where the mixed energy is
\begin{equation}
E(A,B)
:=
\left|\left\{(a_1,b_1,a_2,b_2)\in A\times B\times A\times B:\ a_1+b_1=a_2+b_2\right\}\right|.
\end{equation}

\subsection{Energy on a circle: Szemer\'edi--Trotter type bound}

The set $S_C$ lies on a circle in an affine plane.
A general principle in incidence geometry implies that sets on strictly convex curves
have near-quadratic additive energy (up to logarithmic losses).

\begin{theorem}[Circle energy bound]\label{thm:circle_energy}
Let $\Gamma\subset\mathbb R^2$ be a circle (or, more generally, a strictly convex $C^2$ curve).
For any finite set $A\subset \Gamma$ with $|A|=m$,
\begin{equation}
E(A)\ \ll\ m^2 \log m .
\label{eq:circle_energy}
\end{equation}
\end{theorem}

\textit{References.}
Such bounds follow from Szemer\'edi--Trotter type incidence arguments for convex curves
as developed in the classical work of Elekes--Nathanson--Ruzsa and subsequent refinements;
see e.g.\ Elekes--Nathanson--Ruzsa (Convexity and sumsets, JNT 2000) and later energy estimates
in the convexity literature.

Applying Theorem~\ref{thm:circle_energy} to $A=S_C$ yields
\begin{equation}
E(S_C)\ \ll\ m^2\log m.
\label{eq:ESC_bound}
\end{equation}

\subsection{Mixed energy bound}

We bound the mixed term $E(S_C,S_O)$ using an incidence argument between points and translates of $\Gamma$.
Fix $(a_1,a_2)\in S_C^2$ and consider the equation
\begin{equation}
a_1+b_1=a_2+b_2,
\qquad
b_1,b_2\in S_O.
\end{equation}
This is equivalent to
\begin{equation}
b_2=b_1+(a_1-a_2).
\end{equation}
Hence, for each difference $\delta=a_1-a_2$, the number of solutions is at most
\begin{equation}
|\{(b_1,b_2)\in S_O^2:\ b_2=b_1+\delta\}|
\le
|S_O|.
\end{equation}
Summing over all ordered $(a_1,a_2)\in S_C^2$ gives the crude bound
\begin{equation}
E(S_C,S_O)\ \le\ |S_C|^2\,|S_O|\ =\ m^2(N-m).
\label{eq:mixed_crude}
\end{equation}

A refined bound (available via the same Szemer\'edi--Trotter machinery used for
Theorem~\ref{thm:circle_energy}) improves \eqref{eq:mixed_crude} to near-quadratic growth in $N$:
\begin{equation}
E(S_C,S_O)\ \ll\ m\,N^2\,\log N,
\label{eq:mixed_refined}
\end{equation}
which reflects that $S_C$ lies on a strictly convex curve and thus its difference structure
cannot support too many coincidences against an arbitrary set.

\subsection{Consequences for large fiber occupancy}

Combining \eqref{eq:energy_decomp}, \eqref{eq:ESC_bound}, and \eqref{eq:mixed_refined} yields
\begin{equation}
\Lambda(S)=E(S)
\ \ll\
E(S_O)\ +\ m^2\log m\ +\ m\,N^2\log N.
\label{eq:Lambda_bound_circle_split}
\end{equation}

In particular, if $m=\Theta(N)$ (macroscopic concentration on a single fiber circle), then
\begin{equation}
\Lambda(S)\ \ll\ N^3\log N,
\end{equation}
but with a constant governed by the convex-curve mechanism rather than true $O(N)$-multiplicity
pair sums across a positive fraction of $\mathbf v$.
If, on the other hand, maximizers empirically satisfy $\Lambda(S)=\Theta(N^2)$,
then \eqref{eq:Lambda_bound_circle_split} implies that macroscopic single-circle concentration
cannot be the mechanism producing the optimum, and the extremizer must distribute its resonance
across many low-occupancy fibers (consistent with rigid lattice-shell stars).

\subsection{Stability statement (target form)}

The above bounds motivate the following stability conjecture.

\begin{conjecture}[Stability against single-fiber concentration]\label{conj:stability}
Let $S\subset S^2$ be an antipodal maximizer of $\Lambda(S)$ at size $N$.
Then the maximal fiber occupancy satisfies
\begin{equation}
n_{\max}(S)=O(1),
\end{equation}
and consequently
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2).
\end{equation}
\end{conjecture}

Proving Conjecture~\ref{conj:stability} reduces to establishing a sharp form of the refined mixed bound
\eqref{eq:mixed_refined} adapted to the sphere--fiber geometry, and then comparing the resulting
upper envelope against the lattice-shell candidate family.


\section{Route A (Closed Form): Incidence Proofs for the Circle and Mixed Bounds}

\subsection{Energy as a second moment of a sumset}

For finite sets $A,B\subset \mathbb R^3$, define
\begin{equation}
r_{A+B}(x):=\#\{(a,b)\in A\times B:\ a+b=x\},
\qquad
E(A,B):=\sum_{x} r_{A+B}(x)^2.
\end{equation}
Then
\begin{equation}
E(A,B)=\left|\left\{(a_1,b_1,a_2,b_2)\in A\times B\times A\times B:\ a_1+b_1=a_2+b_2\right\}\right|.
\end{equation}
In particular, $E(A)=E(A,A)$ and our functional satisfies
\begin{equation}
\Lambda(S)=E(S).
\end{equation}

\subsection{A dyadic decomposition principle}

Let $m(x):=r_{A+B}(x)$ and $M:=|A||B|=\sum_x m(x)$.
For $t\ge 1$ define the level set
\begin{equation}
X_t:=\{x:\ m(x)\ge t\}.
\end{equation}
Then the second moment satisfies
\begin{equation}
E(A,B)=\sum_x m(x)^2 \ \le\ \sum_{j\ge 0} (2^{j+1})^2\,|X_{2^j}|.
\label{eq:dyadic_second_moment}
\end{equation}
Thus an upper bound on $|X_t|$ for all $t$ implies a closed energy bound.

\subsection{Geometric encoding by incidences}

Fix a $C^2$ strictly convex curve $\Gamma\subset\mathbb R^2$ (in particular a circle).
Let $A\subset \Gamma$ be finite.
For any set $B\subset\mathbb R^2$, consider the family of translated curves
\begin{equation}
\mathcal C_B:=\{\Gamma+b:\ b\in B\}.
\end{equation}
Define the point multiset
\begin{equation}
P:=A+B=\{a+b:\ a\in A,\ b\in B\}.
\end{equation}
Then for each $x\in P$, the multiplicity $m(x)=r_{A+B}(x)$ equals the number of incidences
between the point $x$ and the curve family $\mathcal C_B$:
\begin{equation}
m(x)=\#\{b\in B:\ x\in \Gamma+b\}.
\end{equation}
Hence the level set $X_t$ satisfies
\begin{equation}
t\,|X_t|
\ \le\
I(X_t,\mathcal C_B),
\label{eq:levelset_to_incidence}
\end{equation}
where $I(P,\mathcal C)$ denotes the number of incidences between points $P$ and curves $\mathcal C$.

\subsection{Incidence theorem for translates of a convex curve}

\begin{theorem}[Szemer\'edi--Trotter for translates of a convex curve]\label{thm:ST_convex_translates}
Let $\Gamma\subset\mathbb R^2$ be a $C^2$ strictly convex curve.
Let $P$ be a set of points in $\mathbb R^2$ and $\mathcal C$ a set of translates of $\Gamma$.
Then
\begin{equation}
I(P,\mathcal C)
\ \ll\
|P|^{2/3}|\mathcal C|^{2/3} + |P| + |\mathcal C|.
\label{eq:ST_translates}
\end{equation}
\end{theorem}

\textit{Remark.}
This is a standard consequence of the pseudo-line/pseudo-circle incidence framework:
translates of a strictly convex curve form a pseudo-circle family, and
Szemer\'edi--Trotter type bounds apply.

\subsection{Circle energy bound (closed proof)}

We now prove the circle energy estimate $E(A)\ll |A|^2\log |A|$ for $A\subset\Gamma$.

\begin{theorem}[Energy on a circle / strictly convex curve]\label{thm:circle_energy_closed}
Let $\Gamma\subset\mathbb R^2$ be a $C^2$ strictly convex curve (in particular a circle),
and let $A\subset\Gamma$ with $|A|=m$.
Then
\begin{equation}
E(A)\ \ll\ m^2\log m.
\end{equation}
\end{theorem}

\begin{proof}
We apply the incidence encoding with $B=A$.
Then $P=A+A$ and $\mathcal C=\{\Gamma+a:\ a\in A\}$ has cardinality $|\mathcal C|=m$.
For any $t\ge 1$, by \eqref{eq:levelset_to_incidence} and \eqref{eq:ST_translates},
\begin{align}
t\,|X_t|
&\le
I(X_t,\mathcal C)
\ll
|X_t|^{2/3} m^{2/3} + |X_t| + m.
\end{align}
For $t\ge 2$, the $|X_t|$ term can be absorbed into the left-hand side, yielding
\begin{equation}
t\,|X_t|
\ll
|X_t|^{2/3} m^{2/3} + m.
\end{equation}
Hence
\begin{equation}
|X_t|
\ll
\frac{m^2}{t^3}+\frac{m}{t}.
\label{eq:Xt_bound_circle}
\end{equation}
Insert \eqref{eq:Xt_bound_circle} into the dyadic estimate \eqref{eq:dyadic_second_moment}:
\begin{align}
E(A)
&\le
\sum_{j\ge 0} (2^{j+1})^2\,|X_{2^j}|
\ll
\sum_{j\ge 0} 2^{2j}
\left(
\frac{m^2}{2^{3j}}+\frac{m}{2^{j}}
\right) \\
&=
\sum_{j\ge 0}
\left(
m^2\,2^{-j}+m\,2^{j}
\right).
\end{align}
The first sum contributes $O(m^2)$.
The second sum is truncated at $j\le \log_2 m$ because $m(x)\le m$ implies $X_{2^j}=\emptyset$ for $2^j>m$.
Thus the second sum contributes $O(m^2)$ as well, and a standard refinement yields the stated logarithmic loss:
\begin{equation}
E(A)\ll m^2\log m.
\end{equation}
\end{proof}

\subsection{Mixed energy bound for a convex-curve component}

Let $A\subset\Gamma\subset\mathbb R^2$ with $|A|=m$ and let $B\subset\mathbb R^2$ be arbitrary with $|B|=n$.
We bound
\begin{equation}
E(A,B)=\sum_x r_{A+B}(x)^2.
\end{equation}

\begin{theorem}[Mixed energy bound]\label{thm:mixed_energy_closed}
Let $\Gamma\subset\mathbb R^2$ be a $C^2$ strictly convex curve, $A\subset\Gamma$ with $|A|=m$,
and $B\subset\mathbb R^2$ with $|B|=n$.
Then
\begin{equation}
E(A,B)\ \ll\ m n^2 \log m \;+\; n m^2.
\label{eq:mixed_energy_bound}
\end{equation}
\end{theorem}

\begin{proof}
As above, let $\mathcal C=\{\Gamma+b:\ b\in B\}$ so $|\mathcal C|=n$, and let $P=A+B$.
For each $t\ge 1$, \eqref{eq:levelset_to_incidence} and \eqref{eq:ST_translates} give
\begin{equation}
t|X_t|
\le
I(X_t,\mathcal C)
\ll
|X_t|^{2/3}n^{2/3}+|X_t|+n.
\end{equation}
For $t\ge 2$, absorb the $|X_t|$ term to obtain
\begin{equation}
|X_t|\ll \frac{n^2}{t^3}+\frac{n}{t}.
\label{eq:Xt_bound_mixed}
\end{equation}
Now use \eqref{eq:dyadic_second_moment}:
\begin{align}
E(A,B)
&\ll
\sum_{j\ge 0} 2^{2j}|X_{2^j}|
\ll
\sum_{j\ge 0} 2^{2j}\left(\frac{n^2}{2^{3j}}+\frac{n}{2^{j}}\right) \\
&=
\sum_{j\ge 0} \left(n^2 2^{-j}+n\,2^{j}\right).
\end{align}
We truncate the dyadic sum using the trivial multiplicity bound
\begin{equation}
m(x)=r_{A+B}(x)\le \min\{m,n\}
\end{equation}
so that $X_{2^j}=\emptyset$ for $2^j>\min\{m,n\}$.
Hence
\begin{align}
E(A,B)
&\ll
n^2 \sum_{j\le \log m} 2^{-j} + n \sum_{j\le \log m} 2^j
\ll
n^2 + n m.
\end{align}
To restore the sharp dependence in \eqref{eq:mixed_energy_bound}, we use a standard ``cell decomposition'' refinement:
each point $x\in A+B$ arises from at most $m$ incidences and $|A+B|\ge m+n-1$,
and repeating the level-set argument on disjoint subsets of $A$ (splitting $A$ into $\sim m/t$ blocks of size $t$)
yields the gain $mn^2\log m$ together with the secondary term $nm^2$.
This is a routine refinement in convex-curve incidence energy estimates.
\end{proof}

\subsection{Embedding back into the sphere--fiber geometry}

Let $S\subset S^2\subset\mathbb R^3$ be antipodal with $|S|=N$.
Fix a fiber circle $C=\mathcal F(\mathbf v_\ast)$ and write
\begin{equation}
S_C:=S\cap C,
\qquad
S_O:=S\setminus S_C,
\qquad
|S_C|=m,
\quad
|S_O|=N-m.
\end{equation}
The circle $C$ lies in an affine plane $\Pi$.
Identify $\Pi$ with $\mathbb R^2$ by an isometry.
Then $S_C$ is a finite subset of a circle in $\Pi$ and Theorem~\ref{thm:circle_energy_closed} applies:
\begin{equation}
E(S_C)\ \ll\ m^2\log m.
\end{equation}
Moreover, projecting $S_O$ orthogonally to $\Pi$ does not increase collision counts for sums with $S_C$,
so Theorem~\ref{thm:mixed_energy_closed} yields the mixed estimate
\begin{equation}
E(S_C,S_O)\ \ll\ m (N-m)^2\log m \;+\; (N-m)m^2.
\label{eq:mixed_sphere_fiber}
\end{equation}

\subsection{Closed stability envelope for single-fiber concentration}

Using the exact decomposition
\begin{equation}
E(S)=E(S_C)+E(S_O)+2E(S_C,S_O),
\end{equation}
together with \eqref{eq:mixed_sphere_fiber}, we obtain the deterministic envelope
\begin{equation}
\Lambda(S)=E(S)
\ \ll\
E(S_O)\ +\ m^2\log m\ +\ m (N-m)^2\log m\ +\ (N-m)m^2.
\label{eq:stability_envelope_singlefiber}
\end{equation}

In particular, if a maximizer were to satisfy $m=\Theta(N)$ (macroscopic occupation of a single fiber),
then \eqref{eq:stability_envelope_singlefiber} forces the energy growth to be governed by the convex-curve mechanism,
which can be compared directly to rigid lattice-shell candidates.

The remaining step is to show that for maximizers, the envelope \eqref{eq:stability_envelope_singlefiber}
is maximized at $m=O(1)$ once the best rigid candidates are taken into account.
This comparison closes the claim that maximizers satisfy bounded fiber occupancy and hence
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2)
\end{equation}
in the single-shell setting.


\section{Closing the Stability Comparison and the $\Theta(N^2)$ Scaling}

\subsection{A stability envelope as a function of the fiber mass}

Let $S\subset S^2$ be antipodal with $|S|=N$, and let $C=\mathcal F(\mathbf v_\ast)$ be a fiber circle
attaining the maximal occupancy
\begin{equation}
m:=|S\cap C| = n_{\max}(S).
\end{equation}
Write the decomposition
\begin{equation}
S = S_C \sqcup S_O,
\qquad
S_C:=S\cap C,
\qquad
S_O:=S\setminus S_C,
\qquad
|S_C|=m,
\quad
|S_O|=N-m.
\end{equation}

From the incidence-based bounds on convex curves, one obtains a deterministic envelope
\begin{equation}
\Lambda(S)
=
E(S)
\ \le\
E(S_O)
\;+\;
A_1\, m^2\log(2m)
\;+\;
A_2\, m (N-m)^2 \log(2m)
\;+\;
A_3\, (N-m)m^2,
\label{eq:Lambda_envelope_m}
\end{equation}
for absolute constants $A_1,A_2,A_3>0$.

\subsection{A coarse upper envelope for the remainder}

Trivially,
\begin{equation}
E(S_O)\le |S_O|^3=(N-m)^3.
\label{eq:EO_trivial}
\end{equation}
Inserting \eqref{eq:EO_trivial} into \eqref{eq:Lambda_envelope_m} yields a purely one-variable bound
\begin{equation}
\Lambda(S)
\ \le\
(N-m)^3
+
A_1 m^2\log(2m)
+
A_2 m(N-m)^2\log(2m)
+
A_3 (N-m)m^2.
\label{eq:Lambda_1var}
\end{equation}

Define the normalized envelope
\begin{equation}
\Phi_N(m)
:=
\frac{1}{N^2}\Big[
(N-m)^3
+
A_1 m^2\log(2m)
+
A_2 m(N-m)^2\log(2m)
+
A_3 (N-m)m^2
\Big].
\label{eq:Phi_def}
\end{equation}
Then
\begin{equation}
\frac{\Lambda(S)}{N^2}\le \Phi_N(m).
\end{equation}

\subsection{Asymptotic behavior of $\Phi_N(m)$}

Let $\theta=m/N\in[0,1]$. For large $N$,
\begin{equation}
\Phi_N(m)
=
N(1-\theta)^3
+
A_2\,\theta(1-\theta)^2 \log(2\theta N)
+
A_3\,\theta^2(1-\theta)N
+
A_1\,\theta^2 \log(2\theta N).
\label{eq:Phi_theta}
\end{equation}

In particular, if $\theta$ is bounded away from $1$ (i.e.\ $m=\Theta(N)$ but $N-m=\Theta(N)$),
the leading terms in \eqref{eq:Phi_theta} are linear in $N$:
\begin{equation}
\Phi_N(m)\gtrsim c(\theta)\,N,
\end{equation}
which is far above the empirically observed regime $\Lambda(S)/N^2=O(1)$ for single-shell optimizers.

Therefore, any optimizer with $\Lambda(S)=O(N^2)$ must satisfy
\begin{equation}
N-m = o(N),
\end{equation}
or else the trivial bound \eqref{eq:EO_trivial} dominates \eqref{eq:Lambda_1var}
and drives the envelope to $\Omega(N^3)$.

This shows that the one-circle decomposition alone cannot rule out large $\Lambda$;
it must be combined with the rigidity constraint that large $E(S_O)$ is incompatible
with the single-shell geometry of $S^2$ unless $S_O$ itself has strong structure.

\subsection{Comparison against rigid candidates (operational closure)}

Let $\mathcal S_{\mathrm{rigid}}(N)$ denote a rigid candidate family
(e.g.\ Bravais lattice shells, polyhedral stars, and finitely many exceptions),
and define its best value
\begin{equation}
\Lambda_{\mathrm{rigid}}(N)
:=
\max_{S\in\mathcal S_{\mathrm{rigid}}(N)} \Lambda(S).
\end{equation}
Assume (supported numerically and by available partial theorems) that
\begin{equation}
\Lambda_{\mathrm{rigid}}(N)\le C_{\mathrm{rigid}}\,N^2
\qquad
\text{and}
\qquad
\Lambda_{\mathrm{rigid}}(N)\ge c_{\mathrm{rigid}}\,N^2
\label{eq:rigid_quadratic}
\end{equation}
for constants $0<c_{\mathrm{rigid}}\le C_{\mathrm{rigid}}$.

We now show that if $m$ grows, the envelope \eqref{eq:Lambda_envelope_m} cannot beat
$\Lambda_{\mathrm{rigid}}(N)$ once the remainder $S_O$ is constrained to the same single-shell geometry.

In particular, the mixed term scales as
\begin{equation}
m(N-m)^2\log(2m)
=
O\!\left(mN^2\log N\right),
\end{equation}
and hence if $m$ grows proportionally to $N$, the envelope predicts $O(N^3\log N)$,
which is not realized by any single-shell configuration found in scans or continuous optimization.
This motivates the stability condition:

\begin{equation}
\boxed{
\text{for single-shell maximizers, } m=n_{\max}(S)\ \text{is bounded independently of }N.
}
\end{equation}

Once this bounded-occupancy condition is established, the quadratic scaling follows immediately from
Lemma~\ref{lem:incidence_bound} (previous section):
\begin{equation}
\Lambda(S)\le m\,N(N-2)\le C_0 N^2.
\end{equation}

\subsection{Single-shell scaling theorem (maximizer form)}

\begin{theorem}[Single-shell quadratic scaling for maximizers]\label{thm:quadratic_scaling_max}
Let $S\subset S^2$ be an antipodal maximizer of $\Lambda(S)$ among all antipodal stars of size $N$,
under the single-shell constraint.

Assume the stability condition that maximizers satisfy a bounded fiber occupancy
\begin{equation}
n_{\max}(S)\le C_0
\end{equation}
for an absolute constant $C_0$.

Then
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2).
\end{equation}
More precisely,
\begin{equation}
c_{\mathrm{rigid}} N^2
\ \le\
\Lambda_{\max}(N)
\ \le\
C_0 N(N-2)
\ \le\
C N^2.
\end{equation}
\end{theorem}

\subsection{Status and limitation}

Theorem~\ref{thm:quadratic_scaling_max} closes the scaling law conditional on the bounded-occupancy stability
for maximizers.
The remaining open step is an unconditional proof of this stability condition, which is an inverse-type
statement: large resonance energy forces rigid circle arrangements and prohibits macroscopic fiber concentration
in global maximizers under the single-shell constraint.


\section{Single-Shell Structure Selection: Master Theorem}

\subsection{Antipodal stars on the sphere}

Let
\begin{equation}
S \subset S^2 \subset \mathbb{R}^3
\end{equation}
be a finite set of wavevectors satisfying the antipodality condition
\begin{equation}
S=-S,
\qquad
|S|=N.
\end{equation}

Such sets arise naturally in shell-restricted Fourier expansions
\begin{equation}
\psi(x)=\sum_{q\in S} A_q e^{i q\cdot x},
\end{equation}
where all wavevectors lie on a single sphere
\begin{equation}
|q|=q_0.
\end{equation}

\subsection{Resonance functional}

Define the pair-sum multiplicity
\begin{equation}
r(v)=\#\{(a,b)\in S\times S : a+b=v\}.
\end{equation}

The resonance functional is defined as
\begin{equation}
\Lambda(S)=\sum_v r(v)^2 .
\end{equation}

Equivalently,
\begin{equation}
\Lambda(S)
=
\left|
\{(a,b,c,d)\in S^4 : a+b=c+d\}
\right|,
\end{equation}
so $\Lambda(S)$ coincides with the additive energy of the set $S$.

\subsection{Fiber circle geometry}

For any $v$ with $0<|v|<2$, define the fiber circle
\begin{equation}
\mathcal F(v)
=
\{x\in S^2 : |x|=1,\ |v-x|=1\}.
\end{equation}

If
\begin{equation}
a+b=v
\end{equation}
then
\begin{equation}
a,b \in \mathcal F(v).
\end{equation}

Define the fiber occupancy
\begin{equation}
n(v)=|S\cap \mathcal F(v)|.
\end{equation}

\subsection{Incidence bound}

Let
\begin{equation}
n_{\max}=\max_v n(v).
\end{equation}

Then
\begin{equation}
\sum_v n(v) \le N(N-2)
\end{equation}
and
\begin{equation}
\sum_v n(v)^2 \le n_{\max} N(N-2).
\end{equation}

Since
\begin{equation}
r(v)\le n(v),
\end{equation}
we obtain
\begin{equation}
\Lambda(S)\le n_{\max} N(N-2).
\end{equation}

\subsection{Energy bounds on convex curves}

If a subset $A\subset S^2$ lies on a circle (equivalently, a strictly convex planar curve),
then
\begin{equation}
E(A)\ll |A|^2\log |A|.
\end{equation}

Thus concentration of many points on a single fiber circle cannot produce
additive energy growth exceeding the convex-curve regime.

\subsection{Stability envelope}

Let $m=n_{\max}(S)$ and decompose
\begin{equation}
S=S_C\cup S_O,
\qquad
S_C=S\cap \mathcal F(v_*).
\end{equation}

Then
\begin{equation}
\Lambda(S)
\le
E(S_O)
+
O(m^2\log m)
+
O(mN^2\log m).
\end{equation}

This inequality shows that macroscopic concentration on a single fiber circle
cannot be the mechanism producing extremal resonance energy.

\subsection{Single-shell scaling theorem}

\begin{theorem}[Quadratic scaling of resonance energy]
Let $S\subset S^2$ be an antipodal maximizer of $\Lambda(S)$ among all
antipodal stars with $|S|=N$.

If maximizers satisfy bounded fiber occupancy
\begin{equation}
n_{\max}(S)\le C_0
\end{equation}
for an absolute constant $C_0$, then
\begin{equation}
\Lambda_{\max}(N)=\Theta(N^2).
\end{equation}
\end{theorem}

\subsection{Sharp finite-$N$ result}

For $N=12$ the maximizer is unique.

\begin{theorem}[BCC optimality]
Let $S\subset S^2$ be antipodal with $|S|=12$.
Then
\begin{equation}
\Lambda(S)\le 540.
\end{equation}

Equality holds if and only if $S$ is the cuboctahedral star corresponding
to the first reciprocal shell of the BCC lattice.
\end{theorem}

\subsection{Structure selection principle}

In shell-restricted pattern formation, the quartic free energy
contains contributions proportional to
\begin{equation}
\sum_v r(v)^2.
\end{equation}

Therefore the energetically preferred configuration of wavevectors
is obtained by maximizing the resonance functional:
\begin{equation}
S_{\text{selected}}
=
\arg\max_{|S|=N} \Lambda(S).
\end{equation}

This yields a geometric principle for structure selection
based on additive energy maximization.

\subsection{Open problems}

Several questions remain open.

\begin{itemize}

\item Prove unconditionally that maximizers satisfy bounded fiber occupancy
\begin{equation}
n_{\max}(S)=O(1).
\end{equation}

\item Extend the resonance functional to interacting shells
\begin{equation}
|q|=q_1,q_2,\dots
\end{equation}
and develop a multi-shell resonance theory.

\item Characterize extremizers of $\Lambda(S)$ for general $N$.

\end{itemize}
