\section*{Math36-Addendum-B: Multi-branch extension of the Math36-extractor pipeline}

\subsection*{1.\ Motivation}

Math91 reframed Pillar~4 Q2 from single $g(\Lambda)$ flow to branch-resolved $\{g_i(\Lambda)\}_{i \in \mathcal{I}}$ family. The Math36-AddA Theorem `thm:Q2-pipeline-correctness' was scope-restricted to single-branch case. This addendum extends the 5-step pipeline to handle the multi-branch case, providing the branch-discovery and per-branch RG-extraction algorithm that closes Q2 in implementation.

\subsection*{2.\ Algorithmic specification (6-step pipeline)}

\paragraph*{Inputs.}
\begin{itemize}
\item Cold-start ensemble $\{\Psi^{*,(k)}\}_{k=1}^{K}$ from Math82-I-proper (full-12 random-phase, $K=8$ replicas per $\mu^2$).
\item Spectral filter window $[\Lambda_{\rm UV}, \Lambda_{\rm IR}]$ and number of scales $n_\Lambda$.
\item Clustering threshold $\delta_{\rm Hamming} = 0.1$ (relative-difference threshold for two converged $\Psi^*$'s to be classified as same branch).
\end{itemize}

\paragraph*{Step 0 (NEW: pre-processing — converged-ensemble validation).}
For each $k$, verify Newton-convergence: $\|F(\Psi^{*,(k)})\|/\sqrt{N} < 10^{-7}$. Exclude any non-converged replicas (relegate to a NEWTON\_FAILURE bucket, do not enter clustering).

\paragraph*{Step 1 (NEW: branch clustering).}
Compute pairwise Hamming-like distances:
\[
d^{(k,k')} \;=\; \frac{\|\Psi^{*,(k)} - \Psi^{*,(k')}\|_{2}}{\frac{1}{2}\big(\|\Psi^{*,(k)}\|_{2} + \|\Psi^{*,(k')}\|_{2}\big)}.
\]
Apply hierarchical agglomerative clustering with single-linkage at threshold $\delta_{\rm Hamming}$. Output: branch index set $\mathcal{I} \subseteq \{1,\ldots,K\}$ with $|\mathcal{I}|$ branches and a representative $\Psi^{*,(i)}$ per branch.

\paragraph*{Step 2 (Math88 §6 Step 1, per-branch).} Spectral filter $\hat\Psi^{(i)}_\Lambda(k) = \hat\Psi^{*,(i)}(k) \cdot \chi_{[0,\Lambda]}(|k|)$.

\paragraph*{Step 3 (Math88 §6 Step 2, per-branch).} Raw moments $W_0^{(i)}, W_2^{(i)}, M_2^{(i)}, G_4^{(i)}$ at scale $\Lambda$.

\paragraph*{Step 4 (Math88 §6 Step 3, per-branch).} Directional decomposition: $g_c^{(i)}, g_E^{(i)}, g_O^{(i)}$ from $G_4^{(i)}$ along (cube, edge, octahedron) BCC orbits.

\paragraph*{Step 5 (Math88 §6 Step 4, per-branch).} Flow reconstruction: $\Lambda \mapsto (g_c^{(i)}, g_E^{(i)}, g_O^{(i)})$ for each $i \in \mathcal{I}$.

\paragraph*{Step 6 (Math88 §6 Step 5, per-branch).} Fixed-point detection per branch:
\[
\bigg|\frac{d g_\alpha^{(i)}}{d\log\Lambda}\bigg| < \epsilon_{\rm FP} \quad\text{for }\Lambda \in [\Lambda_{\rm IR}, 2\Lambda_{\rm IR}],\ \forall \alpha \in \{c, E, O\}.
\]
Output per branch: \texttt{FIXED\_POINT\_FOUND} + values $(g_c^{*,(i)}, g_E^{*,(i)}, g_O^{*,(i)})$, or \texttt{NO\_FIXED\_POINT}.

\subsection*{3.\ Theorem `thm:Q2-pipeline-branch-resolved-correctness`}

\begin{theorem}[Multi-branch Q2 reducibility]
\label{thm:Q2-pipeline-branch-resolved-correctness}
Under the hypotheses of Math36-AddA Theorem `thm:Q2-pipeline-correctness' (PDE complete + Math36 dictionary complete + extractor formulas complete) plus the Math91 branch-resolved reframing, the 6-step pipeline of §2 above is sufficient for Pillar~4 Q2 closure if and only if:
\begin{enumerate}
\item[(i)] At least one branch $i^{*} \in \mathcal{I}$ yields \texttt{FIXED\_POINT\_FOUND} with stable $(g_c^{*,(i^{*})}, g_E^{*,(i^{*})}, g_O^{*,(i^{*})})$.
\item[(ii)] The fixed-point ratios match the SM coupling pattern within Math36-AddA F2 tolerance (factor of 2):
\[
2.5 \le g_c^{*,(i^{*})}/g_E^{*,(i^{*})} \le 4.5,\qquad 2.0 \le g_E^{*,(i^{*})}/g_O^{*,(i^{*})} \le 6.0.
\]
\item[(iii)] The branch $i^{*}$ has condensate amplitude $\|\Psi^{*,(i^{*})}\|/\sqrt{N} > 10^{-3}$ (non-vacuum branch).
\end{enumerate}
\end{theorem}

\paragraph*{Proof sketch.} Math36-AddA Theorem `thm:Q2-pipeline-correctness' establishes that ANY single $\Psi^{*}$ satisfying (iii) admits the Math88 §6 5-step extraction yielding a single $g(\Lambda)$ flow. Per Math91 §3, applying this 5-step extraction to a representative $\Psi^{*,(i)}$ of each branch $i$ yields the per-branch flow $g^{(i)}(\Lambda)$. Step 1 of §2 (clustering) is well-defined under the Hamming-like metric (positive-definite + symmetric + triangle inequality satisfied for $\|\cdot\|_2$ norm). Single-linkage hierarchical clustering at threshold $\delta_{\rm Hamming}$ produces a deterministic $\mathcal{I}$ given the ensemble. Conditions (i)--(iii) then identify the physically meaningful branch (gauge-emergent, non-vacuum, SM-pattern-matching). Q2 reducibility is achieved because the algorithm is finite, deterministic, and produces an output that either confirms gauge emergence (i*+ii+iii satisfied) or refutes it (no $i^{*}$ exists).

\subsection*{4.\ Pre-registered execution gates (CLAUDE.md §6.3.3)}

\paragraph*{Multi-branch F-gates (extending Math36-AddA F1-F3).}
\begin{itemize}
\item \textbf{F4 (branch-existence)}: $|\mathcal{I}| \ge 1$ for at least one $\mu^2$ value in the run (otherwise Math82-I cold-start has failed structurally).
\item \textbf{F5 (branch-coverage)}: Hoeffding bound on $K=8$ replicas guarantees $\Pr(\text{all branches with } p_i \ge 0.25 \text{ are sampled}) \ge 90\%$. If $|\mathcal{I}| = 1$ across all $\mu^2$, the system is genuinely single-branch (Math91 reframing reduces to Math88 §6 single-flow).
\item \textbf{F6 (gauge-pattern match)}: at least one $i^{*}$ satisfying Theorem~\ref{thm:Q2-pipeline-branch-resolved-correctness} (i)--(iii). Failure of F6 means TECT does NOT realise SM gauge emergence in the BCC condensate sector.
\end{itemize}

\subsection*{5.\ Devil's-advocate self-test (CLAUDE.md §6.3 mandatory)}

\paragraph*{$\alpha$.} \emph{The clustering threshold $\delta_{\rm Hamming} = 0.1$ is arbitrary; different choices give different $|\mathcal{I}|$.}\\
\textbf{Verdict: VALID with empirical bound.} Cluster stability over $\delta_{\rm Hamming} \in [0.05, 0.2]$ is the sensitivity test; if $|\mathcal{I}|$ varies by more than $\pm 1$ over this range, branch identification is unstable. Mitigation: report $|\mathcal{I}|$ as a function of $\delta_{\rm Hamming}$; require monotone-decreasing, plateau-stable behaviour.

\paragraph*{$\beta$.} \emph{Hamming-like distance in mode space may be insensitive to phase rotations; two physically-equivalent BCC variants (related by a $G_{\rm SM}$ gauge transformation) may be miscounted as distinct branches.}\\
\textbf{Verdict: VALID with mitigation.} Compute distance modulo $G_{\rm SM}$-action: $d^{(k,k')} := \min_{g \in G_{\rm SM}} \|\Psi^{*,(k)} - g\cdot\Psi^{*,(k')}\|_2 / \cdots$. The minimum-over-group operation requires $|G_{\rm SM}| \cdot K^2 / 2$ distance evaluations per $\mu^2$; for $|G_{\rm SM}|^{\rm discrete}$ practical (modulo continuous gauge degrees), feasible in $O(K^2)$. Implementation flagged for Math36-AddC if needed.

\paragraph*{$\gamma$.} \emph{Theorem~\ref{thm:Q2-pipeline-branch-resolved-correctness} establishes reducibility but not numerical execution; the actual $\mathcal{I}$, $i^{*}$, and gauge-pattern match are unknown until Math82-I-proper runs.}\\
\textbf{Verdict: VALID (this is an honest scope statement).} The theorem is a logical-reducibility statement: IF the algorithm is run and IF conditions (i)--(iii) are satisfied THEN Q2 is closed. The numerical execution is independent and is the Math92 §8 trigger \#2 + \#3.
