\begin{abstract}
The Stage $\alpha$ closure (Math64 v1.1) and the post-closure routing
verdict (Math63 v1.7) reduced the anti-Hermitian structural mass of
the BCC residual Jacobian to a single operator block $F_{\mathrm{cII}}$,
and mandated a pre-surgery decisive diagnostic
$\Delta_{\mathrm{cII}} := \lVert F_{\mathrm{cII}}^{\mathrm{impl}} -
F_{\mathrm{cII}}^{\mathrm{grad}}\rVert_{F}$ to classify the defect
as Case 1 (assembly inconsistency, local rewrite),
Case 2 (EL-incompatible design, in-solver symmetrisation), or
Case 3 (inconclusive).  Execution of that test requires a scalar
functional $E_{\mathrm{cII}}[\Psi]$ whose variational derivative
defines $F_{\mathrm{cII}}^{\mathrm{grad}}$; however, the current
executable backend implements $F_{\mathrm{cII}}^{\mathrm{impl}}$
directly as a projected-divergence residual, with no pre-existing
$E_{\mathrm{cII}}$ in the module.  The present skeleton provides
(i) the structural audit of $F_{\mathrm{cII}}^{\mathrm{impl}}$
showing which factors plausibly arise from an Euler--Lagrange
derivation and which constitute a phenomenological projector
closure; (ii) three increasingly rigorous candidate functionals
$E_{\mathrm{cII}}^{(A)}, E_{\mathrm{cII}}^{(B)}, E_{\mathrm{cII}}^{(C)}$;
(iii) a Helmholtz--Hodge-type obstruction lemma identifying the
anti-Hermitian mass $a_{\mathrm{cII}} = 4.357\times 10^{-7}$ with a
curl-component of the projected current; and (iv) the formal hook
$E_{\mathrm{cII}}^{\mathrm{cand}}$ consumed by
\texttt{Tools/check\_jacobian\_blocks.py v1.3}
(\texttt{--cII-grad-check}).  All theorem/proposition content is
marked SKELETON and is deferred to v0.2 pending the decisive-test
outcome $\Delta_{\mathrm{cII}}, r^{\mathrm{impl}}, r^{\mathrm{grad}}$.

\section{Frame, scope, and critical finding}
\label{sec:frame}

Throughout this note we work on the real complex Hilbert space
$\mathcal{H} = \mathbb{C}^{3 \otimes N^{3}} \simeq \mathbb{R}^{2 \cdot 3 N^{3}}$
carrying the inner product $\langle u, v\rangle_{\mathbb{R}} =
\operatorname{Re}\sum_{\mathbf{x},a} u_{a}(\mathbf{x})^{\ast} v_{a}(\mathbf{x})$,
in agreement with Math63 §2A.1 and Math64 §2.  The three Gell-Mann
generators $\{T_{1}, T_{2}, T_{3}\}$ embedded in $\mathrm{SU}(3)$
act on the internal colour index $a \in \{1,2,3\}$, and
$\nabla$ denotes the BCC spectral derivative supplied by
\texttt{\_spectral\_derivative\_scalar\_t} (Math57 Addendum A).

\begin{definition}[$F_{\mathrm{cII}}^{\mathrm{impl}}$, the executable
implementation; Math65-Def-impl]
\label{def:math65-cii-impl}
Let $\rho(\Psi) = \sum_{a}|\Psi_{a}|^{2} + 10^{-12}$.  Define, for
each generator $T \in \{T_{1}, T_{2}, T_{3}\}$, the scalar
modulation
\begin{equation}
m_{T}(\Psi) \;=\; \Psi^{\dagger} T \Psi,
\qquad
q_{T}(\Psi) \;=\; m_{T}(\Psi) / \rho(\Psi),
\label{eq:math65-mq}
\end{equation}
and the projected-channel vector field
$\mathcal{J}_{T}^{i}(\Psi) = \partial_{i} m_{T}(\Psi)$,
$\mathcal{K}_{T}^{i}(\Psi) = \mathcal{J}_{T}^{i}(\Psi)
- q_{T}(\Psi)\,\partial_{i}\rho(\Psi)$.
The executable Class II residual is
\begin{equation}
F_{\mathrm{cII}}^{\mathrm{impl}}(\Psi)
\;=\;
\sum_{T}
\Bigl[
\mathrm{pref}_{JJ}\,
\bigl(\nabla\!\cdot\!\mathcal{J}_{T}\bigr)
\;+\;
\mathrm{pref}_{JK}\,
\bigl(\nabla\!\cdot\!\mathcal{K}_{T}\bigr)
\Bigr]
\bigl(T\Psi - q_{T}(\Psi)\,\Psi\bigr),
\label{eq:math65-impl}
\end{equation}
with prefactors
$\mathrm{pref}_{JJ} = c_{JJ} \alpha_{X}^{2}/M_{X}^{2}$,
$\mathrm{pref}_{JK} = c_{JK} \alpha_{X}\beta_{X}/M_{X}^{2}$.
This is an \emph{exact transcription} of lines 423--461 of
\texttt{PDE/real\_backend\_pt\_bcc\_mixed\_v3.py} (v3 tagged
at \texttt{Math63-Solver-Redesign-v2p5-2026-04-22}).
\end{definition}

\begin{remark}[Critical finding: absence of a scalar potential in the
backend; Math65-Rem-critical]
\label{rmk:math65-critical-finding}
A static audit of \texttt{PDE/real\_backend\_pt\_bcc\_mixed\_v3.py}
(performed 2026-04-22) shows that the module \emph{exposes no
scalar functional} \texttt{E\_cII}, \texttt{action\_cII},
\texttt{energy\_cII}, \texttt{lagrangian\_cII}, or any analogue
thereof.  The module docstring of
\texttt{\_classII\_effective\_term\_t} explicitly describes the
block as an ``executable proxy for the integrated-out Class II
sector'', constructed at the level of the residual, not from a
variational principle.  Consequently, the Math64 §6 decisive test
\begin{equation}
F_{\mathrm{cII}}^{\mathrm{grad}}(\Psi)
\;:=\;
\Bigl[\nabla_{\Psi}\, E_{\mathrm{cII}}[\Psi]\Bigr]_{\text{Wirtinger}}
\label{eq:math65-grad-def}
\end{equation}
cannot be evaluated until a candidate $E_{\mathrm{cII}}$ is
supplied.  Supplying such a candidate is the principal object of
this note.
\end{remark}

\begin{remark}[Scope restriction]
\label{rmk:math65-scope}
This note concerns \emph{only} the construction of
$E_{\mathrm{cII}}^{\mathrm{cand}}[\Psi]$ and the obstruction
analysis that fixes the interpretation of a nonzero
$\Delta_{\mathrm{cII}}$.  The physical derivation of the Class II
sector (integration-out of the heavy $X$-doublet, Math50 Pillar 4
scaffold) is \emph{not} revisited here; we take the executable
prefactors $\{\mathrm{pref}_{JJ}, \mathrm{pref}_{JK}\}$ and the
projector structure $T\Psi - q_{T}(\Psi)\,\Psi$ as given, and
search within that fixed structural template for the scalar
functional, if any, whose gradient reproduces
\eqref{eq:math65-impl}.
\end{remark}

\section{Three candidate functionals}
\label{sec:candidates}

We introduce three progressively richer candidates and, for each,
record the status (proven Euler--Lagrange parent, plausible parent
modulo boundary terms, or demonstrably non-variational).

\subsection{Candidate A: bilinear channel-gradient functional}
\label{sec:candA}

\begin{definition}[$E_{\mathrm{cII}}^{(A)}$]
\label{def:math65-EcII-A}
\begin{equation}
E_{\mathrm{cII}}^{(A)}[\Psi]
\;=\;
\tfrac{1}{2}
\sum_{T}\int_{V}\!d^{3}x\;
\Bigl[
\mathrm{pref}_{JJ}\,|\nabla m_{T}(\Psi)|^{2}
+
\mathrm{pref}_{JK}\,|\nabla m_{T}(\Psi)|^{2}
\Bigr].
\label{eq:math65-EA}
\end{equation}
\end{definition}

\begin{proposition}[Candidate A status; Math65-Prop-A]
\label{prop:math65-A-status}
\textsc{SKELETON.}
The Wirtinger derivative
$\nabla_{\Psi^{\ast}} E_{\mathrm{cII}}^{(A)}$ reproduces the
\emph{first} factor $\nabla\!\cdot\!\mathcal{J}_{T}$ of
\eqref{eq:math65-impl} but fails to reproduce (i) the
$q_{T}$-projected term $\nabla\!\cdot\!\mathcal{K}_{T}$, and
(ii) the projector $(T\Psi - q_{T}\Psi)$ in the outer factor:
the EL derivative of \eqref{eq:math65-EA} is proportional to
$T\Psi$ alone.  Hence $E_{\mathrm{cII}}^{(A)}$ is an
\emph{incomplete} parent; it captures the pure-$JJ$ sector only.
\end{proposition}

\subsection{Candidate B: density-normalised channel functional}
\label{sec:candB}

\begin{definition}[$E_{\mathrm{cII}}^{(B)}$]
\label{def:math65-EcII-B}
\begin{equation}
E_{\mathrm{cII}}^{(B)}[\Psi]
\;=\;
\tfrac{1}{2}
\sum_{T}\int_{V}\!d^{3}x\;
\mathrm{pref}_{JJ}\,\rho(\Psi)\,
\bigl|\nabla q_{T}(\Psi)\bigr|^{2}
\;+\;
\tfrac{1}{2}
\sum_{T}\int_{V}\!d^{3}x\;
\mathrm{pref}_{JK}\,
\nabla\!\cdot\!\mathcal{K}_{T}\,\cdot\,q_{T}.
\label{eq:math65-EB}
\end{equation}
\end{definition}

\begin{proposition}[Candidate B status; Math65-Prop-B]
\label{prop:math65-B-status}
\textsc{SKELETON.}
Using the identity $\nabla q_{T} = \mathcal{K}_{T}/\rho$ and
$\rho |\nabla q_{T}|^{2} = \mathcal{K}_{T}\!\cdot\!\nabla q_{T}$,
the Wirtinger derivative of the first term of
\eqref{eq:math65-EB} generates
$\nabla\!\cdot\!\mathcal{J}_{T} \cdot (T\Psi - q_{T}\Psi)$
\emph{after} integration by parts, which is the $\mathrm{pref}_{JJ}$
half of \eqref{eq:math65-impl}.  However, the mixed $JK$-term in
\eqref{eq:math65-EB} is not symmetric under
$T \leftrightarrow T^{\dagger}$ and introduces an anti-Hermitian
artifact of order $\mathrm{pref}_{JK}^{2}\|\nabla q\|^{2}$; this
is the candidate Case-2 pathology and its numerical magnitude is
to be compared with $a_{\mathrm{cII}} = 4.357\times 10^{-7}$
in the v0.2 closure.
\end{proposition}

\subsection{Candidate C: fully EL-consistent projected current}
\label{sec:candC}

\begin{definition}[$E_{\mathrm{cII}}^{(C)}$]
\label{def:math65-EcII-C}
\begin{equation}
E_{\mathrm{cII}}^{(C)}[\Psi]
\;=\;
\tfrac{1}{2}
\sum_{T}\int_{V}\!d^{3}x\;
\rho(\Psi)\,
\Bigl[
\mathrm{pref}_{JJ}\,|\nabla q_{T}(\Psi)|^{2}
+
\mathrm{pref}_{JK}\,\nabla q_{T}(\Psi)\!\cdot\!\nabla\log\rho(\Psi)
\Bigr].
\label{eq:math65-EC}
\end{equation}
\end{definition}

\begin{proposition}[Candidate C status; Math65-Prop-C]
\label{prop:math65-C-status}
\textsc{SKELETON.}
$E_{\mathrm{cII}}^{(C)}$ is manifestly real, the integrand is
$\mathrm{Herm}(T)$-symmetric, and its Wirtinger derivative,
computed via the chain rule on $q_{T} = \Psi^{\dagger}T\Psi /
\Psi^{\dagger}\Psi$, produces a residual of the same functional
form as \eqref{eq:math65-impl}, up to a boundary term that
vanishes on the periodic BCC torus.  The two hypotheses
\emph{Case 1} (assembly consistency --- $E_{\mathrm{cII}}^{(C)}$
is the exact parent, $\Delta_{\mathrm{cII}} = O(\varepsilon_{\mathrm{fd}})$)
and \emph{Case 2} (design mismatch ---
$\Delta_{\mathrm{cII}} \sim a_{\mathrm{cII}}$) both remain open;
their resolution is the content of Math64 §6.
\end{proposition}

\section{Helmholtz--Hodge obstruction analysis}
\label{sec:hodge}

The key structural question is: \emph{is the projected-divergence
vector field
$\mathbf{V}_{T}(\Psi) := \mathrm{pref}_{JJ}\,\mathcal{J}_{T}
+ \mathrm{pref}_{JK}\,\mathcal{K}_{T}$
the gradient of a scalar potential on $\mathcal{H}$?}

By the Helmholtz--Hodge decomposition on the 3-torus $(\mathbb{T}^{3})$,
\begin{equation}
\mathbf{V}_{T} \;=\; \nabla\phi_{T} \;+\; \nabla\!\times\!\mathbf{A}_{T}
\;+\; \mathbf{h}_{T},
\qquad
\nabla\!\cdot\!\mathbf{h}_{T} = 0,\ \nabla\!\times\!\mathbf{h}_{T} = 0,
\label{eq:math65-hh}
\end{equation}
and a variational parent $E_{\mathrm{cII}}$ whose gradient equals
$\nabla\!\cdot\!\mathbf{V}_{T}\,\cdot\,(T\Psi - q_{T}\Psi)$ exists
\emph{iff} the curl component $\nabla\!\times\!\mathbf{A}_{T}$
vanishes in the $L^{2}$ sense.

\begin{proposition}[Hodge obstruction $\Leftrightarrow$
$a_{\mathrm{cII}}$; Math65-Prop-hodge]
\label{prop:math65-hodge}
\textsc{SKELETON.}
Let
$\mathcal{O}_{T}(\Psi) := \lVert \nabla\!\times\!
\mathbf{V}_{T}(\Psi) \rVert_{L^{2}(\mathbb{T}^{3})}$.
Then
\begin{equation}
a_{\mathrm{cII}}(\Psi)
\;=\;
\lVert B_{\mathrm{cII}}(\Psi) - B_{\mathrm{cII}}(\Psi)^{\top}
\rVert_{F}
\;\ge\;
c_{0}\,
\sum_{T}\mathcal{O}_{T}(\Psi)
\label{eq:math65-obstruction}
\end{equation}
for an explicit geometric constant $c_{0} > 0$ depending only on
the BCC spectral kernel of Math57 Add.~A.  Equality holds iff the
prefactor matrix $(\mathrm{pref}_{JJ}, \mathrm{pref}_{JK})$ is
rank-deficient.
\end{proposition}

\begin{corollary}[Structural reading; Math65-Cor-reading]
\label{cor:math65-reading}
\textsc{SKELETON.}
Under Proposition~\ref{prop:math65-hodge},
the thermal-seed observation
$a_{\mathrm{cII}} = 4.357\times 10^{-7}$ (Math64 v1.1 Theorem
\textsc{math64-full-sympd}) quantifies the $L^{2}$-norm of the
Hodge curl component of $\mathbf{V}_{T}$ at that state; this is
the \emph{operationally inert but structurally present} curl
charge identified in the Math63 v1.7 Stage-$\alpha$ closure
(Corollary \textsc{math63-2A3-solver-routing}).
\end{corollary}

\section{Decisive-test hook for Tools/check\_jacobian\_blocks.py v1.3}

\begin{definition}[Canonical candidate and API contract; Math65-Def-API]
\label{def:math65-cand-API}
The canonical candidate fed to the v1.3 decisive probe is
$E_{\mathrm{cII}}^{\mathrm{cand}} := E_{\mathrm{cII}}^{(C)}$.
The tool exposes a pluggable override
\texttt{--cII-energy-module MODULE:FUNC} which imports
\texttt{FUNC(Psi\_t, params) -> torch.Tensor (real scalar)};
default resolution points to
\texttt{Tools.cII\_energy\_candidates.E\_cII\_C}.
Given a candidate $E$, the tool computes
\begin{equation}
F_{\mathrm{cII}}^{\mathrm{grad}}(\Psi)
\;=\;
2\,\frac{\partial E[\Psi;\mathrm{params}]}
       {\partial \Psi^{\ast}}
\;=\;
\texttt{torch.autograd.grad}(E, \Psi^{\ast})[0]
\label{eq:math65-autograd}
\end{equation}
(the factor $2$ enforcing the Wirtinger normalisation of
Math63 §2A.2 Lemma 2), and emits the observables
$(\Delta_{\mathrm{cII}}, r^{\mathrm{impl}}, r^{\mathrm{grad}})$
of Math64 §6 Eq.~\textsc{math64-decisive-observables}.
\end{definition}

\begin{remark}[Fallback ordering and acceptance]
\label{rmk:math65-fallback}
The tool runs all three candidates $\{A, B, C\}$ in sequence, and
the test is considered \emph{conclusive} only if at least one
candidate achieves
$\max_{\Psi\in\{\mathrm{seed},\mathrm{thermal}\}}
\Delta_{\mathrm{cII}}(\Psi)
\le 10\,\varepsilon_{\mathrm{fd}}$,
classifying Case 1 (assembly consistency, parent identified).
If all three candidates yield
$\Delta_{\mathrm{cII}} \sim a_{\mathrm{cII}}$, the decision flips
to Case 2 (design defect, no variational parent in the bilinear
+ Hodge-compatible hypothesis class) and escalates to Math66
(in-solver symmetrisation vs.\ sector rewrite).
\end{remark}

\section{Candidate $C$ post-run analysis (R-2026-04-22-004)}
\label{sec:math65-run-C}

The decisive test was executed via \texttt{Tools/check\_jacobian\_blocks.py}
v1.3.1 with
\[

on the thermal seed of

$(\Delta_{\mathrm{cII}}, r^{\mathrm{impl}}, r^{\mathrm{grad}})$ of
Eq.~\texttt{math64-decisive-observables} returned
\begin{equation}
\boxed{\;
\lVert F_{\mathrm{cII}}^{\mathrm{impl}}\rVert_{F} = 1.265\times 10^{-4},\qquad
\lVert F_{\mathrm{cII}}^{\mathrm{grad}(C)}\rVert_{F} = 1.057\times 10^{-2},\qquad
\Delta_{\mathrm{cII}}^{(C)} = 1.057\times 10^{-2},
\;}
\label{eq:math65-run-C-norms}
\end{equation}
\begin{equation}
r_{\mathrm{impl}}^{(C)} = \Delta_{\mathrm{cII}}^{(C)} / \lVert F^{\mathrm{impl}}\rVert = 83.59,\qquad
r_{\mathrm{grad}}^{(C)} = \Delta_{\mathrm{cII}}^{(C)} / \lVert F^{\mathrm{grad}(C)}\rVert = 1.00009.
\label{eq:math65-run-C-ratios}
\end{equation}

\begin{remark}[Near-orthogonality of the gradient-vs-implementation pair]
\label{rem:math65-C-orthogonality}
The value $r_{\mathrm{grad}}^{(C)} = 1.00009$ is not a small perturbation
around a variational limit but its structural opposite. The polarisation
identity
\[
\Delta_{\mathrm{cII}}^{2} = \lVert F^{\mathrm{impl}}\rVert^{2}
                         + \lVert F^{\mathrm{grad}}\rVert^{2}
                         - 2\,\mathrm{Re}\langle F^{\mathrm{impl}},F^{\mathrm{grad}}\rangle
\]
together with Eqs.~\eqref{eq:math65-run-C-norms},~\eqref{eq:math65-run-C-ratios}
pins
\begin{equation}
\mathrm{Re}\bigl\langle F_{\mathrm{cII}}^{\mathrm{impl}},\,F_{\mathrm{cII}}^{\mathrm{grad}(C)}\bigr\rangle
\;\approx\; -2\times 10^{-9},\qquad
\cos\angle(F^{\mathrm{impl}},F^{\mathrm{grad}(C)})
  \;=\; \frac{\mathrm{Re}\langle F^{\mathrm{impl}},F^{\mathrm{grad}(C)}\rangle}
             {\lVert F^{\mathrm{impl}}\rVert\,\lVert F^{\mathrm{grad}(C)}\rVert}
  \;\approx\; -1.5\times 10^{-3}.
\label{eq:math65-C-cosine}
\end{equation}
Candidate $C$'s gradient is therefore essentially $L^{2}$-orthogonal to
$F_{\mathrm{cII}}^{\mathrm{impl}}$ at the thermal seed. This rules out any
interpretation of $E_{\mathrm{cII}}^{(C)}$ as "the variational parent modulo
a small anti-Hermitian defect" and falsifies the a priori conjecture of
Prop.~\texttt{math65-C-status} that $(C)$ would sit in the Case 1 / Case 2
boundary.
\end{remark}

\begin{remark}[Magnitude stratification against the design-defect band]
\label{rem:math65-C-magnitude}
Proposition \texttt{math64-cii-surgery-gate} was calibrated to
$\Delta_{\mathrm{cII}} \sim a_{\mathrm{cII}} = 4.357\times 10^{-7}$
(Math64 Thm.~\texttt{math64-full-sympd}). The observed
$\Delta_{\mathrm{cII}}^{(C)}$ exceeds this reference by
\[
\Delta_{\mathrm{cII}}^{(C)}/a_{\mathrm{cII}}
 \;\approx\; 2.4\times 10^{4}.
\]
The "Case 2 — design defect" label emitted by the classifier is mechanically
correct under the current threshold logic but semantically understates the
severity: the signal lies four decades above the envisaged design-defect band
and three decades above $\lVert F^{\mathrm{impl}}\rVert$ itself. This regime
is designated \textbf{Case 2$^{\ast}$ (structural class mismatch)}, and must
not be treated as interchangeable with the "small-ish anti-Hermitian defect"
scenario originally imagined by Prop.~\texttt{math64-cii-surgery-gate}(2).
\end{remark}

\section{Triangulation mandate (T-A, T-B) and measure audit (M1--M3)}
\label{sec:math65-triangulation}

Remark~\ref{rem:math65-C-orthogonality} rules out $(C)$ as the variational
parent; it does \emph{not}, on its own, rule out $(A)$ or $(B)$, nor does it
establish that the scalar-$E_{\mathrm{cII}}$ class is globally insufficient.
The Math66 operator-surgery escalation (in-solver symmetrisation, Math64
§5(X)) is \emph{held} pending the following two-step closure.

\begin{definition}[Candidate triangulation test, (T-A) and (T-B)]
\label{def:math65-triangulation}
The decisive test of Def.~\texttt{math64-decisive-test} is executed on each
of
\[
E_{\mathrm{cII}}^{(A)}\quad\text{and}\quad E_{\mathrm{cII}}^{(B)}
\]
with the same $(N,\mu^{2},\text{seed},\varepsilon_{\mathrm{fd}},
a_{\mathrm{cII}}^{\mathrm{ref}})$ as Eq.~\eqref{eq:math65-run-C-norms}.
Denote the outputs
$(\Delta^{(X)}, r^{\mathrm{impl},(X)}, r^{\mathrm{grad},(X)})$ for
$X\in\{A,B\}$, and the implementation--gradient overlap cosines
$\cos\theta^{(X)} = \mathrm{Re}\langle F^{\mathrm{impl}},F^{\mathrm{grad}(X)}\rangle
 /(\lVert F^{\mathrm{impl}}\rVert\lVert F^{\mathrm{grad}(X)}\rVert)$.
\end{definition}

\begin{proposition}[Triangulation verdict logic]
\label{prop:math65-triangulation-verdict}
The Case classification of the $\{E_{\mathrm{cII}}^{(A,B,C)}\}$ class is
sealed by the following trichotomy on $(T\text{-}A, T\text{-}B)$:
\begin{enumerate}
\item[(R)] \textbf{Rescue of the scalar-$E$ class.}  If
  $\min_{X\in\{A,B\}} r^{\mathrm{impl},(X)} \le 10$ and $\Delta^{(X)} \le

  variational parent and candidate $C$ is retracted; Math65 v0.2 proceeds on

  (A) and (B) fail in the same regime as $(C)$), the scalar-$E$ class
  constructed in §2 does not contain $F_{\mathrm{cII}}^{\mathrm{impl}}$.
  Math66 operator surgery (in-solver symmetrisation, Math64 §5(X)) is sealed

\item[(S)] \textbf{Split with measure signature.}  If exactly one of
  $(A),(B)$ yields $\cos\theta \approx \pm 1$ (alignment or anti-alignment)

  measure audit (Remark~\ref{rem:math65-measure-audit}) is the governing

\end{enumerate}
\end{proposition}

\subsection{Branch selection of Prop.~\ref{prop:math65-triangulation-verdict}}

\begin{proposition}[Class insufficiency of the scalar-$E_{\mathrm{cII}}$ family]
\label{prop:math65-class-insufficient}
Let $\mathcal{E} := \{E^{(A)}[\Psi], E^{(B)}[\Psi], E^{(C)}[\Psi]\}$
denote the three candidates constructed in §\ref{sec:math65-candidates}.
Define the variational-gradient image
$\mathcal{G}_{\mathcal{E}}[\Psi] := \{(\partial E/\partial \Psi^{\ast})[\Psi]
: E \in \mathcal{E}\}$ evaluated at the stored seed $\Psi_{0}$.
Then
\[
\min_{F \in \mathcal{G}_{\mathcal{E}}[\Psi_{0}]}
\bigl|\cos\angle(F,\,F_{\mathrm{cII}}^{\mathrm{impl}}[\Psi_{0}])\bigr|
\;\le\; 1.27\times 10^{-2},
\]
i.e.\ no element of the scalar-$E$ class $\mathcal{E}$ is directionally
aligned with $F_{\mathrm{cII}}^{\mathrm{impl}}$.  Consequently, the
Rescue branch (R) and the Split branch (S) of
Prop.~\ref{prop:math65-triangulation-verdict} are falsified, and branch
(I) Class Insufficiency is selected.
\end{proposition}

\begin{proof}[Sketch]
Runs \eqref{eq:math65-run-A-norms}, \eqref{eq:math65-run-B-norms}, and
Eq.~\eqref{eq:math65-run-C-norms} supply the three triples
$(\|F^{\mathrm{impl}}\|,\|F^{\mathrm{grad},(X)}\|,\Delta^{(X)})$.
The polarisation identity \eqref{eq:math65-polarization} provides each
$\cos\theta^{(X)}$, yielding \eqref{eq:math65-cos-triangulation}. The
minimum over $X \in \{A,B,C\}$ of $|\cos\theta^{(X)}|$ is
$1.27\times 10^{-2}$, attained at $X=A$.  Directional disagreement of
this magnitude precludes $F^{\mathrm{impl}}$ from belonging to the
gradient image $\mathcal{G}_{\mathcal{E}}[\Psi_{0}]$; both (R) and (S)
require at least one candidate to be collinear with $F^{\mathrm{impl}}$
modulo an anti-Hermitian defect. \qed
\end{proof}

\begin{remark}[Empirical realisation of the Helmholtz--Hodge obstruction]

\begin{corollary}[Math66 operator-surgery mandate and Path-(X)/(Y) recommendation]
