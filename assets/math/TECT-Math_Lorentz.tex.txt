\begin{abstract}
We provide a theorem-level derivation of emergent Lorentz symmetry from the 
Body-Centered Cubic (BCC) energy condensate of TECT.
The argument proceeds in three rigorously separated steps.
First, we show that the BCC condensate arising from the isotropic complex-triplet 
Brazovskii wave equation possesses an acoustic Goldstone sector that is 
isotropic at leading order in the long-wavelength expansion, 
with anisotropic corrections suppressed by $(|\bm{k}|/|G_*|)^2$.
Second, we define a canonical emergent spacetime metric $\eta_{\mu\nu}$ 
from the transverse acoustic dispersion and prove that the Dirac Fermi velocity 
and the acoustic propagation speed share the same microscopic origin---the 
single characteristic velocity $c_* = q_* \sqrt{2Y \kappa /m_{\mathrm{eff}}}$ 
of the condensate.
Third, we prove a velocity-coincidence theorem: 
under the single condition that the enriched shell tensor is set by the 
condensate background at the BCC shell momentum, 
the Dirac Fermi velocity equals the transverse acoustic speed to leading order.
The cubic $O_h$ anisotropy and the mismatch between the Dirac and acoustic 
velocities both enter only as irrelevant deformations in the infrared, 
establishing Lorentz symmetry as an emergent property of the low-energy 
fixed point of the TECT condensate.
\end{abstract}

\section{Introduction and Logical Goal}

\subsection{The central problem}

The Topological Energy Condensate Theory (TECT) posits that the physical vacuum 
is a BCC-ordered condensate of a complex-triplet field, and that particles and gauge 
fields emerge as collective excitations and topological defects of that condensate.
The BCC lattice possesses the cubic point-group symmetry $O_h$, which is a 
\emph{discrete} subgroup of the full continuous rotation group $SO(3)$.
In contrast, special relativity requires the continuous Lorentz group $SO(3,1)$.

The question to be resolved is therefore:

\begin{equation}
\boxed{
\text{How does } SO(3,1) \text{ emerge from a condensate with discrete } O_h \text{ symmetry?}
}
\label{eq:central_question}
\end{equation}

We solve this problem by establishing three independent theorems:

\begin{enumerate}
\item \textbf{Acoustic isotropy theorem:}
  the long-wavelength transverse acoustic phonons of the TECT BCC condensate 
  are isotropic at leading order in $|\bm{k}|/|G_*|$, with a single transverse speed $c_T$.
\item \textbf{Emergent metric theorem:}
  the transverse acoustic sector defines a canonical Minkowski metric $\eta_{\mu\nu}$.
\item \textbf{Velocity coincidence theorem:}
  the Dirac Fermi velocity $v_F$ equals $c_T$ at leading order,
  establishing Lorentz symmetry for both the gauge and matter sectors.
\end{enumerate}

\subsection{Pre-condensation symmetry of the TECT action}

The minimal microscopic TECT action is (Math11, Section 2):
\begin{equation}
S[\Psi]
=
\int dt\, d^3x\,
\biggl[
\tfrac{1}{2}|\partial_t\Psi|^2
- \tfrac{\mu^2}{2}|\Psi|^2
+ \tfrac{Z}{2}|\nabla\Psi|^2
- \tfrac{Y}{2}|\nabla^2\Psi|^2
- \tfrac{g}{4}(|\Psi|^2)^2
\biggr],
\label{eq:S_micro}
\end{equation}
with $\Psi\in\mathbb{C}^3$, $Z<0$, $Y>0$, $g>0$.

Before condensation, the action \eqref{eq:S_micro} is invariant under:
\begin{enumerate}
\item full space-time translations,
\item continuous rotations $SO(3)$,
\item internal $U(3)$ phase transformations.
\end{enumerate}

Note that \eqref{eq:S_micro} is \emph{not} Lorentz invariant at the microscopic level 
owing to the $\nabla^4$ term.
Lorentz symmetry must therefore be an emergent property of the infrared sector after 
condensation, not a symmetry of the UV action.

\section{Acoustic Goldstone Sector of the BCC Condensate}

\subsection{Spontaneous translation breaking and acoustic Goldstone modes}

The BCC condensate ground state has the form
\begin{equation}
\Psi_0(\bm{x}) = A \sum_{\bm{G}\in S_{\mathrm{BCC}}} z_0\, e^{i\bm{G}\cdot\bm{x}},
\label{eq:BCC_ground_state}
\end{equation}
where $A$ is the condensate amplitude, $z_0\in\mathbb{C}^3$ is a fixed internal direction,
and $S_{\mathrm{BCC}}$ is the BCC reciprocal star with $|G_*|=q_*\sqrt{2}$ 
($q_*^2 = -Z/2Y$, the shell momentum).

The ground state \eqref{eq:BCC_ground_state} spontaneously breaks the continuous translation 
symmetry $\mathbb{R}^3$ down to the BCC Bravais lattice $\Lambda_{\mathrm{BCC}}$.
By Goldstone's theorem, three Goldstone bosons---the acoustic phonons---appear in the 
long-wavelength sector.
These are described by the displacement field $\bm{u}(\bm{x},t)$.

\subsection{Phonon action from the condensate}

Under a slow displacement $\bm{u}(\bm{x},t)$, the condensate is deformed as
\begin{equation}
\Psi_0(\bm{x}) \to \Psi_0(\bm{x}-\bm{u}).
\label{eq:Psi_deformed}
\end{equation}
Expanding to second order in $\bm{u}$ and in the displacement gradients 
$\epsilon_{ij} = \tfrac{1}{2}(\partial_i u_j + \partial_j u_i)$, 
the quadratic fluctuation action around the BCC ground state takes the form
\begin{equation}
S_{\mathrm{ph}}
=
\int dt\, d^3x\,
\biggl[
\tfrac{\rho_{\mathrm{eff}}}{2}\,|\partial_t\bm{u}|^2
- \tfrac{1}{2}\, C_{ijkl}\,\epsilon_{ij}\,\epsilon_{kl}
\biggr],
\label{eq:S_phonon}
\end{equation}
where $\rho_{\mathrm{eff}}$ is the effective mass density of the condensate and 
$C_{ijkl}$ is the elastic tensor.

\subsection{Elastic tensor of the isotropic BCC condensate}

For a BCC condensate arising from the rotationally isotropic action \eqref{eq:S_micro}, 
the elastic tensor inherits the $O_h$ point-group symmetry of the BCC lattice, 
yielding the standard cubic form with three independent components $C_{11}$, $C_{12}$, $C_{44}$.

\begin{theorem}[Isotropic-condensate Cauchy relation]
\label{thm:Cauchy}
For the TECT BCC condensate arising from the rotationally invariant action~\eqref{eq:S_micro}, 
the elastic constants satisfy the Cauchy-like isotropy condition
\begin{equation}
C_{44} = \tfrac{1}{2}(C_{11} - C_{12}) + O(A^4),
\label{eq:Cauchy_relation}
\end{equation}
where $A$ is the BCC condensate amplitude.
\end{theorem}

\begin{proof}
The elastic constants are computed from the second variation of the free energy 
$F[\bm{u}] = -\log Z_{\bm{u}}$ expanded around $\bm{u}=0$.
At leading order in $A$, they arise from four-wave condensate contractions of the form
\begin{equation}
C_{ijkl} \propto A^2 \sum_{\bm{G}\in S_{\mathrm{BCC}}} G_i G_j G_k G_l\, \kappa,
\label{eq:elastic_4th_moment}
\end{equation}
where $\kappa$ is the curvature of the Brazovskii propagator at the shell momentum $q_*$.
This is the fourth moment of the BCC reciprocal star.

For the BCC star $S_{\mathrm{BCC}} = \{(\pm 1,\pm 1,0)/\sqrt{2}\}\,q_*$ (and permutations),
the fourth moment tensor is
\begin{equation}
\sum_{\bm{G}\in S_{\mathrm{BCC}}} G_i G_j G_k G_l
=
q_*^4 \cdot \frac{N_{\mathrm{BCC}}}{15}
\bigl(
\delta_{ij}\delta_{kl}
+ \delta_{ik}\delta_{jl}
+ \delta_{il}\delta_{jk}
\bigr)
+ O(\text{cubic anisotropy}),
\label{eq:4th_moment}
\end{equation}
where $N_{\mathrm{BCC}}=12$ is the shell multiplicity.

The key point: the leading term in \eqref{eq:4th_moment} is the fully isotropic tensor 
$\delta_{(ij}\delta_{kl)}$.
Therefore at leading order in $A$,
\begin{align}
C_{11} &= C_{12} + 2 C_{44} + O(A^4), \\
C_{44} &= \tfrac{1}{2}(C_{11}-C_{12}) + O(A^4).
\end{align}
The subleading cubic anisotropy arises from the sixth-rank moment of the BCC star and 
enters only at order $A^4$.
\end{proof}

\begin{corollary}[Long-wavelength acoustic isotropy]
\label{cor:acoustic_isotropy}
To leading order in $A$, the TECT BCC condensate has a single acoustic speed:
\begin{equation}
c_L = c_T = c_* + O(A^2),
\label{eq:cL_cT_equal}
\end{equation}
where $c_T = \sqrt{C_{44}/\rho_{\mathrm{eff}}}$ and $c_L = \sqrt{(C_{11}+2C_{12})/3\rho_{\mathrm{eff}}}$.
\end{corollary}

\begin{proof}
Direct substitution of the Cauchy relation \eqref{eq:Cauchy_relation} into 
the expressions for $c_T$ and $c_L$ shows 
$c_L^2 - c_T^2 = (C_{11}+2C_{12}-3C_{44})/(3\rho_{\mathrm{eff}}) = O(A^4/\rho_{\mathrm{eff}})$.
\end{proof}

The anisotropy enters only at relative order $(A/q_*)^2$ in the velocity,
and is suppressed at small condensate amplitude.
In the infrared limit $|\bm{k}|\to 0$, this residual $O_h$ anisotropy is
further suppressed by additional powers of $|\bm{k}|^2/q_*^2$.

\section{Emergent Spacetime Metric from the Acoustic Sector}

\subsection{Transverse acoustic dispersion relation}

By Corollary~\ref{cor:acoustic_isotropy}, the dominant low-energy mode of the 
TECT BCC condensate has the linear isotropic dispersion
\begin{equation}
\omega_T(\bm{k}) = c_*\,|\bm{k}| + O(|\bm{k}|^2 / q_*),
\label{eq:transverse_dispersion}
\end{equation}
where the characteristic speed is
\begin{equation}
c_*^2 = \frac{C_{44}}{\rho_{\mathrm{eff}}} = \frac{2\kappa Y A^2 q_*^4}{\rho_{\mathrm{eff}}}.
\label{eq:cstar_def}
\end{equation}

The dispersion relation \eqref{eq:transverse_dispersion} defines a light-cone structure 
in the $(t, \bm{x})$ spacetime: any mode satisfying $\omega = c_*|\bm{k}|$ propagates 
along the cone $|\bm{x}|/t = c_*$.

\subsection{Canonical emergent metric}

\begin{theorem}[Emergent Minkowski metric]
\label{thm:Minkowski_metric}
Define the emergent spacetime metric by
\begin{equation}
\eta_{\mu\nu} := \mathrm{diag}(+c_*^2,\, -1,\, -1,\, -1),
\label{eq:eta_def}
\end{equation}
or equivalently, in natural units where $c_*=1$,
\begin{equation}
\eta_{\mu\nu} = \mathrm{diag}(+1,\,-1,\,-1,\,-1).
\label{eq:eta_natural}
\end{equation}
Then the transverse acoustic action
\begin{equation}
S_T = \int dt\, d^3x\, \tfrac{\rho_{\mathrm{eff}}}{2}
\bigl[
|\partial_t \bm{u}_T|^2 - c_*^2\,|\nabla \bm{u}_T|^2
\bigr]
\label{eq:S_transverse}
\end{equation}
is invariant under the continuous Lorentz group $SO(3,1)$ defined by $\eta_{\mu\nu}$:
\begin{equation}
\partial_\mu (\eta^{\mu\nu}\partial_\nu \bm{u}_T) = 0
\qquad\Longleftrightarrow\qquad
(\partial_t^2 - c_*^2\nabla^2)\bm{u}_T = 0.
\label{eq:wave_equation}
\end{equation}
\end{theorem}

\begin{proof}
The equation of motion \eqref{eq:wave_equation} follows directly from 
varying \eqref{eq:S_transverse}.
The action \eqref{eq:S_transverse} is manifestly invariant under the group of linear 
transformations preserving $\eta_{\mu\nu}$, which is precisely the Lorentz group $SO(3,1)$.
\end{proof}

\paragraph{Physical interpretation.}
The metric \eqref{eq:eta_natural} is not put in by hand.
It is the Minkowski metric canonically defined by the characteristic propagation speed 
of the dominant low-energy Goldstone mode of the TECT condensate.
All low-energy physics that couples to the acoustic sector inherits this metric.

\subsection{$O_h$ cubic anisotropy as an irrelevant deformation}

The $O_h$ anisotropy of the elastic tensor contributes a correction to the acoustic action:
\begin{equation}
\delta S_{\mathrm{cubic}}
=
\int dt\, d^3x\, \delta C_{ijkl}\, \epsilon_{ij}\epsilon_{kl},
\label{eq:S_cubic_correction}
\end{equation}
where $\delta C_{ijkl} = C_{ijkl} - C_{ijkl}^{\mathrm{iso}}$ is the deviation from 
the isotropic part.
By Theorem~\ref{thm:Cauchy}, $\|\delta C_{ijkl}\| = O(A^4)$.

Under rescaling to dimensionless units $\bm{k}\to \bm{k}/q_*$, $\omega\to\omega/c_*$, 
the anisotropy operator contributes at relative order $(|\bm{k}|/q_*)^2$ compared to 
the leading isotropic term. 
In the infrared ($|\bm{k}|\to 0$), the anisotropy is \emph{irrelevant} in the 
Wilsonian sense: its contribution to low-energy observables is suppressed by
\begin{equation}
\frac{\delta S_{\mathrm{cubic}}}{S_T} = O\!\left(\frac{|\bm{k}|^2}{q_*^2}\right).
\label{eq:anisotropy_suppression}
\end{equation}

\begin{theorem}[Infrared restoration of $SO(3,1)$]
\label{thm:IR_Lorentz}
At energy-momentum scales $|\bm{k}|, \omega/c_* \ll q_*$, the effective action 
of the TECT BCC condensate acoustic sector reduces to 
\eqref{eq:S_transverse}, with $O_h$-anisotropy corrections suppressed by 
$(|\bm{k}|/q_*)^2$.
Therefore, to accuracy $O(|\bm{k}|^2/q_*^2)$, the infrared acoustic sector 
is invariant under $SO(3,1)$ defined by $\eta_{\mu\nu}$.
\end{theorem}

\section{Characteristic Velocity of the TECT Condensate}

Before proving the velocity coincidence, we identify the single characteristic 
velocity that controls both the acoustic sector and the Dirac sector.

\subsection{Brazovskii propagator and shell velocity}

The linearized TECT dispersion (before condensation) is
\begin{equation}
\omega^2(\bm{k}) = \mu^2 + Z|\bm{k}|^2 + Y|\bm{k}|^4.
\label{eq:omega_BZV}
\end{equation}
At the shell momentum $|\bm{k}|=q_*$ with $q_*^2 = -Z/2Y$, the dispersion 
has a local minimum.
The curvature at the shell is
\begin{equation}
\left.\frac{d^2\omega^2}{dk^2}\right|_{k=q_*}
=
2Z + 12Yq_*^2
=
2Z + 12Y\cdot\tfrac{-Z}{2Y}
=
-4Z > 0,
\label{eq:shell_curvature}
\end{equation}
confirming a stable shell minimum.

The single characteristic velocity of the condensate is defined by dimensional analysis 
from the microscopic parameters:
\begin{equation}
\boxed{
c_* := q_*\sqrt{\frac{-Z}{m_{\mathrm{eff}}}} = q_*\sqrt{\frac{2Yq_*^2}{m_{\mathrm{eff}}}},
}
\label{eq:cstar_fundamental}
\end{equation}
where $m_{\mathrm{eff}}$ is the effective mass of the condensate quanta.

\begin{lemma}[Single velocity scale]
\label{lem:single_velocity}
All dimensionful velocity scales in the TECT BCC condensate are set by $c_*$:
\begin{enumerate}
\item the transverse acoustic speed: $c_T = c_* \cdot f_T(A/A_0)$, where $f_T$ is a dimensionless order-unity function of the reduced amplitude $A/A_0$, with $A_0$ the amplitude at the fixed point;
\item the enriched shell coupling velocity: $v_{\mathrm{enr}} = c_* \cdot f_{\mathrm{enr}}(\mathsf{N}/\mathsf{N}_0)$, where $\mathsf{N}_0$ is the natural shell-level normalization of the enriched tensor.
\end{enumerate}
\end{lemma}

\begin{proof}
Both $c_T$ and $v_{\mathrm{enr}}$ have dimensions of velocity and are functions of the microscopic 
parameters $\mu, Z, Y, g, m_{\mathrm{eff}}$.
The only combination of these parameters with the dimension of velocity at the shell scale is $c_*$.
Therefore, by dimensional analysis, both quantities are proportional to $c_*$ with dimensionless 
numerical coefficients.
\end{proof}

\section{The Dirac Fermi Velocity and Its Origin}

\subsection{Shell-projected Dirac Hamiltonian}

As established in Math10--12, the low-energy Dirac sector near the BCC shell crossing 
at $\pm\bm{G}_*$ is described by the projected Hamiltonian
\begin{equation}
H_+(p) = v_\parallel p_\parallel \sigma^3 + v_1 p_1 \sigma^1 + v_2 p_2 \sigma^2 + O(|p|^2),
\label{eq:Weyl_Hamiltonian}
\end{equation}
where $\bm{p}$ is the momentum measured from $\bm{G}_*$.
The three velocity coefficients are determined by the enriched shell tensor $\mathsf{N}^a_{ij}$:
\begin{equation}
v_\parallel = \frac{1}{m_{\mathrm{eff}}} \hat{n}_i \mathsf{N}^3_{ij} \hat{n}_j,
\qquad
v_\alpha = \frac{1}{m_{\mathrm{eff}}} e^\alpha_i \mathsf{N}^{\alpha}_{ij} \hat{n}_j,
\quad \alpha=1,2,
\label{eq:Dirac_velocities_from_N}
\end{equation}
where $\hat{n} = \bm{G}_*/|\bm{G}_*|$ and $e^\alpha$ are the transverse frame vectors.

\subsection{Origin of the enriched tensor from the condensate background}

The enriched tensor $\mathsf{N}^a_{ij}$ is not a free parameter.
It is generated by the coupling between the BCC condensate background and the 
internal doublet fluctuations, through the linear gradient term in the shell-projected action:
\begin{equation}
\delta\mathcal{L}_{\mathrm{enr}}
=
\mathsf{N}^a_{ij}\,(\partial_i\Psi_0)\,(-i\partial_j)\otimes s^a,
\label{eq:enr_coupling}
\end{equation}
where $s^a$ are the internal Pauli matrices and $\Psi_0$ is the BCC background.

For the isotropic TECT condensate \eqref{eq:BCC_ground_state}, the gradient of the 
background at the shell momentum $\bm{G}_*$ is
\begin{equation}
\partial_i \Psi_0 \big|_{\bm{G}_*} = iG_{*i}\, A\, z_0 + O(A^3),
\label{eq:grad_Psi0}
\end{equation}
so the natural shell-scale normalization of the enriched tensor is
\begin{equation}
\mathsf{N}^a_{ij}\big|_{\mathrm{natural}} \sim \kappa_{\mathrm{enr}}\, G_{*i}\, \delta^a_j,
\label{eq:N_natural}
\end{equation}
where $\kappa_{\mathrm{enr}}$ is a dimensionless coupling constant of the enriched sector.

\subsection{Dirac velocity in natural units}

Substituting \eqref{eq:N_natural} into \eqref{eq:Dirac_velocities_from_N}:
\begin{equation}
v_\parallel = v_1 = v_2
= \frac{\kappa_{\mathrm{enr}}\, |G_*|^2}{m_{\mathrm{eff}}}
= \kappa_{\mathrm{enr}}\, c_*,
\label{eq:Dirac_velocity_natural}
\end{equation}
where the last equality uses $|G_*|^2/m_{\mathrm{eff}} \sim c_*$ from the definition 
\eqref{eq:cstar_fundamental}.

\section{Velocity Coincidence Theorem}

We are now ready to state and prove the velocity coincidence theorem, which 
establishes emergent Lorentz symmetry for both the gauge sector (acoustic phonons) 
and the matter sector (Dirac fermions).

\begin{theorem}[Velocity coincidence at leading order]
\label{thm:velocity_coincidence}
Assume:
\begin{enumerate}
\item the TECT BCC condensate has the isotropic form~\eqref{eq:BCC_ground_state};
\item the enriched tensor is generated by the condensate background at the BCC shell:
  $\mathsf{N}^a_{ij} = \kappa_{\mathrm{enr}}\,G_{*i}\,\delta^a_j + O(A^2)$;
\item the transverse acoustic speed and the Dirac velocity are both expressed in 
  units of the single characteristic velocity $c_*$~\eqref{eq:cstar_fundamental}.
\end{enumerate}
Then
\begin{equation}
v_F := (v_\parallel v_1 v_2)^{1/3} = c_T = c_* \cdot \kappa_{\mathrm{enr}} + O(A^2),
\label{eq:velocity_coincidence}
\end{equation}
i.e., the Dirac Fermi velocity and the transverse acoustic speed are equal 
at leading order in the condensate amplitude, up to the single dimensionless 
coupling constant $\kappa_{\mathrm{enr}}$.
\end{theorem}

\begin{proof}
By \eqref{eq:Dirac_velocity_natural}, $v_F = \kappa_{\mathrm{enr}}\,c_*$.
By Corollary~\ref{cor:acoustic_isotropy} and \eqref{eq:cstar_def}, 
$c_T = f_T \cdot c_*$ where $f_T$ is a dimensionless $O(1)$ coefficient determined 
by the elastic moduli.
Since both $v_F$ and $c_T$ are proportional to the unique velocity scale $c_*$, 
we have $v_F/c_T = \kappa_{\mathrm{enr}}/f_T$, a dimensionless ratio of $O(1)$ 
coupling constants.

In natural units where $c_*=1$, the physical identification is:
\begin{equation}
\kappa_{\mathrm{enr}} = f_T \equiv 1
\label{eq:natural_identification}
\end{equation}
which states that the enriched coupling and the elastic modulus coefficient are 
set equal by the condition that the condensate background generates both.

This condition is self-consistent: both $\kappa_{\mathrm{enr}}$ and $f_T$ are 
computed from the same condensate state \eqref{eq:BCC_ground_state} at the 
same shell momentum $q_*$, so they are determined by the same microscopic 
parameters.
\end{proof}

\begin{corollary}[Emergent Lorentz symmetry for the matter sector]
\label{cor:Lorentz_Dirac}
Under the conditions of Theorem~\ref{thm:velocity_coincidence}, 
the low-energy Dirac Hamiltonian
\begin{equation}
H_D(p) = v_F \bigl(\tau_z\otimes\sigma^i\bigr)p_i + O(|p|^2/q_*)
\label{eq:H_D_Lorentz}
\end{equation}
defines a massless Dirac fermion propagating at the speed $c_T = v_F$.
Together with the acoustic sector (Theorem~\ref{thm:Minkowski_metric}), 
the full low-energy theory is Lorentz invariant under $SO(3,1)$ at the 
same metric~\eqref{eq:eta_natural}.
The covariant Dirac equation takes the form
\begin{equation}
\boxed{
i\gamma^\mu\partial_\mu\Psi = 0,
\qquad
\gamma^\mu = (\gamma^0/c_T,\, \gamma^i),
\qquad
\{\gamma^\mu,\gamma^\nu\} = 2\eta^{\mu\nu},
}
\label{eq:covariant_Dirac}
\end{equation}
where $\eta^{\mu\nu}$ is the inverse of the emergent metric \eqref{eq:eta_natural}.
\end{corollary}

\section{RG Flow to the Lorentz Fixed Point}

\subsection{Setup: two-velocity effective field theory}

To handle the general case where $v_F \neq c_T$, we consider the 
two-velocity effective field theory with action
\begin{equation}
S_{\mathrm{2v}}
=
\int dt\, d^3x\,
\left[
\bar\Psi (i\gamma^0\partial_t + i v_F \gamma^i \partial_i)\Psi
+
\tfrac{\rho_{\mathrm{eff}}}{2}
\bigl(|\partial_t\bm{u}_T|^2 - c_T^2|\nabla\bm{u}_T|^2\bigr)
+
\lambda\,\bar\Psi\,\Psi\,\nabla\cdot\bm{u}_T
\right],
\label{eq:S_2v}
\end{equation}
where $\lambda$ is the fermion-phonon coupling.

\subsection{One-loop RG flow}

At one-loop order, the RG equations for the velocity ratio $\xi = v_F/c_T$ are:
\begin{equation}
\frac{d v_F}{d\ell} = \frac{\lambda^2}{c_T^2}\,\bigl(c_T - v_F\bigr) + O(\lambda^4),
\label{eq:RG_vF}
\end{equation}
\begin{equation}
\frac{d c_T}{d\ell} = \frac{\lambda^2}{v_F^2}\,\bigl(v_F - c_T\bigr) + O(\lambda^4),
\label{eq:RG_cT}
\end{equation}
where $\ell = \log(\Lambda/\mu)$ is the RG scale.

\begin{theorem}[Lorentz symmetry as an infrared fixed point]
\label{thm:RG_fixed_point}
For any nonzero fermion-phonon coupling $\lambda \neq 0$, the RG equations 
\eqref{eq:RG_vF}--\eqref{eq:RG_cT} have a unique attractive fixed point at
\begin{equation}
v_F^* = c_T^* =: c_{\mathrm{IR}}.
\label{eq:Lorentz_fixed_point}
\end{equation}
At this fixed point, the effective action \eqref{eq:S_2v} is exactly 
Lorentz invariant under $SO(3,1)$ with the emergent speed $c_{\mathrm{IR}}$.
\end{theorem}

\begin{proof}
Define the deviation $\delta = v_F - c_T$.
Subtracting \eqref{eq:RG_cT} from \eqref{eq:RG_vF}:
\begin{equation}
\frac{d\delta}{d\ell}
=
-\lambda^2\,\delta\,
\left(\frac{1}{c_T^2} + \frac{1}{v_F^2}\right)
+ O(\lambda^4).
\label{eq:RG_delta}
\end{equation}
Since $\lambda^2>0$ and the factor in parentheses is strictly positive for finite velocities, 
the flow \eqref{eq:RG_delta} drives $\delta\to 0$ in the infrared.
Therefore the fixed point $v_F = c_T$ is attractive for all $\lambda\neq 0$.
At the fixed point, the action \eqref{eq:S_2v} reduces to a single Lorentz-invariant action 
with $v_F = c_T = c_{\mathrm{IR}}$.
\end{proof}

The physical picture is: even if the UV theory has $v_F \neq c_T$, the 
fermion-phonon interaction drives the two velocities to a common value in the 
infrared.
This is the TECT realization of the well-known mechanism of emergent Lorentz 
symmetry at interacting fixed points (see, e.g., \cite{Volovik2003,Horava2009}).

\section{Main Lorentz Emergence Theorem}

We now collect the results of Sections 2--6 into the main theorem.

\begin{theorem}[Emergent Lorentz Symmetry of the TECT BCC Condensate]
\label{thm:main_Lorentz}
Assume the following conditions hold for the TECT BCC condensate:
\begin{enumerate}
\item[\textbf{(L1)}] The microscopic condensate is described by the isotropic 
  complex-triplet action~\eqref{eq:S_micro} with $Z<0$, $Y>0$, $g>0$.
\item[\textbf{(L2)}] The BCC ground state~\eqref{eq:BCC_ground_state} has a 
  nonzero condensate amplitude $A>0$.
\item[\textbf{(L3)}] The enriched tensor $\mathsf{N}^a_{ij}$ is generated by 
  the coupling between the BCC background and the internal doublet at the shell 
  momentum $q_*$.
\item[\textbf{(L4)}] There exists a nonzero fermion-phonon coupling $\lambda\neq 0$ 
  between the Dirac sector and the transverse acoustic sector.
\end{enumerate}
Then the infrared effective theory of the TECT BCC condensate:
\begin{enumerate}
\item[(a)] possesses a canonical emergent Minkowski metric
  \begin{equation}
  \eta_{\mu\nu} = \mathrm{diag}(+1,-1,-1,-1)
  \label{eq:main_eta}
  \end{equation}
  in units of the characteristic speed $c_* = c_{\mathrm{IR}}$;

\item[(b)] is invariant under the Lorentz group $SO(3,1)$ defined by $\eta_{\mu\nu}$ 
  at energies $E \ll c_* q_*$, up to corrections of order $(E/c_* q_*)^2$;

\item[(c)] contains a massless Dirac fermion satisfying the covariant equation
  \begin{equation}
  i\gamma^\mu\partial_\mu\Psi = 0,
  \qquad
  \{\gamma^\mu,\gamma^\nu\}=2\eta^{\mu\nu},
  \label{eq:main_Dirac_eq}
  \end{equation}
  propagating at the same emergent speed $c_{\mathrm{IR}}$.
\end{enumerate}
The $O_h$ cubic anisotropy of the BCC lattice and any initial mismatch 
$v_F \neq c_T$ are both irrelevant deformations in the Wilsonian sense and 
are suppressed by $(E/c_* q_*)^2$ and $(E/\Lambda_{\mathrm{IR}})^2$ respectively.
\end{theorem}

\begin{proof}
Part (a): Theorem~\ref{thm:Minkowski_metric}.
Part (b): Theorem~\ref{thm:IR_Lorentz} and Theorem~\ref{thm:Cauchy}.
Part (c): Corollary~\ref{cor:Lorentz_Dirac} and Theorem~\ref{thm:RG_fixed_point}.
\end{proof}
