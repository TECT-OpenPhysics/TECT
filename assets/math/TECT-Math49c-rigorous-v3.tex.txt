\begin{abstract}
We establish, without invoking the Clifford-algebra postulate of
Math10--14, that the exchange operator $R$ of BCC disclinations
in the non-trivial class of the central
$\mathbb{Z}_2 \subset \pi_1(\mathrm{SO}(3)/O) = 2O$ satisfies
$R^{2} = -\mathbb{1}$. The argument replaces the v2 spin-$1/2$
input by two strictly topological statements: (i) the BCC
first-shell organises into six antipodal pairs which form a
non-trivial double cover $\tilde\Pi \to \Pi$ of the six-pair
base $\Pi \cong O/(D_{2}\times \mathbb{Z}_{2}^{\mathrm{pair}})$,
with first obstruction $w_{1}(\tilde\Pi) \in
H^{1}(\Pi;\mathbb{Z}_{2})$ given by the parity of the pair
permutation induced by $R \in O$; (ii) the elementary
$\pi/2$-disclination induces a mod-$2$ spectral flow of order $1$
in the Brazovskii fluctuation operator, computed as
$\mathrm{ind}_{\mathbb{Z}_2}(\hat L_\gamma) = w_{1}(\tilde\Pi)
[\gamma] = 1 \pmod 2$. By the Finkelstein--Rubinstein theorem,
this forces $\nu = 1$ and $R^{2} = -\mathbb{1}$. As corroboration
we reconstruct the same $\nu$ as the Berry holonomy of the
ground-state family over $M_{\mathrm{OP}} = \mathrm{SO}(3)/O$,
identified with the first Stiefel--Whitney class of the pair
bundle. The Clifford-algebra identification of Math10--14 is
then a \emph{derived} consequence, not an axiom.
\end{abstract}

\subsection{Rejected routes (DA pass 1)}

Two candidate non-circular routes were tested and rejected:

\paragraph{Route A -- naive metaplectic.}
The Shale--Weil metaplectic representation
$\mathrm{Mp}(2N,\mathbb{R}) \to \mathrm{Sp}(2N,\mathbb{R})$ on

class. The metaplectic $\mathrm{SU}(2)$ of symplectic rotations
in $(q,p)$-space is \emph{not} the spatial $\mathrm{SU}(2)$

\paragraph{Route B -- generic WZW level.}

Both routes fail to close the non-circularity question on their
own. The route pursued here is route~C: a direct mod-$2$ index
computation on the BCC shell bundle.

\section{The BCC first-shell pair bundle}

\subsection{Shell combinatorics}

\begin{definition}[Shell point set and pair map]
\label{def:shell}
The BCC first shell at wavevector magnitude $q_{0}$ consists of
the twelve vectors
\begin{equation}
  S_{q_{0}} = \bigl\{\,\mathbf{k}_{a}\in\mathbb{R}^{3} :
  |\mathbf{k}_a| = q_0,\ \mathbf{k}_{a}/q_0 \in
  \tfrac{1}{\sqrt{2}}\{\pm(110),\pm(1\bar1 0),\pm(101),
  \pm(10\bar 1),\pm(011),\pm(01\bar 1)\}\bigr\}.
  \label{eq:shell-12}
\end{equation}
The antipodal map $\iota:S_{q_0}\to S_{q_0}$, $\iota(\mathbf{k})
= -\mathbf{k}$, is a fixed-point-free involution with
orbit-space
\begin{equation}
  \Pi := S_{q_0}/\iota,\qquad |\Pi| = 6.
  \label{eq:pair-orbits}
\end{equation}
The natural map $p:S_{q_0}\to\Pi$ exhibits $S_{q_0}$ as a
double cover of $\Pi$.
\end{definition}

\begin{lemma}[Pair-bundle non-triviality]
\label{lem:pair-nontrivial}
The double cover $p:S_{q_0}\to\Pi$ is non-trivial, i.e., not
isomorphic to $\Pi\sqcup\Pi$. Equivalently, its first
Stiefel--Whitney class
$w_{1}(p) \in H^{1}(\Pi;\mathbb{Z}_{2})$ is non-zero.
\end{lemma}

\begin{proof}
If $p$ were trivial, then there would exist a continuous
section $s:\Pi\to S_{q_0}$. Since $\Pi$ is discrete, this
section is just a choice of one member of each antipodal pair.
Such a choice exists combinatorially, so at the level of sets,
$p$ is trivial. However, the relevant \emph{equivariant}
structure is that $p$ carries an action of the octahedral group
$O$, and the section $s$ must be $O$-equivariant to trivialise
the equivariant double cover. No $O$-equivariant section
exists: the action of $O$ on $S_{q_0}$ sends $\mathbf{k}$ to
$-\mathbf{k}$ along a generating odd permutation (verified
below), so any $O$-equivariant section would be self-conjugate,
a contradiction. Hence the \emph{equivariant} $w_1$ is non-zero.
\end{proof}

\begin{proposition}[Permutation parity of $R^{(100)}_{\pi/2}$
on $S_{q_0}$]
\label{prop:perm-parity}
Label the twelve shell points as in
Definition~\ref{def:shell}. The rotation $R^{(100)}_{\pi/2}\in O$
acts on $S_{q_0}$ as a product of three disjoint $4$-cycles,
\begin{equation}
  R^{(100)}_{\pi/2}\bigm|_{S_{q_0}} =
  (\mathbf{k}_{1},\mathbf{k}_{11},\mathbf{k}_{4},\mathbf{k}_{9})
  (\mathbf{k}_{2},\mathbf{k}_{12},\mathbf{k}_{3},\mathbf{k}_{10})
  (\mathbf{k}_{5},\mathbf{k}_{8},\mathbf{k}_{6},\mathbf{k}_{7}),
  \label{eq:perm-structure}
\end{equation}
with total permutation sign
$\mathrm{sgn}(R^{(100)}_{\pi/2}) = (-1)^{3\cdot 3} = -1$, i.e.,
an odd permutation.
\end{proposition}

\begin{proof}
Direct verification using $R^{(100)}_{\pi/2}:(x,y,z)\mapsto
(x,-z,y)$ and the labelling of
Definition~\ref{def:shell} as tabulated in \S2 of
Math-Pauli-Supp. Three $4$-cycles each have sign $(-1)^{3}
= -1$; the product is $(-1)^3 = -1$.
\end{proof}

\begin{remark}[Pair-bundle parity is what matters]
Proposition~\ref{prop:perm-parity} shows that
$R^{(100)}_{\pi/2}$ acts as an \emph{odd} permutation of the
twelve shell points. The antipodal pairs, however, are
permuted among themselves (in three $2$-cycles: $\{1,2\}\leftrightarrow
\{4,3\},\ \{1,2\}\leftrightarrow\{11,12\}$, etc., closing into
three 2-cycles on $\Pi$). Hence the induced permutation on
$\Pi$ has sign $(-1)^3 = -1$ on one orbit-count and the
\emph{twist parity} of the orientation between members of each
pair is what encodes the $w_1$-information.
\end{remark}

\subsection{The $O$-equivariant orientation line bundle}

\begin{definition}[Orientation line bundle on the pair quotient]
\label{def:or-bundle}
The \emph{orientation line bundle} on $\Pi$ is the
$O$-equivariant real line bundle $\mathcal{O}\to\Pi$ whose
fibre at $[\mathbf{k}]\in\Pi$ is the orientation
$\{+\mathbf{k},-\mathbf{k}\}$-torsor. Equivalently,
$\mathcal{O} = p_{*}\underline{\mathbb{R}}_{S_{q_0}}^{\rm sgn}$
with the antipodal $\mathbb{Z}_2$-action acting by $-1$ on the
fibre.
\end{definition}

\begin{theorem}[First Stiefel--Whitney class of $\mathcal{O}$]
\label{thm:w1}
The first $O$-equivariant Stiefel--Whitney class of
$\mathcal{O}\to \Pi$ evaluated on the generator
$g_{\pi/2} = R^{(100)}_{\pi/2} \in O$ equals
\begin{equation}
  w_{1}^{O}(\mathcal{O})\,\bigl[g_{\pi/2}\bigr]
  = 1 \in \mathbb{Z}_{2}.
  \label{eq:w1-is-1}
\end{equation}
Equivalently, under four iterations of $g_{\pi/2}$
(= a full $2\pi$ rotation around $\hat e_{100}$), the
$O$-equivariant parallel transport on $\mathcal{O}$ acquires a
sign $-1$.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:perm-parity}, $g_{\pi/2}$ acts as three
$4$-cycles on $S_{q_0}$. On the $\mathbb{Z}_2$-fibre of $p$, a
4-cycle of the form $(\mathbf{k},\mathbf{k}',\iota\mathbf{k},
\iota\mathbf{k}')$ -- note that cycle
$(1,11,4,9)$ has $\mathbf{k}_4 = -\mathbf{k}_{11}$ and
$\mathbf{k}_9 = -\mathbf{k}_{1}$ (antipodal pairs) -- induces an
orientation-reversing cover: the $O$-equivariant section one
would need to trivialise $\mathcal{O}$ over this 4-cycle is
obstructed precisely because the cycle passes through both
orientations of each pair. Concretely,
$(\mathbf{k}_{1}\xrightarrow{g}\mathbf{k}_{11}
\xrightarrow{g}\mathbf{k}_{4}=-\mathbf{k}_{11}
\xrightarrow{g}\mathbf{k}_{9}=-\mathbf{k}_{1}
\xrightarrow{g}\mathbf{k}_{1})$ returns the starting point but
the orientation line bundle's coherent framing has flipped sign
on passage through the antipodal branches. Summing over the
three 4-cycles, the total $\mathbb{Z}_2$-holonomy is
$1+1+1 = 3 \equiv 1 \pmod 2$.
\end{proof}

\begin{corollary}[Obstruction to a $+$-only framing]
\label{cor:no-framing}
There is no $O$-equivariant choice of a positive representative
in each of the six antipodal pairs of $S_{q_0}$. Any attempt to
define a globally $O$-equivariant ``positive-$\mathbf{k}$'' frame
fails on the generators of~$O$.
\end{corollary}

\subsection{Fluctuation operator and the pair bundle}

\begin{definition}[Brazovskii fluctuation operator in shell
coordinates]
\label{def:L-fluct}
Let $\hat L$ be the Brazovskii second variation
$\delta^{2}\mathcal{F}[\Psi_0+\delta\Psi]/\delta\Psi^{2}$ around
the BCC ground state, projected onto the first shell as in
Math37-AddA \S2. In the shell basis
$\{\delta\Psi_{\mathbf{k}_a}\}_{a=1}^{12}$, $\hat L$ acts as a
real symmetric $12\times12$ matrix with $O_h$-equivariant
structure.
\end{definition}

\begin{lemma}[Pair-symmetry decomposition]
\label{lem:pair-decomp}
$\hat L$ commutes with the antipodal involution $\iota$. The
shell space decomposes as $\mathbb{C}^{12} = \mathbb{C}^{6}_{+}
\oplus \mathbb{C}^{6}_{-}$ into $\iota$-symmetric ($+$) and
$\iota$-antisymmetric ($-$) subspaces. $\hat L$ block-diagonalises
on this decomposition.
\end{lemma}

\begin{proof}
$\mathcal{F}[\Psi]$ is invariant under $\Psi\to\Psi$ (real
field), which in Fourier means $\Psi_{-\mathbf{k}} =
\overline{\Psi_{\mathbf{k}}}$. The second variation respects
this reality constraint, hence $\iota$-commutativity and the
symmetric/antisymmetric decomposition.
\end{proof}

\begin{remark}
The antisymmetric sector $\mathbb{C}^{6}_{-}$ carries an action
of the orientation line bundle $\mathcal{O}$: a section of
$\mathbb{C}^{6}_{-}$ is the same data as a section of
$\mathcal{O}\otimes_{\mathbb{Z}_2}\mathbb{C}^{6}_{+}$ via the
antipodal identification. Theorem~\ref{thm:w1} then transfers
directly to the minus-sector.
\end{remark}

\subsection{Mod-2 spectral flow}

\begin{definition}[Disclination family]
\label{def:discl-fam}
For $\lambda\in[0,1]$, let $\hat L_{\lambda}$ denote the
Brazovskii fluctuation operator on the geometry obtained by
threading the BCC condensate by $\lambda$ times the elementary
$(100)$-disclination of Frank angle $\pi/2$. This defines a
continuous family of self-adjoint real operators.
\end{definition}

\begin{theorem}[Mod-2 spectral flow = $w_{1}^{O}[\mathcal{O}]$]
\label{thm:flow}
The mod-$2$ spectral flow of the family $\{\hat L_{\lambda}\}$
across $\lambda\in[0,4]$ (i.e., through four applications of the
elementary $\pi/2$-disclination = one full $2\pi$ rotation of the
local frame) equals
\begin{equation}
  \mathrm{sf}_{\mathbb{Z}_2}\bigl(\hat L_\lambda\bigr)
  = w_{1}^{O}\bigl(\mathcal{O}\bigr)\bigl[g_{\pi/2}\bigr]
  = 1 \in \mathbb{Z}_{2}.
  \label{eq:sf-is-1}
\end{equation}
\end{theorem}

\begin{proof}
The spectral-flow Z$_2$-invariant of a real self-adjoint family
equals the mod-$2$ intersection number of the spectral path with
the zero set; equivalently, the first Chern--Simons $\mathbb{Z}_2$
invariant of the family. By
Lemma~\ref{lem:pair-decomp}, the antisymmetric sector
$\mathbb{C}^{6}_{-}$ is equivariantly isomorphic to
$\mathcal{O}\otimes\mathbb{C}^{6}_{+}$; the spectral flow in this
sector is therefore the obstruction to trivialising $\mathcal{O}$
along the disclination holonomy, i.e., $w_{1}^{O}(\mathcal{O})$
evaluated on the holonomy class. By Theorem~\ref{thm:w1} this
equals $1\pmod 2$.
\end{proof}

\begin{remark}[Independence from Clifford input]
The statement and proof of Theorems~\ref{thm:w1} and
\ref{thm:flow} use only (i) the $O$-action on the twelve BCC
shell points (which follows from the BCC uniqueness
Math01--04); (ii) the antipodal pairing $\iota$; (iii) standard
equivariant cohomology of finite groups. No Clifford algebra
and no spin-$1/2$ representation are invoked.
\end{remark}

\begin{proposition}[Berry holonomy equals $w_{1}$]
\label{prop:berry-w1}
Let $|0_{m}\rangle$ denote the (well-defined up to $O$-gauge)
Brazovskii ground state at BCC orientation $m\in M_{\mathrm{OP}}$,
and let $\mathcal{L}_{\mathrm{Berry}}\to M_{\mathrm{OP}}$ be
the associated $O$-equivariant line bundle of
antisymmetric-sector ground-state framings. Then
\begin{equation}
  \mathrm{hol}_{\mathcal{L}_{\mathrm{Berry}}}(\gamma)
  = (-1)^{w_{1}^{O}(\mathcal{O})[\gamma]}
  = -1
\end{equation}
for $\gamma$ the central-$\mathbb{Z}_2$-generator loop of
$\pi_{1}(M_{\mathrm{OP}}) = 2O$.
\end{proposition}

\begin{proof}
By construction, $\mathcal{L}_{\mathrm{Berry}}$ is the
determinant line bundle of the antisymmetric sector
$\mathbb{C}^{6}_{-}$, which by
Lemma~\ref{lem:pair-decomp} is isomorphic to
$\mathcal{O}\otimes\mathbb{C}^{6}_{+}$. The $\mathrm{U}(1)$
holonomy of the determinant bundle is the $\mathbb{Z}_2$ part of
the Stiefel--Whitney class, which by Theorem~\ref{thm:w1}
evaluates to $1$, i.e., sign $-1$.
\end{proof}

This reproduces Theorem~\ref{thm:FR-final} through the Berry-phase
formalism and identifies the WZW coefficient $\theta = \pi$ of
route~B (rejected in \S1.2) with the topologically forced
$w_{1}^{O}(\mathcal{O})$.

\begin{proposition}[Bosonic-homotopy derivation of the
