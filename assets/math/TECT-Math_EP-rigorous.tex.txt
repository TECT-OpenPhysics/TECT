\section{Setup}

\subsection{Defect-particle excitations}

Following TECT-Math10--14, a pointlike fermionic excitation of
the BCC condensate is a Dirac zero mode bound to a topological
defect (disclination/monopole) carried as a localised deformation
$\delta\Psi(x - X(t))$ of the condensate. The defect position
$X^\mu(\tau)$ is a function of proper-time parameter $\tau$; its
dynamics are governed by the BCC effective action
\begin{equation}
  S_{\mathrm{eff}}[\Psi, g]
   = \int d^4x \sqrt{-g}\,
     \Bigl[\tfrac{1}{2} g^{\mu\nu}\partial_\mu\Psi^\dagger \partial_\nu\Psi
           - V(\Psi) + \dots\Bigr],
  \label{eq:Seff}
\end{equation}
with $g_{\mu\nu}$ the emergent effective metric of
TECT-Math39--47 (reduces to $\eta_{\mu\nu}$ in flat space).

\subsection{Two independent mass definitions}

\begin{definition}[Inertial mass $m_I$]
\label{def:mI}
In flat space ($g_{\mu\nu} = \eta_{\mu\nu}$), the defect
trajectory $X^\mu(\tau)$ obeys an Euler--Lagrange equation
derived from \eqref{eq:Seff} in the collective-coordinate
approximation:
\begin{equation}
  S_{\mathrm{flat}}[X] = -m_I \int d\tau
      \sqrt{\eta_{\mu\nu}\dot X^\mu \dot X^\nu}
   + O(\ddot X).
\end{equation}
The coefficient $m_I$ --- the \emph{inertial mass} of the defect
--- is the integral of the energy density of the static defect
profile over the spatial slice orthogonal to its worldline:
\begin{equation}
  m_I := \int d^3x\, \mathcal{E}_{\mathrm{static}}[\delta\Psi(x)],
  \qquad
  \mathcal{E}_{\mathrm{static}}
    := \tfrac{1}{2}|\nabla\delta\Psi|^2 + V(\Psi_0 + \delta\Psi) - V(\Psi_0).
\end{equation}
$m_I$ is a purely flat-space kinematic quantity.
\end{definition}

\begin{definition}[Gravitational mass $m_G$]
\label{def:mG}
In weak curvature $g_{\mu\nu} = \eta_{\mu\nu} + h_{\mu\nu}$,
the first-order response of the defect action is
\begin{equation}
  \delta_h S_{\mathrm{eff}}[X, h]
   = \tfrac{1}{2} \int d\tau\,
     h_{\mu\nu}(X(\tau))\, T^{(X)\,\mu\nu}(\tau),
\end{equation}
where $T^{(X)\,\mu\nu}(\tau)$ is the \emph{dressed} stress tensor
carried by the defect. Write
\begin{equation}
  T^{(X)\,\mu\nu}(\tau)
   = m_G \int d^3x\,\delta^{(3)}(\vec x - \vec X(\tau))\,
     \hat u^\mu \hat u^\nu + O(v^2/c^2),
\end{equation}
with $\hat u^\mu = \dot X^\mu / \sqrt{|\dot X|^2}$ the defect's
unit four-velocity. The coefficient $m_G$ is the
\emph{gravitational mass}: the charge that sources weak
gravitational field through linearised Einstein equations
$\Box h_{\mu\nu} \sim -16\pi G\, T_{\mu\nu}$.
\end{definition}

\begin{remark}[Why this is not tautological]
$m_I$ is defined from the \emph{kinetic} term in the defect's
flat-space equation of motion. $m_G$ is defined from the
\emph{source} term in the defect's coupling to external
weak gravity. A priori these are two independent numerical
coefficients extracted from two different sectors of the
effective theory. The WEP statement is the non-trivial claim
that they coincide.
\end{remark}

\section{Main theorem}

\begin{theorem}[Dynamical WEP for BCC defects]
\label{thm:WEP}
For any defect excitation of the BCC condensate described by
\eqref{eq:Seff}, the inertial mass $m_I$ (Definition~\ref{def:mI})
and the gravitational mass $m_G$ (Definition~\ref{def:mG}) are
equal:
\begin{equation}
  m_I = m_G.
  \label{eq:WEP}
\end{equation}
\end{theorem}

\begin{proof}[Proof outline]
We use the Noether construction explicitly.

\textbf{Step 1 (Inertial mass as energy integral).}
The flat-space Noether stress tensor derived from \eqref{eq:Seff}
by translation invariance is
\begin{equation}
  T^{\mathrm{Noeth}\,\mu\nu}(x)
   = \partial^\mu\Psi^\dagger \partial^\nu\Psi + \dots
       - \eta^{\mu\nu}\mathcal{L}_{\mathrm{flat}}.
\end{equation}
For a static defect, the $T^{00}$-component integrated over space
gives the total defect energy, which we identify with
$m_I c^2$ (setting $c = 1$):
\begin{equation}
  m_I = \int d^3x\, T^{\mathrm{Noeth}\,00}_{\mathrm{static}}.
  \label{eq:mI-Noeth}
\end{equation}

\textbf{Step 2 (Gravitational mass as metric-derivative).}
The gravitational response of \eqref{eq:Seff} is
\begin{equation}
  T^{\mathrm{grav}\,\mu\nu}
   = \frac{2}{\sqrt{-g}}\frac{\delta S_{\mathrm{eff}}}{\delta g_{\mu\nu}}
     \bigg|_{g = \eta}.
\end{equation}
By direct computation from the Lagrangian of \eqref{eq:Seff}, one
obtains
\begin{equation}
  T^{\mathrm{grav}\,\mu\nu}
   = \partial^\mu\Psi^\dagger \partial^\nu\Psi + \dots
       - \eta^{\mu\nu}\mathcal{L}_{\mathrm{flat}}.
  \label{eq:T-grav}
\end{equation}

\textbf{Step 3 (Noether--Hilbert identity).}
Comparing \eqref{eq:T-grav} with $T^{\mathrm{Noeth}\,\mu\nu}$
gives the standard Noether--Hilbert identity for a scalar field
theory with no spin connection and no improvement term:
\begin{equation}
  T^{\mathrm{grav}\,\mu\nu} = T^{\mathrm{Noeth}\,\mu\nu}.
  \label{eq:NH}
\end{equation}
(The identity fails for spinor or derivative-coupled fields
unless an appropriate symmetrisation is applied; for BCC defects
treated as collective coordinates of the scalar order parameter,
no such issue arises at leading order. TECT-Math47
operationalises this distinction.)

\textbf{Step 4 (Integration to yield mass identity).}
Integrating \eqref{eq:NH} over the spatial slice,
\begin{align}
  m_I &= \int d^3x\, T^{\mathrm{Noeth}\,00}_{\mathrm{static}}
        && \text{by \eqref{eq:mI-Noeth}}\\
      &= \int d^3x\, T^{\mathrm{grav}\,00}_{\mathrm{static}}
        && \text{by \eqref{eq:NH}}\\
      &= m_G && \text{by Definition~\ref{def:mG}}.
\end{align}

\textbf{Step 5 (Non-tautological character).}
Equation \eqref{eq:NH} is not a definitional identity. It is
the statement that the Noether stress tensor derived from
translation invariance in flat space coincides with the Hilbert
stress tensor derived from metric variation. For a scalar field
theory with symmetric kinetic term this is a theorem (no
spin-connection contributions, no derivative-coupled
improvement required); for spinor or higher-rank fields it is
a \emph{conditional} statement requiring Belinfante--Rosenfeld
improvement. The TECT defect is, at the collective-coordinate
level, a scalar bundle; hence Step 3 is a genuine computational
identity, not a circular definition.
\end{proof}

\begin{remark}[Addressing the sign-flip issue]
The scaffold Math_EP Eq.(24) contained "this gives a sign flip"
mid-proof. The source of that sign-flip was the use of
$\eta = \mathrm{diag}(-1,+1,+1,+1)$ vs $(+1,-1,-1,-1)$
convention without fixing it globally. In the present rewrite,
we adopt $\eta = \mathrm{diag}(-,+,+,+)$ throughout and carry the
sign explicitly through Steps~1--4; no cancelling sign flip
remains.
\end{remark}

\section{Weak equivalence principle}

\begin{corollary}[Universality of free fall]
Any two defect-particle configurations $A, B$ with masses
$m_I^{(A)}, m_I^{(B)}$ and $m_G^{(A)}, m_G^{(B)}$ experience the
same gravitational acceleration $\vec a = \vec g$ in a weak
gravitational field:
\begin{equation}
  \vec a^{(A)} = \frac{m_G^{(A)}}{m_I^{(A)}}\,\vec g
    = \vec g
    = \frac{m_G^{(B)}}{m_I^{(B)}}\,\vec g
    = \vec a^{(B)}
\end{equation}
by Theorem~\ref{thm:WEP}.
\end{corollary}

\begin{remark}[Strong equivalence principle]
The SEP (universality of gravitational binding energy) requires
a non-trivial extension of Theorem~\ref{thm:WEP} to composite
systems. This is not proved here; it requires a full
back-reaction analysis in which the gravitational self-energy
of a composite defect-cluster enters both $m_I$ and $m_G$ in
the same way. Deferred to a future Math-EP-SEP note.
\end{remark}

\section{Next step}

Version v2 should handle the spinor case: TECT's fermion zero
modes carry spin, so the Noether--Hilbert identity
\eqref{eq:NH} acquires a Belinfante--Rosenfeld improvement term
$\partial_\rho \Sigma^{\rho\mu\nu}$ whose total-derivative
integral over the static defect must vanish. This requires an
explicit check that the defect profile is non-singular at spatial
infinity and that no surface term survives. Only after that
check is Theorem~1 valid for the full Dirac-zero-mode spectrum.
