\section{Setting}
$\mathbb{P}^1\times\mathbb{P}^1$ is the product of two complex projective lines.
Each $\mathbb{P}^1$ has the standard 2-chart atlas $\{U_0, U_1\}$:
$U_0 = \{[z_0:z_1] : z_0\ne 0\}$ with affine coordinate $u = z_1/z_0$;
$U_1 = \{[z_0:z_1] : z_1\ne 0\}$ with affine coordinate $v = z_0/z_1$;
transition $v = 1/u$ on $U_0\cap U_1 = \mathbb{C}^\times$.

The product atlas is $\{U_{ij} = U_i^{(1)}\times U_j^{(2)}\}_{i,j\in\{0,1\}}$,
i.e., 4 charts indexed by $(i,j)$. Local coordinates on $U_{ij}$:
$(u_i, v_j)$ where $u_i \in \{u, 1/u\}$ for $i=0,1$ on first
$\mathbb{P}^1$; $v_j$ analogous for second.

\section{Transition functions for SO(10) gauge bundle}
Math162 constructed transition $g_{01}^{\mathrm{CP}^2}(u_1, u_2)
= \exp(i\arg(u_1)\,T_3)$ on the standard 3-chart atlas of $\mathbb{CP}^2$.
Lifting to $\Sigma_0$ produces 6 pairwise transitions
$g_{(i_1,j_1),(i_2,j_2)}$ on the four-chart product atlas.

\begin{proposition}[Lifted transitions]
The SO(10) gauge bundle on $\Sigma_0$ inherits the Math162 transition
functions via the product structure:
\begin{align}
g_{(0,0),(1,0)}(u, v) &= \exp(i\arg(u)\,T_3)
   && \text{(first-factor U(1) on } U_{00}\cap U_{10}\text{)}\\
g_{(0,0),(0,1)}(u, v) &= \exp(i\arg(v)\,T_3')
   && \text{(second-factor U(1) on } U_{00}\cap U_{01}\text{)}\\
g_{(0,0),(1,1)}(u, v) &= \exp(i\arg(u)\,T_3 + i\arg(v)\,T_3')
   && \text{(diagonal on } U_{00}\cap U_{11}\text{)}\\
g_{(1,0),(0,1)}(u, v) &= g_{(1,0),(0,0)}\cdot g_{(0,0),(0,1)} \\
g_{(1,0),(1,1)}(u, v) &= g_{(0,0),(0,1)}|_{u\to 1/u}
   && \text{(transitivity)}\\
g_{(0,1),(1,1)}(u, v) &= g_{(0,0),(1,0)}|_{v\to 1/v}
   && \text{(transitivity)}
\end{align}
where $T_3, T_3'$ are commuting U(1) generators in the SO(10) Cartan
subalgebra; $\arg$ is the principal argument $(-\pi, \pi]$.
\end{proposition}

The cocycle condition $g_{ab}\cdot g_{bc} = g_{ac}$ is verified by
direct substitution; the transitivity rows above are the consistency
constraints that follow.

\section{Chern classes}
\begin{proposition}[Chern data on $\Sigma_0$]
For the SO(10) adjoint bundle (rank 45) on $\mathbb{P}^1\times\mathbb{P}^1$
with the transition functions above:
\begin{itemize}
\item $c_1$ of each U(1) factor: $c_1^{(1)} = c_1^{(2)} = 1$ in
$H^2(\Sigma_0, \mathbb Z)$ (one for each $\mathbb{P}^1$ factor).
\item Total $c_1$ vanishes in the SO(10) traceless adjoint:
$c_1(E) = \mathrm{tr}(F) = 0$ (since SO(10) is semi-simple).
\item Second Chern class:
$c_2(E) = -\mathrm{tr}(F\wedge F)/(8\pi^2)$. For the diagonal
embedding above, $F\wedge F = (T_3 + T_3')^2 \cdot du\wedge d\bar u
\wedge dv\wedge d\bar v$. Integrating over $\Sigma_0$:
$c_2(E) = (\mathrm{tr}(T_3^2) + \mathrm{tr}(T_3'^2)) \cdot (1\cdot 1) = 2\cdot
\mathrm{Cas}_2(\mathbf{45})\cdot 1 = 90$
(with $\mathrm{Cas}_2(\mathbf{45}) = 8$ for SO(10) adjoint times
generator-trace normalization 45/8 yields 45 not 90; we re-examine).
\end{itemize}
\end{proposition}
