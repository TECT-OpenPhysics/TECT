\begin{abstract}
We execute the recommended closure pathway of \texttt{TECT-Math80}
(Strategies 2 + 3 combined) and obtain a Lie-algebraic and
representation-theoretic theorem for the 10-defect-moduli count
appearing in the Pillar 6 dimension chain
\[
24\;\text{(\(G_{\mathrm{SM}}\) gauge-field moduli, Math75-Q3)}
\;+\;
11\;\text{(BCC amplitude relative phases)}
\;+\;
10\;\text{(higher-charge defect moduli)}
\;=\;
45 \;=\; \dim \mathrm{SO}(10).
\]
The 10 is identified \emph{uniquely} as $\dim \mathfrak{so}(10) -
\dim \mathfrak{so}(9) = 10$, equivalently $\dim \mathbf{10}_{\mathrm{vec}}$
(the defining $10$-dimensional vector representation of
$\mathrm{SO}(10)$). Two independent derivations are presented and
shown to agree exactly. The result removes the principal
audit-flagged defect of \texttt{Math77-Q6a-Q6b-closure} (the 10 was
previously a conjecture). Pillar 6 remains \texttt{PARTIAL-ADVANCED}
because the equivariant identification of the BCC defect bundle with
this 10-dim coset --- the topological half of Q6a --- is not closed in
this addendum and is reserved for the next sub-task (Strategy 1 of
Math80 \S 4).
\end{abstract}

\section{Setup and convention}

\subsection{Standing assumptions}

\begin{itemize}
\item $\mathrm{SO}(10)$ is the simply-connected real-form Lie group
       used in standard $\mathrm{SO}(10)$ Grand Unification, $\dim_{\mathbb{R}} \mathfrak{so}(10) = 45$.
\item $\mathrm{Spin}(7) \subset \mathrm{SO}(10)$ is realised as the
       maximal compact subgroup that stabilises a unit vector
       $v_0 \in \mathbb{R}^{10}$ along with the $\mathrm{Spin}(7)$
       structure on its orthogonal complement $v_0^{\perp} \cong \mathbb{R}^9$
       carrying the standard $\mathrm{Spin}(7)$-action on $\mathbb{R}^9$.
       (The relevant subgroup chain is
       $\mathrm{SO}(10) \supset \mathrm{SO}(9) \supset \mathrm{Spin}(7)$;
       only the first inclusion is used here.)
\item The Pati--Salam decomposition $G_{\mathrm{PS}} = \mathrm{SU}(4)_C \times \mathrm{SU}(2)_L \times \mathrm{SU}(2)_R$ is identified inside $\mathrm{SO}(10)$ via the standard isomorphisms $\mathfrak{su}(4) \cong \mathfrak{so}(6)$, $\mathfrak{su}(2)\oplus\mathfrak{su}(2) \cong \mathfrak{so}(4)$.
\item The Standard Model gauge group is $G_{\mathrm{SM}} = (\mathrm{SU}(3)_C \times \mathrm{SU}(2)_L \times \mathrm{U}(1)_Y)/\mathbb{Z}_6$, $\dim_{\mathbb{R}} \mathfrak{g}_{\mathrm{SM}} = 8 + 3 + 1 = 12$.
\end{itemize}

\subsection{Working definitions of the three moduli classes}

\begin{itemize}
\item \emph{$G_{\mathrm{SM}}$ gauge-field moduli} ($24$-dim): the
       symplectic-reduced Brazovskii first-shell amplitude phase
       space, per Math75-Q3-symplectic-reduction theorem. Local
       identification with $T^* G_{\mathrm{SM}}$ at the reduced level
       is proved (Math75-Q3 \S 4).
\item \emph{BCC amplitude relative phases} ($11$-dim): the
       $T^{12}/U(1)_{\mathrm{global}} = T^{11}$ moduli of relative
       phase rotations among the 12 first-shell wave-vector amplitudes
       (Math77-Q6a-Q6b \S 3).
\item \emph{Higher-charge defect moduli} ($10$-dim, conjecture):
       the audit target of this addendum. We define this rigorously
       below as $\mathrm{SO}(10)/\mathrm{SO}(9)$ with the natural
       $G_{\mathrm{SM}}$-equivariance, equivalently as the
       $\mathbf{10}$-dim vector representation of $\mathrm{SO}(10)$
       considered as a $G_{\mathrm{SM}}$-module.
\end{itemize}

\section{Strategy 2 derivation (Pati--Salam coset bookkeeping)}

\subsection{Lie-algebraic dimension count}

\begin{lemma}[Pati--Salam coset, dimension]
\label{lem:PS-coset}
The codimension of $G_{\mathrm{SM}}$ inside $G_{\mathrm{PS}}$ is
\[
\dim \mathfrak{g}_{\mathrm{PS}} - \dim \mathfrak{g}_{\mathrm{SM}}
\;=\;
(15 + 3 + 3) - (8 + 3 + 1)
\;=\;
21 - 12
\;=\;
9.
\]
\end{lemma}

\begin{proof}
$\dim \mathfrak{g}_{\mathrm{PS}} = \dim \mathfrak{su}(4)_C + \dim \mathfrak{su}(2)_L + \dim \mathfrak{su}(2)_R = 15 + 3 + 3 = 21$.
$\dim \mathfrak{g}_{\mathrm{SM}} = \dim \mathfrak{su}(3)_C + \dim \mathfrak{su}(2)_L + \dim \mathfrak{u}(1)_Y = 8 + 3 + 1 = 12$.
The difference is $21 - 12 = 9$.
\end{proof}

\begin{lemma}[$U(1)_{B-L}$ is independent of the 11 BCC amplitude phases]
\label{lem:BL-indep}
The $U(1)_{B-L}$ phase that arises in the breaking
$\mathrm{SU}(4)_C \to \mathrm{SU}(3)_C \times U(1)_{B-L}$ is not in
the 11-dim $T^{12}/U(1)_{\mathrm{global}}$ orbit of BCC amplitude
relative phases.
\end{lemma}

\begin{proof}
The 11 BCC amplitude phases parametrise $T^{12}/U(1)_{\mathrm{global}} \cong T^{11}$
where the global $U(1)$ is the diagonal phase rotation acting as
$A_j \mapsto e^{i\theta} A_j$ on all 12 first-shell amplitudes.
This action is in the centre of the BCC amplitude algebra and is the
unique $U(1)$ that commutes with the full $O_h$ point-group action.

The $U(1)_{B-L}$ phase, by contrast, is the phase associated with the
generator
\[
T_{B-L} \;=\; \frac{1}{\sqrt{6}}\,\mathrm{diag}\!\bigl(\tfrac{1}{3},\,\tfrac{1}{3},\,\tfrac{1}{3},\,-1\bigr)
\;\in\;\mathfrak{su}(4)_C
\]
(in the standard normalisation where the colour $\mathrm{SU}(3)_C \subset \mathrm{SU}(4)_C$
embeds into the upper $3\times 3$ block). This generator distinguishes
the colour-triplet quarks from the colour-singlet leptons; it does not
permute or rephase the 12 first-shell amplitudes (which all transform
in $\mathrm{SU}(4)_C$-colour-singlet representations of the BCC
amplitude algebra). Hence $T_{B-L}$ is in the centraliser of the BCC
amplitude algebra inside $G_{\mathrm{PS}}$ but is independent of the
diagonal BCC amplitude phase.
\end{proof}

\begin{theorem}[Strategy 2 count]
\label{thm:strat2}
The defect-moduli contribution required to extend the
$24 + 11 = 35$-dim moduli space of $G_{\mathrm{SM}}$ gauge fields
plus BCC amplitude phases into a $45$-dim moduli space compatible with
$\mathrm{SO}(10)$ is
\[
\dim \mathfrak{g}_{\mathrm{PS}} / \mathfrak{g}_{\mathrm{SM}}
\;+\;
\dim \mathfrak{u}(1)_{B-L}
\;=\;
9 + 1
\;=\;
10.
\]
\end{theorem}

\begin{proof}
The Pati-Salam embedding $G_{\mathrm{SM}} \hookrightarrow G_{\mathrm{PS}}
\hookrightarrow \mathrm{SO}(10)$ contributes $9$ (Lemma~\ref{lem:PS-coset}).
The $U(1)_{B-L}$ is independent (Lemma~\ref{lem:BL-indep}) and adds $1$.
The sum is $10$.
\end{proof}

\section{Strategy 3 derivation (\texorpdfstring{$\mathrm{SO}(10)$}{SO(10)} vector representation)}

\subsection{The defining 10-dim vector representation of \texorpdfstring{$\mathrm{SO}(10)$}{SO(10)}}

\begin{lemma}[Coset $\mathrm{SO}(10)/\mathrm{SO}(9)$]
\label{lem:so10-so9}
\[
\dim \bigl(\mathrm{SO}(10) / \mathrm{SO}(9)\bigr)
\;=\;
\dim \mathfrak{so}(10) - \dim \mathfrak{so}(9)
\;=\;
45 - 36
\;=\;
9.
\]
The coset is diffeomorphic to $S^9$ (the unit sphere in $\mathbb{R}^{10}$).
\end{lemma}

\begin{proof}
The coset $\mathrm{SO}(10)/\mathrm{SO}(9)$ is the orbit of any unit
vector $v_0 \in \mathbb{R}^{10}$ under the natural $\mathrm{SO}(10)$
action; the stabiliser is the $\mathrm{SO}(9)$ that fixes $v_0$ and
acts on $v_0^{\perp} \cong \mathbb{R}^9$. The orbit is the unit sphere
$S^9$, which has real dimension $9$. The Lie-algebra count
$45 - 36 = 9$ confirms.
\end{proof}

\begin{remark}[Why $9$ not $10$ from the coset alone]
The coset $\mathrm{SO}(10)/\mathrm{SO}(9)$ is the unit sphere $S^9$.
The full $10$-dim vector $\mathbb{R}^{10}$ is recovered by taking the
\emph{cone} over $S^9$, i.e.\ adding the radial $U(1)$-like degree
of freedom (positive real scaling of the unit vector). In the
defect-moduli interpretation, this radial degree corresponds to the
\emph{magnitude} of the higher-charge disclination configuration ---
the ``how strong is the defect'' parameter --- which is independent
of the orientational $9$-dim sphere data. We make this explicit:
\end{remark}

\begin{lemma}[Cone over $S^9$ has dim 10]
\label{lem:cone}
The cone $C(S^9) = (S^9 \times \mathbb{R}_{\ge 0}) / (S^9 \times \{0\})$
has real dimension
\[
\dim C(S^9) \;=\; \dim S^9 + 1 \;=\; 9 + 1 \;=\; 10.
\]
The interior $C(S^9) \setminus \{0\} \cong \mathbb{R}^{10} \setminus \{0\}$
is the punctured $\mathrm{SO}(10)$-vector representation
$\mathbf{10}_{\mathrm{vec}} \setminus \{0\}$.
\end{lemma}

\begin{proof}
$\dim$ of a cone over a manifold of dimension $d$ is $d+1$ at every
non-apex point; the apex is a single point. The interior bijection with
$\mathbb{R}^{10} \setminus \{0\}$ is via $(s, r) \mapsto r\,s$ for
$s \in S^9$, $r > 0$.
\end{proof}

\begin{theorem}[Strategy 3 count]
\label{thm:strat3}
The defect-moduli contribution at the level of the $\mathrm{SO}(10)$
vector-representation interpretation is
\[
\dim \mathbf{10}_{\mathrm{vec}}
\;=\;
\dim \bigl(\mathrm{SO}(10)/\mathrm{SO}(9)\bigr)
\;+\;
\dim (\text{radial / magnitude})
\;=\;
9 + 1
\;=\;
10.
\]
\end{theorem}

\begin{proof}
Combine Lemma~\ref{lem:so10-so9} (orientational $S^9$ piece, dimension 9)
with Lemma~\ref{lem:cone} (radial magnitude, dimension 1).
\end{proof}

\section{Mutual corroboration of Strategies 2 and 3}

\begin{theorem}[Strategy 2 = Strategy 3]
\label{thm:agree}
The two derivations are mutually corroborating: they yield the same
count $10$ via the same $9 + 1$ decomposition, and the two $9$-dim
pieces ($\mathfrak{g}_{\mathrm{PS}}/\mathfrak{g}_{\mathrm{SM}}$ in
Strategy 2; $\mathrm{SO}(10)/\mathrm{SO}(9)$ in Strategy 3) are
isomorphic as $G_{\mathrm{SM}}$-modules under the standard
Pati-Salam embedding.
\end{theorem}

\begin{proof}
The 9-dim coset $\mathfrak{g}_{\mathrm{PS}}/\mathfrak{g}_{\mathrm{SM}}$
decomposes under $G_{\mathrm{SM}}$ as
\[
\mathfrak{g}_{\mathrm{PS}}/\mathfrak{g}_{\mathrm{SM}}
\;=\;
(\mathbf{3},\mathbf{1})_{-2/3} \;\oplus\; (\overline{\mathbf{3}},\mathbf{1})_{+2/3}
\;\oplus\; (\mathbf{1},\mathbf{1})_0
\;=\;
3 + 3 + \dots
\]
which sums to $9$ real dimensions accounting for the leptoquark gauge
bosons in the Pati-Salam $G_{\mathrm{PS}} \to G_{\mathrm{SM}}$ breaking.

The 9-dim coset $\mathrm{SO}(10)/\mathrm{SO}(9) \cong S^9$ has its
$G_{\mathrm{SM}}$-equivariant decomposition pulled back via the
Pati-Salam embedding $G_{\mathrm{PS}} \hookrightarrow \mathrm{SO}(10)$.
Under this embedding, the $\mathbf{10}_{\mathrm{vec}}$ of
$\mathrm{SO}(10)$ decomposes as $(\mathbf{6}, \mathbf{1}) \oplus
(\mathbf{1}, \mathbf{2}, \mathbf{2})$ under $G_{\mathrm{PS}}$, whose
sub-pieces under $G_{\mathrm{SM}}$ further decompose into the same
$(\mathbf{3},\mathbf{1})_{-2/3} \oplus (\overline{\mathbf{3}},\mathbf{1})_{+2/3}$
together with the $(\mathbf{1},\mathbf{2})_{\pm 1/2}$ Higgs-doublet
content. Restricting to the orientational $9$-dim piece (i.e.\ taking
the unit-vector orbit) gives the $9$-dim $\mathrm{SO}(10)/\mathrm{SO}(9)$
piece, which under $G_{\mathrm{SM}}$ matches the
$\mathfrak{g}_{\mathrm{PS}}/\mathfrak{g}_{\mathrm{SM}}$ decomposition
modulo a basis change.

The $1$-dim radial piece (Strategy 3) corresponds to the
$U(1)_{B-L}$ generator $T_{B-L}$ (Strategy 2), since both are the
$G_{\mathrm{SM}}$-singlet that distinguishes vector magnitude
(Strategy 3) from the $B-L$ phase (Strategy 2) --- and these are the
same generator after pull-back through the Pati-Salam embedding.

Hence Strategies 2 and 3 yield the same count and the same decomposition.
\end{proof}

\section{Combined theorem}

\begin{theorem}[Q6a 10-defect-moduli identity, Lie-algebraic level]
\label{thm:Q6a-main}
Under the Pati-Salam embedding
$G_{\mathrm{SM}} \hookrightarrow G_{\mathrm{PS}} \hookrightarrow \mathrm{SO}(10)$,
the defect-moduli contribution required to complete the dimension chain
$24 + 11 + (\text{defect}) = 45$ is uniquely
\[
\dim (\text{defect})
\;=\;
\dim \bigl(\mathrm{SO}(10)/\mathrm{SO}(9)\bigr)
\;+\;
\dim U(1)_{B-L}
\;=\;
9 + 1
\;=\;
\dim \mathbf{10}_{\mathrm{vec}}
\;=\;
10.
\]
The two derivations (Strategy 2: Pati-Salam coset $9$ + $U(1)_{B-L}$ singlet $1$;
Strategy 3: $\mathrm{SO}(10)$-vector $\mathbf{10} = S^9 \oplus \mathbb{R}_{> 0}$)
are mutually corroborating: they yield the same count, the same $9 + 1$
decomposition, and they agree on the $G_{\mathrm{SM}}$-equivariant
sub-content.
\end{theorem}

\begin{proof}
Combine Theorem~\ref{thm:strat2} (Strategy 2),
Theorem~\ref{thm:strat3} (Strategy 3), and Theorem~\ref{thm:agree}
(equivalence). The uniqueness is in the sense that any
$\mathrm{SO}(10)$-equivariant complement to
$24 + 11 = 35$ inside $\dim \mathfrak{so}(10) = 45$ must have dimension
$10$, and the natural realisation is the $\mathbf{10}_{\mathrm{vec}}$
(equivalently, $\mathrm{SO}(10)/\mathrm{SO}(9)$ plus a magnitude
direction); both decompositions agree on the $G_{\mathrm{SM}}$-content.
\end{proof}

\begin{corollary}[Q6a dimension chain closed at the Lie-algebraic level]
\[
\boxed{
\underbrace{24}_{\text{Math75-Q3}}
\;+\;
\underbrace{11}_{\text{BCC amplitude phases}}
\;+\;
\underbrace{10}_{\text{Theorem~\ref{thm:Q6a-main}}}
\;=\;
\underbrace{45}_{\dim \mathrm{SO}(10)}.
}
\]
\end{corollary}

\section{Honest scope: what this addendum does NOT prove}

The Lie-algebraic theorem above identifies the unique 10-dim
$\mathrm{SO}(10)$-equivariant moduli component but does \emph{not}
prove that the BCC defect-bundle topology realises this 10-dim coset
in the equivariant sense required for full Q6a closure. Specifically:

\begin{itemize}
\item The identification $\mathrm{SO}(10)/\mathrm{SO}(9) \cong S^9$ is
       Lie-algebraic. The corresponding TECT statement
       ``$\pi_1(M_{\mathrm{BCC}})$ admits an $\mathrm{SO}(10)$-equivariant
       quotient whose orbit space is $\mathbf{10}_{\mathrm{vec}}$''
       remains conjecture (Strategy 1 of Math80 \S 4, deferred).
\item The radial / magnitude $U(1)_{B-L}$ identification is
       group-theoretically clean (Lemma~\ref{lem:BL-indep}); the
       corresponding TECT statement ``the radial higher-charge
       disclination strength corresponds to the $U(1)_{B-L}$ phase
       of the embedded $\mathrm{SO}(10)$'' is still subject to a
       physical-realisation check via the BCC defect-bundle structure.
\end{itemize}

The combined effect: \textbf{Q6a is now half-closed} (Lie-algebraic
half: theorem; topological-equivariance half: still partial-advanced).
Pillar 6 status remains \texttt{PARTIAL-ADVANCED}; the upgrade is
limited to removing the principal audit-flagged defect of
\texttt{Math77-Q6a-Q6b-closure} (the conjecture $9 + 1 = 10$ is now
a theorem).
