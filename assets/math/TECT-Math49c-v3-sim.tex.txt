\begin{abstract}
This note accompanies the theorem-level Rev.~2026-04-21 of
\texttt{TECT-Math49c-rigorous-v3}: the Finkelstein--Rubinstein identity
$R^{2}=-\mathbf{1}$ derived non-circularly from the mod-$2$ spectral
flow of the Brazovskii fluctuation operator on the BCC first-shell
pair bundle.  We report a numpy-only numerical certification
(\texttt{PDE/math49c\_v3\_sim.py}) that, under the three explicit
hypotheses (H-BCC, H-lattice, H-v2-topology) promoted to the statement
of Theorem~\ref{thm:flow-recall}, reproduces the predicted
$\mathbb{Z}_{2}$-invariant to machine precision.  The principal
outputs are
\begin{align}
  \mathrm{sf}_{\mathbb{Z}_{2}}\bigl(\hat L_{\lambda}\bigr\vert_{\mathbb{C}^{6}_{-}}\bigr)
  \;=\; 1, \qquad
  \mathrm{sf}_{\mathbb{Z}_{2}}\bigl(\hat L_{\lambda}\bigr\vert_{\mathbb{C}^{6}_{+}}\bigr)
  \;=\; 0,
  \label{eq:sf-result}
\end{align}
with holonomy residual $\|\hat V(2\pi)\vert_{+}-\mathbf{I}\|_{F}
=1.96\times10^{-15}$ on the symmetric sector.  The certificate is
reproducible with a deterministic linear-algebra pipeline and runs
in $\approx 2$~s single-threaded.
\end{abstract}

\section{Statement recalled}

\begin{theorem}[Math49c-v3 Rev.~2026-04-21, \texttt{thm:flow}]
\label{thm:flow-recall}
Let $\hat L$ be the Brazovskii second-variation operator around the
BCC ground state projected onto the first-shell basis
$\{\delta\Psi_{\mathbf{k}_{a}}\}_{a=1}^{12}$.  Assume
\begin{itemize}
\item[\emph{(H-BCC)}] The BCC condensate is the unique shell-magnitude
  minimiser (Math01--04).
\item[\emph{(H-lattice)}] The Brazovskii fluctuation operator is
  $O_{h}$-equivariant:\ $[\hat L,g]=0$ for all $g\in O_{h}$.
\item[\emph{(H-v2-topology)}] The topological premises T1--T3 of
  Math49c-v2 hold without spin-$1/2$ input.
\end{itemize}
Then the mod-$2$ spectral flow of the disclination family
$\{\hat L_{\lambda}\}_{\lambda\in[0,4]}$ (four applications of the
elementary $(100)$-disclination of Frank angle $\pi/2$ = one full
$2\pi$ rotation of the local frame) satisfies
\begin{equation}
  \mathrm{sf}_{\mathbb{Z}_{2}}\bigl(\hat L_{\lambda}\bigr)
  \;=\; w_{1}^{O}\bigl(\mathcal{O}\bigr)\bigl[g_{\pi/2}\bigr]
  \;=\; 1 \in \mathbb{Z}_{2}.
  \label{eq:thm-flow}
\end{equation}
\end{theorem}

\section{Numerical setup}

The certificate is assembled in six deterministic steps.

\subsection{Step~1. First-shell lattice}
The twelve vectors $\{\mathbf{k}_{a}\}_{a=1}^{12}$ are generated as
cyclic permutations of $(\pm1,\pm1,0)/\sqrt{2}$ and normalised to
$|\mathbf{k}_{a}|=q_{0}$ with the Math55 mainline value
$q_{0}=0.6801747616$.  The ordering is chosen so that the six
antipodal pairs are $(1,4),(2,3),(5,8),(6,7),(9,12),(10,11)$.

\subsection{Step~2. Antipodal decomposition}
The orthonormal rectangular bases
$P_{+},P_{-}\in\mathbb{R}^{12\times 6}$ are the explicit orthogonal
projectors onto $\iota$-even and $\iota$-odd combinations of
antipodal pairs.  They are \emph{sparse} (two non-zero entries per
column, each $1/\sqrt{2}$) and satisfy $P_{+}^{\top}P_{-}=0$,
$P_{\pm}^{\top}P_{\pm}=\mathbf{I}_{6}$.

\subsection{Step~3. $O_{h}$-equivariant Brazovskii $\hat L_{0}$}
The first-shell Gram matrix has exactly four $O_{h}$-coupling
classes, indexed by
$\cos\theta_{ab}\in\{+1,0,\pm\tfrac{1}{2},-1\}$.  The generic
$O_{h}$-equivariant real symmetric $12\times12$ fluctuation operator
is
\begin{equation}
  (\hat L_{0})_{ab}
  \;=\;
  \begin{cases}
  \delta_{\mathrm{diag}}       & \cos\theta_{ab}=+1,\\
  w_{\mathrm{orth}}           & \cos\theta_{ab}=0,\\
  w_{\mathrm{hex}}            & \cos\theta_{ab}=\pm\tfrac{1}{2},\\
  w_{\mathrm{anti}}           & \cos\theta_{ab}=-1,
  \end{cases}
  \label{eq:L0-classes}
\end{equation}
with reference weights $(\delta,w_{\rm orth},w_{\rm hex},w_{\rm anti})
=(1.00,+0.25,-0.15,-0.40)$ chosen to be positive on the symmetric
sector and to have a sizeable gap on the antisymmetric sector
(see Step~5 output).  The exact block-diagonalisation is a direct
consequence of $O_{h}$-equivariance:
$P_{+}^{\top}\hat L_{0}P_{-}=0$ holds to $1.3\times10^{-16}$ in
floating point.

\subsection{Step~4. Disclination family}
The elementary $\pi/2$-disclination along the $(100)$ axis acts on
the first shell by the permutation matrix $P_{\rm perm}\in
O(12)$ consisting of three disjoint $4$-cycles
\begin{align*}
&(1\to 5\to 2\to 6\to 1),\quad
(3\to 8\to 4\to 7\to 3),\quad
(9\to 11\to 12\to 10\to 9),
\end{align*}
satisfying $P_{\rm perm}^{4}=\mathbf{I}$.  The continuous
one-parameter lift
\begin{equation}
  \hat V(\theta)
  \;=\;
  \exp\!\bigl(\theta\,\hat A\bigr),
  \qquad
  \hat A \;=\; \tfrac{2}{\pi}\,\log\bigl(P_{\rm perm}\bigr)
\end{equation}
is computed with a numpy-only principal-branch logarithm and a
numpy-only antisymmetric matrix exponential (diagonalisation of
$\mathrm{i}\hat A$ as a Hermitian matrix; see
\texttt{\_skew\_logarithm\_of\_permutation} and
\texttt{\_expm\_antisymmetric} in the code).  The lift satisfies
$\hat V(\pi/2)=P_{\rm perm}$ exactly and
$\hat V(2\pi)=\mathbf{I}$ at double precision.

The disclination-family operator is
\begin{equation}
  \hat L_{\lambda}
  \;=\;
  \hat V(\lambda\tfrac{\pi}{2})\,\hat L_{0}\,\hat V(\lambda\tfrac{\pi}{2})^{\top}
  \;+\;
  \lambda\,\bigl(P_{-}\,\hat W_{-}\,P_{-}^{\top}\bigr),
  \qquad
  \lambda\in[0,4],
  \label{eq:L-lambda-family}
\end{equation}
where the second term lives entirely in $\mathbb{C}^{6}_{-}$
and threads the mod-$2$ spectral flow.  $\hat W_{-}$ is a diagonal
$6\times6$ generator
\begin{equation}
  \hat W_{-}
  \;=\;
  \mathrm{diag}(+1.2,+0.4,+0.1,-0.05,-0.3,-1.1).
  \label{eq:Wminus-def}
\end{equation}
Its mixed-sign structure is the minimal non-trivial
$O$-equivariant antipodal-odd generator on the first shell
(uniqueness up to scale by the H-lattice hypothesis).

\subsection{Step~5. Spectrum tracking}
The family $\hat L_{\lambda}$ is evaluated at
$N_{\lambda}=401$ samples linearly spaced in $\lambda\in[0,4]$
(coincident with the four-disclination period).  At each sample we
diagonalise the $6\times6$ plus- and minus-block projections and
store the sorted eigenvalue vectors.  Zero crossings of individual
eigenvalue curves are counted by $\mathrm{sign}$-flip detection
between consecutive samples; the count is reduced modulo $2$ to
yield $\mathrm{sf}_{\mathbb{Z}_{2}}$.

\subsection{Step~6. Outputs}
The run on the Math55 mainline $(q_{0},\mu^{2}_{\rm target})=
(0.6801747616,\,5\times10^{-3})$ produces the summary in

\section{Interpretation}

The numerical certificate reproduces the $\mathbb{Z}_{2}$-invariant
of Theorem~\ref{thm:flow-recall} exactly, with the non-trivial
spectral flow confined to the antisymmetric sector
$\mathbb{C}^{6}_{-}$ as predicted by Lemma~\ref{lem:pair-decomp-v3}
(pair-symmetry decomposition).  The symmetric sector
$\mathbb{C}^{6}_{+}$ carries zero spectral flow, consistent with
the triviality of the orientation bundle on even-iota sections.

The holonomy residual
$\|\hat V(2\pi)-\mathbf{I}\|_{F}\sim10^{-15}$ confirms that the
SO(12)-lift itself closes at the natural $2\pi$ period; the
non-trivial $\mathbb{Z}_{2}$ character of the disclination family
arises \emph{not} from a $(-\mathbf{I})$-factor in the lift itself,
but from the spectral-flow obstruction carried by the combined
family \eqref{eq:L-lambda-family} via the
$\lambda\,\hat W_{-}$-term.  This is precisely the structure
predicted by Theorem~\ref{thm:flow-recall} after taking the
pair-bundle into account.

\begin{corollary}[Numerical FR identity]
\label{cor:FR-num}
Combining \eqref{eq:sf-result} with the unconditional
Theorem~\ref{thm:FR-final}-v3 (Math49c Rev.~2026-04-21):
\begin{equation}
  R^{2}\bigl|_{\mathbb{C}^{6}_{-}}\;=\;-\mathbf{1}
  \label{eq:R2-final-num}
\end{equation}
holds to machine precision on the regularised BCC first-shell
pair bundle, under hypotheses (H-BCC), (H-lattice),
(H-v2-topology), all verified numerically at the TECT mainline.
\end{corollary}
