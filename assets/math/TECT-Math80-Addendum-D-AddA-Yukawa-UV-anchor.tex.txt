\section{RG running from $M_{\rm GUT}$ to $M_Z$}

\subsection{Yukawa $\beta$-functions in the SM}

At the GUT scale, all Yukawa couplings are equal: $Y_t(M_{\rm GUT}) = Y_b(M_{\rm GUT})
= Y_\tau(M_{\rm GUT}) = Y_{\rm SO(10)}^{\rm TECT} \approx 0.27$. As the energy scale
runs downward to the electroweak scale, the SM 1-loop Yukawa $\beta$-functions
govern the evolution:

\begin{equation}
16\pi^2 \frac{dY_t}{dt} = Y_t \left( 4.5 Y_t^2 + 1.5 Y_b^2 + Y_\tau^2
- 8 g_3^2 - \frac{9}{4} g_2^2 - \frac{17}{12} g_1^2 \right),
\label{eq:betat}
\end{equation}

\begin{equation}
16\pi^2 \frac{dY_b}{dt} = Y_b \left( 4.5 Y_b^2 + 1.5 Y_t^2 + Y_\tau^2
- 8 g_3^2 - \frac{9}{4} g_2^2 - \frac{1}{12} g_1^2 \right),
\label{eq:betab}
\end{equation}

\begin{equation}
16\pi^2 \frac{dY_\tau}{dt} = Y_\tau \left( 3 Y_\tau^2 + 4.5 Y_\tau^2 + 1.5 Y_\tau^2
- \frac{9}{4} g_2^2 - \frac{9}{4} g_1^2 \right),
\label{eq:betatau}
\end{equation}

where $t = \ln \mu$ and $(g_3, g_2, g_1)$ are the SU(3), SU(2), U(1) gauge couplings.

\subsection{Running of gauge couplings}

The gauge couplings run independently under their own $\beta$-functions:
\begin{align}
16\pi^2 \frac{dg_3}{dt} &= \left( -11 + \frac{2n_f}{3} \right) g_3^3,\\
16\pi^2 \frac{dg_2}{dt} &= \left( -\frac{22}{3} + \frac{4n_f}{3} \right) g_2^3,\\
16\pi^2 \frac{dg_1}{dt} &= \left( \frac{41}{10} + \frac{2n_f}{15} \right) g_1^3,
\end{align}
with $n_f = 3$ (three fermion families below the GUT scale).

\subsection{Observation: dominant running is the top Yukawa}

\begin{observation}
The top Yukawa $Y_t$ dominates the RG evolution because (i) it is the
largest of the three Yukawa couplings at all scales, and (ii) the strong
and electroweak gauge couplings pump negative $\beta$-function contributions.
The net effect is that $Y_t$ decreases slowly as the scale runs downward,
while $Y_b$ and $Y_\tau$ run more steeply (they are suppressed by the absence
of the large $4.5 Y_t^2$ term).
\end{observation}
