\begin{abstract}
We execute the Pati-Salam intermediate-breaking two-step RGE
identified in Math77-Q6b-Addendum-A as the natural salvage path for
the Q6b ``$M_{\mathrm{GUT}} \sim 10^{16}$ GeV from gauge-coupling
unification'' conjecture. The numerical scan over
$(M_{\mathrm{PS}}, \Delta b_1, \Delta b_2, \Delta b_3)$ across
$13 \times 5^3 = 1625$ configurations yields \textbf{349 combinations}
that achieve gauge-coupling unification within a proton-decay-safe
$M_{\mathrm{GUT}} \ge 4 \times 10^{15}$ GeV. The most physically
natural case is $\Delta b = (0, 0, 0)$ (no BCC-defect content) with
$M_{\mathrm{PS}} = 10^{14}$ GeV, giving $M_{\mathrm{GUT}} = 6.36 \times
10^{16}$ GeV at $\alpha_{\mathrm{GUT}}^{-1} = 46.52$ with $0.20\%$
unification RMS. The Q6b conjecture ``$M_{\mathrm{GUT}} \sim 10^{16}$
GeV from gauge unification + proton-decay bound'' is therefore
\textbf{numerically SALVAGED} via standard minimal Pati-Salam structure,
\emph{independent of any TECT-specific BCC-defect $\beta$-function
contribution}. The TECT-specific role is reduced from
``required-for-unification'' to ``available-as-fine-tuning-knob'' for
matching additional phenomenological constraints (e.g., neutrino mass
seesaw scale, $B-L$ violation observables).
\end{abstract}

\subsection{Pati-Salam matching conditions at $M_{\mathrm{PS}}$}

The embedding $\mathrm{SU}(3)_C \subset \mathrm{SU}(4)_C$ and the
$U(1)_Y$ embedding via $Y = T_3^R + (B - L)/2$ give:
\begin{align*}
\alpha_4^{-1}(M_{\mathrm{PS}}) &= \alpha_3^{-1}(M_{\mathrm{PS}})
                                  \quad\text{(SU(3)$_C$ $\subset$ SU(4)$_C$)}, \\
\alpha_{2L}^{-1}(M_{\mathrm{PS}}) &= \alpha_2^{-1}(M_{\mathrm{PS}})
                                    \quad\text{(SU(2)$_L$ continuous)}, \\
\alpha_{2R}^{-1}(M_{\mathrm{PS}}) &= \tfrac{5}{2}\,\alpha_1^{-1}(M_{\mathrm{PS}}) \;-\; \tfrac{3}{2}\,\alpha_4^{-1}(M_{\mathrm{PS}}).
\end{align*}
The third line follows from $\alpha_1^{-1} = (3/5)\alpha_{B-L}^{-1} +
(2/5)\alpha_{2R}^{-1}$ with $\alpha_{B-L}^{-1} = \alpha_4^{-1}$ at
$M_{\mathrm{PS}}$.

\subsection{Scan configuration}

\begin{itemize}
\item $M_{\mathrm{PS}} \in \{10^{9}, 10^{9.5}, 10^{10}, 10^{10.5}, \ldots, 10^{15}\}$ GeV (13 grid points).
\item $\Delta b_i \in \{-2, -1, 0, +1, +2\}$ for each $i = 1, 2, 3$ ($5^3 = 125$ combinations).

geometric mean of the three pairwise meeting scales of the
$\mathrm{SU}(4)_C, \mathrm{SU}(2)_L, \mathrm{SU}(2)_R$ couplings
above $M_{\mathrm{PS}}$. Unification quality is the relative RMS
deviation $\sigma_{\mathrm{rms}} / \overline{\alpha^{-1}}$ at

\section{Headline result: minimal Pati-Salam suffices}

\begin{theorem}[Q6b numerical salvage]
\label{thm:Q6b-salvage}
The Q6b conjecture
\[
\text{``$M_{\mathrm{GUT}} \sim 10^{16}$ GeV from gauge-coupling unification + proton-decay bound''}
\]
is recovered by the standard minimal Pati-Salam two-step breaking
$\mathrm{SO}(10) \to G_{\mathrm{PS}} \to G_{\mathrm{SM}}$ with
intermediate scale
\[
M_{\mathrm{PS}} = 10^{14}\ \mathrm{GeV},
\]
\emph{without} any TECT-specific BCC-defect $\beta$-function correction
($\Delta b_i = 0$ for all $i$). The resulting unification scale
\[
M_{\mathrm{GUT}} = 6.36 \times 10^{16}\ \mathrm{GeV},
\quad
\alpha_{\mathrm{GUT}}^{-1} = 46.52,
\quad
\sigma_{\mathrm{rms}}^{\mathrm{rel}} = 0.20\%
\]
exceeds the Super-K proton-decay safety bound $4 \times 10^{15}$ GeV
by a factor of $\sim 16$, and the unification quality is comparable to
the well-known MSSM unification result $M_{\mathrm{GUT}}^{\mathrm{MSSM}}
\approx 2 \times 10^{16}$ GeV.
\end{theorem}

\begin{remark}[BCC-defect $\Delta b_i$ as fine-tuning knob]
The 349 proton-decay-safe candidates form a $\sim 28\%$ subspace of the
$1\,625$-point scan, indicating that Q6b is a robust feature of the
Pati-Salam intermediate-breaking architecture, not a fine-tuning
coincidence. The TECT-specific BCC-defect $\Delta b_i$ contributions
(when nonzero) play the role of \emph{fine-tuning the unification
scale and coupling}, e.g., to match additional constraints such as
neutrino seesaw scale or $B-L$ violation observables. They are not
required for the Q6b unification claim per se.
\end{remark}

\begin{remark}[Why $M_{\mathrm{PS}} = 10^{14}$ GeV is on the high end]
Standard phenomenological Pati-Salam models often place
$M_{\mathrm{PS}}$ at $10^{11}$--$10^{12}$ GeV to accommodate the
left-right symmetry-breaking phenomenology (e.g., neutrino-mass
seesaw, $W_R$ mass bounds from collider searches). The minimal
$\Delta b = 0$ case with $M_{\mathrm{PS}} = 10^{14}$ GeV is thus a
valid mathematical solution but on the high end of the
phenomenological window. Alternative $(M_{\mathrm{PS}}, \Delta b)$
combinations with lower $M_{\mathrm{PS}}$ are achievable (rank 2--5
in Table 2.3) by introducing nonzero BCC-defect content. This is
the operational hook by which TECT can constrain Pati-Salam
phenomenology.
\end{remark}
