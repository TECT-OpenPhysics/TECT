\subsection*{1. Ingredient 1 -- $\mathbb{CP}^{2}$ order manifold from BCC
$\{110\}$ star}

\paragraph{Claim.} The reduced internal-amplitude space of one
coherent BCC condensation cell is $\mathbb{CP}^{2}$.

\paragraph{Argument.} The BCC $\{110\}$ reciprocal star contains six
primitive vectors $\pm\mathbf{Q}_{1},\pm\mathbf{Q}_{2},\pm\mathbf{Q}_{3}$

condensate amplitude).  The associated coherent amplitudes form a
complex 3-vector $z=(z_{1},z_{2},z_{3})\in\mathbb{C}^{3}$ with the
BCC lock-condition $|z|^{2}=1$.  The global phase $z\to e^{i\chi}z$
is physically redundant.  Hence
$$
\mathcal{P}_{\rm cell}\;=\;S^{5}/U(1)\;=\;\mathbb{CP}^{2}.
$$

\paragraph{Devil's-advocate (\S 6.3.1)}
\begin{itemize}

   not three.  \textbf{Verdict: VALID with mitigation.}  The folding
   $\mathbf{Q}\leftrightarrow-\mathbf{Q}$ is correct for a real

   to leading order.  Higher-order anisotropy lifts $\mathbb{CP}^{2}$

\end{itemize}

\subsection*{3. Ingredient 3 -- Spin-2 elastic $\to$ EH matching with
explicit $16\pi$}

\paragraph{Claim.}
$$
\rho_{\rm cond}\;=\;\frac{c^{4}}{16\pi G\,a_{\rm BCC}^{2}},
\qquad
a_{\rm BCC}\;=\;4\sqrt{\pi}\,\ell_{\rm P}.
$$

\paragraph{Argument.}  The traceless symmetric strain
$h_{ij}=\partial_{i}u_{j}+\partial_{j}u_{i}-(2/3)\delta_{ij}\partial_{k}u_{k}$
identifies with the spin-2 metric perturbation.  Coarse-graining the
elastic shear-energy density $\mu \varepsilon_{ij}^{T}\varepsilon_{T}^{ij}$
over one BCC cell yields a Fierz--Pauli kinetic term
$$
S_{2}^{\rm el}
=
\frac{\mu a_{\rm BCC}^{2}}{4}\int d^{4}x\,\partial_{\alpha}h_{\mu\nu}^{TT}\partial^{\alpha}h_{TT}^{\mu\nu}.
$$
Matching to the linearised EH action at TT gauge,
$S_{\rm EH}^{(2)}|_{TT}=\frac{c^{3}}{64\pi G}\int\partial h \partial h$,
gives
$$
\mu\,a_{\rm BCC}^{2}=\frac{c^{3}}{16\pi G}.
$$
Using the relativistic stiffness identification $\mu=\rho_{\rm cond}\cdot c^{2}$,
$$
\rho_{\rm cond}\,a_{\rm BCC}^{2}\;=\;\frac{c^{4}}{16\pi G}.
$$
By the Weinberg spin-2 self-consistency theorem (Lorentz invariance +
linear gauge invariance + universal stress-energy coupling
$\Rightarrow$ unique nonlinear EH completion), this matching extends
nonlinearly.

\paragraph{Combining with Math98 main formula.}
$\hbar_{\rm eff}=\rho_{\rm cond}a_{\rm BCC}^{4}/c$ together with
$\rho_{\rm cond}=c^{4}/(16\pi G a_{\rm BCC}^{2})$ gives
$$
\hbar_{\rm eff}\;=\;\frac{c^{3}a_{\rm BCC}^{2}}{16\pi G},
\qquad
a_{\rm BCC}^{2}\;=\;\frac{16\pi G \hbar}{c^{3}},
\qquad
a_{\rm BCC}\;=\;4\sqrt{\pi}\,\ell_{\rm P}\approx 7.09\,\ell_{\rm P}.
$$
