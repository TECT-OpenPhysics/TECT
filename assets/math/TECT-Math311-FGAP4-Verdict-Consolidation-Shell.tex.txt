\section{Theorem 311.1 statement}

\begin{theorem}[F-GAP4 verdict-consolidation determinism]
\label{thm:consol}
The F-GAP4-DEFECT-MASS verdict $V \in \{\text{PASS}, \text{FAIL}, \text{DEFER}\}$
arriving 2026-05-14 maps to a unique canonical-record update
$U(V)$ specified in §\ref{sec:outcomes}, comprising:
(i) Pillar 4 sub-task 2 status update,
(ii) Pillar 4 atomic composite update (carve-out per Math302),
(iii) Stage-2 composite update,
(iv) Stage-1 scoreboard update,
(v) Math246 contingency activation (FAIL only),
(vi) Math311-AddA-{PASS,FAIL,DEFER} archive note creation,
(vii) Math305 $c_2$ followup trigger (PASS only).
The map $V \mapsto U(V)$ is functional, not relational; exactly one
update fires per verdict.
\end{theorem}

\begin{proof}
The three outcomes are mutually exclusive (Math300 Definition).
Each outcome's canonical-record update is enumerated in §\ref{sec:outcomes}
without overlap. Therefore the $V \mapsto U(V)$ mapping is single-valued
and total. \qed
\end{proof}
