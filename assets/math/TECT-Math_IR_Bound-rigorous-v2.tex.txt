\section{Setting}

\subsection{Effective action near the Brazovskii fixed point}

The TECT condensate near the Brazovskii first-order transition is
governed by the Ginzburg--Landau free energy
\begin{equation}
  \mathcal{F}[\Psi]
   = \int d^3x\,\Bigl[
      \tfrac{r}{2}|\Psi|^2
    + \tfrac{Z}{2}|\nabla\Psi|^2
    + \tfrac{Y}{2}|\nabla^2\Psi|^2
    + \tfrac{\lambda}{4}|\Psi|^4
    + \tfrac{\gamma}{6}|\Psi|^6\Bigr],
  \label{eq:F-brazovskii}
\end{equation}
with locked parameters
$(\mu^2,\lambda,\gamma) = (0.26,-0.43,1.62)$ and shell wavevector
$q_0 = \sqrt{-Z/(2Y)} = 0.6802$ at $Y = 1.0$.

\subsection{$O_h$-invariant derivative expansion}

At fourth order in derivatives, the two independent
$O_h$-invariant operators are
\begin{align}
  \mathcal{O}^{(\mathrm{iso})}_4
    &= (\nabla^2\Psi)^2
     = \bigl(\textstyle\sum_{i=1}^{3}\partial_i^2\Psi\bigr)^{2}, \\
  \mathcal{O}^{(c)}_4
    &= \sum_{i=1}^{3}(\partial_i\Psi)^{4}
     - \tfrac{1}{3}\Bigl(\sum_{i=1}^{3}(\partial_i\Psi)^{2}\Bigr)^{2}.
  \label{eq:Oc4-correct}
\end{align}
Only $\mathcal{O}^{(c)}_4$ is \emph{cubic} and distinguishes $O_h$
from $SO(3)$.

\begin{remark}[Correction to scaffold]
The scaffold Math\_IR\_Bound Eq.(3) defined
$\mathcal{O}^{(c)}_4 := (\partial_i\partial_j\Psi)^2
-\tfrac{1}{3}(\nabla^2\Psi)^2$, a spin-$2$ quadrupole (isotropic
under $SO(3)$). That operator is inherently \emph{isotropic} and
cannot distinguish $O_h$ from $SO(3)$. The correct cubic-$O_h$
invariant is \eqref{eq:Oc4-correct}.
\end{remark}

\section{Canonical dimensions (corrected)}

\subsection{Field dimension at the Gaussian fixed point}

At the Gaussian fixed point with canonical scaling $[\Psi] =
(d-2)/2 = 1/2$ for $d = 3$.

\begin{lemma}[Canonical dimension of $\mathcal{O}^{(c)}_4$]
\label{lem:dim-gauss}
The canonical dimension of the cubic operator
\eqref{eq:Oc4-correct} at the Gaussian fixed point is
\begin{equation}
  [\mathcal{O}^{(c)}_4]
   = 4\cdot[\partial] + 4\cdot[\Psi]
   = 4\cdot 1 + 4\cdot\tfrac{1}{2}
   = 6.
  \label{eq:Oc4-dim}
\end{equation}
The associated coupling $g$ in the effective action
$\int d^3x\,g\,\mathcal{O}^{(c)}_4$ has dimension
\begin{equation}
  [g] = d - [\mathcal{O}^{(c)}_4] = 3 - 6 = -3.
  \label{eq:g-dim}
\end{equation}
Under RG flow with cut-off scale $\mu$, $g(\mu) \sim \mu^{-3}
g(\mu_0) \to 0$ as $\mu \to 0$. Therefore
$\mathcal{O}^{(c)}_4$ is an \emph{irrelevant} (IR-suppressed)
operator at the Gaussian fixed point.
\end{lemma}

\begin{proof}
Field dimension from the kinetic term
$(\partial\Psi)^2 \sim k^{d+2}$ at the Gaussian fixed point gives
$[\Psi] = (d-2)/2$. Four derivatives contribute $4\cdot[\partial] =
4$ and four fields contribute $4\cdot[\Psi] = 2$, summing to
$[\mathcal{O}^{(c)}_4] = 6$. The coupling dimension
\eqref{eq:g-dim} follows from the requirement that
$\int d^d x\,g\,\mathcal{O}$ be dimensionless. Since $[g] < 0$,
$g$ runs to zero in the IR.
\end{proof}

\begin{remark}[Correction to v1]
The v1 Lemma 1 read ``dim $6 - 3 = 3$, i.e.\ relevant operator at
Gaussian fixed point''; this is inverted. The correct logical
chain is: operator dimension $\Delta = 6$ $\Rightarrow$ coupling
dimension $[g] = d - \Delta = -3$ $\Rightarrow$ $g$ shrinks in
the IR $\Rightarrow$ operator is \emph{irrelevant}, which is the
intended conclusion. The scaffold's instinct was right; the v1
derivation path through the inversion was wrong.
\end{remark}

\subsection{Field dimension at the Brazovskii fixed point}
\label{sec:brazovskii-scaling}

\emph{This section is new in v2}, addressing defect N11 of the
devil's-advocate pass 2.

Near the shell minimum $|k| = q_0$, write
$\vec k = q_0\hat r + \delta \vec k$, decomposed into radial
($\delta k_\parallel$) and tangential ($\delta k_\perp$) components
relative to the shell, with $\delta k_\perp$ living on the
$(d-1)$-dimensional tangent space to the sphere $|k|=q_0$
($d - 1 = 2$ in $d = 3$). Expanding the dispersion
$\omega(k) = r + Z k^2 + Y k^4$ around $|k| = q_0$ with
$Z = -2 Y q_0^2$ gives
\begin{align}
  k^2 - q_0^2
  &= 2 q_0\,\delta k_\parallel
     + \delta k_\perp^{2} + \delta k_\parallel^{2}, \\
  \omega(k) - r
  &= Y\,(k^2 - q_0^2)^{2}
  = Y\bigl(2 q_0\,\delta k_\parallel
           + \delta k_\perp^{2} + \cdots\bigr)^{2}.
  \label{eq:braz-disp}
\end{align}
The Brazovskii fixed point balances the two leading terms in the
bracket of \eqref{eq:braz-disp} at
$\delta k_\parallel \sim \delta k_\perp^{2}/(2 q_0)$, which
defines the anisotropic scaling
\begin{equation}
  [\delta k_\perp] = \mu, \qquad
  [\delta k_\parallel] = \mu^{2}.
  \label{eq:braz-aniso}
\end{equation}

\begin{lemma}[Brazovskii canonical scaling]
\label{lem:dim-brazovskii}
At the Brazovskii fixed point with anisotropic scaling
\eqref{eq:braz-aniso}, the effective $d$-dimensional integration
measure carries dimension
\begin{equation}
  [d^3 k] = [\delta k_\perp^{2}][\delta k_\parallel]
           = \mu^{2}\cdot\mu^{2}
           = \mu^{4},
\end{equation}
so the effective Brazovskii dimension is $d_{\mathrm{eff}} = 4$.
The Brazovskii field dimension (with quartic shell dispersion
$\omega \sim Y \delta k_\perp^{4}$ at the shell minimum) is
\begin{equation}
  [\Psi]_B = (d_{\mathrm{eff}} - 4)/2 = 0,
\end{equation}
using the propagator constraint $[\Psi^2][\omega^{-1}] = [d^dk]$
rearranged for a $k^4$-type dispersion. The cubic-anisotropy
operator then has dimension
\begin{equation}
  [\mathcal{O}^{(c)}_4]_B
  = 4\cdot[\delta k]_{\mathrm{eff}} + 4\cdot[\Psi]_B,
\end{equation}
where $[\delta k]_{\mathrm{eff}}$ is a mixed power depending on
whether the derivative is tangential or radial:
$[\partial_\perp] = \mu$, $[\partial_\parallel] = \mu^{2}$. For
cubic $O_h$-invariant combinations built from $\partial_i$ along
cubic axes, which sit in generic directions relative to any one
shell point, the average dimension is
$\langle[\partial_i]\rangle = (2\cdot\mu + \mu^{2})/3$ per
derivative; the leading large-distance ($\mu \to 0$) behaviour
is dominated by $[\partial_i] = \mu$, giving
$[\mathcal{O}^{(c)}_4]_B \geq 4$ and hence coupling dimension
$[g]_B = d_{\mathrm{eff}} - [\mathcal{O}^{(c)}_4]_B \leq 0$.
The operator remains \emph{IR-irrelevant at the Brazovskii fixed
point}, with irrelevance weakened relative to the Gaussian case
by the enhanced $d_{\mathrm{eff}} = 4$.
\end{lemma}

\begin{proof}
The anisotropic scaling \eqref{eq:braz-aniso} is a direct
consequence of the shell minimum of the Brazovskii dispersion.
The engineering-dimension count of a multi-derivative operator
decomposes each $\partial_i$ into its tangential and radial
components relative to each shell point; averaging over the
sphere gives the stated mixed dimension. The IR limit
($\mu \to 0$) is dominated by the smallest dimension among the
component derivatives, namely $[\partial_\perp] = \mu$. Therefore
$[\mathcal{O}^{(c)}_4]_B \geq 4 \cdot \mu \cdot 1 = 4$ (exact
equality requires no field anomalous dimension, i.e.\ Gaussian
Brazovskii), matching
$[g]_B \leq 0$.
\end{proof}

\begin{remark}[Physical interpretation]
The Brazovskii fixed point enhances the effective dimension from
$d = 3$ to $d_{\mathrm{eff}} = 4$ via the quartic shell
dispersion. This is the same mechanism that makes the Brazovskii
transition effectively weakly first-order (Brazovskii, Sov.\ Phys.\
JETP 41 (1975) 85): the enhanced dimension pushes the theory
closer to its upper critical dimension $d_c = 4$. The
cubic-anisotropy operator remains irrelevant, but its IR
suppression is weaker than the Gaussian estimate (power $-3$
replaced by a power between $-2$ and $-3$ depending on the
derivative-anisotropy decomposition).
\end{remark}

\subsection{Suppression from BCC shell structure}

\begin{proposition}[Shell-projection suppression]
\label{prop:shell}
At the Brazovskii fixed point, the condensate $\Psi$ is supported
on a thin shell $|k| \in [q_0-\Delta, q_0+\Delta]$ with width
$\Delta \sim \sqrt{|r|/Y} \ll q_0$. The matrix element of
$\mathcal{O}^{(c)}_4$ between shell modes is suppressed by
\begin{equation}
  \bigl\langle \mathcal{O}^{(c)}_4 \bigr\rangle_{\mathrm{shell}}
   \sim (\Delta/q_0)^{2}\,|\Psi|_{\mathrm{shell}}^{4}.
\end{equation}
This is the genuine BCC-specific small number.
\end{proposition}

\begin{proof}[Sketch]
The matrix element is
\begin{equation}
  \int d^3k_1\cdots d^3k_4\,\Psi(k_1)\cdots\Psi(k_4)\,
  T^{(c)}(k_1,\dots,k_4)\,\delta(k_1+\cdots+k_4),
\end{equation}
with kernel
$T^{(c)}(k_1,\dots,k_4) = \sum_i k_{1i}k_{2i}k_{3i}k_{4i} -
\tfrac{1}{3}(\sum_i k_{1i}k_{2i})(\sum_j k_{3j}k_{4j})$. On the
12 BCC nearest-neighbour wavevectors, $O_h$ symmetry makes this
kernel vanish identically (the cubic cancellation built into
\eqref{eq:Oc4-correct}). Shell-width perturbations
$k_i = \mathbf G_{\alpha_i}+\delta k_i$ produce the
leading $(\delta k/q_0)^{2} \sim (\Delta/q_0)^{2}$ correction.
\end{proof}

\section{1-loop anomalous dimension}

\subsection{Diagrammatic setup}

The anomalous dimension $\eta$ of $\mathcal{O}^{(c)}_4$ at the
Brazovskii fixed point is computed from the 1-loop correction to
the operator's two-point function in the condensate background.
The relevant diagrams are:
(i) self-energy insertion into each external derivative leg;
(ii) vertex correction from $\lambda|\Psi|^4$;
(iii) ``tadpole'' insertion from $\gamma|\Psi|^6$.

\begin{proposition}[1-loop $\eta$ structure]
\label{prop:eta}
At 1-loop, the anomalous dimension of $\mathcal{O}^{(c)}_4$ at
the Brazovskii fixed point is
\begin{equation}
  \eta^{(c)} = a_1\,\lambda + a_2\,\gamma\,|\Psi_0|^{2}
              + O(\lambda^{2}, \lambda\gamma),
\end{equation}
with coefficients
\begin{align}
  a_1 &= \int_{\mathrm{BZ}} \frac{d^3k}{(2\pi)^3}\,
          \frac{T^{(c)}(k,-k,k,-k)}{[\omega(k)]^{2}}, \\
  a_2 &= \int_{\mathrm{BZ}} \frac{d^3k}{(2\pi)^3}\,
          \frac{T^{(c)}(k,-k,k,-k)}{[\omega(k)]^{3}},
\end{align}
both $O_h$-invariant BZ integrals. Numerical evaluation at the
locked Brazovskii parameters is deferred to v3.
\end{proposition}

\begin{remark}
The order of magnitude is
$\eta^{(c)} \sim |\lambda|/(4\pi)^{2}\sim 10^{-3}$. The scaffold
asserted $\eta = +0.02$ without this structure; v2 states the
computable integrals explicitly and flags the numerical value as
pending.
\end{remark}

\section{SME Lorentz-violation bound}

\begin{proposition}[SME bound, conditional]
\label{prop:SME}
Given Proposition~\ref{prop:shell}, the effective SME coefficient
for fermion dispersion at observable wavevector $k_{\mathrm{obs}}$
is bounded by
\begin{equation}
  |\kappa^{(c)}|
  \lesssim (\Delta/q_0)^{2}\cdot(k_{\mathrm{obs}}/q_0)^{2}.
  \label{eq:SME-bound}
\end{equation}
\end{proposition}

\begin{remark}[Order of magnitude]
With $q_0 \sim M_P$, $\Delta/q_0 \sim 10^{-2}$, and
$k_{\mathrm{obs}} \sim 100\,\mathrm{GeV}$ (LHC scale),
\begin{equation}
  |\kappa^{(c)}|
  \lesssim 10^{-4}\cdot 10^{-34}
  = 10^{-38}.
\end{equation}
This is weaker than the scaffold's asserted $10^{-70}$ bound but
still far below current SME observational limits ($10^{-20}$ to
$10^{-30}$). We honestly state: the $10^{-70}$ figure was
incorrect; the derivable bound is $\sim 10^{-38}$.
\end{remark}
