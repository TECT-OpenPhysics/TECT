\section{Left-handed Weyl spectrum per generation}

Following TECT-Math49-rigorous, the flavour bundle
$E_L \to \mathrm{Gr}(2,5)$ contains per generation the
Standard-Model Weyl fermions
\begin{equation}
  \mathcal{F}_1 = \bigl\{ Q_L,\ u_R^c,\ d_R^c,\ L_L,\ e_R^c \bigr\},
  \label{eq:weyl-set}
\end{equation}
written in the conventional \emph{all-left-handed} basis (right-handed
fields charge-conjugated to left-handed). Their $G_{\mathrm{SM}}
= SU(3)_c \times SU(2)_W \times U(1)_Y$ quantum numbers and Weyl
multiplicities follow directly from the branching rule
$\mathbf{5} = (\mathbf{3},\mathbf{1})_{-1/3} \oplus
(\mathbf{1},\mathbf{2})_{+1/2}$ of TECT-Math49-rigorous Eq.(5):
\begin{equation}
  \begin{array}{|c|c|c|c|c|}
  \hline
  \text{Field} & SU(3)_c & SU(2)_W & U(1)_Y & \text{Weyl mult.}\ n_f \\
  \hline
  Q_L   & \mathbf{3} & \mathbf{2} & +1/6  & 3 \times 2 = 6 \\
  u_R^c & \bar{\mathbf{3}} & \mathbf{1} & -2/3  & 3 \\
  d_R^c & \bar{\mathbf{3}} & \mathbf{1} & +1/3  & 3 \\
  L_L   & \mathbf{1} & \mathbf{2} & -1/2  & 2 \\
  e_R^c & \mathbf{1} & \mathbf{1} & +1    & 1 \\
  \hline
  \end{array}
  \label{eq:SM-table}
\end{equation}
Total Weyl multiplicity per generation:
$6 + 3 + 3 + 2 + 1 = 15$, consistent with the $SU(5)$ unification
representation $\overline{\mathbf{5}} \oplus \mathbf{10}$.

\begin{lemma}[Hypercharge derivation]
The hypercharges in \eqref{eq:SM-table} are the unique solutions
to the branching-rule constraints
\begin{equation}
  Y(\mathbf{5}\text{ part of } (\mathbf{3},\mathbf{1})) = -\tfrac{1}{3},
  \qquad
  Y(\mathbf{5}\text{ part of } (\mathbf{1},\mathbf{2})) = +\tfrac{1}{2},
\end{equation}
together with the $\mathbf{10} = \mathbf{5}\wedge\mathbf{5}$
antisymmetric square giving hypercharges from the addition
$Y(f_1 \wedge f_2) = Y(f_1) + Y(f_2)$ under $G_{\mathrm{SM}}$.
Explicitly, from $\mathbf{10} = (\bar{\mathbf{3}},\mathbf{1})_{-2/3}
\oplus (\mathbf{3},\mathbf{2})_{+1/6} \oplus
(\mathbf{1},\mathbf{1})_{+1}$ in the $SU(5)$ GUT embedding
(Georgi--Glashow, PRL 32 (1974) 438), and
$\overline{\mathbf{5}} = (\bar{\mathbf{3}},\mathbf{1})_{+1/3} \oplus
(\mathbf{1},\mathbf{2})_{-1/2}$, one reads off the Standard-Model
hypercharge column of \eqref{eq:SM-table}.
\end{lemma}

\section{The six triangle-anomaly coefficients}

\subsection{General form}

For a chiral gauge theory with left-handed Weyl fields in
representations $R_f$ of the gauge group $G$, the triangle-anomaly
coefficient for a triangle with three gauge-boson legs of types
$T^a, T^b, T^c$ is
\begin{equation}
  \mathcal{A}^{abc}_{[G_1 G_2 G_3]}
   = \sum_{f \in \mathcal{F}} n_f \,
     \mathrm{Tr}_{R_f}\bigl(T^a\{T^b, T^c\}\bigr),
  \label{eq:anom-general}
\end{equation}
with $T^a$ acting in the appropriate factor. For the gravitational
anomaly ($\mathrm{Grav}^2 U(1)$), the trace reduces to the
hypercharge sum
$\sum_f n_f \dim(R_f) Y_f$. For the Abelian cube ($U(1)_Y^3$):
$\sum_f n_f \dim(R_f) Y_f^3$. The Standard-Model group has six
independent anomalies to check per generation:
\begin{equation}
  \mathrm{SU}(3)_c^3,\quad
  SU(2)_W^3,\quad
  U(1)_Y^3,\quad
  SU(3)_c^2 U(1)_Y,\quad
  SU(2)_W^2 U(1)_Y,\quad
  \mathrm{Grav}^2 U(1)_Y.
\end{equation}

\subsection{$SU(3)_c^3$}

Only colour-charged fields contribute: $Q_L$, $u_R^c$, $d_R^c$.
\begin{equation}
  \mathcal{A}[SU(3)_c^3]
   = 6\, A(\mathbf{3}) + 3\,A(\bar{\mathbf{3}}) + 3\,A(\bar{\mathbf{3}}),
\end{equation}
where $A(R)$ is the cubic Casimir index normalised so that
$A(\mathbf{3}) = +1$ and $A(\bar{\mathbf{3}}) = -1$. Using the
Weyl multiplicities $n_f$ of \eqref{eq:SM-table}:
\begin{equation}
  n_{Q_L}\,A(\mathbf{3}) = 6\cdot\bigl(\tfrac{1}{1}\bigr)
   \cdot (+1)\cdot \dim_{SU(2)_W}(\mathbf{2})\ ?
\end{equation}
The cleaner route is to track each Weyl field individually.
Per Weyl field we have: $Q_L$ is colour $\mathbf{3}$ with
multiplicity (over $SU(2)_W$ indices and over the family) $n = 6$
\emph{but the $SU(2)_W$ index is already absorbed into $n_f$},
so as a pure $SU(3)_c$ triplet it contributes $n_f A(\mathbf{3})$
with $n_f^{(\text{colour})} = 2 \cdot 1 = 2$ (the factor $2$ counts
the $SU(2)_W$ doublet index, the factor $1$ counts the generation).

To avoid this subtle double-count, we restate \eqref{eq:SM-table}
factored only over the \emph{colour} index:
\begin{equation}
  \begin{array}{|c|c|c|}
  \hline
  \text{Colour-rep.\ of Weyl} & \text{Colour-weyl count } n_c & A(R_c) \\
  \hline
  Q_L \in \mathbf{3} & 2 \text{ (from SU(2)_W index)}  & +1 \\
  u_R^c \in \bar{\mathbf{3}} & 1 & -1 \\
  d_R^c \in \bar{\mathbf{3}} & 1 & -1 \\
  \hline
  \end{array}
\end{equation}
\begin{equation}
  \boxed{\ \mathcal{A}[SU(3)_c^3] = 2 \cdot (+1) + 1 \cdot (-1)
                                 + 1 \cdot (-1) = 0\ }
  \label{eq:A333}
\end{equation}

\subsection{$SU(2)_W^3$}

The relevant structure is
$\mathrm{Tr}(T^a\{T^b,T^c\}) = \tfrac{1}{2} d^{abc} + \dots$ in
the fundamental. For $\mathfrak{su}(2)$, the totally symmetric
tensor $d^{abc}$ vanishes identically:
\begin{equation}
  d^{abc}_{\mathfrak{su}(2)} \equiv 0.
  \label{eq:dabc-vanish}
\end{equation}
This is the standard group-theoretic fact that $\mathfrak{su}(2)$
(unlike $\mathfrak{su}(N)$ for $N \geq 3$) has no cubic Casimir
invariant, because its representations are all pseudo-real (or
real) and the symmetric product $\mathrm{Sym}^3$ of the adjoint
does not contain a singlet. Hence
\begin{equation}
  \boxed{\ \mathcal{A}[SU(2)_W^3] = 0 \text{ identically,
          for any chiral matter content, by } d^{abc} = 0\ }
  \label{eq:A222}
\end{equation}
\emph{Correction to scaffold Math49b}: the scaffold claimed this
vanishing "follows because $e_R$ is included"; that reasoning is
wrong. $e_R$ is an $SU(2)_W$-singlet and never enters the trace;
the true reason is the group-theoretic identity
\eqref{eq:dabc-vanish}.

\subsection{$U(1)_Y^3$}

\begin{equation}
  \mathcal{A}[U(1)_Y^3] = \sum_f n_f Y_f^3,
\end{equation}
with sum over all Weyl fields with multiplicities $n_f$:
\begin{align}
  \mathcal{A}[U(1)_Y^3] &=
    \underbrace{6\cdot\bigl(\tfrac{1}{6}\bigr)^3}_{Q_L}
  + \underbrace{3\cdot\bigl(-\tfrac{2}{3}\bigr)^3}_{u_R^c}
  + \underbrace{3\cdot\bigl(+\tfrac{1}{3}\bigr)^3}_{d_R^c}
  + \underbrace{2\cdot\bigl(-\tfrac{1}{2}\bigr)^3}_{L_L}
  + \underbrace{1\cdot(+1)^3}_{e_R^c} \nonumber \\
  &= 6\cdot\tfrac{1}{216}
   + 3\cdot\bigl(-\tfrac{8}{27}\bigr)
   + 3\cdot\tfrac{1}{27}
   + 2\cdot\bigl(-\tfrac{1}{8}\bigr)
   + 1 \nonumber \\
  &= \tfrac{1}{36} - \tfrac{8}{9} + \tfrac{1}{9}
   - \tfrac{1}{4} + 1.
\end{align}
Common denominator $36$:
\begin{equation}
  \tfrac{1}{36} - \tfrac{32}{36} + \tfrac{4}{36}
  - \tfrac{9}{36} + \tfrac{36}{36}
  = \tfrac{1 - 32 + 4 - 9 + 36}{36}
  = \tfrac{0}{36} = 0.
\end{equation}
\begin{equation}
  \boxed{\ \mathcal{A}[U(1)_Y^3] = 0\ }
  \label{eq:AYYY}
\end{equation}

\subsection{$SU(3)_c^2 U(1)_Y$}

Only colour-charged fields contribute. The index
$\mathrm{Tr}(T^a T^b)$ in $\mathbf{3}$ is $T(\mathbf{3})
\delta^{ab}$ with $T(\mathbf{3}) = 1/2$ and
$T(\bar{\mathbf{3}}) = 1/2$. The anomaly coefficient, with signs
absorbed into the chirality (all Weyl fields left-handed),
reduces to the hypercharge sum weighted by colour index:
\begin{align}
  \mathcal{A}[SU(3)_c^2 U(1)_Y]
  &\propto \sum_{f \in \text{coloured}} n_f^{(\text{colour})}
           \epsilon_f Y_f \\
  &= 2 \cdot (+1) \cdot \bigl(+\tfrac{1}{6}\bigr)
   + 1 \cdot (-1) \cdot \bigl(-\tfrac{2}{3}\bigr)
   + 1 \cdot (-1) \cdot \bigl(+\tfrac{1}{3}\bigr) \\
  &= \tfrac{1}{3} + \tfrac{2}{3} - \tfrac{1}{3} = \tfrac{2}{3} ?
\end{align}
This cannot be correct; the anomaly must vanish. The resolution
is that the factor $\epsilon_f = \pm 1$ distinguishing
$\mathbf{3}$ and $\bar{\mathbf{3}}$ is already encoded in the
\emph{sign of the hypercharge of the charge-conjugated Weyl
field}: for a right-handed antiquark $u_R$ with hypercharge
$+2/3$, the charge-conjugated left-handed Weyl $u_R^c$ has
hypercharge $-2/3$ and lives in $\bar{\mathbf{3}}$. So with all
Weyl fields left-handed and the trace
$T(R) = +1/2$ for both $\mathbf{3}$ and $\bar{\mathbf{3}}$ (the
sign flip being absorbed in $Y_f$):
\begin{align}
  \mathcal{A}[SU(3)_c^2 U(1)_Y]
  &= \sum_f n_f^{(c)} T(R_f) Y_f \\
  &= 2\cdot\tfrac{1}{2}\cdot(+\tfrac{1}{6})
   + 1\cdot\tfrac{1}{2}\cdot(-\tfrac{2}{3})
   + 1\cdot\tfrac{1}{2}\cdot(+\tfrac{1}{3}) \\
  &= \tfrac{1}{6} - \tfrac{1}{3} + \tfrac{1}{6}
   = \tfrac{1}{6} + \tfrac{1}{6} - \tfrac{1}{3}
   = \tfrac{2}{6} - \tfrac{2}{6} = 0.
\end{align}
\begin{equation}
  \boxed{\ \mathcal{A}[SU(3)_c^2 U(1)_Y] = 0\ }
  \label{eq:A33Y}
\end{equation}

\subsection{$SU(2)_W^2 U(1)_Y$}

Only $SU(2)_W$-doublets contribute: $Q_L$ and $L_L$. With
$T(\mathbf{2}) = 1/2$ and colour multiplicity $n_c(Q_L) = 3$,
$n_c(L_L) = 1$:
\begin{align}
  \mathcal{A}[SU(2)_W^2 U(1)_Y]
  &= n_c(Q_L)\cdot\tfrac{1}{2}\cdot Y_{Q_L}
   + n_c(L_L)\cdot\tfrac{1}{2}\cdot Y_{L_L} \\
  &= 3\cdot\tfrac{1}{2}\cdot\bigl(+\tfrac{1}{6}\bigr)
   + 1\cdot\tfrac{1}{2}\cdot\bigl(-\tfrac{1}{2}\bigr) \\
  &= \tfrac{1}{4} - \tfrac{1}{4} = 0.
\end{align}
\begin{equation}
  \boxed{\ \mathcal{A}[SU(2)_W^2 U(1)_Y] = 0\ }
  \label{eq:A22Y}
\end{equation}

\subsection{$\mathrm{Grav}^2 U(1)_Y$}

The mixed gravitational--gauge anomaly reduces to the
$\dim$-weighted hypercharge sum over all Weyl fields:
\begin{align}
  \mathcal{A}[\mathrm{Grav}^2 U(1)_Y]
  &= \sum_f n_f Y_f \\
  &= \underbrace{6\cdot(+\tfrac{1}{6})}_{Q_L}
   + \underbrace{3\cdot(-\tfrac{2}{3})}_{u_R^c}
   + \underbrace{3\cdot(+\tfrac{1}{3})}_{d_R^c}
   + \underbrace{2\cdot(-\tfrac{1}{2})}_{L_L}
   + \underbrace{1\cdot(+1)}_{e_R^c} \\
  &= 1 - 2 + 1 - 1 + 1 = 0.
\end{align}
\begin{equation}
  \boxed{\ \mathcal{A}[\mathrm{Grav}^2 U(1)_Y] = 0\ }
  \label{eq:AggY}
\end{equation}

\section{Main theorem}

\begin{theorem}[Per-generation anomaly cancellation, rigorous]
Under the TECT branching rule
$\mathbf{5} = (\mathbf{3},\mathbf{1})_{-1/3}
\oplus (\mathbf{1},\mathbf{2})_{+1/2}$ of
TECT-Math49-rigorous, all six independent triangle-anomaly
coefficients of the Standard-Model gauge group vanish per
generation:
\begin{equation}
  \mathcal{A}[SU(3)_c^3]
  = \mathcal{A}[SU(2)_W^3]
  = \mathcal{A}[U(1)_Y^3]
  = \mathcal{A}[SU(3)_c^2 U(1)_Y]
  = \mathcal{A}[SU(2)_W^2 U(1)_Y]
  = \mathcal{A}[\mathrm{Grav}^2 U(1)_Y]
  = 0.
\end{equation}
\end{theorem}

\begin{proof}
By \eqref{eq:A333}, \eqref{eq:A222}, \eqref{eq:AYYY},
\eqref{eq:A33Y}, \eqref{eq:A22Y}, \eqref{eq:AggY}. Each
individual identity is an explicit Weyl-multiplicity-weighted
arithmetic sum over the table \eqref{eq:SM-table}, with
hypercharges derived from the $SU(5)$-branching rule
(Lemma~1). The $SU(2)_W^3$ vanishing follows from
$d^{abc} = 0$ in $\mathfrak{su}(2)$, a group-theoretic
identity independent of matter content.
\end{proof}

\begin{corollary}
The sum over three generations gives $3 \cdot 0 = 0$, so the
Standard Model spectrum in TECT is free of gauge and
gauge--gravitational triangle anomalies.
\end{corollary}
