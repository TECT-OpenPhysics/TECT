\paragraph{Anomaly coefficients (definitions).} For a chiral
representation $R$ of $G_{\rm SM}$, the perturbative gauge-anomaly
coefficient triples we must compute are
\begin{align}
\mathcal A_3 &\equiv
A_3(R)\;\text{(cubic Dynkin index of SU(3) part of }R\text{)}, \\
\mathcal A_{32Y} &\equiv
T_3(R)\,Y, \quad
\mathcal A_{22Y} \equiv T_2(R)\,Y, \\
\mathcal A_{YYY} &\equiv Y^3, \quad
\mathcal A_{\text{grav}^2 Y} \equiv Y, \\
\mathcal A_{222} &\equiv 0\;\text{(SU(2) is anomaly-free intrinsically;
no }d^{abc}\text{).}
\end{align}
$T_n(R)$ denotes the SU($n$) Dynkin index of the SU($n$) part of $R$,
with $T_n(\text{fund}) = 1/2$ and $T_n(\text{anti-fund})=1/2$. The
cubic SU(3) index uses
$A_3(\mathbf{3})=+1,\, A_3(\overline{\mathbf{3}})=-1,\,
A_3(\mathbf{1})=0$. The SU(2) cubic index is identically zero because
SU(2) admits no symmetric invariant $d^{abc}$.

A representation $R$ is \emph{anomaly-free} iff all five
non-trivially-defined coefficients vanish.

\subsection*{2. The six anomaly coefficients on a single $\mathbf{16}$}

\subsubsection*{2.1 SU(3)$_C^3$ anomaly}

\begin{align}
\mathcal A_3(\mathbf{16}) &=
\sum_{\psi}\, A_3(R_3(\psi))\,\dim_2(\psi),
\end{align}

\begin{equation}
\mathcal A_3 \;=\; \underbrace{2\cdot(+1)}_{Q:\,\mathbf{2}\text{ doublets of }\mathbf{3}}
\;+\;\underbrace{1\cdot(-1)}_{u^c:\,\overline{\mathbf{3}}}
\;+\;\underbrace{1\cdot(-1)}_{d^c:\,\overline{\mathbf{3}}}
\;=\; 2 - 1 - 1 \;=\; \boxed{0}.
\label{eq:A3-result}
\end{equation}

\subsubsection*{2.3 SU(3)$_C^2 \times$ U(1)$_Y$ anomaly}

\begin{align}

&= \sum_{\psi}\, T_3(R_3(\psi))\,\dim_2(\psi)\, Y(\psi) \notag \\

\;+\;\underbrace{1\cdot\tfrac{1}{2}\cdot(-\tfrac{2}{3})}_{u^c}

\label{eq:A22Y-result}

\subsection*{3. SO(10)-side cross-check}

\paragraph{Perturbative SO(10) cubic anomaly.} For Lie algebras
$\mathfrak{so}(2N)$ with $N\ne 4$, the symmetric invariant
$d^{abc}\equiv \mathrm{Tr}\,T^a\{T^b,T^c\}$ vanishes
representation-independently (Slansky 1981, \S5). In particular,
for $\mathrm{SO}(10)$ ($N=5$),
\begin{equation}
\mathrm{Tr}_{\mathbf{16}}\, T^a\{T^b,T^c\} \;=\; 0
\quad\text{(no symmetric cubic Casimir)},
\end{equation}
so the $\mathrm{SO}(10)^3$ perturbative anomaly is automatically zero
on any representation, including $\mathbf{16}$.

\paragraph{Witten global anomaly.}
$\pi_4(\mathrm{SO}(10)) = 0$, so there is no $\mathbb{Z}_2$ Witten
anomaly. The $\mathbf{16}$ representation is therefore admissible as a
chiral fermion content of an $\mathrm{SO}(10)$ gauge theory at both the
perturbative and global level.

\paragraph{Three-generation closure.}
The branching is linear in the number of generations $N_g$. Since each
$\mathbf{16}$ contributes zero to every anomaly coefficient,
\begin{equation}
\mathcal A^{(N_g)} = N_g \cdot 0 = 0
\quad \forall\, N_g \in \mathbb{Z}_{\ge 0}.
\end{equation}
$N_g = 3$ as in the SM is admissible; the framework imposes no
generation-counting constraint via anomaly cancellation alone (this is
why generations remain a Pillar 6 input).
