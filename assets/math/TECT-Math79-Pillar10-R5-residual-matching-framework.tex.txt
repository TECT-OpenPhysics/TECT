\begin{abstract}
We define a fourth-and-final-style route, denoted R5, for the Pillar 10
$\hbar$-origin problem of TECT. Routes R1--R3 (canonical quantization,
geometric quantization on $M_{\mathrm{red}}$, $\Lambda=0$ consistency,
$SO(10)$ scale $\times$ lattice spacing) all fail to derive $\hbar$
internally; the user-suggested R4 (BCC inter-lattice spacing $\times$
Brazovskii stiffness wave $\times$ Hubble factor) reduces to
dimensional-monomial enumeration which is unlikely to succeed by direct
matching. R5 reframes the question: rather than deriving $\hbar$ from
nothing, we ask whether the systematic residuals between exact classical
TECT predictions and observed precision physics can all be explained by
a single universal completion scale $\chi_*$. We define
$\chi_* := \hbar/S_{\mathrm{BCC}}$ with $S_{\mathrm{BCC}} = m_e c \cdot a_{\mathrm{BCC}}$
the natural classical action scale, fix four independent residual
channels (cosmological constant $\Lambda$, Casimir vacuum-energy, free
electron Compton wavelength, electron $g{-}2$ anomaly), and lay out the
five admissibility criteria a successful audit must pass. The framework
is consistent with R1--R4 (it does not contradict the existing
no-go statements) but elevates the discussion from
``$\hbar$ derivable / not derivable'' to
``does a single $\chi_*$ control the gap between exact classical TECT
and real physics?'' We do not perform the numerical extraction in this
note; we only fix the framework.
\end{abstract}

\section{Motivation and reformulation}

The earlier Pillar 10 notes (Math59, Math59-v2, Math59-v3) reach a
common conclusion: classical-field-theoretic structure alone cannot fix
$\hbar$. The remaining freedom is the symplectic rescaling
$\omega \mapsto \lambda \omega$, which translates to an undetermined
overall action scale.

A na\"ive read of this conclusion is that Pillar 10 is permanently
closed in the negative. But there is a finer question, articulated in
the 2026-04-24 reviewer audit: the no-go theorems address the
\emph{derivation} of $\hbar$, not the \emph{universality} of an
external completion parameter. Specifically, if a single dimensionless
ratio
\[
\chi_* := \frac{\hbar}{S_{\mathrm{BCC}}},
\qquad
S_{\mathrm{BCC}} := m_e \, c \, a_{\mathrm{BCC}}
\]
can be consistently inferred from \emph{multiple independent} residual
mismatches between exact classical TECT predictions $Q_i^{\mathrm{TECT,cl}}$
and observed values $Q_i^{\mathrm{obs}}$, then this $\chi_*$ is a
\emph{phenomenologically universal} parameter even if not derivable.
That is the content of R5.

\section{General framework}

\begin{definition}[Residual channel]
A \emph{residual channel} is an ordered triple $(Q_i^{\mathrm{TECT,cl}}, Q_i^{\mathrm{obs}}, C_i)$ where
\begin{enumerate}
\item $Q_i^{\mathrm{TECT,cl}}$ is a sharp classical TECT prediction, either an exact zero or a tightly-constrained value, derived from one of the closed pillar-level theorem notes;
\item $Q_i^{\mathrm{obs}}$ is the experimentally measured value (or a precision bound) of the same observable;
\item $C_i = C_i\bigl(a_{\mathrm{BCC}}, q_0, \mu^2, \lambda, \gamma, Y, c, \rho_{\mathrm{BCC}}, \dots\bigr)$ is a coefficient computable using only classical TECT quantities, with no implicit insertion of $\hbar$.
\end{enumerate}
\end{definition}

\begin{definition}[Residual ansatz]
For a residual channel $(Q_i^{\mathrm{TECT,cl}}, Q_i^{\mathrm{obs}}, C_i)$, define
\[
\delta Q_i \;:=\; Q_i^{\mathrm{obs}} - Q_i^{\mathrm{TECT,cl}},
\qquad
\chi_*^{(i)} \;:=\; \frac{\delta Q_i}{C_i}.
\]
The leading-order R5 ansatz is
\[
\delta Q_i \;=\; C_i \, \chi_* \;+\; O(\chi_*^2).
\]
R5 succeeds iff $\chi_*^{(i)}$ is approximately the same constant across multiple independent channels.
\end{definition}

\begin{definition}[Five admissibility criteria]
A residual channel is \emph{admissible} for R5 iff:
\begin{description}
\item[(C1) Sharpness.]    $Q_i^{\mathrm{TECT,cl}}$ is sharp (exact zero or precise classical-TECT prediction).
\item[(C2) Observability.] $Q_i^{\mathrm{obs}}$ is precision-measured.
\item[(C3) Non-circularity.] $C_i$ is computable from classical TECT alone, without inserting $\hbar$.
\item[(C4) Robustness.]   $\chi_*^{(i)}$ is invariant under reasonable changes of regulator, box size, defect-core profile, or renormalization convention.
\item[(C5) Canonical uplift.] $\chi_*^{(i)}$ has an explicit candidate identification with $\hbar/S_{\mathrm{BCC}}$ for a defined classical action scale $S_{\mathrm{BCC}}$.
\end{description}
\end{definition}

\section{Selection of the canonical four-channel set}

Six candidate channels were considered:
\[
\{\,\Lambda,\;\; F_{\mathrm{Casimir}},\;\; a_e,\;\; R_\infty,\;\; \lambda_{\mathrm{Compton}},\;\; \sigma_{\mathrm{SB}}/b_{\mathrm{Wien}}\,\}.
\]
\begin{itemize}
\item Stefan--Boltzmann $\sigma_{\mathrm{SB}}$ and Wien displacement $b_{\mathrm{Wien}}$ \emph{fail} (C1) because classical TECT does not yield a native thermal-radiation spectrum without invoking an external statistical-mechanical postulate; the resulting $C_i$ cannot be honestly written.
\item Rydberg constant $R_\infty$ partially passes (C1)--(C2) but is dropped in favour of the Compton wavelength because the latter has a strictly cleaner classical-TECT companion length scale $a_{\mathrm{BCC}}$ and avoids the emergent-Coulomb subtlety of Pillar 4.
\end{itemize}

\begin{proposition}[Fixed R5 channel set]
The canonical R5 audit uses
\[
\boxed{
\bigl\{\,\delta\Lambda,\;\; \delta F_{\mathrm{Casimir}},\;\; \delta\lambda_{\mathrm{Compton}},\;\; \delta a_e\,\bigr\}.
}
\]
\end{proposition}

\section{Channel R5-A --- Cosmological constant residual}

\begin{align*}
Q^{\mathrm{TECT,cl}}_\Lambda
   &\;=\; 0
   \qquad
   \text{(Pillar 11 chain: Math58-v2 + v3 + v4-sublemma + v5)},\\
Q^{\mathrm{obs}}_\Lambda
   &\;\approx\; 1.1088 \times 10^{-52}\ \mathrm{m}^{-2}
   \qquad
   \text{(LCDM, Planck 2018)},\\
\delta\Lambda
   &\;\approx\; +1.1088 \times 10^{-52}\ \mathrm{m}^{-2},\\
C_\Lambda^{(\mathrm{TECT,cl})}
   &\;\sim\; \frac{Y\,q_0^4}{a_{\mathrm{BCC}}^{\,2}}\,,\\
\chi_*^{(\Lambda)}
   &\;=\; \frac{\delta\Lambda}{C_\Lambda^{(\mathrm{TECT,cl})}}\,.
\end{align*}

\begin{remark}
The dependence on the cosmological model enters through the LCDM
inversion of dark-energy density to a $\Lambda$ value. Any future
extraction must quote $\Omega_\Lambda$-scheme explicitly. (C4) is
tested by recomputing under conservative wCDM bounds.
\end{remark}

\section{Channel R5-B --- Casimir vacuum-energy residual}

For two parallel conducting plates at separation $d$ in the
$\mathrm{BCC}$ background:
\begin{align*}
Q^{\mathrm{TECT,cl}}_{F/A}(d)
   &\;=\; 0
   \qquad
   \text{(Pillar 11 Math58-v5 BCC sector }\Delta\Lambda_{\mathrm{BCC}}=0\text{; Math58-v6 Dirac-sector contact-term absorption)},\\
Q^{\mathrm{obs}}_{F/A}(d)
   &\;=\; -\,\frac{\pi^2\,\hbar c}{240\,d^4}
   \qquad
   \text{(precision-measured)},\\
\delta F/A(d)
   &\;=\; -\,\frac{\pi^2\,\hbar c}{240\,d^4},\\
C_{\mathrm{Cas}}^{(\mathrm{TECT,cl})}(d)
   &\;\sim\; \frac{Y\,q_0^{\,2}}{d^{\,2}}
   \qquad
   \text{(classical defect-mediated boundary stress between plates)},\\
\chi_*^{(\mathrm{Cas})}(d)
   &\;=\; \frac{\delta F/A(d)}{C_{\mathrm{Cas}}^{(\mathrm{TECT,cl})}(d)}\,.
\end{align*}

\begin{remark}
The $d$-dependence enables a within-channel (C4) test: $\chi_*^{(\mathrm{Cas})}(d)$
must be $d$-independent over the experimentally accessible range
$d \in [10\,\mathrm{nm},\,10\,\mu\mathrm{m}]$ for the channel to qualify.
\end{remark}

\section{Channel R5-C --- Electron Compton-wavelength residual}

\begin{align*}
Q^{\mathrm{TECT,cl}}_{\ell_e}
   &\;=\; a_{\mathrm{BCC}} \;=\; 2\pi/q_0^{\mathrm{phys}}
   \qquad
   \text{(classical defect characteristic length, Pillar 1 + Pillar 6)},\\
Q^{\mathrm{obs}}_{\lambda_C}
   &\;=\; \frac{h}{m_e c} \;=\; 2.4263 \times 10^{-12}\ \mathrm{m},\\
\delta\ell
   &\;:=\; \lambda_C - a_{\mathrm{BCC}},\\
C_{\mathrm{Comp}}^{(\mathrm{TECT,cl})}
   &\;=\; a_{\mathrm{BCC}}
   \qquad
   \text{(natural classical length normaliser)},\\
\chi_*^{(\mathrm{Comp})}
   &\;=\; \frac{\delta\ell}{a_{\mathrm{BCC}}}
   \;\;\stackrel{\lambda_C \gg a}{\approx}\;\;
   \frac{\lambda_C}{a_{\mathrm{BCC}}}
   \;=\;
   \frac{\hbar\,q_0^{\mathrm{phys}}}{2\pi\,m_e c}.
\end{align*}

\begin{proposition}[Calibration role of R5-C]
The Compton-wavelength channel is the calibration channel for $\chi_*$:
\[
\chi_*^{(\mathrm{Comp})} \;=\; \frac{\hbar\,q_0^{\mathrm{phys}}}{2\pi\,m_e\,c}
\;=\;
\frac{\hbar}{m_e c \cdot a_{\mathrm{BCC}}}
\;=\;
\frac{\hbar}{S_{\mathrm{BCC}}^{(e)}},
\qquad
S_{\mathrm{BCC}}^{(e)} := m_e c \cdot a_{\mathrm{BCC}}.
\]
This is precisely the canonical-uplift definition. R5-A, R5-B, R5-D
must yield $\chi_*$-values consistent with R5-C for the audit to succeed.
\end{proposition}

\section{Channel R5-D --- Electron $g{-}2$ anomaly residual}

\begin{align*}
Q^{\mathrm{TECT,cl}}_{g}
   &\;=\; 2
   \qquad
   \text{(Pillar 7 spinor algebra; Pillar 5 Dirac structure; classical Land\'e factor)},\\
Q^{\mathrm{obs}}_{g}
   &\;=\; 2.002\,319\,304\,361\,\dots
   \qquad
   \text{(precision-measured to }10^{-12}\text{)},\\
\delta a_e
   &\;:=\; \frac{g_{\mathrm{obs}}-2}{2} \;=\; 1.1597 \times 10^{-3},\\
C_{a_e}^{(\mathrm{TECT,cl})}
   &\;=\; \frac{1}{2\pi}\,\frac{e^{\,2}}{4\pi\varepsilon_0\,Y\,q_0^{\,2}\,a_{\mathrm{BCC}}}
   \qquad
   \text{(classical defect-loop self-coupling)},\\
\chi_*^{(g{-}2)}
   &\;=\; \frac{\delta a_e}{C_{a_e}^{(\mathrm{TECT,cl})}}\,.
\end{align*}

\begin{remark}[Non-circularity audit for R5-D]
The classical coupling $C_{a_e}$ is built only from $e$, $\varepsilon_0$,
$Y$, $q_0$, and $a_{\mathrm{BCC}}$ --- never from
$\alpha = e^2/(4\pi\varepsilon_0\hbar c)$. The Schwinger value
$\alpha/(2\pi)$ for $a_e$ is \emph{not} imported; it is precisely the
combination we wish to recover phenomenologically by extracting
$\chi_*$. If the recovered $\chi_*^{(g{-}2)}$ matches
$\chi_*^{(\mathrm{Comp})}$ to within a factor of two, R5-D corroborates
the universality.
\end{remark}

\section{Pre-registered success and failure criteria}

\begin{theorem}[R5 success criterion]
Define $\chi_*^{(\mathrm{Comp})}$ from R5-C as the calibration value.
The R5 audit succeeds iff the three independent ratios
\[
\rho_\Lambda \;:=\; \chi_*^{(\Lambda)}\,/\,\chi_*^{(\mathrm{Comp})},
\quad
\rho_{\mathrm{Cas}} \;:=\; \chi_*^{(\mathrm{Cas})}\,/\,\chi_*^{(\mathrm{Comp})},
\quad
\rho_{g{-}2} \;:=\; \chi_*^{(g{-}2)}\,/\,\chi_*^{(\mathrm{Comp})}
\]
all lie in the interval $[0.5,\;2.0]$ simultaneously, with
$\chi_*^{(\mathrm{Cas})}(d)$ furthermore $d$-independent across the
accessible separation range.
\end{theorem}

\begin{theorem}[R5 failure (= negative outcome) criterion]
The R5 audit fails iff at least one of $\rho_\Lambda, \rho_{\mathrm{Cas}}, \rho_{g{-}2}$
differs from unity by a factor of $10$ or more, or if
$\chi_*^{(\mathrm{Cas})}(d)$ exhibits power-law $d$-dependence.
\end{theorem}

A failed R5 audit reinforces \texttt{Pillar 10 = OPEN-NEGATIVE REFINED}:
even phenomenological universality breaks down. A successful R5 audit
yields the phenomenological-universality statement boxed in the
abstract: TECT does not derive $\hbar$ from pure classical first
principles, but a single external completion scale $\chi_*$ is
consistently inferred from independent residual mismatches.
