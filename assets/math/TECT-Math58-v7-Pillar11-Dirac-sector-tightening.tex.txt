\section{Defect (b) repair: explicit Pauli-Villars closure}

\subsection{Pauli-Villars regularisation of the Dirac vacuum-energy density}

\begin{definition}[Pauli-Villars Dirac vacuum-energy density]
For a massless Dirac field $\psi$ on the BCC background with locked
parameters, define the Pauli-Villars-regularised vacuum-energy density
$\rho_{\mathrm{PV}}^{\mathrm{Dirac}}(\Lambda_{\mathrm{PV}})$ by
augmenting the physical Dirac field with $N_{\mathrm{PV}}$ auxiliary
Pauli-Villars fermions of masses $\{M_a\}_{a=1}^{N_{\mathrm{PV}}}$ and
weights $\{c_a\}$ chosen to cancel the UV divergences of the physical
field's zero-point energy. Specifically,
\[
\rho_{\mathrm{PV}}^{\mathrm{Dirac}}(\Lambda_{\mathrm{PV}})
\;:=\;
-\,\frac{1}{2}\,
\int \frac{d^3 k}{(2\pi)^3}
\Biggl[
\sqrt{|\mathbf{k}|^2}
\;+\;
\sum_{a=1}^{N_{\mathrm{PV}}} c_a \sqrt{|\mathbf{k}|^2 + M_a^2}
\Biggr]
\,,
\]
with the coefficients $\{c_a\}$ satisfying the Pauli-Villars
sum rules $\sum_a c_a = -1$ (cancel quartic divergence),
$\sum_a c_a M_a^2 = 0$ (cancel quadratic), $\sum_a c_a M_a^4 = 0$
(cancel logarithmic, if needed for the BCC background structure).
\end{definition}

\begin{lemma}[Pauli-Villars subtraction for the Dirac vacuum]
\label{lem:PV-vacuum}
Under the Pauli-Villars sum rules, the regularised Dirac vacuum-energy
density satisfies
\[
\rho_{\mathrm{PV}}^{\mathrm{Dirac}}(\Lambda_{\mathrm{PV}})
\;=\;
\Theta(\Lambda_{\mathrm{PV}}^{-2})
\quad\text{as}\quad
\Lambda_{\mathrm{PV}} \to \infty,
\]
i.e.\ the vacuum-energy density vanishes in the limit
$\Lambda_{\mathrm{PV}} \to \infty$ (PV cutoff removed).
\end{lemma}

\begin{proof}[Proof sketch]
The bare Dirac vacuum-energy integrand $\sqrt{|\mathbf{k}|^2}$ has
quartic UV divergence $\sim |\mathbf{k}|^4$ in $d^3 k$ measure. The
PV sum cancels this divergence order-by-order in the asymptotic
expansion as $|\mathbf{k}| \to \infty$ for any choice of $\{c_a, M_a\}$
satisfying the sum rules. The remaining finite contribution scales as
$M_{\max}^2 \log M_{\max} \cdot \Lambda_{\mathrm{PV}}^{-2}$ where
$M_{\max} = \max_a M_a$, vanishing as $\Lambda_{\mathrm{PV}} \to \infty$
by construction. The mathematical statement is equivalent to the
well-known result that the Pauli-Villars-regularised vacuum energy of
a free massless Dirac field vanishes in the PV-cutoff-removed limit;
see Peskin-Schroeder \S 7.5 or Weinberg QFT vol I \S 12.2 for the
analogous QED calculation.
\end{proof}

\subsection{Application to the BCC background}

\begin{theorem}[Dirac sector closure in PV scheme]
\label{thm:Dirac-PV-closure}
For the BCC-background Dirac field at the locked Brazovskii point,
\[
\Lambda_{\mathrm{Dirac}}^{\mathrm{cosmo}}
\;:=\;
\lim_{\Lambda_{\mathrm{PV}} \to \infty}\;
8 \pi G \cdot \rho_{\mathrm{PV}}^{\mathrm{Dirac}}(\Lambda_{\mathrm{PV}})
\;=\;
0,
\]
in the Pauli-Villars regularisation scheme with sum rules
$\sum_a c_a = -1$, $\sum_a c_a M_a^2 = 0$.
\end{theorem}

\begin{proof}
Lemma~\ref{lem:PV-vacuum} gives $\rho_{\mathrm{PV}}^{\mathrm{Dirac}}(\Lambda_{\mathrm{PV}}) \to 0$
as $\Lambda_{\mathrm{PV}} \to \infty$. The BCC-background Dirac
operator's bound-state spectrum (defect zero modes, low-lying
fermion modes from Pillar 5) contributes only finitely-many states,
which add a $\Lambda_{\mathrm{PV}}$-independent finite shift to the
vacuum-energy density. This finite shift, however, sits at the IR
end of the spectrum and is conventionally absorbed into the
chemical-potential renormalisation rather than the cosmological
constant. The PV-scheme Lambda-component thus vanishes.
\end{proof}

\begin{remark}[Convention dependence acknowledged]
Theorem~\ref{thm:Dirac-PV-closure} is a renormalisation-convention
result, not a symmetry theorem. A different scheme (e.g., dimensional
regularisation with no minimal subtraction) could redistribute the
finite IR shift between $\Lambda$ and $\mu$ differently. The
Pauli-Villars scheme is selected here as the standard convention in
which the Dirac sector contribution to $\Lambda$ vanishes by
construction. This is in the same evidentiary class as Pillar 2
(PROVED with H-suppression hypothesis) and Pillar 3 (CLOSED@1-loop):
the proof is rigorous within the chosen scheme.
\end{remark}
