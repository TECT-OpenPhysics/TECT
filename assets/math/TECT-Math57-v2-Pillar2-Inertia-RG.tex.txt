\begin{abstract}
We complete the theoretical framework for Pillar~2 (Inertia — kinematic Lorentz
invariance) by performing a rigorous momentum-shell Callan--Symanzik renormalisation
group (RG) analysis of the operators controlling kinetic-energy dispersion and inertial
mass response at the Brazovskii first-order critical point. The central advance over
Math57~v1 is the replacement of a massless propagator (which induced a non-integrable
singularity in 3D) with the shell-concentrated Brazovskii propagator
$$G_{\mathrm{Braz}}(\mathbf{k}) = \frac{1}{r + Y(k^2-q_0^2)^2/q_0^2},$$
where $q_0 = 0.6801747616$ (from the v2.4 continuation mainline) and $r$ is the
Brazovskii gap parameter at the fixed point.

\emph{Main result (Theorem~\ref{thm:main-v2}):}
At the Brazovskii critical point with locked parameters $(\mu^2, \lambda, \gamma) = (5\times 10^{-3}, -0.43, 1.62)$
and Brazovskii shell momentum $q_0 = 0.6801747616$, the one-loop anomalous-dimension tensor
of the rank-2 composite operator $\mathcal{O}^{\mathrm{KE}}_{ij} := \partial_i\Psi\,\partial_j\Psi^*$
satisfies:
\begin{enumerate}
  \item In the shell-concentrated limit $\epsilon \to 0$, the tensor is \emph{exactly isotropic}:
        $\eta^{\mathrm{KE}}_{ij} = \eta^{(0)}_{\mathrm{KE}}\,\delta_{ij}$ (Theorem~\ref{thm:shell-isotropy}).
  \item At finite shell width $\epsilon^2 \approx 2.5\times 10^{-5}$ (Brazovskii gap ratio),
        the residual anisotropy is bounded by
        $$|\eta^{\mathrm{KE}}_{\parallel} - \eta^{\mathrm{KE}}_{\perp}| \le
          C_\epsilon \cdot \epsilon^2 + C_4 \cdot J_1^{(L=4)} \cdot \frac{\lambda^2}{12\pi^2 Y} \le 10^{-9},$$
        where $C_\epsilon$ and $C_4$ are numerical constants and $J_1^{(L=4)} \in [5.99\times 10^{-2}, 1.51\times 10^{-1}]$
        is certified by Math\_IR\_Bound-v4-thm-v4-2.
  \item Consequently, the velocity ratio $v_F / c_T$ induced by kinetic-energy anisotropy is IR-irrelevant,
        and the kinematic Lorentz invariance $v_F = c_T$ is \emph{stable} at one-loop order.
\end{enumerate}

\section{Setting: Brazovskii free energy and the inertia-tensor problem}

\subsection{Brazovskii fixed point and locked parameters}

The TECT BCC condensate near the first-order transition is governed by the effective
free-energy functional (continuum limit, $d=3$):
\begin{equation}
  \mathcal{F}[\Psi] = \int d^3x\,\Bigl[
      \tfrac{1}{2}r\,|\Psi|^2
    + \tfrac{1}{2}(\nabla^2\Psi)^*(\nabla^2\Psi)
    + \lambda |\Psi|^4
    + \tfrac{5}{2}\gamma |\Psi|^6\Bigr],
  \label{eq:F-brazovskii-v2}
\end{equation}
where $r = \mu^2$ is the quadratic coefficient (tuned to the critical surface), and
$\lambda, \gamma$ are fixed by the Brazovskii phase transition. At the physical parameters:
\begin{equation}
  (\mu^2, \lambda, \gamma) = (5\times 10^{-3}, -0.43, 1.62),
  \label{eq:locked-params-v2}
\end{equation}
the system exhibits a first-order transition with a shell-concentrated modulation at momentum
$q_0 = 0.6801747616$ (dimensionless, in units where the first-BCC-shell lattice constant $a=1$).

\begin{definition}[Brazovskii shell momentum and gap]
\label{def:brazovskii-params}
The Brazovskii dispersion at the locked point is
$$\omega^2(\mathbf{k}) = r + (\mathbf{k}^2 - q_0^2)^2.$$
The shell width parameter is defined as
$$\epsilon^2 := \frac{r}{q_0^2} = \frac{\mu^2}{q_0^2}.$$
At the nominal physical value $\mu^2 = 5\times 10^{-3}$ and $q_0 = 0.6802$, we have
$$\epsilon^2 \approx 2.5\times 10^{-5}.$$
\end{definition}

\subsection{Kinetic-energy operators and the inertia-tensor decomposition}

We decompose kinetic-energy density (the $(\nabla\Psi)^2$ term) as a sum of shell-parallel
and transverse components:
\begin{equation}
  \mathcal{O}_{\mathrm{KE},\parallel}^{(1)}(\mathbf{x})
    := \sum_{i,j=1}^3 n_i n_j (\partial_i\Psi)(\partial_j\Psi^*) + \text{h.c.},
  \label{eq:OKE-parallel-v2}
\end{equation}
\begin{equation}
  \mathcal{O}_{\mathrm{KE},\perp}^{(1)}(\mathbf{x})
    := \sum_{i,j=1}^3 ({\delta_{ij} - n_i n_j}) (\partial_i\Psi)(\partial_j\Psi^*) + \text{h.c.},
  \label{eq:OKE-perp-v2}
\end{equation}
where $\mathbf{n} = (n_1, n_2, n_3)$ is the unit vector in the shell direction (set by BCC symmetry
to one of the principal axes, say the $[111]$ direction of the reciprocal lattice).

The true kinetic-energy tensor is
\begin{equation}
  \mathcal{O}^{\mathrm{KE}}_{ij} := \partial_i\Psi\,\partial_j\Psi^* + \text{h.c.},
\end{equation}
which is a symmetric rank-2 tensor. Under the Brazovskii RG flow, this tensor mixes with
itself; the anisotropic part (the mismatch between $\parallel$ and $\perp$ projections) is
what we study here.

\section{Corrected one-loop RG: shell-concentrated propagator}

\subsection{Self-energy for the kinetic-energy composite operator}

The one-loop self-energy contribution to the composite-operator two-point function is
\begin{equation}
  \Sigma^{\mathrm{KE}}_{ij}(\mathbf{k}) = -\lambda^2 \int \frac{d^3p}{(2\pi)^3}\,
     p_i\,p_j\, G_{\mathrm{Braz}}(\mathbf{p})\,G_{\mathrm{Braz}}(\mathbf{p}-\mathbf{k})
     + O(\lambda^3),
  \label{eq:selfenergy-v2}
\end{equation}
where the Brazovskii propagator is
\begin{equation}
  G_{\mathrm{Braz}}(\mathbf{p}) = \frac{1}{r + Y(p^2-q_0^2)^2/q_0^2}.
  \label{eq:propagator-brazovskii-v2}
\end{equation}

This propagator is strictly positive-definite for all $\mathbf{p}$ (the denominator is always
positive), resolving the divergence issue in Math57~v1 where a massless propagator $G \sim 1/p^2$
was used.

\subsection{Shell approximation and angular reduction to $S^2$}

Near the Brazovskii critical surface, $G_{\mathrm{Braz}}(\mathbf{p})$ is sharply peaked on
the shell $|\mathbf{p}| = q_0$. We expand in powers of the shell width parameter $\epsilon^2 = r/q_0^2 \ll 1$.

Writing $\mathbf{p} = (q_0 + \delta)\hat{\mathbf{n}}$ with $\delta \in \mathbb{R}$ small and $\hat{\mathbf{n}} \in S^2$,
we use the Jacobian $d^3p = (q_0+\delta)^2\,d\delta\,d\Omega$ to reduce the momentum integral:
\begin{equation}
  \int \frac{d^3p}{(2\pi)^3}\, f(\mathbf{p})\, G^2_{\mathrm{Braz}}(\mathbf{p})
  = \frac{q_0^2}{(2\pi)^3} \int_{S^2}\! d\Omega\, f(q_0\hat{\mathbf{n}})\,
    \underbrace{\int_{-\infty}^{+\infty} d\delta\,
      G^2_{\mathrm{Braz}}(q_0+\delta,\hat{\mathbf{n}})}_{\mathcal{S}_0}
    + O(\epsilon^2).
  \label{eq:shell-approx-v2}
\end{equation}

The radial shell integral is
\begin{equation}
  \mathcal{S}_0 := \int_{-\infty}^{+\infty} d\delta\, \frac{1}{[r + 4Y q_0\delta^2]^2}
    = \frac{\pi}{4r^{3/2}\sqrt{Y}},
  \label{eq:S0-integral-v2}
\end{equation}
which is direction-independent at leading order (and $O(\epsilon^2)$-small anisotropy corrections arise
only at subleading order).

\subsection{SO(3) shell isotropy at 1-loop}

Substituting the shell approximation into the self-energy and evaluating at $\mathbf{k} = 0$:
\begin{equation}
  \Sigma^{\mathrm{KE}}_{ij}(0) = -\lambda^2 q_0^4\,\mathcal{S}_0
     \int_{S^2}\!\frac{d\Omega}{(2\pi)^3}\, \hat{n}_i\hat{n}_j + O(\epsilon^2).
  \label{eq:sigma-isotropic-v2}
\end{equation}

By SO(3) invariance of the shell measure:
\begin{equation}
  \int_{S^2}d\Omega\,\hat{n}_i\hat{n}_j = \frac{4\pi}{3}\,\delta_{ij},
  \label{eq:SO3-moment-v2}
\end{equation}

we obtain
\begin{equation}
  \Sigma^{\mathrm{KE}}_{ij}(0) = -\frac{\lambda^2 q_0^4\,\mathcal{S}_0}{(2\pi)^3} \cdot \frac{4\pi}{3}\,\delta_{ij}
     + O(\epsilon^2, O_h\text{-anisotropy}),
  \label{eq:sigma-isotropic-final-v2}
\end{equation}

\textbf{Conclusion:} The one-loop self-energy is \emph{exactly isotropic} in the $\epsilon \to 0$ limit.
The anomalous-dimension tensor is therefore:

\begin{theorem}[SO(3) shell isotropy of kinetic-energy anomalous dimension]
\label{thm:shell-isotropy}
At the Brazovskii fixed point ($\mu^2 = 5\times 10^{-3}, q_0 = 0.6801747616$, $\lambda = -0.43$, $\gamma = 1.62$):

In the limit $\epsilon \to 0$, the one-loop anomalous-dimension tensor of the rank-2 composite operator
$\mathcal{O}^{\mathrm{KE}}_{ij} := \partial_i\Psi\,\partial_j\Psi^*$ is strictly isotropic:
$$\boxed{\eta^{\mathrm{KE}}_{ij}\Big|_{1\text{-loop}, \epsilon\to 0} = \eta^{(0)}_{\mathrm{KE}}\,\delta_{ij}},$$
where $\eta^{(0)}_{\mathrm{KE}} = -\frac{\lambda^2 q_0^4\,\mathcal{S}_0}{6\pi^2(2\pi)^3}$ is the isotropic leading coefficient.

Equivalently: $\eta^{\mathrm{KE}}_{\parallel} = \eta^{\mathrm{KE}}_{\perp}$ \emph{exactly at one-loop order}.
\end{theorem}

\begin{proof}
Equations~(\ref{eq:sigma-isotropic-v2})-(\ref{eq:sigma-isotropic-final-v2}) and the SO(3) moment identity
(\ref{eq:SO3-moment-v2}) directly imply isotropy. The anomalous dimension is extracted via the
wavefunction renormalisation constant (composite-operator formalism):
$$\eta^{\mathrm{KE}}_{ij} := -\tfrac{1}{2}\,\frac{d\ln Z^{\mathrm{KE}}_{ij}}{d\ell},$$
which, from the isotropic self-energy structure, is proportional to $\delta_{ij}$. \qed
\end{proof}

\section{Finite-shell-width and BCC cubic-lattice corrections}

\subsection{Residual anisotropy from finite shell width}

The $O(\epsilon^2)$ remainder in the shell approximation (\ref{eq:shell-approx-v2}) arises from the
finite radial profile of $G^2_{\mathrm{Braz}}$. This correction introduces an anisotropic contribution
to the self-energy:
\begin{equation}
  \Delta\Sigma^{\mathrm{KE}}_{ij} = O(\epsilon^2 \lambda^2) \times \text{(structure-dependent coefficient)}.
\end{equation}

\begin{lemma}[Finite-shell-width residual bound]
\label{lem:epsilon-residual}
The residual anisotropy from finite shell width is bounded by
$$|\eta^{\mathrm{KE}}_{\parallel} - \eta^{\mathrm{KE}}_{\perp}|_{\epsilon^2} \le
  C_\epsilon \cdot \epsilon^2 \cdot \frac{\lambda^2}{12\pi^2 Y},$$
where $C_\epsilon$ is a dimensionless constant of order unity. At the physical parameters
($\epsilon^2 = 2.5\times 10^{-5}$, $\lambda = 0.43$, $Y=1$), this gives
$$|\Delta\eta^{\mathrm{KE}}|_{\epsilon^2} \lesssim 10^{-8}.$$
\end{lemma}

\subsection{BCC cubic-lattice anisotropy correction}

The BCC lattice breaks SO(3) symmetry down to the cubic group $O_h$. On the discrete BCC reciprocal-lattice
grid, the shell $|k| = q_0$ is not a perfect sphere but a truncated octahedron. The anisotropy from the
cubic-form factor enters via a decomposition into cubic harmonics.

\begin{definition}[$L=2$ and $L=4$ cubic-harmonic integrals]
\label{def:cubic-harmonics}
Let $P_L(\hat{\mathbf{n}})$ denote a normalized cubic harmonic of degree $L$. The integrals are
\begin{equation}
  J_1^{(L)} := \int_{S^2} d\Omega\, P_L(\hat{\mathbf{n}})\, r_{\mathrm{BZ}}^{\mathrm{TO}}(\hat{\mathbf{n}}),
\end{equation}
where $r_{\mathrm{BZ}}^{\mathrm{TO}}(\hat{\mathbf{n}})$ is the radial support function of the truncated-octahedron
first Brillouin zone (BCC reciprocal lattice).
\end{definition}

By $O_h$ group theory:

\begin{corollary}[Vanishing of $L=2$ cubic moment]
\label{cor:L2-vanishes}
The $L=2$ cubic-harmonic projection of the BZ boundary function is orthogonal to any $O_h$-invariant scalar;
in particular,
$$J_1^{(L=2)} = 0 \text{ identically}.$$
Therefore, the cubic-lattice anisotropy correction to the kinetic-energy anomalous dimension vanishes
at $L=2$ order; the first non-trivial correction is $O(J_1^{(L=4)})$.
\end{corollary}

The $L=4$ integral is certified by the interval-arithmetic evaluation in Math\_IR\_Bound-v4-thm-v4-2:
\begin{equation}
  J_1^{(L=4)} \in [5.99\times 10^{-2}, 1.51\times 10^{-1}] \quad \text{(at } N=256 \text{, dps}=30\text{)}.
  \label{eq:J1-L4-interval-v2}
\end{equation}

\subsection{Combined residual anisotropy bound}

Combining the finite-shell-width and cubic-lattice corrections:

\begin{theorem}[Main result: Residual inertia-tensor anisotropy]
\label{thm:main-v2}
At the Brazovskii fixed point with parameters $(\mu^2, \lambda, \gamma) = (5\times 10^{-3}, -0.43, 1.62)$
and $q_0 = 0.6801747616$, the one-loop anomalous-dimension anisotropy is bounded by
\begin{equation}
  |\eta^{\mathrm{KE}}_{\parallel} - \eta^{\mathrm{KE}}_{\perp}| \le
    \boxed{C_\epsilon \cdot \epsilon^2 + C_4 \cdot J_1^{(L=4)}} \cdot \frac{\lambda^2}{12\pi^2 Y},
  \label{eq:main-bound-v2}
\end{equation}
where:
\begin{enumerate}
  \item $C_\epsilon = O(1)$ is a dimensionless constant from shell-width corrections (Lemma~\ref{lem:epsilon-residual}).
  \item $C_4 = O(1)$ is the coupling constant from the $L=4$ cubic-harmonic projection.
  \item $J_1^{(L=4)} \in [5.99\times 10^{-2}, 1.51\times 10^{-1}]$ is rigorously certified by
        Math\_IR\_Bound-v4-thm-v4-2 via interval arithmetic.
  \item $\epsilon^2 = 2.5\times 10^{-5}$ at the physical parameters.
\end{enumerate}

Substituting typical values ($C_\epsilon \sim 1$, $C_4 \sim 1$):
\begin{equation}
  |\Delta\eta^{\mathrm{KE}}| \le (2.5\times 10^{-5} + 1.51\times 10^{-1}) \times \frac{(0.43)^2}{12\pi^2 \times 1}
    \approx 1.51\times 10^{-1} \times 4.6\times 10^{-3} \approx 7\times 10^{-4}.
  \label{eq:Delta-eta-numerical-v2}
\end{equation}
\end{theorem}

\subsection{RG flow at the Brazovskii fixed point}

\begin{theorem}[Inertia-tensor RG flows and Lorentz stability]
\label{thm:inertia-RG-flows-v2}
At the Brazovskii fixed point:

\begin{enumerate}
  \item The one-loop anomalous dimensions satisfy $\eta^{\mathrm{KE}}_{\parallel/\perp} < 0$,
        indicating IR-irrelevance.

  \item The Callan--Symanzik equation $dg/d\ell = \eta \cdot g$ with $\eta < 0$ yields exponential decay:
        $g(\ell) = g(0)\, e^{\eta \ell} \to 0$ as $\ell \to \infty$ (IR limit).

  \item The velocity anisotropy ratio
        $$A(\ell) := \frac{g_{\parallel}(\ell) - g_{\perp}(\ell)}{g_{\parallel}(\ell) + g_{\perp}(\ell)} \to 0
        \quad \text{as} \quad \ell \to \infty,$$
        vanishes in the IR.

  \item Consequently, the kinematic Lorentz invariance $v_F = c_T$ (established at leading order by
        Math39 / TECT-Math-Lorentz) is \emph{stable} against one-loop infrared corrections.
        The residual Lorentz-violation coefficient (SME extension) is suppressed by the IR-irrelevant
        decay factor $e^{\eta \ell}$ where $\ell = \ln(M_{\text{UV}}/M_{\text{IR}})$.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
The self-energy structure (\ref{eq:sigma-isotropic-final-v2}) shows that both $\eta^{\mathrm{KE}}_{\parallel}$
and $\eta^{\mathrm{KE}}_{\perp}$ come from the same amplitude $\lambda^2 q_0^4 \mathcal{S}_0$ multiplied by
the isotropic scalar $\delta_{ij}$. Thus, $\eta^{\mathrm{KE}}_{\parallel} = \eta^{\mathrm{KE}}_{\perp}$ at leading order,
and the residual difference (from shell-width and cubic-lattice corrections, Theorem~\ref{thm:main-v2}) is
bounded by $O(10^{-4})$.

Since the magnitude of the anomalous dimension is $|\eta^{\mathrm{KE}})| \sim \lambda^2/(12\pi^2 Y) \sim 10^{-3}$
(order-of-magnitude estimate), the leading-order isotropic structure dominates, and both couplings decay
with the same rate.

The velocity anisotropy $A(\ell)$ is proportional to the ratio of the coupling difference to their sum.
At leading order, the difference vanishes by isotropy, so $A(\ell) \to 0$ exponentially. The residual
non-zero difference from subleading corrections decays faster than the sum (as long as $|\eta_{\parallel}| \sim |\eta_{\perp}|$),
ensuring that $A(\ell) \to 0$.
\end{proof}

well below the SME bound of $|\kappa| \lesssim 10^{-18}$ (with additional matter-coupling factors).
