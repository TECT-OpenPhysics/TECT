\subsection*{3. First-principles re-derivation of $K_{4}(\Lambda)$ and audit of Math383 §2}

The single-shell ansatz $\Psi(\mathbf r) = A\sum_{j=1}^{n}\cos(\mathbf q_{j}\cdot\mathbf r)$ with all $|\mathbf q_{j}|=q_{0}$ implies (by elementary trig product-to-sum + volume averaging):
\begin{align}
  \langle \Psi^{2}\rangle &= \tfrac{n}{2}A^{2}, \\
  \langle \Psi^{4}\rangle &= A^{4}\,\bigg\langle\Big(\sum_{j}\cos\theta_{j}\Big)^{4}\bigg\rangle, \qquad \theta_{j}\equiv \mathbf q_{j}\cdot\mathbf r,
\end{align}
where the volume average $\langle\,\cdot\,\rangle$ of $\cos\theta_{j_{1}}\cos\theta_{j_{2}}\cos\theta_{j_{3}}\cos\theta_{j_{4}}$ is non-zero iff some sign assignment $\epsilon_{1},\ldots,\epsilon_{4}\in\{\pm 1\}$ satisfies $\sum_{i}\epsilon_{i}\mathbf q_{j_{i}}=0$. The rule, derived from $\cos\theta = \tfrac{1}{2}(e^{i\theta}+e^{-i\theta})$, is
\begin{equation}
  \langle\cos\theta_{j_{1}}\cos\theta_{j_{2}}\cos\theta_{j_{3}}\cos\theta_{j_{4}}\rangle
  = \frac{1}{16}\,\#\big\{(\epsilon_{1},\ldots,\epsilon_{4})\in\{\pm 1\}^{4}\,\big|\,\sum_{i}\epsilon_{i}\mathbf q_{j_{i}}=0\big\}.
\end{equation}
The sum is over ordered tuples $(j_{1},j_{2},j_{3},j_{4})$ with $j_{i}\in\{1,\ldots,n\}$.

\paragraph{Lamellar} ($n=1$, single mode $\mathbf q_{1}$).
Only $(j_{1}j_{2}j_{3}j_{4}) = (1111)$ contributes; the sign-zero count is $\binom{4}{2}=6$ (choose 2 positive of 4); contribution $\frac{6}{16}=\frac{3}{8}$. Total $\langle\Psi^{4}\rangle/A^{4}=\frac{3}{8}$.
$\langle\Psi^{2}\rangle^{2}=A^{4}/4$. Hence
\begin{equation}
  K_{4}^{\rm lam} \;=\; \frac{\langle\Psi^{4}\rangle}{\langle\Psi^{2}\rangle^{2}}
  \;=\; \frac{3/8}{1/4}\;=\;\boxed{1.5}\quad (\text{Math383 §2 had }3.0;\text{ wrong by }2\times).
\end{equation}

\paragraph{HEX} ($n=3$, modes at $120^{\circ}$ in plane satisfying $\mathbf q_{1}+\mathbf q_{2}+\mathbf q_{3}=0$).
\begin{itemize}
\item $(jjjj)$: $n\cdot\frac{3}{8} = \frac{9}{8}$.
\item $(jjkk j\neq k)$: $\binom{3}{2}\cdot 6\cdot\frac{1}{4} = 3\cdot 6\cdot\frac{1}{4}=\frac{9}{2}$.
\item 4-distinct: only 3 modes; not possible.
\item 3-distinct (e.g., $(jklj)$): vanishes (odd cosine integrals).
\end{itemize}
