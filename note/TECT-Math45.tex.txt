\subsection*{1. Shell displacement field}

Let $\psi(x)=\phi_{0}\,e^{i\theta(x)}+\delta\psi(x)$ denote the
locked Brazovskii order parameter of Math40 (\ref{eq:Feff}), with
$\theta(x)$ the phase field on the shell $\Sigma_{q_{0},\Delta}$.

\begin{lem}[Shell phase-to-displacement extraction]
\label{lem:u-extract}
The shell phase decomposes as
\begin{equation}
\theta(x) \;=\; q_{0}\,\hat{n}\!\cdot\!u(x)
 \;+\; \theta_{\perp}(x)
 \;+\; O(u^{2}),
\label{eq:phase-disp}
\end{equation}
where $\hat{n}$ is the shell's local propagation direction,
$u:\mathbb{R}^{3}\to\mathbb{R}^{3}$ is the emergent displacement
field, and $\theta_{\perp}$ collects modes orthogonal to the shell
normal. The decomposition is unique up to global shifts.
\end{lem}

\begin{proof}
The shell constraint $|k|\in[q_{0}-\Delta,q_{0}+\Delta]$ forces the
phase gradient $\partial_{i}\theta$ to lie predominantly along the
shell normal. Projecting onto $\hat{n}$ and defining $u$ as the
integrated displacement recovers (\ref{eq:phase-disp}).
\end{proof}

\subsection*{2. Displacement kernel $K_{ij}(p)$}

\begin{defn}[Quadratic displacement kernel]
The quadratic part of the Brazovskii functional expanded around the
locked state gives
\begin{equation}
F_{\mathrm{quad}}^{(u)}[u]
 \;=\;\tfrac{1}{2}\int\!\frac{d^{3}p}{(2\pi)^{3}}\,
      u_{i}(-p)\,K_{ij}(p)\,u_{j}(p),
\qquad
K_{ij}(p)
 \;=\; A(p)\,p_{i}p_{j} \;+\; B(p)\,\delta_{ij}\,p^{2},
\label{eq:Kij}
\end{equation}
where $A,B$ are scalar functions of $|p|$ determined by the
Brazovskii kinetic $(Z,Y)$ and the shell geometry.
\end{defn}

\begin{lem}[Longitudinal--transverse spectrum]
\label{lem:LT}
$K_{ij}(p)$ has eigenvalues
$\lambda_{L}(p)=p^{2}(A+B)$ on the longitudinal subspace and
$\lambda_{T}(p)=p^{2}B$ on the transverse subspace. At one loop
on the locked triple,
\begin{equation}
A(p) \;=\; |Z| \;+\; O(p^{2}),
\qquad
B(p) \;=\; Y\,p^{2} \;+\; O(p^{4}),
\end{equation}
so that, by Math40 Thm 1, $\lambda_{L}(p)\,/\,p^{2}\lambda_{T}(p)
=|Z|/(Y p^{2})=2q_{0}^{2}/p^{2}$ at low $p$.
\end{lem}

\begin{proof}
Direct substitution of the shell gradient expansion of Math40
Lemma 1 into the quadratic functional, followed by
eigen-decomposition of the rank-1 tensor $p_{i}p_{j}$.
\end{proof}

\subsection*{3. Strain tensor and TT projection}

\begin{defn}[Strain and TT projector]
The symmetrised strain is
$\varepsilon_{ij}=\tfrac{1}{2}(\partial_{i}u_{j}+\partial_{j}u_{i})$,
and the transverse-traceless projector on $\mathbb{R}^{3}$ is
\begin{equation}
P^{TT}_{ij,kl}(p)
 \;=\; \tfrac{1}{2}\bigl(P_{ik}P_{jl}+P_{il}P_{jk}\bigr)
       - \tfrac{1}{2}\,P_{ij}P_{kl},
\qquad
P_{ij}=\delta_{ij}-\hat{p}_{i}\hat{p}_{j}.
\end{equation}
The spin-2 component is $h_{ij}^{TT}(p)=P^{TT}_{ij,kl}(p)\,
\varepsilon_{kl}(p)$.
\end{defn}

\begin{lem}[Hessian block decomposition]
\label{lem:H-blocks}
The quadratic functional of the strain admits the decomposition
\begin{equation}
F_{\mathrm{quad}}^{(\varepsilon)}
 \;=\;\tfrac{1}{2}\int\!\frac{d^{3}p}{(2\pi)^{3}}\,
   \Bigl[\,
     H_{TT}(p)\,|h^{TT}(p)|^{2}
     + H_{\mathrm{tr}}(p)\,|\mathrm{tr}\,\varepsilon|^{2}
     + H_{V}(p)\,|h^{V}(p)|^{2}
   \,\Bigr],
\end{equation}
where $h^{V}$ is the transverse-longitudinal (vector) component.
At one loop on the locked triple,
\begin{equation}
H_{TT}(p) \;=\; Y\,p^{4} \;+\; O(p^{6}),
\qquad
H_{\mathrm{tr}}(p) \;=\; (Z+4Y\,p^{2}/3)\,p^{2},
\qquad
H_{V}(p) \;=\; |Z|\,p^{2}.
\end{equation}
\end{lem}

\begin{proof}
Expand $\varepsilon_{kl}(p)$ onto the orthonormal basis
$\{P^{TT},P^{\mathrm{tr}},P^{V}\}$ and collect by block.
\end{proof}

\subsection*{4. Spin-2 purity, Einstein normalisation, universal
coupling (finite audit)}

\begin{thm}[C2 computable purity]
\label{thm:purity}
On the locked triple, and for $|p|\ll q_{0}$,
\begin{equation}
\frac{H_{TT}(p)}{H_{\mathrm{tr}}(p) + H_{V}(p)}
 \;=\;\frac{Y p^{2}}{|Z|}\,\bigl(1+O(p^{2}/q_{0}^{2})\bigr)
 \;=\;\frac{p^{2}}{2 q_{0}^{2}}\,\bigl(1+O(p^{2}/q_{0}^{2})\bigr),
\end{equation}
so the spin-2 channel decouples in the IR at rate $(p/q_{0})^{2}$.
The audit passes if the numerically extracted ratio matches this
scaling within $1\%$ across three refinement grids.
\end{thm}

\begin{proof}
Substitute Lemma \ref{lem:H-blocks} into the ratio and use
Math40 Thm 1.
\end{proof}

\begin{thm}[C2 Einstein-type kinetic form (computable)]
\label{thm:Einstein}
The TT Hessian $H_{TT}(p)=Y p^{4}$ is the Fourier-space form of the
position-space kinetic operator
\begin{equation}
\mathcal{L}_{\mathrm{grav}}^{TT}
 \;=\;\tfrac{1}{2}\,Y\,(\Box h^{TT}_{ij})\,(\Box h^{TT,ij})
 \;\stackrel{\mathrm{IR}}{\longrightarrow}\;
 \tfrac{1}{2\kappa_{G}^{2}}\,(\partial h^{TT}_{ij})^{2},
\qquad
\kappa_{G}^{-2} \;=\; Y\,q_{0}^{2}
 \;=\;\tfrac{1}{2}|Z|,
\end{equation}
where the IR reduction uses the Brazovskii shell projection
$\Box\to -q_{0}^{2}$ on near-shell modes.
\end{thm}

\begin{proof}
$Y p^{4}=Y q_{0}^{2}\cdot p^{2}+Y(p^{4}-q_{0}^{2}p^{2})$, and the
second term is $O(p^{2}(p^{2}-q_{0}^{2}))$ which vanishes on-shell.
Identification with the canonical graviton kinetic term gives the
stated $\kappa_{G}^{-2}$.
\end{proof}

\begin{thm}[C2 universal coupling (species-independent)]
\label{thm:univ}
Let $\alpha$ label any condensate species (Class-I, Class-II, or
Class-III fluctuation) and define the coupling to the TT mode by
the bilinear
\begin{equation}
\mathcal{L}_{\mathrm{int}}^{\alpha}
 \;=\; \kappa_{\alpha}\,h^{TT}_{ij}\,T^{\alpha,ij},
\end{equation}
where $T^{\alpha,ij}$ is the species stress-energy tensor. Then, on
the locked triple,
\begin{equation}
\kappa_{\alpha} \;=\; \kappa_{G}
 \;=\;\bigl(Y q_{0}^{2}\bigr)^{-1/2},
\qquad\text{independent of }\alpha.
\end{equation}
\end{thm}

\begin{proof}
The coupling $\kappa_{\alpha}$ arises from the bilinear
$\partial_{i}u_{j}\,T^{\alpha,ij}$ in the matter action, which,
after TT projection, carries the universal normalisation
$(Y q_{0}^{2})^{-1/2}$ set by the $h^{TT}$ kinetic term. Species
independence follows because the projection $P^{TT}$ acts on the
matter stress-energy without referring to $\alpha$.
\end{proof}

\begin{thm}[C2 actual finite audit]
\label{thm:audit}
The C2 sector closes as an Einstein-type emergent gravity, at
one-loop leading log on the Brazovskii-locked triple, iff the
following three numerical extractor tests pass on three grid
refinements:
\begin{enumerate}
\item[(T1)] Purity: the ratio of Theorem \ref{thm:purity} scales as
$(p/q_{0})^{2}$ with relative error $<1\%$ per refinement.
\item[(T2)] Einstein normalisation: $\kappa_{G}^{-2}$ extracted from
the TT kinetic matches $Y q_{0}^{2}=|Z|/2$ with relative error
$<1\%$.
\item[(T3)] Universality: $\kappa_{\alpha}/\kappa_{G}-1<1\%$ across
all $\alpha\in\{\text{Class-I, Class-II, Class-III}\}$.
\end{enumerate}
\end{thm}
