\section{V1 — Mathematica gauge-invariance verification}
\textbf{Targets}: Math47-48 (Ward identity per generation), Math157
(SO(10) $\mathbf{16}$ anomaly trace, six SM coefficients exact zero).

\textbf{Protocol}:
\begin{enumerate}
\item Construct SO(10) $\mathbf{16}$ representation via Mathematica
\texttt{LieART} package.
\item Compute the six SM anomaly traces $\mathrm{Tr}(T_a T_b T_c)$ for
$a,b,c \in \{Y, T_2, T_3, T_8\}$ symbolically.
\item Verify each trace evaluates to exact zero (modulo Mathematica
symbolic-simplification floor).
\item Cross-check via independent representation construction
(\texttt{GroupTheory} package) if available.
\end{enumerate}

\textbf{Acceptance gate}: All six traces $\equiv 0$ symbolically, OR
$|\mathrm{Tr}| < 10^{-15}$ numerically. Math157's claim ``exact
zero'' must be verified.

\textbf{Estimated effort}: 1-2 days for setup + verification on a
mathematician with Mathematica access.

\section{V2 — SageMath Chern-class verification}
\textbf{Targets}: Math174 ($c_2(E) = -40$ on $\mathbb{CP}^2$),
Math305 ($c_2(E)$ on $\Sigma_0 = \mathbb{P}^1\times\mathbb{P}^1$).

\textbf{Protocol}:
\begin{enumerate}
\item Use SageMath \texttt{RiemannSurface} or \texttt{ToricVariety}
machinery to construct $\mathbb{CP}^2$ + bundle $E$ from Math174
characterization.
\item Compute $c_2(E)$ via direct Chern-class arithmetic
(\texttt{c2(bundle)} or via splitting principle).
\item Verify result against Math174 claim ($-40$).
\item Repeat for $\Sigma_0 = \mathbb{P}^1 \times \mathbb{P}^1$ (Math305
predicts $\sim 90$ for SO(10) adjoint, convention-dependent).
\end{enumerate}

\textbf{Acceptance gate}: SageMath result matches Math174 / Math305
to integer precision (Chern classes are integers).

\textbf{Estimated effort}: 2-3 days for setup + verification.

\section{V3 — Sage GAP SO(10) representation verification}
\textbf{Targets}: Math229 (Cartan-forcing breaking chain), Math242
(cubic-sublattice forcing of SO(10) → SU(5)×U(1)$_\chi$).

\textbf{Protocol}:
\begin{enumerate}
\item Use Sage GAP (or standalone GAP) to construct SO(10) Lie algebra +
$\mathbf{16}$ spinor representation.
\item Verify maximal subgroup decomposition: SO(10) → SU(5) × U(1)$_\chi$,
SU(5) × SU(5), Pati-Salam, Spin(6) × U(1) (rule out non-maximal).
\item For each candidate maximal subgroup, compute the $\mathbf{16}$
restriction (e.g., SO(10) $\mathbf{16}$ → SU(5) $\mathbf{10}+\bar{\mathbf{5}}+\mathbf{1}$).
\item Verify cubic-sublattice O$_h$ commutativity with each maximal
subgroup.
\item Confirm Math242 conclusion: only SU(5) × U(1)$_\chi$ has
O$_h$-commuting Cartan.
\end{enumerate}

\textbf{Acceptance gate}: GAP confirms the $\mathbf{16}$ branching for
SU(5) × U(1)$_\chi$ + the O$_h$ Cartan-commutativity ⇒ SU(5) × U(1)$_\chi$
unique among Math242 candidates.

\textbf{Estimated effort}: 3-5 days for setup + verification (GAP has
substantial learning curve for SO(10) representation theory).

\section{Theorem 319.1 statement}

\begin{theorem}[External-tool verification protocol]
\label{thm:protocol}
Math319 specifies V1-V3 verification protocols. Each Vi is independent
of TECT-internal Math notes (i.e., uses external symbolic engines +
representation libraries). Joint verification (all three PASS) provides
strong third-party confirmation of the load-bearing Math47-48, Math157,
Math174, Math229, Math242, Math305 theorems.
\end{theorem}

\begin{proof}
V1 is independent because Mathematica's anomaly computation uses its
own representation-theory library (LieART or GroupTheory), not Math157.
V2 is independent because SageMath's Chern-class arithmetic uses its
own toric/algebraic-geometry libraries, not Math174's splitting-principle
derivation. V3 is independent because GAP uses its own SO(10) Lie algebra
construction, not Math229/242's O$_h$-commutativity arguments.
The three together cover symbolic computation (V1), algebraic geometry
(V2), and group/representation theory (V3) — three distinct mathematical
domains with three distinct external-engine implementations. Joint PASS
rules out single-engine bugs. \qed
\end{proof}
