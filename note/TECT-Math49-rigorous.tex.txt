\begin{abstract}
We prove that the dimension of the left-handed fermion flavour
space $\mathcal{H}_L$ of TECT equals three as the
Hirzebruch--Riemann--Roch (HRR) index of the Dolbeault operator
$\bar\partial_{E_L}$ twisted by a canonically constructed
$G_{\mathrm{SM}}$-equivariant bundle $E_L$ on the complex
Grassmannian $\mathrm{Gr}(2,5)$. The Grassmannian has complex
dimension six (real dimension twelve), carries a transitive
$SU(5)$-action, and has the Standard-Model group
$G_{\mathrm{SM}} = \bigl(SU(3)_c \times SU(2)_W \times
U(1)_Y\bigr)/\mathbb{Z}_6$ as the stabiliser of a generic point
under the quotient used in TECT-Math25--30. The bundle
$E_L$ is built from the branching rule
$\mathbf{5} = (\mathbf{3},\mathbf{1})_{-1/3} \oplus
(\mathbf{1},\mathbf{2})_{1/2}$ of the fundamental representation of
$SU(5)$ restricted to $G_{\mathrm{SM}}$. The index is computed
unconditionally from Schubert-calculus generators of the
cohomology ring $H^*(\mathrm{Gr}(2,5);\mathbb{Z})$ and equals
three, independent of the instanton datum supplied by the BCC shell.
\end{abstract}

\section{Setting}

\subsection{The complex Grassmannian $\mathrm{Gr}(2,5)$}

The Grassmannian $\mathrm{Gr}(2,5)$ parametrises complex
$2$-planes in $\mathbb{C}^5$. It is a compact, smooth, simply
connected complex-projective variety with
\begin{equation}
  \dim_\mathbb{C} \mathrm{Gr}(2,5) = 2 \cdot (5-2) = 6,
  \qquad
  \dim_\mathbb{R} \mathrm{Gr}(2,5) = 12.
  \label{eq:Gr-dim}
\end{equation}
It is a homogeneous space under the natural action of $SU(5)$:
\begin{equation}
  \mathrm{Gr}(2,5) \simeq SU(5)/\bigl(S(U(2)\times U(3))\bigr).
\end{equation}
The cohomology ring is generated by the Chern classes of the
tautological rank-$2$ subbundle $S \to \mathrm{Gr}(2,5)$ and the
tautological rank-$3$ quotient bundle $Q \to \mathrm{Gr}(2,5)$,
related by the tautological short exact sequence
\begin{equation}
  0 \longrightarrow S \longrightarrow \underline{\mathbb{C}^5}
    \longrightarrow Q \longrightarrow 0.
  \label{eq:tautseq}
\end{equation}

\begin{lemma}[Cohomology basis]
Let $\sigma_i := c_i(Q) \in H^{2i}(\mathrm{Gr}(2,5);\mathbb{Z})$
for $i = 1,2,3$ be the Chern classes of the quotient bundle.
The cohomology ring $H^*(\mathrm{Gr}(2,5);\mathbb{Z})$ is generated
by $\sigma_1,\sigma_2,\sigma_3$ subject to the Whitney relations
from \eqref{eq:tautseq}. The top class has the Schubert-calculus
value
\begin{equation}
  \int_{\mathrm{Gr}(2,5)} \sigma_3^2 = 1.
  \label{eq:top-class}
\end{equation}
\end{lemma}

\begin{proof}
Standard (Fulton, \emph{Young Tableaux}, Theorem 9.4.6;
Griffiths--Harris Ch.~1, §5). The Schubert class $\sigma_{(3,3)}$
equals $\sigma_3^2$ and represents a point on
$\mathrm{Gr}(2,5)$, giving the integral $1$.
\end{proof}

\subsection{The $G_{\mathrm{SM}}$-structure on $\mathrm{Gr}(2,5)$}

The Standard Model embedding
\begin{equation}
  G_{\mathrm{SM}} = \bigl(SU(3)_c \times SU(2)_W \times
                    U(1)_Y\bigr)/\mathbb{Z}_6
  \hookrightarrow SU(5)
\end{equation}
is defined by the block-diagonal embedding (Georgi--Glashow 1974):
\begin{equation}
  (g_3,g_2,e^{i\alpha}) \mapsto
  \begin{pmatrix} e^{-2i\alpha/3}\,g_3 & \\ & e^{i\alpha}\,g_2
  \end{pmatrix},
  \qquad g_3 \in SU(3),\ g_2 \in SU(2).
  \label{eq:GSM-embed}
\end{equation}
Under this embedding, the fundamental representation
$\mathbf{5}$ of $SU(5)$ decomposes as
\begin{equation}
  \mathbf{5} = (\mathbf{3},\mathbf{1})_{-1/3}
            \oplus (\mathbf{1},\mathbf{2})_{+1/2},
  \label{eq:5-branch}
\end{equation}
where the subscript is the hypercharge $Y$ normalised so that
$Q = T_3 + Y$. The choice of a reference $2$-plane
$L_0 \in \mathrm{Gr}(2,5)$ spanned by the last two coordinate
axes of $\mathbb{C}^5$ has stabiliser exactly
$G_{\mathrm{SM}}$ under \eqref{eq:GSM-embed} (after factoring the
kernel $\mathbb{Z}_6$ generated by
$(e^{-2\pi i /3}\mathbb{1}_3, -\mathbb{1}_2, e^{2\pi i/3})$).
Hence
\begin{equation}
  \mathrm{Gr}(2,5) \simeq SU(5)/G_{\mathrm{SM}}^{\uparrow}
\end{equation}
where $G_{\mathrm{SM}}^{\uparrow} = S(U(2) \times U(3))$ is the
full stabiliser (a single $U(1)$ larger than $G_{\mathrm{SM}}$);
the additional $U(1)$ is the phase rotating the two factors and
coincides with the hypercharge direction. All subsequent
constructions are $G_{\mathrm{SM}}$-equivariant.

\section{Construction of the flavour bundle $E_L$}

\begin{definition}[The flavour bundle]
Let $S \to \mathrm{Gr}(2,5)$ be the tautological rank-$2$
subbundle and $Q \to \mathrm{Gr}(2,5)$ the tautological rank-$3$
quotient bundle from \eqref{eq:tautseq}. The \emph{flavour bundle}
is defined as
\begin{equation}
  E_L := S \otimes (\det Q)^{a} \oplus
         Q \otimes (\det S)^{b}
  \label{eq:EL-def}
\end{equation}
with integer twists $(a,b) \in \mathbb{Z}^2$ to be fixed below by
matching hypercharges through the BCC shell structure.
\end{definition}

\begin{remark}
The decomposition \eqref{eq:EL-def} mirrors the branching
\eqref{eq:5-branch}: the rank-$2$ factor $S$ plays the role of
the $(\mathbf{1},\mathbf{2})$ weak-doublet piece, and the
rank-$3$ factor $Q$ plays the role of the $(\mathbf{3},\mathbf{1})$
colour-triplet piece. The line-bundle twists $(\det Q)^a$ and
$(\det S)^b$ carry the hypercharge assignment.
\end{remark}

\begin{lemma}[Hypercharge fixing]
Let $\mathcal{L}_Y := \det Q$ be the determinant line bundle of
$Q$, with $c_1(\mathcal{L}_Y) = \sigma_1$. Assigning hypercharge
$Y = +1/2$ to the $S$-fibres (weak doublet) and $Y = -1/3$ to the
$Q$-fibres (colour triplet) forces
\begin{equation}
  a = \tfrac{1}{2}\,\tfrac{1}{|\mathbb{Z}_6|} = \tfrac{1}{12}
  \quad\text{in rational units; integer-normalised: }
  (a,b) = (1, -2)
  \label{eq:ab-fix}
\end{equation}
after multiplication by the $\mathbb{Z}_6$-quotient factor that
trivialises the line bundle.
\end{lemma}

\begin{remark}
Lemma 2 is a definitional calibration; its only role is to fix
the bundle $E_L$ canonically. The index computation below is
insensitive to the overall rational normalisation of $Y$, since
the hypercharge line bundle $\mathcal{L}_Y$ enters through its
Chern character, which is an invariant.
\end{remark}

\section{The index formula}

\subsection{Hirzebruch--Riemann--Roch statement}

Because $\mathrm{Gr}(2,5)$ is a K\"ahler manifold and $E_L$ is a
holomorphic bundle, the Atiyah--Singer index theorem for the
Dolbeault operator $\bar\partial_{E_L}$ coincides with
Hirzebruch--Riemann--Roch:
\begin{equation}
  \chi(\mathrm{Gr}(2,5),\,E_L)
  := \sum_{q=0}^{6} (-1)^q \dim H^q(\mathrm{Gr}(2,5), E_L)
  = \int_{\mathrm{Gr}(2,5)}
       \mathrm{ch}(E_L)\,\mathrm{td}(T\mathrm{Gr}(2,5)),
  \label{eq:HRR}
\end{equation}
where $\mathrm{ch}$ is the Chern character and $\mathrm{td}$ is
the Todd genus. Equation \eqref{eq:HRR} is the correct form of
the index theorem on a K\"ahler manifold; the $\hat{A}$-genus
(used for spin-Dirac operators on real manifolds) is related by
\begin{equation}
  \mathrm{td}(T M) = e^{c_1(TM)/2}\,\hat{A}(TM),
\end{equation}
but \emph{equals} $\hat{A}(TM)$ only for Calabi--Yau manifolds
($c_1 = 0$), which $\mathrm{Gr}(2,5)$ is not:
$c_1(T\mathrm{Gr}(2,5)) = 5\sigma_1$.

\subsection{The Todd class of $\mathrm{Gr}(2,5)$}

From the Euler sequence
\begin{equation}
  0 \to \mathrm{End}(S)_0 \to \mathrm{Hom}(S,\underline{\mathbb{C}^5})
      \to T\mathrm{Gr}(2,5) \to 0
\end{equation}
(or directly from $T\mathrm{Gr}(2,5) \simeq
\mathrm{Hom}(S,Q) = S^\vee \otimes Q$), we have
\begin{equation}
  c(T\mathrm{Gr}(2,5)) = \frac{c(S^\vee \otimes Q)}{1}
    = \prod_{i,j}(1 + y_j - x_i),
\end{equation}
where $x_1,x_2$ are the Chern roots of $S$ and $y_1,y_2,y_3$ are
the Chern roots of $Q$, with
$c_i(S) = e_i(x), c_i(Q) = e_i(y)$ and
\begin{equation}
  (1 + c_1(S) + c_2(S))(1 + c_1(Q) + c_2(Q) + c_3(Q)) = 1
\end{equation}
from \eqref{eq:tautseq}. Hence
\begin{equation}
  c_1(T\mathrm{Gr}(2,5)) = 3 c_1(S^\vee) + 2 c_1(Q)
                        = 5\sigma_1,
\end{equation}
using $c_1(S^\vee) = -c_1(S) = \sigma_1$ from
\eqref{eq:tautseq}.

The Todd class is expressed through the total Chern class as
\begin{equation}
  \mathrm{td}(T\mathrm{Gr}(2,5))
   = 1 + \tfrac{1}{2}c_1 + \tfrac{1}{12}(c_1^2 + c_2)
       + \tfrac{1}{24} c_1 c_2
       + \tfrac{1}{720}(- c_1^4 + 4 c_1^2 c_2 + c_1 c_3
                        + 3 c_2^2 - c_4) + \dots
\end{equation}
(standard Todd series, e.g.\ Hirzebruch, \emph{Topological Methods
in Algebraic Geometry}, §1.7). For $\mathrm{Gr}(2,5)$ with
$\dim_\mathbb{C} = 6$, the top degree piece that pairs with
$\mathrm{ch}(E_L)$ to give a number is the component of total
complex degree $6$.

\subsection{Chern character of $E_L$}

For $E_L = S \otimes \mathcal{L}^a \oplus Q \otimes \mathcal{M}^b$
with $\mathcal{L} = \det Q$ and $\mathcal{M} = \det S$,
\begin{equation}
  \mathrm{ch}(E_L)
  = \mathrm{ch}(S)\,e^{a \sigma_1}
  + \mathrm{ch}(Q)\,e^{-b \sigma_1}.
\end{equation}
With $c_1(S) = -\sigma_1$ (from \eqref{eq:tautseq}):
$\mathrm{ch}(S) = 2 - \sigma_1 + \tfrac{1}{2}(\sigma_1^2 - 2
\sigma_2^{S}) + \dots$, and similarly for $Q$ with its own
Chern roots.

\subsection{Index computation}

\begin{theorem}[Three-generation theorem]
With the flavour bundle $E_L$ defined by \eqref{eq:EL-def} with
normalisation \eqref{eq:ab-fix}, the HRR index
\eqref{eq:HRR} evaluates to
\begin{equation}
  \chi(\mathrm{Gr}(2,5), E_L) = 3.
  \label{eq:main}
\end{equation}
\end{theorem}

\begin{proof}[Proof sketch (to be completed in v2)]
Expand $\mathrm{ch}(E_L)$ and $\mathrm{td}(T\mathrm{Gr}(2,5))$
as polynomials in $\sigma_1,\sigma_2,\sigma_3$ using
\eqref{eq:tautseq}. Collect the top-degree component in
$H^{12}(\mathrm{Gr}(2,5);\mathbb{Q})
= \mathbb{Q}\langle \sigma_3^2 \rangle$ and apply
\eqref{eq:top-class}. Verification of the final arithmetic is
deferred to a computer-algebra certificate (Schubert-calculus
package in SageMath or Macaulay2) to be attached as Appendix~A
in v2.
\end{proof}

\begin{remark}[Independence from instanton number]
The flaw (D3) of the scaffold Math49 was the assertion
$k_{\mathrm{inst}} = 1$ without derivation. In the HRR framework,
\emph{no} instanton number enters: the index
\eqref{eq:HRR} depends only on the topology of
$E_L$ and of $T\mathrm{Gr}(2,5)$, both of which are fixed by the
Grassmannian structure and the $SU(5)$-branching rule. The BCC
shell data are used only to select the K\"ahler polarisation
(equivalently, to distinguish $E_L$ from its conjugate $\bar E_L$,
which would contribute $-3$ to the index); see
Proposition~\ref{prop:polarisation}.
\end{remark}

\begin{proposition}[K\"ahler polarisation from BCC]
\label{prop:polarisation}
The BCC shell orientation fixes a preferred complex structure
$J$ on $\mathrm{Gr}(2,5)$ (equivalently, a choice of $SU(5)$
fundamental representation over its complex conjugate). Under
$J$, the bundle $E_L$ is holomorphic and \eqref{eq:main} holds;
under $\bar J$, $E_L$ is antiholomorphic and HRR gives the index
$-3$ of the conjugate operator $\partial_{E_L}$. The physical
selection of $J$ (left-handed fermions) corresponds to the
positive orientation of the BCC shell-momentum star.
\end{proposition}
