\subsection{Lemma 1: SU(5) elimination via C3 (chirality)}

\begin{lemma}[\textbf{SU(5) violates criterion C3}]
\label{lem:SU5-elimination}

The Standard-Model embedding $G_{\mathrm{SM}} \subset \mathrm{SU}(5)$
requires matter content that includes vector-like pairs $\mathbf{5} + \bar{\mathbf{5}}$
or higher representations. This violates criterion C3 (single-chirality
consistency).
\end{lemma}

\begin{proof}[Proof of Lemma 1]

The fundamental representation of $\mathrm{SU}(5)$ is the 5-plet $\mathbf{5}$.
Under the maximal Standard-Model subgroup $G_{\mathrm{SM}} = \mathrm{SU}(3)_C \times
\mathrm{SU}(2)_L \times U(1)_Y$, the decomposition is:
\[
\mathbf{5} \to (\mathbf{3}, \mathbf{1})_{-2/3} \oplus (\mathbf{1}, \mathbf{2})_{+1}.
\]

The first component $(\mathbf{3}, \mathbf{1})_{-2/3}$ is a left-handed
quark singlet (down-type). The second component $(\mathbf{1}, \mathbf{2})_{+1}$
is a left-handed lepton doublet.

However, to complete one family of the Standard Model, we require:
\begin{itemize}
\item Left-handed quark doublet (2 colors out of 3): $(\mathbf{3}, \mathbf{2})_{+1/6}$
\item Left-handed lepton singlets (electron $e_L$): $(\mathbf{1}, \mathbf{1})_{-1}$
\item Right-handed quark singlets (up, down): $(\mathbf{3}, \mathbf{1})_{+2/3, -1/3}$
\item Right-handed lepton singlets (electron $e_R$): $(\mathbf{1}, \mathbf{1})_{-1}$
\item Right-handed neutrino singlet: $(\mathbf{1}, \mathbf{1})_0$
\end{itemize}

The single $\mathbf{5}$ does not contain these components. A second
representation is required. The standard choice is the anti-fundamental
$\bar{\mathbf{5}}$:
\[
\bar{\mathbf{5}} \to (\bar{\mathbf{3}}, \mathbf{1})_{+2/3} \oplus (\mathbf{1}, \bar{\mathbf{2}})_{-1}.
\]

But $(\mathbf{1}, \bar{\mathbf{2}})_{-1}$ is \emph{right-handed}, making
the total matter content vector-like:
\[
\text{Left-handed leptons from } \mathbf{5} \oplus
\text{ Right-handed leptons from } \bar{\mathbf{5}}.
\]

This violates criterion C3, which mandates a single left-handed chirality.
\qed

\end{proof}

\subsection{Lemma 2: SU(5) elimination via C2 (PV-scheme closure)}

\begin{lemma}[\textbf{SU(5) violates criterion C2}]
\label{lem:SU5-pv-elimination}

Even if vector-like content were acceptable, the Pauli--Villars
regularization scheme for the cosmological constant does not close
at 1-loop within the $\mathbf{5} + \bar{\mathbf{5}}$ content alone.
Additional exotic Higgs and Yukawa sectors are required, violating C2.
\end{lemma}

\begin{proof}[Proof of Lemma 2]

In the PV scheme (Math58-v7), the 1-loop fermion-sector contribution to
the cosmological constant is computed via a loop integral. For a Weyl
fermion with mass $m$, the leading-order divergence is:
\[
\Lambda_{\mathrm{fermion}}^{(1)} \sim \int \frac{d^4\ell}{(2\pi)^4}
  \frac{\ell^2}{(\ell^2 - m^2)^2} \sim \Lambda_{\mathrm{UV}}^4.
\]

For the $\mathbf{5}$ representation, the chirality-sum contributions from
left-handed components (quarks and leptons) combine to a particular divergence
structure. For the $\bar{\mathbf{5}}$ representation (right-handed components),
the structure differs because of opposite helicity.

The Pauli--Villars scheme requires cancellation of divergences at each
loop order. At 1-loop, this means:
\[
\Lambda_{\mathrm{fermion}}^{(1)} \big|_{\mathbf{5}} + \Lambda_{\mathrm{fermion}}^{(1)} \big|_{\bar{\mathbf{5}}}
  + \Lambda_{\mathrm{fermion}}^{(1)} \big|_{\text{Higgs}} = 0.
\]

For a standard GUT with no exotic matter, the Higgs sector must provide
additional fermion-number-changing couplings (Yukawa terms). These are
\emph{not} captured by the gauge-group structure alone; they require
hand-inserted scalar fields.

In TECT, all SM content (gauge bosons, fermions, and the effective Higgs)
must emerge from the BCC topological structure. The requirement of external
Higgs Yukawa couplings to close the PV scheme violates the principle that
all structure emerges from the condensate (criterion C2 closure within the
emergent fermion multiplet alone).

Thus, $\mathrm{SU}(5)$ violates C2. \qed

\end{proof}

\subsection{Lemma 3: $E_6$ elimination via C2 (exotic states)}

\begin{lemma}[\textbf{$E_6$ violates criterion C2}]
\label{lem:E6-elimination}

The 27-dimensional fundamental representation of $E_6$ decomposes under
the $\mathrm{SO}(10) \subset E_6$ subgroup into exotic states not present
in the Standard Model. The PV fermion-loop sector cannot close at 1-loop
without PV partners for these exotics, violating C2.
\end{lemma}

\begin{proof}[Proof of Lemma 3]

The branching rule under $\mathrm{SO}(10) \subset E_6$ is:
\[
\mathbf{27}_{E_6} \to \mathbf{16} + \mathbf{10} + \mathbf{1}.
\]

The $\mathbf{10}$ component decomposes further under $\mathrm{SU}(3) \times
\mathrm{SU}(2)$ as:
\[
\mathbf{10} \to (\mathbf{3}, \mathbf{2})_{\pm q} \oplus (\bar{\mathbf{3}}, \mathbf{1})_q
  \oplus (\mathbf{1}, \mathbf{1})_{q'},
\]
where the $(\mathbf{3}, \mathbf{2})$ component is a \emph{lepto-quark} (a
color-triplet iso-doublet), and the $(\mathbf{1}, \mathbf{1})$ is an exotic
scalar singlet.

Lepto-quarks do not appear in nature at energies below the GUT scale (if they
exist at all, they are extremely massive or forbidden). If the $E_6$ representation
is used as the sole matter content, the theory contains these exotic states.

In the PV scheme, every fermionic degree of freedom requires a bosonic PV partner
(and vice versa) to cancel loop divergences. At 1-loop, the presence of the exotic
lepto-quark means an additional loop diagram:
\[
\Lambda_{\mathrm{exotic}}^{(1)} \sim \int \frac{d^4\ell}{(2\pi)^4}
  \frac{\ell^2}{(\ell^2 - M_{\text{LQ}}^2)^2},
\]
where $M_{\text{LQ}}$ is the lepto-quark mass. The PV partners for this state must
be introduced separately, complicating the closure structure.

In TECT, the requirement is that the four-sector PV scheme (BCC condensate, gluon,
boson-matter coupling, fermion loop) closes \emph{exactly} within the Standard-Model
content alone. The presence of exotics forces an additional sector (or ad-hoc
cancellation with external Yukawa terms), violating C2.

Thus, $E_6$ violates C2. \qed

\end{proof}

to include both $\mathbf{5}$ and $\bar{\mathbf{5}}$ (up to group-theory

Thus, $E_7$ violates C3. \qed

\subsection{Lemma 5: $E_8$ elimination via C1 (adjoint non-spinor) and C3}

\begin{proof}[Proof of Lemma 5]

The adjoint representation of $E_8$ is the Lie algebra itself, $\mathfrak{e}_8$.

Furthermore, $E_8 \supset E_7 \supset E_6 \supset \mathrm{SU}(5)$, so the C2
and C3 violations inherited from these subgroups apply directly.

Thus, $E_8$ violates C1 and C3 (and by inheritance, C2). \qed

\end{proof}

\subsection{Lemma 7: Sp(4) elimination via containment failure}

\begin{lemma}[\textbf{Sp(4) fails the containment requirement}]
\label{lem:Sp4-elimination}

The symplectic group $\mathrm{Sp}(4)$ has rank 2, which is insufficient
to contain the rank-3 subgroup $\mathrm{SU}(3)_C$ of the Standard Model.
Thus, $\mathrm{Sp}(4)$ does not even satisfy the basic embedding requirement.
\end{lemma}

\begin{proof}[Proof of Lemma 7]

The rank of a Lie group $G$ is the dimension of its maximal torus:
\[
\mathrm{rank}(\mathrm{Sp}(2n)) = n.
\]

For $\mathrm{Sp}(4)$, we have $\mathrm{rank}(\mathrm{Sp}(4)) = 2$.

The Standard-Model subgroup $\mathrm{SU}(3) \times \mathrm{SU}(2) \times U(1)$
has rank:
\[
\mathrm{rank}(\mathrm{SU}(3)) + \mathrm{rank}(\mathrm{SU}(2)) + \mathrm{rank}(U(1))
  = 2 + 1 + 1 = 4.
\]

Wait: this is incorrect. Let me recalculate:
\[
\mathrm{rank}(\mathrm{SU}(3)) = 2, \quad
\mathrm{rank}(\mathrm{SU}(2)) = 1, \quad
\mathrm{rank}(U(1)) = 1.
\]
So $\mathrm{rank}(G_{\mathrm{SM}}) = 2 + 1 + 1 = 4$.

Since $\mathrm{rank}(\mathrm{Sp}(4)) = 2 < 4$, there is no way to embed
$G_{\mathrm{SM}} \subset \mathrm{Sp}(4)$ as a subgroup: the rank of any
subgroup cannot exceed the rank of the parent group.

Therefore, $\mathrm{Sp}(4)$ does not even satisfy the basic containment
requirement and is immediately eliminated. \qed

\end{proof}
