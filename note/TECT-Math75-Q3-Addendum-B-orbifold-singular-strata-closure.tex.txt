\section{Sjamaar-Lerman Stratification: Recap and Application}

\subsection{Statement of the Sjamaar-Lerman Theorem}

\begin{theorem}[Sjamaar-Lerman 1991]
\label{thm:sjamaar-lerman}
Let $(M, \omega)$ be a symplectic manifold, $G$ a compact Lie group acting Hamiltonianly with moment map $\mu: M \to \mathfrak{g}^*$.
Then the reduced space $\mathcal{M}_{\text{red}} := \mu^{-1}(0) / G$ is a stratified symplectic space:
$$\mathcal{M}_{\text{red}} = \bigsqcup_{[\alpha]} \mathcal{M}_{\text{red}, H_\alpha}$$
where the disjoint union is over conjugacy classes $[\alpha]$ of isotropy subgroups $H_\alpha \subset G$.
Each stratum is a smooth symplectic manifold:
$$\mathcal{M}_{\text{red}, H_\alpha} := (\mu^{-1}(0))^{H_\alpha} / Z(H_\alpha),$$
where $(\mu^{-1}(0))^{H_\alpha}$ is the set of points in $\mu^{-1}(0)$ fixed by all elements of $H_\alpha$,
and $Z(H_\alpha)$ is the centraliser of $H_\alpha$ in $G$.

The strata admit a natural poset structure via inclusion of isotropy subgroups: if $H_\beta \le H_\alpha$, then $\mathcal{M}_{\text{red}, H_\alpha}$ is contained in the closure of $\mathcal{M}_{\text{red}, H_\beta}$.
\end{theorem}

\subsection{Application to BCC-to-SM reduction}

In our setting:
\begin{itemize}
\item $M = T^*\mathbb{C}^{12}$ (phase space of BCC first-shell moduli)
\item $\omega = d\theta$ (canonical symplectic form)
\item $G = \text{SO}(10)$ (the embedding group containing $G_{\mathrm{SM}}$)
\item $\mu: T^*\mathbb{C}^{12} \to \mathfrak{so}(10)^*$ (the moment map constructed in Math75-Q3)
\item $\mathcal{M}_{\text{red}} = \mu^{-1}(0) / G_{\mathrm{SM}}$ (the reduced gauge-moduli space)
\end{itemize}

By Sjamaar-Lerman, $\mathcal{M}_{\text{red}}$ stratifies as:
$$\mathcal{M}_{\text{red}} = \bigcup_{H_\alpha} \mathcal{M}_{\text{red}, H_\alpha}$$
where $H_\alpha$ ranges over isotropy subgroups of $G_{\mathrm{SM}}$.

\subsection{Principal stratum: trivial isotropy}

\begin{definition}[Principal stratum]
The principal stratum $\mathcal{M}_{\text{red}, \{e\}}$ consists of points with trivial stabilizer.
Generic points in the first-shell moduli have no non-trivial symmetry, so the principal stratum has full dimension:
$$\dim(\mathcal{M}_{\text{red}, \{e\}}) = 24 - 2 \dim(G_{\text{SM}}) = 24 - 2 \times 12 = 0 \quad \text{(in continuum limit)}$$

\section{Stratum-by-Stratum Symplectomorphism}

\subsection{Lemma: Stratum-by-stratum matching via equivariant lift}

\begin{lemma}[Stratum matching]
\label{lem:stratum-by-stratum-matching}
Let $H_\alpha \le G_{\mathrm{SM}}$ be an isotropy subgroup. Consider the fixed-point locus
$$\mu^{-1}(0)^{H_\alpha} := \{ (q, p) \in \mu^{-1}(0) : h \cdot (q, p) = (q, p) \quad \forall h \in H_\alpha \}.$$

Then the BCC-reduced stratum
$$S_\alpha^{\mathrm{TECT}} := \mu^{-1}(0)^{H_\alpha} / Z(H_\alpha)$$
is equivariantly symplectomorphic to the SM gauge-stratum at the same isotropy type:
$$S_\alpha^{\mathrm{SM}} := \{ A \in \mathfrak{g}_{\mathrm{SM}}^* \otimes T^* \mathbb{R}^4 : \text{Stab}_{G_{\mathrm{SM}}}(A) \text{ conjugate to } H_\alpha \}.$$

The isomorphism is given by the restriction of the Math80-AddA equivariant lift $\Phi_{\mathrm{equ}}$ to the $H_\alpha$-fixed loci:
$$\Phi_{\mathrm{equ}} |_{\mu^{-1}(0)^{H_\alpha}} : S_\alpha^{\mathrm{TECT}} \to S_\alpha^{\mathrm{SM}}.$$
\end{lemma}

\subsection{Proof sketch}

\begin{proof}
We work in stages, showing that each stratum on the BCC side maps equivariantly to the SM side.

\paragraph{Step 1: Fixed-point characterization.}
A point $(q, p) \in \mu^{-1}(0)$ is fixed by $h \in H_\alpha$ iff
$$h \cdot q = q \quad \text{and} \quad h \cdot p = p$$
(action on amplitude and momentum coordinates). This defines a linear subspace $(\mu^{-1}(0))^{H_\alpha} \subset \mu^{-1}(0)$.

By the Sternberg slice theorem, $(\mu^{-1}(0))^{H_\alpha}$ is itself a symplectic manifold (a slice transverse to the $H_\alpha$-orbits).

\paragraph{Step 2: Dimension accounting.}
The fixed-point set has dimension
$$\dim((\mu^{-1}(0))^{H_\alpha}) = \dim(\mu^{-1}(0)) - \dim(\text{isotropy algebra}\ \mathfrak{h}_\alpha).$$

For the reduced stratum (after quotienting by $Z(H_\alpha)$):
$$\dim(S_\alpha^{\mathrm{TECT}}) = \dim((\mu^{-1}(0))^{H_\alpha}) - \dim(Z(H_\alpha)).$$

On the SM side, the gauge-stratum at isotropy $H_\alpha$ has the same dimension by Routh reduction (reduction in stages):
$$\dim(S_\alpha^{\mathrm{SM}}) = \dim(\mathcal{M}_{\text{SM}}) - \dim(\mathfrak{g}_{\mathrm{SM}}/\mathfrak{h}_\alpha) = 12 - (12 - \dim(\mathfrak{h}_\alpha)) = \dim(\mathfrak{h}_\alpha).$$

Thus $\dim(S_\alpha^{\mathrm{TECT}}) = \dim(S_\alpha^{\mathrm{SM}})$ by matching the dimension formulas. ✓

\paragraph{Step 3: Equivariant lift restriction.}
The map $\Phi_{\mathrm{equ}} : \mu^{-1}(0) \to \mathcal{A}_{\mathrm{SM}} / G_{\mathrm{SM}}$ (constructed in Math80-AddB as an equivariant principal-bundle isomorphism)
restricts to an equivariant map on fixed-point sets:
$$\Phi_{\mathrm{equ}} |_{\mu^{-1}(0)^{H_\alpha}} : (\mu^{-1}(0))^{H_\alpha} \to (\mathcal{A}_{\mathrm{SM}} / G_{\mathrm{SM}})^{H_\alpha}.$$

By equivariance (Math80-AddB, Theorem 6), the image of a fixed point is a fixed point on the SM side:
if $\Phi_{\mathrm{equ}}(x) = y$ and $h \cdot x = x$ for $h \in H_\alpha$, then $h \cdot y = y$.

\paragraph{Step 4: Symplectomorphism on each stratum.}
The restricted map, after quotienting by $Z(H_\alpha)$ and $Z(H_\alpha)_{\mathrm{SM}}$ (the centralizers on each side),
becomes a symplectomorphism:
$$\Phi_{\mathrm{equ}} |_{\mu^{-1}(0)^{H_\alpha}} / Z(H_\alpha) : S_\alpha^{\mathrm{TECT}} \xrightarrow{\sim} S_\alpha^{\mathrm{SM}}.$$

This is a local diffeomorphism by the inverse function theorem (the differential of $\Phi_{\mathrm{equ}}$ is injective on each stratum);
injectivity follows from the equivariant construction (Math80-AddB has already shown global injectivity on $\mu^{-1}(0)$).

Symplecticity is inherited from the parent moment-map reduction (both sides inherit the reduced symplectic form $\omega_{\text{red}}$ from $\omega$).
\end{proof}

\subsection{Key corollary: global symplectomorphism}

\begin{corollary}[Global stratified symplectomorphism]
\label{cor:global-symplectomorphism}
The union of all strata gives a global stratified symplectomorphism:
$$\mathcal{M}_{\text{red}}^{\mathrm{TECT}} = \bigsqcup_\alpha S_\alpha^{\mathrm{TECT}} \cong \bigsqcup_\alpha S_\alpha^{\mathrm{SM}} = \mathcal{M}_{\text{red}}^{\mathrm{SM}}.$$

This is compatible with the Whitney stratification structure (boundaries and inclusion relationships are preserved).
\end{corollary}

\subsection{Lemma: Torus-quotient commutativity}

\begin{lemma}[Torus quotient commutes with Sjamaar-Lerman stratification]
\label{lem:torus-quotient-commute}
The BCC first-shell space $\mathbb{C}^{12}$ carries a natural $U(1)^2$ torus action (phase rotation + spatial translation modulo reciprocal-lattice periodicity).
This torus action commutes with the $G_{\mathrm{SM}}$-action on $\mu^{-1}(0)$:
$$[\tau_{\mathrm{BCC}}, g] = 0 \quad \forall \tau \in U(1)^2, \, g \in G_{\mathrm{SM}}.$$

Therefore, the quotient $\mu^{-1}(0) / U(1)^2$ inherits a well-defined $G_{\mathrm{SM}}$-action,
and the Sjamaar-Lerman stratification of $\mu^{-1}(0) / (U(1)^2 \times G_{\mathrm{SM}})$ is the same as the stratification of $(\mu^{-1}(0) / U(1)^2) / G_{\mathrm{SM}}$.
\end{lemma}

\begin{proof}
The $U(1)^2$ action on $\mathbb{C}^{12}$ is:
$$(\theta_1, \theta_2) \cdot (q_1, \ldots, q_{12}) = (e^{i\theta_1 + i n_1 \theta_2} q_1, \ldots, e^{i\theta_1 + i n_{12} \theta_2} q_{12}),$$
where $n_j$ encode the reciprocal-lattice indexing.

The $G_{\mathrm{SM}}$ action (via SO(10) embedding) acts on the amplitude moduli $q_j$ by rotation in the 12-dimensional space.

Commutativity holds because:
\begin{itemize}
\item Phase rotation ($\theta_1$) is U(1)-valued and commutes with all group actions (it is a central element in the symplectic-lift of SO(10)).
\item Reciprocal-lattice shifts ($\theta_2 \to n_j \theta_2$) are kinematic and do not interact with the Hamiltonian structure of the moment map.
\end{itemize}

Therefore, $[\tau_{\mathrm{BCC}}, \mu] = 0$ (the moment map is equivariant with respect to the torus action),
and reduction in stages (first by torus, then by $G_{\mathrm{SM}}$) is equivalent to simultaneous reduction.

Sjamaar-Lerman stratification commutes with all symplectic reductions (it is a purely topological statement about stratification).
\end{proof}

\subsection{Objection α: "Sjamaar-Lerman requires the quotient $\mu^{-1}(0)/G$ to be a manifold or orbifold; does it hold for all strata?""}

\textbf{Critique:} Sjamaar-Lerman says the reduced space is **stratified**, but stratification does not automatically imply each stratum is a manifold.
Could there be obstructions to smoothness on the BCC side that are absent on the SM side?

\textbf{Response:}

By the Sternberg slice theorem, each fixed-point set $(\mu^{-1}(0))^{H_\alpha}$ is a smooth symplectic manifold
(the `slice' transverse to the $H_\alpha$-orbit).
The reduced stratum $S_\alpha^{\mathrm{TECT}} = (\mu^{-1}(0))^{H_\alpha} / Z(H_\alpha)$ is a smooth quotient of a smooth manifold by a Lie group action.
Since $Z(H_\alpha)$ (the centralizer) acts freely on the slice (a standard fact in symplectic-reduction theory), the quotient is again a smooth manifold.

On the SM side, the gauge strata $S_\alpha^{\mathrm{SM}}$ are defined as the loci where the stabilizer subgroup is conjugate to $H_\alpha$.
These are manifolds by the same argument (slices transverse to orbits in the space of gauge connections).

Therefore, smoothness holds on both sides. **DISMISSED.** ✓

\subsection{Objection β: "Lemma \ref{lem:stratum-by-stratum-matching} assumes the equivariant lift \( \Phi_{\mathrm{equ}} \) is injective on fixed-point sets. Is this true?""}

\textbf{Critique:} Math80-AddB proves $\Phi_{\mathrm{equ}}$ is injective globally on $\mu^{-1}(0)$. But restriction to a fixed-point set could become degenerate.
For example, if $\Phi_{\mathrm{equ}}(q, p) = \Phi_{\mathrm{equ}}(q', p')$ for two different points on the $H_\alpha$-fixed locus, injectivity is violated.

\textbf{Response:}

Injectivity on fixed-point sets follows from injectivity on the full set $\mu^{-1}(0)$.
If $(q, p), (q', p') \in (\mu^{-1}(0))^{H_\alpha}$ (both fixed by $H_\alpha$) and
$$\Phi_{\mathrm{equ}}(q, p) = \Phi_{\mathrm{equ}}(q', p'),$$
then by injectivity of $\Phi_{\mathrm{equ}}$ on all of $\mu^{-1}(0)$ (Math80-AddB), we have $(q, p) = (q', p')$.

Restriction of an injective map is injective. **DISMISSED.** ✓

\subsection{Objection γ: "O3 is PROVED CONDITIONAL, but the two conditionals (H1) and (H2) involve unverified sub-lemmas. Could they fail?""}

\textbf{Critique:} Conditional (H1) cites Math75-Q3 §2.1, which is in draft status. Conditional (H2) (Lemma \ref{lem:torus-quotient-commute}) is proved in this note but is a novel claim. If either fails, the theorem collapses.
