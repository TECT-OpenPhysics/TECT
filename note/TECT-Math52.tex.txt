\section{Problem statement}
%=====================================================================

The trust-region Newton--Krylov solver (v2.0--v2.1, Math51) for the
Brazovskii shell equation $R(\Psi) = 0$ used the shell free energy
$F[\Psi]$ for both the Armijo line search and the trust-region
acceptance ratio:
\begin{equation}\label{eq:rho-old}
  \rho = \frac{F(\Psi) - F(\Psi + \alpha s)}
              {-\alpha\,\re\ip{g}{s} - \tfrac12 \alpha^2 \re\ip{s}{Hs}},
\end{equation}
where $g = \operatorname{flatten}(R)$ (projected) and $H = P\,J\,P$
is the projected Jacobian operator.

Runtime Phase~1 tests showed $\rho \approx 0.126$ at every Newton step,
independent of trust radius (down to $\Delta = 10^{-12}$), causing
monotonic collapse of the trust region and non-convergence.

%=====================================================================
\section{Diagnosis: backend consistency audit}
%=====================================================================

A five-test diagnostic suite (\texttt{backend\_consistency\_audit.py})
revealed the following facts.

\begin{proposition}[Test~2: $(R,\,J)$ self-consistency]\label{prop:RJ}
  The code's \texttt{hessian\_vec} is the exact Fr\'echet derivative of
  \texttt{residual}: for all tested perturbation vectors~$v$ and finite
  differences $\varepsilon \in [10^{-7},\,10^{-4}]$,
  \[
    \frac{\bigl\|\tfrac{1}{2\varepsilon}[R(\Psi+\varepsilon v)
      - R(\Psi-\varepsilon v)] - Hv\bigr\|}{\|Hv\|}
    < 3\times 10^{-8}.
  \]
  The pair $(R,\,H = J_R)$ is internally consistent to machine precision.
\end{proposition}

\begin{proposition}[Test~1 + Test~5: $R \neq \nabla F$]\label{prop:RF}
  The residual is \emph{not} the gradient of $F$:
  \begin{enumerate}
    \item[(a)] The ratio
      $\displaystyle C(v) \;=\;
        \frac{dF[\Psi + \varepsilon v]/d\varepsilon\big|_{\varepsilon=0}}
             {\re\ip{R}{v}}$
      varies with the direction~$v$
      ($C \in [-2.81,\; 15.97]$ across 5 random probes).
    \item[(b)] At vanishing amplitude ($\|\Psi\|_\mathrm{rms}
      = 10^{-6}$), the ratio is $C = \dV = (L/N)^3$ exactly.
      At operating amplitude, $C = 5.99 \neq \dV = 7.57$.
  \end{enumerate}
\end{proposition}

\begin{proposition}[Wirtinger factor analysis]\label{prop:wirtinger}
  The direction-dependent ratio arises from distinct Wirtinger
  multiplicities.  For a perturbation $\Psi \to \Psi + \varepsilon v$
  of the discrete functional $F = \dV \sum_x f(\Psi(x))$:
  \begin{align}
    \text{Bilinear terms}\quad (r,\,Z,\,Y,\,\text{family},\,\text{lock}):&\quad
      \frac{dF}{d\varepsilon} = \dV\;\re\ip{R_{\mathrm{lin}}}{v},
      \label{eq:wirt-bilinear}\\[4pt]
    \text{Quartic term}\quad (\lambda\,|\Psi|^2\,\Psi):&\quad
      \frac{dF}{d\varepsilon} = 2\,\dV\;\re\ip{R_{\mathrm{q4}}}{v},
      \label{eq:wirt-quartic}\\[4pt]
    \text{Sextic term}\quad (\gamma\,|\Psi|^4\,\Psi):&\quad
      \frac{dF}{d\varepsilon} = 2\,\dV\;\re\ip{R_{\mathrm{q6}}}{v}.
      \label{eq:wirt-sextic}
  \end{align}
  The factor~2 difference between bilinear and nonlinear terms makes
  $dF/d\varepsilon \neq C\;\re\ip{R}{v}$ for any single constant~$C$.
\end{proposition}

\begin{proof}
  For the bilinear term $F_{\mathrm{quad}} = \dV \cdot \tfrac{r}{2}
  \sum_x |\Psi(x)|^2$:
  \[
    \frac{d}{d\varepsilon}\bigg|_0 F_{\mathrm{quad}}(\Psi+\varepsilon v)
    = \dV \cdot \tfrac{r}{2}\cdot 2\sum_x \re(\bar\Psi\, v)
    = \dV \sum_x \re(\overline{r\Psi}\; v)
    = \dV\;\re\ip{R_{\mathrm{quad}}}{v}.
  \]
  For the quartic term $F_{\mathrm{q4}} = \dV \cdot \tfrac{\lambda}{2}
  \sum_x \rho^2$, where $\rho = |\Psi|^2$:
  \[
    \frac{d\rho^2}{d\varepsilon}\bigg|_0
    = 2\rho\cdot 2\re(\bar\Psi\, v) = 4\rho\,\re(\bar\Psi\, v).
  \]
  Hence
  $\frac{d}{d\varepsilon}\big|_0 F_{\mathrm{q4}}
  = \dV\cdot\tfrac{\lambda}{2}\cdot 4\sum_x\rho\,\re(\bar\Psi\, v)
  = 2\,\dV\;\re\ip{\lambda\rho\Psi}{v}
  = 2\,\dV\;\re\ip{R_{\mathrm{q4}}}{v}$.
  The sextic follows identically.
\end{proof}

\begin{remark}
  Since $\lambda = -0.43 < 0$, the nonlinear contribution $\re\ip{R_\mathrm{nl}}{v}$
  can oppose $\re\ip{R_\mathrm{lin}}{v}$.  Setting $a = \re\ip{R_\mathrm{lin}}{v}$,
  $b = \re\ip{R_\mathrm{nl}}{v}$, the effective ratio is
  $C_{\mathrm{eff}} = \dV\,(a + 2b)/(a + b)$.  With $b < 0$ and
  $|b| < |a|$, we obtain $C_{\mathrm{eff}} < \dV$, consistent
  with the observed $C = 5.99 < 7.57 = \dV$.
\end{remark}

%=====================================================================
\section{Fix~1: Merit-function trust-region}
%=====================================================================

Since $(R,\,J)$ is self-consistent (Prop.~\ref{prop:RJ}) but
$(F,\,R)$ is not (Prop.~\ref{prop:RF}), we replace $F$ with the
natural merit function for the root-finding problem $R = 0$:
\begin{equation}\label{eq:merit}
  m(\Psi) \;=\; \tfrac12\,\bigl\|R_{\mathrm{proj}}(\Psi)\bigr\|^2.
\end{equation}

\subsection{Predicted reduction}

Under the linear model $R(\Psi + \alpha s) \approx g + \alpha\,Hs$,
the merit model is
\[
  m_{\mathrm{model}}(\alpha)
  = \tfrac12\,\|g + \alpha\,Hs\|^2
  = \tfrac12\|g\|^2 + \alpha\,\re\ip{g}{Hs}
    + \tfrac12\,\alpha^2\,\|Hs\|^2.
\]
The predicted reduction is
\begin{equation}\label{eq:pred}
  \Delta m_{\mathrm{pred}}(\alpha)
  = m_{\mathrm{old}} - m_{\mathrm{model}}(\alpha)
  = -\alpha\,\re\ip{g}{Hs} - \tfrac12\,\alpha^2\,\|Hs\|^2.
\end{equation}

\subsection{Trust-region ratio}

\begin{equation}\label{eq:rho-new}
  \rho = \frac{m_{\mathrm{old}} - m_{\mathrm{trial}}}
              {\Delta m_{\mathrm{pred}}(\alpha)},
\end{equation}
where $m_{\mathrm{trial}} = \tfrac12\|R_{\mathrm{proj}}(\Psi + \alpha\,s)\|^2$.

\begin{lemma}
  $\rho \to 1$ as $\alpha \to 0^+$, provided $J = DR$ exactly
  (Prop.~\ref{prop:RJ}).
\end{lemma}
\begin{proof}
  $R(\Psi + \alpha s) = g + \alpha\,Js + O(\alpha^2)$.  Since $H = PJP$
  and $g,\,s$ are projected, $m_{\mathrm{trial}} = m_{\mathrm{model}}(\alpha)
  + O(\alpha^2)$, so $\rho = 1 + O(\alpha)$.
\end{proof}

\subsection{Armijo condition}

The directional derivative of the merit along the step direction is
\[
  \frac{dm}{d\alpha}\bigg|_{\alpha=0}
  = \re\ip{g}{Hs} < 0
\]
for a successful Newton step (where $Hs \approx -g$).  The Armijo
sufficient-decrease condition is
\begin{equation}\label{eq:armijo}
  m_{\mathrm{trial}} \;\le\; m_{\mathrm{old}}
    + c_1\,\alpha\,\re\ip{g}{Hs},
  \qquad c_1 = 10^{-4}.
\end{equation}

\section{Fix~2: GMRES inner solve}
%=====================================================================

\begin{proposition}[Test~6: Jacobian asymmetry]\label{prop:asym}
  The Jacobian $J = DR$ satisfies
  \[
    \max_{u,v}\;
    \frac{|\re\ip{u}{Jv} - \re\ip{v}{Ju}|}
         {\max(|\re\ip{u}{Jv}|,\;|\re\ip{v}{Ju}|)}
    = 1.08\times 10^{-2}.
  \]
  The asymmetry source is the Class~II sector's numerical Fr\'echet
  derivative ($\varepsilon_{\mathrm{FD}} = 5\times 10^{-7}$, central
  difference).  All other terms (bilinear, quartic, sextic, family,
  lock, shell bias) are analytically symmetric.
\end{proposition}

\begin{proof}[Proof of bilinear/nonlinear symmetry]
  For the quartic Jacobian $J_{\mathrm{q4}}[v] = \lambda(\rho\,v +
  \delta\rho\;\Psi)$ with $\delta\rho = 2\re(\bar\Psi\,v)$,
  the real inner product $\re\ip{u}{J_{\mathrm{q4}}[v]}
  = \lambda\bigl[\sum\rho\,\re(\bar u\,v)
  + 2\sum\re(\bar\Psi\,v)\,\re(\bar u\,\Psi)\bigr]$ is manifestly
  symmetric under $u\leftrightarrow v$.  Identical structure holds
  for the sextic.  Linear operators ($r - Z\Delta + Y\Delta^2$,
  $M_{\mathrm{fam}}$, $k_{\mathrm{lock}}(I-P_0)$) are self-adjoint.
\end{proof}

Since $J$ is non-symmetric at the $\sim 1\%$ level, CG (which requires
$H = H^T$) is replaced by restarted GMRES (restart parameter~$=50$)
as the default inner Krylov solver.  CG is retained as a fallback via
\texttt{--krylov cg}.

GMRES trust-region enforcement uses simple boundary clipping: if
$\|s_{\mathrm{GMRES}}\| > \Delta$, the solution is rescaled to the
boundary.  This is geometrically suboptimal compared to Steihaug--Toint's
trajectory-based boundary tracking, but the merit-based accept/reject
mechanism~\eqref{eq:rho-new} provides robust globalization regardless.

Key observations:
\begin{enumerate}
  \item $\rho = 0.998 \to 1.000$: the merit-based predicted reduction
    matches the actual reduction to $<0.2\%$ from step~0 onward.
    The previous $\rho \approx 0.126$ pathology is fully resolved.
  \item Superquadratic convergence:
    $\|R\|$ drops 21~orders of magnitude in 5~Newton steps.
  \item $F \to 1.26\times 10^{-14}$: the stationary point $R = 0$
    is also near-stationary for~$F$, confirming physical relevance.
  \item Trust radius expands ($18.2 \to 72.9$), contrasting the
    v2.1 monotonic collapse to $10^{-12}$.
\end{enumerate}

\begin{itemize}
  \item \textbf{Phase~1 (Existence)}: Now converges.
    The solver finds $\Psi^*$ with $\|R(\Psi^*)\| < \mathrm{tol}$.
  \item \textbf{Phase~2 (Stability)}: Unaffected.
    Projected Lanczos uses $J = DR$, which is self-consistent.
  \item \textbf{Phase~3 (Energetics)}: $F(\Psi^*) \approx 0$ observed;
    $\Delta F = F(\Psi^*) - F(0)$ remains well-defined.
  \item \textbf{Phase~4 (Continuum)}: Unaffected.
    $m^{*2}$ is extracted from $J$-spectrum, not $F$.
\end{itemize}
