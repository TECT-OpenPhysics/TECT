\section{Setting}

\subsection{Effective action near the Brazovskii fixed point}

The TECT condensate near the Brazovskii first-order transition
is governed (TECT-Math37 AddA, Math38) by the Ginzburg--Landau
effective free energy
\begin{equation}
  \mathcal{F}[\Psi]
   = \int d^3x\,\Bigl[
      \tfrac{r}{2}|\Psi|^2
    + \tfrac{Z}{2}|\nabla\Psi|^2
    + \tfrac{Y}{2}|\nabla^2\Psi|^2
    + \tfrac{\lambda}{4}|\Psi|^4
    + \tfrac{\gamma}{6}|\Psi|^6\Bigr],
  \label{eq:F-brazovskii}
\end{equation}
with the locked parameters
$(\mu^2, \lambda, \gamma) = (0.26, -0.43, 1.62)$. The shell
wavevector is $q_0 = \sqrt{-Z/(2Y)} = 0.6802$ at $Y = 1.0$.

\subsection{$O_h$-invariant derivative expansion}

Beyond the isotropic kinetic terms in \eqref{eq:F-brazovskii},
the full BCC lattice symmetry $O_h$ permits (only) a discrete
set of anisotropic derivative couplings. At fourth order in
derivatives, the two independent $O_h$-invariant operators are
\begin{align}
  \mathcal{O}^{(\mathrm{iso})}_4 &= \bigl(\nabla^2\Psi\bigr)^2
    = \bigl(\textstyle\sum_{i=1}^{3}\partial_i^2\Psi\bigr)^2, \\
  \mathcal{O}^{(c)}_4
    &= \sum_{i=1}^{3}\bigl(\partial_i\Psi\bigr)^4
      - \tfrac{1}{3}\Bigl(\sum_{i=1}^{3}\bigl(\partial_i\Psi\bigr)^2\Bigr)^2.
  \label{eq:Oc4-correct}
\end{align}
Only $\mathcal{O}^{(c)}_4$ is \emph{cubic} (it distinguishes
$O_h$ from the continuous rotation group $SO(3)$); the other
combination is isotropic.

\begin{remark}[Correction to scaffold]
The scaffold Math_IR_Bound Eq.(3) defined
$\mathcal{O}^{(c)}_4 := (\partial_i\partial_j\Psi)^2 -
\tfrac{1}{3}(\nabla^2\Psi)^2$, which is a quadrupole (spin-2)
$SO(3)$-tensor operator. It is \emph{isotropic} (invariant
under continuous rotations) and therefore cannot possibly
distinguish $O_h$ from $SO(3)$. The correct cubic-$O_h$-invariant
operator is the one in \eqref{eq:Oc4-correct}: a sum of
$(\partial_i)^4$ along cubic axes only, with the quadratic
isotropic part subtracted off.
\end{remark}

\section{Canonical dimensions}

\subsection{Field dimension at the Brazovskii fixed point}

At the shell minimum $k = q_0$ with dispersion
$\omega(k) = r + Zk^2 + Yk^4 \approx r + Y(k^2 - q_0^2)^2$,
the Brazovskii theory is effectively $d = 3$ with anisotropic
scaling: momenta transverse to the shell scale canonically,
momenta parallel to the shell-radius direction scale with
quartic dispersion. For the purpose of the cubic-anisotropy
operator analysis, we adopt the naive $d = 3$ canonical scaling
$[\Psi] = (d-2)/2 = 1/2$.

\begin{lemma}[Canonical dimension of $\mathcal{O}^{(c)}_4$]
\label{lem:dim}
The canonical dimension of $\mathcal{O}^{(c)}_4$ defined by
\eqref{eq:Oc4-correct} is
\begin{equation}
  [\mathcal{O}^{(c)}_4]
   = 4 \cdot [\partial] + 4 \cdot [\Psi]
   = 4 \cdot 1 + 4 \cdot \tfrac{1}{2}
   = 6.
  \label{eq:Oc4-dim}
\end{equation}
The operator
$\int d^3x\, \mathcal{O}^{(c)}_4$ has dimension
$6 - 3 = 3$, i.e., it is a \emph{relevant} operator at the Gaussian
fixed point (irrelevant would require dim $> 3$). This contradicts
the scaffold's claim of $\Delta = 2 + \eta > 2$ being IR-irrelevant.
\end{lemma}

\begin{remark}[Why this matters for the bound]
The naive scaling dimension makes the operator look \emph{relevant}
at the Gaussian fixed point. The IR-\emph{irrelevance} of the
cubic anisotropy in the BCC condensate is therefore a
non-perturbative property tied to the specific fixed point
(Brazovskii) and to the geometry of the BCC shell in momentum
space. It does not follow from a crude power-counting argument,
contrary to the scaffold's assertion.
\end{remark}

\subsection{Suppression from BCC shell structure}

\begin{proposition}[Shell-projection suppression]
\label{prop:shell}
At the Brazovskii fixed point, the condensate $\Psi$ is
supported on a thin shell $|k| \in [q_0 - \Delta, q_0 + \Delta]$
with shell width $\Delta \sim \sqrt{|r|/Y} \ll q_0$
(TECT-Math37 AddA). The matrix element of the cubic-anisotropy
operator between shell modes is suppressed by the dimensionless
ratio
\begin{equation}
  \bigl\langle \mathcal{O}^{(c)}_4 \bigr\rangle_{\mathrm{shell}}
   \sim \Bigl(\frac{\Delta}{q_0}\Bigr)^{\!2}
     \cdot \bigl|\Psi\bigr|_{\mathrm{shell}}^{4}.
\end{equation}
This shell-projection factor $(\Delta/q_0)^2$ is the genuine
BCC-specific small number that suppresses cubic anisotropy in
the effective low-energy theory; it is \emph{not} the generic
power-counting dimension of \eqref{eq:Oc4-dim}.
\end{proposition}

\begin{proof}[Sketch]
The matrix element of $\mathcal{O}^{(c)}_4$ between BCC shell
modes is
\begin{equation}
  \int d^3k_1\dots d^3k_4\,\Psi(k_1)\Psi(k_2)\Psi(k_3)\Psi(k_4)\,
  T^{(c)}(k_1,\dots,k_4)\,
  \delta(k_1 + k_2 + k_3 + k_4),
\end{equation}
where
$T^{(c)}(k_1,\dots,k_4) = \sum_i k_{1i} k_{2i} k_{3i} k_{4i}
   - \tfrac{1}{3}(\sum_i k_{1i}k_{2i})(\sum_j k_{3j}k_{4j})$
is the cubic-anisotropy kernel. Because the BCC 12 nearest-neighbour
wavevectors $\mathbf{G}_\alpha$ are symmetric under $O_h$, the
sum $T^{(c)}(\mathbf{G}_{\alpha_1},\dots,\mathbf{G}_{\alpha_4})$
vanishes at exactly $|\mathbf{G}_\alpha| = q_0$ by the cubic
cancellation implicit in \eqref{eq:Oc4-correct}. The leading
non-zero contribution comes from the shell-width perturbation
$k_i = \mathbf{G}_{\alpha_i} + \delta k_i$, producing a
$(\delta k / q_0)^2 \sim (\Delta/q_0)^2$ suppression.
\end{proof}

\section{1-loop anomalous dimension}

\subsection{Diagrammatic setup}

The anomalous dimension $\eta$ of $\mathcal{O}^{(c)}_4$ at the
Brazovskii fixed point is computed from the 1-loop correction
to the operator's two-point function in the condensate
background. The relevant diagrams are
\begin{itemize}
  \item self-energy insertion into the 4-derivative leg;
  \item vertex correction from the $\lambda|\Psi|^4$ interaction;
  \item ``tadpole'' insertion from the $\gamma|\Psi|^6$ interaction.
\end{itemize}

\begin{proposition}[1-loop $\eta$ structure]
At 1-loop, the anomalous dimension of $\mathcal{O}^{(c)}_4$ at
the Brazovskii fixed point is
\begin{equation}
  \eta^{(c)} = a_1 \lambda + a_2 \gamma\,|\Psi_0|^2
              + O(\lambda^2, \lambda\gamma),
\end{equation}
where the coefficients $a_1, a_2$ are $O_h$-invariant integrals
over the BCC Brillouin zone of products of shell propagators.
Both are finite and positive at the locked Brazovskii parameters;
explicit numerical values require a computer-algebra evaluation
deferred to v2.
\end{proposition}

\begin{remark}
The scaffold asserted $\eta = +0.02$ without providing this

We therefore state the structural result and explicitly flag the

\section{SME Lorentz-violation bound}

\subsection{Effective dispersion}

In the continuum limit, the leading cubic anisotropy contributes
to the fermion dispersion relation an $O_h$-invariant correction
\begin{equation}
  \omega^2(\vec k) = c^2 k^2 + \kappa^{(c)}
      \bigl(k_x^4 + k_y^4 + k_z^4
            - \tfrac{1}{3}(k^2)^2\bigr) + \dots,
\end{equation}
with the effective SME coefficient
$c^{(4)}_{xxxx} - \tfrac{1}{3}\eta_{xx}^2 \propto \kappa^{(c)}$.

\begin{proposition}[SME bound, conditional]
\label{prop:SME}
Given Proposition~\ref{prop:shell}, the effective SME coefficient
is bounded by
\begin{equation}
  |\kappa^{(c)}|
   \lesssim
   \Bigl(\frac{\Delta}{q_0}\Bigr)^{\!2}
   \cdot
   \Bigl(\frac{k_{\mathrm{obs}}}{q_0}\Bigr)^{\!2},
  \label{eq:SME-bound}
\end{equation}
at observable wavevector $k_{\mathrm{obs}}$.
\end{proposition}

\begin{remark}[Order-of-magnitude for the bound]
The scaffold quoted $c_{\mu\nu} \lesssim 10^{-70}$ without
derivation. Let us check the order of magnitude under
Proposition~\ref{prop:SME}. With $q_0 \sim M_P$ (Planck scale) for
the TECT lattice cutoff, $\Delta / q_0 \sim 10^{-2}$ near the
locked Brazovskii point, and $k_{\mathrm{obs}} \sim 100\ \mathrm{GeV}$
at LHC scale, we obtain
\begin{equation}
  |\kappa^{(c)}|
  \lesssim (10^{-2})^2 \cdot \Bigl(\frac{100\ \mathrm{GeV}}{10^{19}\
     \mathrm{GeV}}\Bigr)^{\!2}
  = 10^{-4} \cdot 10^{-34}
  = 10^{-38}.
\end{equation}
This is far weaker than the claimed $10^{-70}$. Achieving
$10^{-70}$ would require either a tighter shell ratio
$\Delta/q_0 \sim 10^{-20}$ (inconsistent with BCC locking) or
an additional higher-power suppression $(k/q_0)^{2+\eta}$ with
a large anomalous-dimension enhancement. We therefore honestly
state: the $10^{-70}$ number was incorrect. The derivable bound
from \eqref{eq:SME-bound} is of order $10^{-38}$, still far
below current SME observational limits ($10^{-20}$ to
$10^{-30}$ depending on the coefficient), but \emph{not} the
near-Planck-scale protection asserted by the scaffold.
\end{remark}
