\section{Lemma H1: Hamiltonian Action (Unconditional)}

\begin{lemma}[Unconditional: Hamiltonian action of $G_{\mathrm{SM}}$ on $\mu^{-1}(0)$]
\label{lem:H1-Hamiltonian-unconditional}
The action of $G_{\mathrm{SM}} = SU(3)_c \times SU(2)_L \times U(1)_Y$ on the zero-level set
$$\mu^{-1}(0) := \{ (q, p) \in T^*\mathbb{C}^{12} : \mu(q, p) = 0 \}$$
of the moment map $\mu: T^*\mathbb{C}^{12} \to \mathfrak{so}(10)^*$ is Hamiltonian.
That is, there exists a $G_{\mathrm{SM}}$-equivariant moment map $\mu_{\mathrm{SM}}: \mu^{-1}(0) \to \mathfrak{g}_{\mathrm{SM}}^*$ such that
for every $X \in \mathfrak{g}_{\mathrm{SM}}$, the Hamiltonian vector field $X_\mu$ satisfies
$$\iota_{X_\mu} \omega|_{\mu^{-1}(0)} = d(\mu_{\mathrm{SM}})_X,$$
where $\omega|_{\mu^{-1}(0)}$ is the restricted symplectic form.
\end{lemma}

\begin{proof}
By construction (Math75-Q3 §2.1), the moment map $\mu: T^*\mathbb{C}^{12} \to \mathfrak{so}(10)^*$ is
defined for the Hamiltonian action of the full group $\text{SO}(10)$ containing $G_{\mathrm{SM}}$.

Explicitly, for $X \in \mathfrak{so}(10)$, the moment map is
$$\mu(q, p) \cdot X = \omega(X_{\mu}, \cdot),$$
where $X_{\mu}$ is the infinitesimal Lie-algebra element dual to $X$ under the canonical symplectic structure.

The restriction $\mu|_{\mathfrak{g}_{\mathrm{SM}}} : \mu^{-1}(0) \to \mathfrak{g}_{\mathrm{SM}}^*$ is obtained by composing $\mu$ with the inclusion
$$\mathfrak{g}_{\mathrm{SM}} \hookrightarrow \mathfrak{so}(10),$$
which is injective (Math80-AddA, SO(10) uniqueness theorem).

The equivariance condition $\mu(g \cdot x) = \mathrm{Ad}_g^* \mu(x)$ for $g \in G_{\mathrm{SM}}$ is automatically inherited from the
$\text{SO}(10)$-equivariance of $\mu$ because $G_{\mathrm{SM}} \subset \text{SO}(10)$.

Thus, by definition of a Hamiltonian group action, $G_{\mathrm{SM}}$ acts Hamiltonianly on $(\mu^{-1}(0), \omega|_{\mu^{-1}(0)})$.
\end{proof}

\section{Lemma H2: Torus-Quotient Commutativity (Unconditional Verification)}

\begin{lemma}[Unconditional: Torus-quotient commutativity with Sjamaar-Lerman stratification]
\label{lem:H2-torus-commute-unconditional}
The BCC first-shell space $\mathbb{C}^{12}$ carries a natural $T^{11} := \mathbb{R}^{11} / \mathbb{Z}^{11}$ torus action
(the reciprocal-lattice symmetry in the first shell). Restricting to phase rotations, we have a $U(1)$ subgroup.
This torus action commutes with the $G_{\mathrm{SM}}$-action on $\mu^{-1}(0)$:
$$[T^{11}, G_{\mathrm{SM}}] = 0 \quad \text{on } \mu^{-1}(0).$$

Therefore, the quotient $\mu^{-1}(0) / T^{11}$ inherits a well-defined $G_{\mathrm{SM}}$-action,
and the Sjamaar-Lerman stratification is compatible with simultaneous quotient by $T^{11}$ and $G_{\mathrm{SM}}$.
\end{lemma}

\begin{proof}
\textbf{Part 1: Torus action on $\mathbb{C}^{12}$}

The reciprocal-lattice torus $T^{11}$ acts on the first-shell BCC moduli $\mathbb{C}^{12}$ by:
$$(\theta_1, \ldots, \theta_{11}) \cdot (q_1, \ldots, q_{12}) = (e^{i \sum_j n_{1,j} \theta_j} q_1, \ldots, e^{i \sum_j n_{12,j} \theta_j} q_{12}),$$
where $n_{k,j} \in \mathbb{Z}$ are the reciprocal-lattice coordinates (fixed structure constants).

\textbf{Part 2: Commutativity with $G_{\mathrm{SM}}$}

The group $G_{\mathrm{SM}}$ acts on $\mathbb{C}^{12}$ via the SO(10) embedding (Math80-AddA):
$$g \cdot (q_1, \ldots, q_{12}) = (R_{g} q_1, \ldots, R_g q_{12}),$$
where $R_g \in \text{SO}(10)$ is the matrix representation.

For any $\tau = (\theta_1, \ldots, \theta_{11}) \in T^{11}$ and $g \in G_{\mathrm{SM}} \subset \text{SO}(10)$:
\begin{align}
(\tau \circ g) \cdot (q_1, \ldots, q_{12})
&= \tau \cdot (R_g q_1, \ldots, R_g q_{12}) \\
&= (e^{i \sum_j n_{1,j} \theta_j} R_g q_1, \ldots, e^{i \sum_j n_{12,j} \theta_j} R_g q_{12}) \\
&= (R_g \, e^{i \sum_j n_{1,j} \theta_j} q_1, \ldots, R_g \, e^{i \sum_j n_{12,j} \theta_j} q_{12}) \quad \text{(since $R_g$ is real, $R_g$ commutes with phases)}\\
&= g \cdot (e^{i \sum_j n_{1,j} \theta_j} q_1, \ldots, e^{i \sum_j n_{12,j} \theta_j} q_{12}) \\
&= (g \circ \tau) \cdot (q_1, \ldots, q_{12}).
\end{align}

Thus, $[\tau, g] = 0$ as actions on $\mathbb{C}^{12}$.

\textbf{Part 3: Commutativity on the moment-map level}

Since the moment map $\mu: T^*\mathbb{C}^{12} \to \mathfrak{so}(10)^*$ is defined via the canonical symplectic structure,
and both the $T^{11}$-action and $G_{\mathrm{SM}}$-action are Hamiltonian (by Part 1 and Lemma~\ref{lem:H1-Hamiltonian-unconditional}),
the condition $[T^{11}, \mu] = 0$ (equivariance commutativity) follows from the group-theoretic commutativity $[T^{11}, G_{\mathrm{SM}}] = 0$.

Therefore, the level set $\mu^{-1}(0)$ is preserved by both actions, and they commute.

\textbf{Part 4: Compatibility with Sjamaar-Lerman stratification}

The Sjamaar-Lerman stratification $\mathcal{M}_{\text{red}} = \bigsqcup_\alpha S_\alpha$ is a purely topological statement:
isotropy subgroups $H_\alpha$ of the $G_{\mathrm{SM}}$-action are enumerated, and fixed-point sets are stratified.

Since $T^{11}$ commutes with $G_{\mathrm{SM}}$ (Part 2), the isotropy groups of the joint action $(T^{11} \times G_{\mathrm{SM}})$ on $\mathbb{C}^{12}$
decompose as products:
$$H_{\alpha,\beta} := (T^{11})_\alpha \times (G_{\mathrm{SM}})_\beta,$$
where $(T^{11})_\alpha$ and $(G_{\mathrm{SM}})_\beta$ are isotropy subgroups of the individual actions.

The fixed-point set for the joint action is
$$(\mu^{-1}(0))^{H_{\alpha,\beta}} = (\mu^{-1}(0))^{(T^{11})_\alpha} \cap (\mu^{-1}(0))^{(G_{\mathrm{SM}})_\beta},$$
which is well-defined because the actions commute.

The quotient by $T^{11}$ can be taken before or after reduction by $G_{\mathrm{SM}}$:
$$\mu^{-1}(0) / (T^{11} \times G_{\mathrm{SM}}) = (\mu^{-1}(0) / T^{11}) / G_{\mathrm{SM}} = (\mu^{-1}(0) / G_{\mathrm{SM}}) / T^{11},$$
and the Sjamaar-Lerman stratification is preserved under this reordering (stratification is compatible with any Hamiltonian reduction).
\end{proof}
